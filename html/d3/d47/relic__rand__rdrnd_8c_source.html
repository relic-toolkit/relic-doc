45 void rand_bytes(uint8_t *buf, int size) {
46  int i = 0, j;
47  unsigned long long r;
48 
49  while (i < size) {
50 #ifdef __RDRND__
51  while (_rdrand64_step(&r) == 0);
52 #else
53 #error "RdRand not available, check your compiler settings."
54 #endif
55  for (j = 0; i < size && j < sizeof(ull_t); i++, j++) {
56  buf[i] = r & 0xFF;
57  }
58  }
59 }
