39void ep2_pck(ep2_t r, const ep2_t p) {
40 bn_t halfQ, yValue;
41
42 bn_null(halfQ);
43 bn_null(yValue);
44
45 RLC_TRY {
46 bn_new(halfQ);
47 bn_new(yValue);
48
49 halfQ->used = RLC_FP_DIGS;
50 dv_copy(halfQ->dp, fp_prime_get(), RLC_FP_DIGS);
51 bn_hlv(halfQ, halfQ);
52
53 fp_prime_back(yValue, p->y[1]);
54
55 int b = bn_cmp(yValue, halfQ) == RLC_GT;
56
57 fp2_copy(r->x, p->x);
58 fp2_zero(r->y);
59 fp_set_bit(r->y[0], 0, b);
60 fp_zero(r->y[1]);
61 fp_set_dig(r->z[0], 1);
62 fp_zero(r->z[1]);
63 r->coord = BASIC;
64 } RLC_CATCH_ANY {
65 RLC_THROW(ERR_CAUGHT);
66 }
67 RLC_FINALLY {
68 bn_free(yValue);
69 bn_free(halfQ);
70 }
71}

73int ep2_upk(ep2_t r, const ep2_t p) {
74 fp2_t t;
75 bn_t halfQ;
76 bn_t yValue;
77 int result = 0;
78
79 fp2_null(t);
80 bn_null(halfQ);
81 bn_null(yValue);
82
83 RLC_TRY {
84 fp2_new(t);
85 bn_new(halfQ);
86 bn_new(yValue);
87
88 ep2_rhs(t, p->x);
89
90 /* t0 = sqrt(x1^3 + a * x1 + b). */
91 result = fp2_srt(t, t);
92
93 if (result) {
94 /* Verify whether the y coordinate is the larger one, matches the
95 * compressed y-coordinate (IETF pairing friendly spec)
96 * sign_F_p^2(y') := { sign_F_p(y'_0) if y'_1 equals 0, else
97 * { 1 if y'_1 > (p - 1) / 2, else
98 * { 0 otherwise.
99 *
100 */
101 halfQ->used = RLC_FP_DIGS;
102 dv_copy(halfQ->dp, fp_prime_get(), RLC_FP_DIGS);
103 bn_hlv(halfQ, halfQ);
104
105 fp_prime_back(yValue, t[1]);
106
107 if (bn_is_zero(yValue)) {
108 fp_prime_back(yValue, t[0]);
109 }
110
111 int sign_fp2y = bn_cmp(yValue, halfQ) == RLC_GT;
112
113 if (sign_fp2y != fp_get_bit(p->y[0], 0)) {
114 fp2_neg(t, t);
115 }
116 fp2_copy(r->x, p->x);
117 fp2_copy(r->y, t);
118 fp_set_dig(r->z[0], 1);
119 fp_zero(r->z[1]);
120 r->coord = BASIC;
121 }
122 }
123 RLC_CATCH_ANY {
124 RLC_THROW(ERR_CAUGHT);
125 }
126 RLC_FINALLY {
127 fp2_free(t);
128 bn_free(yValue);
129 bn_free(halfQ);
130 }
131 return result;
132}
