42inline static void fp9_mul_dxs_unr_lazyr(dv9_t c, const fp9_t a, const fp9_t b) {
43 dv3_t u0, u1, u2, u3;
44 fp3_t t0, t1;
45
46 dv3_null(u0);
47 dv3_null(u1);
48 dv3_null(u2);
49 dv3_null(u3);
50 fp3_null(t0);
51 fp3_null(t1);
52
53 RLC_TRY {
54 dv3_new(u0);
55 dv3_new(u1);
56 dv3_new(u2);
57 dv3_new(u3);
58 fp3_new(t0);
59 fp3_new(t1);
60
61 fp3_muln_low(u0, a[0], b[0]);
62 fp3_muln_low(u1, a[1], b[1]);
63 fp3_addm_low(t0, a[0], a[1]);
64 fp3_addm_low(t1, b[0], b[1]);
65
66 /* c_1 = (a_0 + a_1)(b_0 + b_1) - a_0b_0 - a_1b_1 */
67 fp3_muln_low(u2, t0, t1);
68 fp3_subc_low(u2, u2, u0);
69 fp3_subc_low(c[1], u2, u1);
70
71 /* c_0 = a_0b_0 + E a_2b_1 */
72 fp3_muln_low(u2, a[2], b[1]);
73 fp3_nord_low(c[0], u2);
74 fp3_addc_low(c[0], u0, c[0]);
75
76 /* c_2 = a_0b_2 + a_1b_1 */
77 fp3_muln_low(u2, a[2], b[0]);
78 fp3_addc_low(c[2], u1, u2);
79 } RLC_CATCH_ANY {
80 RLC_THROW(ERR_CAUGHT);
81 } RLC_FINALLY {
82 dv3_free(u0);
83 dv3_free(u1);
84 dv3_free(u2);
85 dv3_free(u3);
86 fp3_free(t0);
87 fp3_free(t1);
88 }
89}

99void fp18_mul_basic(fp18_t c, const fp18_t a, const fp18_t b) {
100 fp9_t t0, t1, t2;
101
102 fp9_null(t0);
103 fp9_null(t1);
104 fp9_null(t2);
105
106 RLC_TRY {
107 fp9_new(t0);
108 fp9_new(t1);
109 fp9_new(t2);
110
111 /* Karatsuba algorithm. */
112
113 /* t0 = a_0 * b_0. */
114 fp9_mul(t0, a[0], b[0]);
115 /* t1 = a_1 * b_1. */
116 fp9_mul(t1, a[1], b[1]);
117 /* t2 = b_0 + b_1. */
118 fp9_add(t2, b[0], b[1]);
119
120 /* c_1 = a_0 + a_1. */
121 fp9_add(c[1], a[0], a[1]);
122
123 /* c_1 = (a_0 + a_1) * (b_0 + b_1) */
124 fp9_mul(c[1], c[1], t2);
125 fp9_sub(c[1], c[1], t0);
126 fp9_sub(c[1], c[1], t1);
127
128 /* c_0 = a_0b_0 + v * a_1b_1. */
129 fp9_mul_art(t1, t1);
130 fp9_add(c[0], t0, t1);
131 } RLC_CATCH_ANY {
132 RLC_THROW(ERR_CAUGHT);
133 } RLC_FINALLY {
134 fp9_free(t0);
135 fp9_free(t1);
136 fp9_free(t2);
137 }
138}

140void fp18_mul_dxs_basic(fp18_t c, const fp18_t a, const fp18_t b) {
141 fp9_t t0, t1, t2;
142
143 fp9_null(t0);
144 fp9_null(t1);
145 fp9_null(t2);
146
147 RLC_TRY {
148 fp9_new(t0);
149 fp9_new(t1);
150 fp9_new(t2);
151
152 /* Karatsuba algorithm. */
153
154 /* t0 = a_0 * b_0. */
155 fp9_mul_dxs(t0, a[0], b[0]);
156#if EP_ADD == BASIC
157 /* t1 = a_1 * b_1. */
158 fp_mul(t2[0][0], a[1][2][0], b[1][1][0]);
159 fp_mul(t2[0][1], a[1][2][1], b[1][1][0]);
160 fp3_mul_nor(t1[0], t2[0]);
161 fp_mul(t1[1][0], a[1][0][0], b[1][1][0]);
162 fp_mul(t1[1][1], a[1][0][1], b[1][1][0]);
163 fp_mul(t1[2][0], a[1][1][0], b[1][1][0]);
164 fp_mul(t1[2][1], a[1][1][1], b[1][1][0]);
165 /* t2 = b_0 + b_1. */
166 fp3_copy(t2[0], b[0][0]);
167 fp_add(t2[1][0], b[0][1][0], b[1][1][0]);
168 fp_copy(t2[1][1], b[0][1][1]);
169#elif EP_ADD == PROJC || EP_ADD == JACOB
170 /* t1 = a_1 * b_1. */
171 fp3_mul(t2[0], a[1][2], b[1][1]);
172 fp3_mul_nor(t1[0], t2[0]);
173 fp3_mul(t1[1], a[1][0], b[1][1]);
174 fp3_mul(t1[2], a[1][1], b[1][1]);
175 /* t2 = b_0 + b_1. */
176 fp3_copy(t2[0], b[0][0]);
177 fp3_add(t2[1], b[0][1], b[1][1]);
178#endif
179 /* c_1 = a_0 + a_1. */
180 fp9_add(c[1], a[0], a[1]);
181
182 /* c_1 = (a_0 + a_1) * (b_0 + b_1) - a_0 * b_0 - a_1 * b_1. */
183 fp9_mul_dxs(c[1], c[1], t2);
184 fp9_sub(c[1], c[1], t0);
185 fp9_sub(c[1], c[1], t1);
186
187 /* c_0 = a_0b_0 + v * a_1b_1. */
188 fp9_mul_art(t1, t1);
189 fp9_add(c[0], t0, t1);
190 } RLC_CATCH_ANY {
191 RLC_THROW(ERR_CAUGHT);
192 } RLC_FINALLY {
193 fp9_free(t0);
194 fp9_free(t1);
195 fp9_free(t2);
196 }
197}

203void fp18_mul_unr(dv18_t c, const fp18_t a, const fp18_t b) {
204 fp9_t t0, t1;
205 dv9_t u0, u1, u2, u3;
206
207 dv9_null(u0);
208 dv9_null(u1);
209 dv9_null(u2);
210 dv9_null(u3);
211 fp9_null(t0);
212 fp9_null(t1);
213
214 RLC_TRY {
215 dv9_new(u0);
216 dv9_new(u1);
217 dv9_new(u2);
218 dv9_new(u3);
219 fp9_new(t0);
220 fp9_new(t1);
221
222 /* Karatsuba algorithm. */
223
224 /* u0 = a_0 * b_0. */
225 fp9_mul_unr(u0, a[0], b[0]);
226 /* u1 = a_1 * b_1. */
227 fp9_mul_unr(u1, a[1], b[1]);
228 /* t1 = a_0 + a_1. */
229 fp9_add(t0, a[0], a[1]);
230 /* t0 = b_0 + b_1. */
231 fp9_add(t1, b[0], b[1]);
232 /* u2 = (a_0 + a_1) * (b_0 + b_1) */
233 fp9_mul_unr(u2, t0, t1);
234 /* c_1 = u2 - a_0b_0 - a_1b_1. */
235 for (int i = 0; i < 3; i++) {
236 fp3_addc_low(u3[i], u0[i], u1[i]);
237 fp3_subc_low(c[1][i], u2[i], u3[i]);
238 }
239 /* c_0 = a_0b_0 + v * a_1b_1. */
240 fp3_nord_low(u2[0], u1[2]);
241 fp3_addc_low(c[0][0], u0[0], u2[0]);
242 fp3_addc_low(c[0][1], u0[1], u1[0]);
243 fp3_addc_low(c[0][2], u0[2], u1[1]);
244 } RLC_CATCH_ANY {
245 RLC_THROW(ERR_CAUGHT);
246 } RLC_FINALLY {
247 dv9_free(u0);
248 dv9_free(u1);
249 dv9_free(u2);
250 dv9_free(u3);
251 fp9_free(t0);
252 fp9_free(t1);
253 }
254}

257 dv18_t t;
258
259 dv18_null(t);
260
261 RLC_TRY {
262 dv18_new(t);

276void fp18_mul_dxs_lazyr(fp18_t c, const fp18_t a, const fp18_t b) {
277 fp9_t t0;
278 dv9_t u0, u1, u2;
279
280 fp9_null(t0);
281 dv9_null(u0);
282 dv9_null(u1);
283 dv9_null(u2);
284
285 RLC_TRY {
286 fp9_new(t0);
287 dv9_new(u0);
288 dv9_new(u1);
289 dv9_new(u2);
290
291 if (ep3_curve_is_twist() == RLC_EP_DTYPE) {
292#if EP_ADD == BASIC
293 /* t0 = a_0 * b_0. */
294 fp_muln_low(u0[0][0], a[0][0][0], b[0][0][0]);
295 fp_muln_low(u0[0][1], a[0][0][1], b[0][0][0]);
296 fp_muln_low(u0[1][0], a[0][1][0], b[0][0][0]);
297 fp_muln_low(u0[1][1], a[0][1][1], b[0][0][0]);
298 fp_muln_low(u0[2][0], a[0][2][0], b[0][0][0]);
299 fp_muln_low(u0[2][1], a[0][2][1], b[0][0][0]);
300 /* t2 = b_0 + b_1. */
301 fp_add(t0[0][0], b[0][0][0], b[1][0][0]);
302 fp_copy(t0[0][1], b[1][0][1]);
303 fp3_copy(t0[1], b[1][1]);
304#elif EP_ADD == PROJC || EP_ADD == JACOB
305 /* t0 = a_0 * b_0. */
306 fp3_muln_low(u0[0], a[0][0], b[0][0]);
307 fp3_muln_low(u0[1], a[0][1], b[0][0]);
308 fp3_muln_low(u0[2], a[0][2], b[0][0]);
309
310 /* t2 = b_0 + b_1. */
311 fp3_add(t0[0], b[0][0], b[1][0]);
312 fp3_copy(t0[1], b[1][1]);
313#endif
314 /* t1 = a_1 * b_1. */
315 fp9_mul_dxs_unr_lazyr(u1, a[1], b[1]);
316 } else {
317 /* t0 = a_0 * b_0. */
318 fp9_mul_dxs_unr_lazyr(u0, a[0], b[0]);
319#if EP_ADD == BASIC
320 /* t0 = a_1 * b_1. */
321 fp_muln_low(u1[1][0], a[1][2][0], b[1][1][0]);
322 fp_muln_low(u1[1][1], a[1][2][1], b[1][1][0]);
323 fp_muln_low(u1[1][2], a[1][2][2], b[1][1][0]);
324 fp3_nord_low(u1[0], u1[1]);
325 fp_muln_low(u1[1][0], a[1][0][0], b[1][1][0]);
326 fp_muln_low(u1[1][1], a[1][0][1], b[1][1][0]);
327 fp_muln_low(u1[1][2], a[1][0][2], b[1][1][0]);
328 fp_muln_low(u1[2][0], a[1][1][0], b[1][1][0]);
329 fp_muln_low(u1[2][1], a[1][1][1], b[1][1][0]);
330 fp_muln_low(u1[2][2], a[1][1][2], b[1][1][0]);
331 /* t2 = b_0 + b_1. */
332 fp3_copy(t0[0], b[0][0]);
333 fp_add(t0[1][0], b[0][1][0], b[1][1][0]);
334 fp_copy(t0[1][1], b[0][1][1]);
335 fp_copy(t0[1][2], b[0][1][2]);
336#elif EP_ADD == PROJC || EP_ADD == JACOB
337 /* t1 = a_1 * b_1. */
338 fp3_muln_low(u1[1], a[1][2], b[1][1]);
339 fp3_nord_low(u1[0], u1[1]);
340 fp3_muln_low(u1[1], a[1][0], b[1][1]);
341 fp3_muln_low(u1[2], a[1][1], b[1][1]);
342 /* t2 = b_0 + b_1. */
343 fp3_copy(t0[0], b[0][0]);
344 fp3_add(t0[1], b[0][1], b[1][1]);
345#endif
346 }
347 /* c_1 = a_0 + a_1. */
348 fp9_add(c[1], a[0], a[1]);
349 /* c_1 = (a_0 + a_1) * (b_0 + b_1) */
350 fp9_mul_dxs_unr_lazyr(u2, c[1], t0);
351 for (int i = 0; i < 3; i++) {
352 fp3_subc_low(u2[i], u2[i], u0[i]);
353 fp3_subc_low(u2[i], u2[i], u1[i]);
354 }
355 fp3_rdcn_low(c[1][0], u2[0]);
356 fp3_rdcn_low(c[1][1], u2[1]);
357 fp3_rdcn_low(c[1][2], u2[2]);
358
359 fp3_nord_low(u2[0], u1[2]);
360 fp3_addc_low(u0[0], u0[0], u2[0]);
361 fp3_addc_low(u0[1], u0[1], u1[0]);
362 fp3_addc_low(u0[2], u0[2], u1[1]);
363 /* c_0 = a_0b_0 + v * a_1b_1. */
364 fp3_rdcn_low(c[0][0], u0[0]);
365 fp3_rdcn_low(c[0][1], u0[1]);
366 fp3_rdcn_low(c[0][2], u0[2]);
367 } RLC_CATCH_ANY {
368 RLC_THROW(ERR_CAUGHT);
369 } RLC_FINALLY {
370 fp9_free(t0);
371 dv9_free(u0);
372 dv9_free(u1);
373 dv9_free(u2);
374 }
375}
