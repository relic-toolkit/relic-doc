42 void fp18_mul_basic(fp18_t c, fp18_t a, fp18_t b) {
43  fp9_t t0, t1, t2;
44 
45  fp9_null(t0);
46  fp9_null(t1);
47  fp9_null(t2);
48 
49  TRY {
50  fp9_new(t0);
51  fp9_new(t1);
52  fp9_new(t2);
53 
54  /* Karatsuba algorithm. */
55 
56  /* t0 = a_0 * b_0. */
57  fp9_mul(t0, a[0], b[0]);
58  /* t1 = a_1 * b_1. */
59  fp9_mul(t1, a[1], b[1]);
60  /* t2 = b_0 + b_1. */
61  fp9_add(t2, b[0], b[1]);
62 
63  /* c_1 = a_0 + a_1. */
64  fp9_add(c[1], a[0], a[1]);
65 
66  /* c_1 = (a_0 + a_1) * (b_0 + b_1) */
67  fp9_mul(c[1], c[1], t2);
68  fp9_sub(c[1], c[1], t0);
69  fp9_sub(c[1], c[1], t1);
70 
71  /* c_0 = a_0b_0 + v * a_1b_1. */
72  fp9_mul_art(t1, t1);
73  fp9_add(c[0], t0, t1);
74  } CATCH_ANY {
75  THROW(ERR_CAUGHT);
76  } FINALLY {
77  fp9_free(t0);
78  fp9_free(t1);
79  fp9_free(t2);
80  }
81 }

87 void fp18_mul_unr(dv18_t c, fp18_t a, fp18_t b) {
88  fp9_t t0, t1;
89  dv9_t u0, u1, u2, u3;
90 
91  dv9_null(u0);
92  dv9_null(u1);
93  dv9_null(u2);
94  dv9_null(u3);
95  fp9_null(t0);
96  fp9_null(t1);
97 
98  TRY {
99  dv9_new(u0);
100  dv9_new(u1);
101  dv9_new(u2);
102  dv9_new(u3);
103  fp9_new(t0);
104  fp9_new(t1);
105 
106  /* Karatsuba algorithm. */
107 
108  /* u0 = a_0 * b_0. */
109  fp9_mul_unr(u0, a[0], b[0]);
110  /* u1 = a_1 * b_1. */
111  fp9_mul_unr(u1, a[1], b[1]);
112  /* t1 = a_0 + a_1. */
113  fp9_add(t0, a[0], a[1]);
114  /* t0 = b_0 + b_1. */
115  fp9_add(t1, b[0], b[1]);
116  /* u2 = (a_0 + a_1) * (b_0 + b_1) */
117  fp9_mul_unr(u2, t0, t1);
118  /* c_1 = u2 - a_0b_0 - a_1b_1. */
119  for (int i = 0; i < 3; i++) {
120  fp3_addc_low(u3[i], u0[i], u1[i]);
121  fp3_subc_low(c[1][i], u2[i], u3[i]);
122  }
123  /* c_0 = a_0b_0 + v * a_1b_1. */
124  fp3_nord_low(u2[0], u1[2]);
125  fp3_addc_low(c[0][0], u0[0], u2[0]);
126  fp3_addc_low(c[0][1], u0[1], u1[0]);
127  fp3_addc_low(c[0][2], u0[2], u1[1]);
128  } CATCH_ANY {
129  THROW(ERR_CAUGHT);
130  } FINALLY {
131  dv9_free(u0);
132  dv9_free(u1);
133  dv9_free(u2);
134  dv9_free(u3);
135  fp9_free(t0);
136  fp9_free(t1);
137  }
