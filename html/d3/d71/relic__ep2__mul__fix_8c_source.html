79 static void ep2_mul_fix_ordin(ep2_t r, ep2_t *table, bn_t k) {
80  int len, i, n;
81  int8_t naf[2 * RLC_FP_BITS + 1], *t;
82 
83  if (bn_is_zero(k)) {
84  ep2_set_infty(r);
85  return;
86  }
87 
88  /* Compute the w-TNAF representation of k. */
89  len = 2 * RLC_FP_BITS + 1;
90  bn_rec_naf(naf, &len, k, EP_DEPTH);
91 
92  t = naf + len - 1;
93  ep2_set_infty(r);
94  for (i = len - 1; i >= 0; i--, t--) {
95  ep2_dbl(r, r);
96 
97  n = *t;
98  if (n > 0) {
99  ep2_add(r, r, table[n / 2]);
100  }
101  if (n < 0) {
102  ep2_sub(r, r, table[-n / 2]);
103  }
104  }
105  /* Convert r to affine coordinates. */
106  ep2_norm(r, r);
107  if (bn_sign(k) == RLC_NEG) {
108  ep2_neg(r, r);
109  }
110 }
