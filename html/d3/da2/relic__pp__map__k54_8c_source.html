40 static void pp_mil_k54(fp54_t r, fp9_t qx, fp9_t qy, ep_t p, bn_t a) {
41  fp54_t l;
42  ep_t _p;
43  fp9_t rx, ry, rz, sx, sy, sz;
44  int i, len = bn_bits(a) + 1;
45  int8_t s[RLC_FP_BITS + 1];
46 
47  fp54_null(l);
48  ep_null(_p);
49  fp9_null(rx);
50  fp9_null(ry);
51  fp9_null(rz);
52  fp9_null(sx);
53  fp9_null(sy);
54  fp9_null(sz);
55 
56  TRY {
57  fp54_new(l);
58  ep_new(_p);
59  fp9_new(rx);
60  fp9_new(ry);
61  fp9_new(rz);
62  fp9_new(sx);
63  fp9_new(sy);
64  fp9_new(sz);
65 
66  fp54_zero(l);
67  fp9_copy(rx, qx);
68  fp9_copy(ry, qy);
69  fp9_set_dig(rz, 1);
70 #if EP_ADD == BASIC
71  ep_neg(_p, p);
72 #else
73  fp_add(_p->x, p->x, p->x);
74  fp_add(_p->x, _p->x, p->x);
75  fp_neg(_p->y, p->y);
76 #endif
77 
78  bn_rec_naf(s, &len, a, 2);
79  for (i = len - 2; i >= 0; i--) {
80  fp54_sqr(r, r);
81  pp_dbl_k54(l, rx, ry, rz, _p);
82  fp54_mul_dxs(r, r, l);
83  if (s[i] > 0) {
84  pp_add_k54(l, rx, ry, rz, qx, qy, p);
85  fp54_mul_dxs(r, r, l);
86  }
87  if (s[i] < 0) {
88  fp9_neg(qy, qy);
89  pp_add_k54(l, rx, ry, rz, qx, qy, p);
90  fp9_neg(qy, qy);
91  fp54_mul_dxs(r, r, l);
92  }
93  }
94  /* Compute f^3. */
95  fp54_sqr(l, r);
96  fp54_mul(r, r, l);
97 
98  fp54_zero(l);
99  fp9_copy(sx, rx);
100  fp9_copy(sy, ry);
101  fp9_copy(sz, rz);
102  pp_dbl_k54(l, sx, sy, sz, _p);
103  fp54_mul_dxs(r, r, l);
104 #if EP_ADD == PROJC
105  fp9_inv(sz, sz);
106  fp9_mul(sx, sx, sz);
107  fp9_mul(sy, sy, sz);
108 #endif
109  pp_add_k54(l, rx, ry, rz, sx, sy, p);
110  fp54_mul_dxs(r, r, l);
111  fp9_frb(rx, qx, 1);
112  fp9_frb(ry, qy, 1);
113  fp9_zero(sz);
114  fp3_set_dig(sz[1], 1);
115  fp9_inv(sz, sz);
116  fp_copy(sz[0][0], sz[2][2]);
117  fp_mul(sz[0][0], sz[0][0], core_get()->fp3_p0[1]);
118  fp_mul(sz[0][0], sz[0][0], core_get()->fp3_p1[3]);
119  fp_mul(sz[0][0], sz[0][0], core_get()->fp3_p1[0]);
120  fp3_mul_nor(sz[0], sz[0]);
121  fp3_mul_nor(sz[0], sz[0]);
122  fp3_mul_nor(sz[0], sz[0]);
123  fp_mul(sz[1][0], sz[0][0], core_get()->fp3_p2[1]);
124 
125  for (int i = 0; i < 3; i++) {
126  fp3_mul(ry[i], ry[i], sz[0]);
127  fp3_mul(rx[i], rx[i], sz[1]);
128  }
129 
130  fp9_frb(sx, qx, 10);
131  fp9_frb(sy, qy, 10);
132  for (int j = 0; j < 10; j++) {
133  for (int i = 0; i < 3; i++) {
134  fp3_mul(sy[i], sy[i], sz[0]);
135  fp3_mul(sx[i], sx[i], sz[1]);
136  }
137  }
138  fp9_set_dig(sz, 1);
139 
140  pp_add_k54(l, sx, sy, sz, rx, ry, p);
141  fp54_mul_dxs(r, r, l);
142  }
143  CATCH_ANY {
144  THROW(ERR_CAUGHT);
145  }
146  FINALLY {
147  fp54_free(l);
148  ep_free(_p);
149  fp9_free(rx);
150  fp9_free(ry);
151  fp9_free(rz);
152  fp9_free(sx);
153  fp9_free(sy);
154  fp9_free(sz);
155  }
156 }

162 void pp_map_k54(fp54_t r, ep_t p, fp9_t qx, fp9_t qy) {
163  bn_t a;
164 
165  bn_null(a);
166

170  fp_prime_get_par(a);

172 
173  if (!ep_is_infty(p) && !(fp9_is_zero(qx) && fp9_is_zero(qy))) {
174  switch (ep_curve_is_pairf()) {
175  case EP_K54:
176  /* r = f_{|a|,Q}(P). */

179  fp54_inv_cyc(r, r);

183  }
184  }
185  }

188  }
