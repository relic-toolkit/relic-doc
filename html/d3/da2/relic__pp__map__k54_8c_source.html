40static void pp_mil_k54(fp54_t r, const fp9_t qx, const fp9_t qy, const ep_t p,
41 const bn_t a) {
42 fp54_t l;
43 ep_t _p;
44 fp9_t rx, ry, rz, sx, sy, sz, qn;
45 size_t len = bn_bits(a) + 1;
46 int i;
47 int8_t s[RLC_FP_BITS + 1];
48
49 fp54_null(l);
50 ep_null(_p);
51 fp9_null(rx);
52 fp9_null(ry);
53 fp9_null(rz);
54 fp9_null(sx);
55 fp9_null(sy);
56 fp9_null(sz);
57
58 RLC_TRY {
59 fp54_new(l);
60 ep_new(_p);
61 fp9_new(rx);
62 fp9_new(ry);
63 fp9_new(rz);
64 fp9_new(sx);
65 fp9_new(sy);
66 fp9_new(sz);
67 fp9_new(qn);
68
69 fp54_zero(l);
70 fp9_copy(rx, qx);
71 fp9_copy(ry, qy);
72 fp9_set_dig(rz, 1);
73#if EP_ADD == BASIC
74 ep_neg(_p, p);
75#else
76 fp_add(_p->x, p->x, p->x);
77 fp_add(_p->x, _p->x, p->x);
78 fp_neg(_p->y, p->y);
79#endif
80 fp9_neg(qn, qy);
81
82 bn_rec_naf(s, &len, a, 2);
83 for (i = len - 2; i >= 0; i--) {
84 fp54_sqr(r, r);
85 pp_dbl_k54(l, rx, ry, rz, _p);
86 fp54_mul_dxs(r, r, l);
87 if (s[i] > 0) {
88 pp_add_k54(l, rx, ry, rz, qx, qy, p);
89 fp54_mul_dxs(r, r, l);
90 }
91 if (s[i] < 0) {
92 pp_add_k54(l, rx, ry, rz, qx, qn, p);
93 fp54_mul_dxs(r, r, l);
94 }
95 }
96 /* Compute f^3. */
97 fp54_sqr(l, r);
98 fp54_mul(r, r, l);
99
100 fp54_zero(l);
101 fp9_copy(sx, rx);
102 fp9_copy(sy, ry);
103 fp9_copy(sz, rz);
104 pp_dbl_k54(l, sx, sy, sz, _p);
105 fp54_mul_dxs(r, r, l);
106#if EP_ADD == PROJC
107 fp9_inv(sz, sz);
108 fp9_mul(sx, sx, sz);
109 fp9_mul(sy, sy, sz);
110#endif
111 pp_add_k54(l, rx, ry, rz, sx, sy, p);
112 fp54_mul_dxs(r, r, l);
113 fp9_frb(rx, qx, 1);
114 fp9_frb(ry, qy, 1);
115 fp9_zero(sz);
116 fp3_set_dig(sz[1], 1);
117 fp9_inv(sz, sz);
118 fp_copy(sz[0][0], sz[2][2]);
119 fp_mul(sz[0][0], sz[0][0], core_get()->fp3_p0[1]);
120 fp_mul(sz[0][0], sz[0][0], core_get()->fp3_p1[3][0]);
121 fp_mul(sz[0][0], sz[0][0], core_get()->fp3_p1[0][0]);
122 fp3_mul_nor(sz[0], sz[0]);
123 fp3_mul_nor(sz[0], sz[0]);
124 fp3_mul_nor(sz[0], sz[0]);
125 fp_mul(sz[1][0], sz[0][0], core_get()->fp3_p2[1][0]);
126
127 for (int i = 0; i < 3; i++) {
128 fp3_mul(ry[i], ry[i], sz[0]);
129 fp3_mul(rx[i], rx[i], sz[1]);
130 }
131
132 fp9_frb(sx, qx, 10);
133 fp9_frb(sy, qy, 10);
134 for (int j = 0; j < 10; j++) {
135 for (int i = 0; i < 3; i++) {
136 fp3_mul(sy[i], sy[i], sz[0]);
137 fp3_mul(sx[i], sx[i], sz[1]);
138 }
139 }
140 fp9_set_dig(sz, 1);
141
142 pp_add_k54(l, sx, sy, sz, rx, ry, p);
143 fp54_mul_dxs(r, r, l);
144 }
145 RLC_CATCH_ANY {
146 RLC_THROW(ERR_CAUGHT);
147 }
148 RLC_FINALLY {
149 fp54_free(l);
150 ep_free(_p);
151 fp9_free(rx);
152 fp9_free(ry);
153 fp9_free(rz);
154 fp9_free(sx);
155 fp9_free(sy);
156 fp9_free(sz);
157 fp9_free(qn);
158 }
159}

165void pp_map_k54(fp54_t r, const ep_t p, const fp9_t qx, const fp9_t qy) {
166 bn_t a;
167
168 bn_null(a);
169
170 RLC_TRY {
171 bn_new(a);
172
173 fp_prime_get_par(a);
174 fp54_set_dig(r, 1);
175
176 if (!ep_is_infty(p) && !(fp9_is_zero(qx) && fp9_is_zero(qy))) {
177 switch (ep_curve_is_pairf()) {
178 case EP_SG54:
179 /* r = f_{|a|,Q}(P). */
180 pp_mil_k54(r, qx, qy, p, a);
181 if (bn_sign(a) == RLC_NEG) {
182 fp54_inv_cyc(r, r);
183 }
184 pp_exp_k54(r, r);
185 break;
186 }
187 }
188 }
189 RLC_CATCH_ANY {
190 RLC_THROW(ERR_CAUGHT);
191 }
192 RLC_FINALLY {
193 bn_free(a);
194 }
195}
