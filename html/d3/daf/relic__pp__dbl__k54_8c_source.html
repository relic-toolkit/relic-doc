40static void ep9_dbl_basic(fp9_t s, fp9_t rx, fp9_t ry) {
41 fp9_t t0, t1, t2;
42
43 fp9_null(t0);
44 fp9_null(t1);
45 fp9_null(t2);
46
47 RLC_TRY {
48 fp9_new(t0);
49 fp9_new(t1);
50 fp9_new(t2);
51
52 /* t0 = 1/(2 * y1). */
53 fp9_dbl(t0, ry);
54 fp9_inv(t0, t0);
55
56 /* t1 = 3 * x1^2 + a. */
57 fp9_sqr(t1, rx);
58 fp9_copy(t2, t1);
59 fp9_dbl(t1, t1);
60 fp9_add(t1, t1, t2);
61
62 /* a = 0. */
63 /* t1 = (3 * x1^2 + a)/(2 * y1). */
64 fp9_mul(t1, t1, t0);
65
66 if (s != NULL) {
67 fp9_copy(s, t1);
68 }
69
70 /* t2 = t1^2. */
71 fp9_sqr(t2, t1);
72
73 /* x3 = t1^2 - 2 * x1. */
74 fp9_dbl(t0, rx);
75 fp9_sub(t0, t2, t0);
76
77 /* y3 = t1 * (x1 - x3) - y1. */
78 fp9_sub(t2, rx, t0);
79 fp9_mul(t1, t1, t2);
80
81 fp9_sub(ry, t1, ry);
82
83 fp9_copy(rx, t0);
84 }
85 RLC_CATCH_ANY {
86 RLC_THROW(ERR_CAUGHT);
87 }
88 RLC_FINALLY {
89 fp9_free(t0);
90 fp9_free(t1);
91 fp9_free(t2);
92 }
93}

146void pp_dbl_k54_projc(fp54_t l, fp9_t rx, fp9_t ry, fp9_t rz, const ep_t p) {
147 fp9_t t0, t1, t2, t3, t4, t5, t6;
148
149 fp9_null(t0);
150 fp9_null(t1);
151 fp9_null(t2);
152 fp9_null(t3);
153 fp9_null(t4);

157 RLC_TRY {
158 fp9_new(t0);
159 fp9_new(t1);
160 fp9_new(t2);
161 fp9_new(t3);
162 fp9_new(t4);
163 fp9_new(t5);
164 fp9_new(t6);
165
166 /* A = x1^2. */
167 fp9_sqr(t0, rx);
168 /* B = y1^2. */
169 fp9_sqr(t1, ry);
170 /* C = z1^2. */
171 fp9_sqr(t2, rz);
172 /* D = 3bC, general b. */
173 fp9_dbl(t3, t2);
174 fp9_add(t3, t3, t2);
175 fp9_zero(t4);
176 fp_copy(t4[1][0], ep_curve_get_b());
177
178 fp9_mul(t3, t3, t4);
179 /* E = (x1 + y1)^2 - A - B. */
180 fp9_add(t4, rx, ry);
181 fp9_sqr(t4, t4);
182 fp9_sub(t4, t4, t0);
183 fp9_sub(t4, t4, t1);
184
185 /* F = (y1 + z1)^2 - B - C. */
186 fp9_add(t5, ry, rz);
187 fp9_sqr(t5, t5);
188 fp9_sub(t5, t5, t1);
189 fp9_sub(t5, t5, t2);
190
191 /* G = 3D. */
192 fp9_dbl(t6, t3);
193 fp9_add(t6, t6, t3);
194
195 /* x3 = E * (B - G). */
196 fp9_sub(rx, t1, t6);
197 fp9_mul(rx, rx, t4);
198
199 /* y3 = (B + G)^2 -12D^2. */
200 fp9_add(t6, t6, t1);
201 fp9_sqr(t6, t6);
202 fp9_sqr(t2, t3);
203 fp9_dbl(ry, t2);
204 fp9_dbl(t2, ry);
205 fp9_dbl(ry, t2);
206 fp9_add(ry, ry, t2);
207 fp9_sub(ry, t6, ry);
208
209 /* z3 = 4B * F. */
210 fp9_dbl(rz, t1);
211 fp9_dbl(rz, rz);
212 fp9_mul(rz, rz, t5);
213
214 /* l11 = D - B. */
215 fp9_sub(l[0][0], t3, t1);
216
217 /* l10 = (3 * xp) * A. */
218 fp_mul(l[2][0][0][0], p->x, t0[0][0]);
219 fp_mul(l[2][0][0][1], p->x, t0[0][1]);
220 fp_mul(l[2][0][0][2], p->x, t0[0][2]);
221 fp_mul(l[2][0][1][0], p->x, t0[1][0]);
222 fp_mul(l[2][0][1][1], p->x, t0[1][1]);
223 fp_mul(l[2][0][1][2], p->x, t0[1][2]);
224 fp_mul(l[2][0][2][0], p->x, t0[2][0]);
225 fp_mul(l[2][0][2][1], p->x, t0[2][1]);
226 fp_mul(l[2][0][2][2], p->x, t0[2][2]);
227
228 /* l00 = F * (-yp). */
229 fp_mul(l[0][1][0][0], p->y, t5[0][0]);
230 fp_mul(l[0][1][0][1], p->y, t5[0][1]);
231 fp_mul(l[0][1][0][2], p->y, t5[0][2]);
232 fp_mul(l[0][1][1][0], p->y, t5[1][0]);
233 fp_mul(l[0][1][1][1], p->y, t5[1][1]);
234 fp_mul(l[0][1][1][2], p->y, t5[1][2]);
235 fp_mul(l[0][1][2][0], p->y, t5[2][0]);
236 fp_mul(l[0][1][2][1], p->y, t5[2][1]);
237 fp_mul(l[0][1][2][2], p->y, t5[2][2]);
238 }
239 RLC_CATCH_ANY {
240 RLC_THROW(ERR_CAUGHT);
241 }
242 RLC_FINALLY {
243 fp9_free(t0);
244 fp9_free(t1);
245 fp9_free(t2);
246 fp9_free(t3);
247 fp9_free(t4);
248 fp9_free(t5);
249 fp9_free(t6);
250 }
251}
