42 void fp54_sqr_basic(fp54_t c, fp54_t a) {
43  fp18_t t0, t1, t2, t3, t4;
44 
45  fp18_null(t0);
46  fp18_null(t1);
47  fp18_null(t2);
48  fp18_null(t3);
49  fp18_null(t4);
50 
51  TRY {
52  fp18_new(t0);
53  fp18_new(t1);
54  fp18_new(t2);
55  fp18_new(t3);
56  fp18_new(t4);
57 
58  /* t0 = a_0^2 */
59  fp18_sqr(t0, a[0]);
60 
61  /* t1 = 2 * a_1 * a_2 */
62  fp18_mul(t1, a[1], a[2]);
63  fp18_dbl(t1, t1);
64 
65  /* t2 = a_2^2. */
66  fp18_sqr(t2, a[2]);
67 
68  /* c_2 = a_0 + a_2. */
69  fp18_add(c[2], a[0], a[2]);
70 
71  /* t3 = (a_0 + a_2 + a_1)^2. */
72  fp18_add(t3, c[2], a[1]);
73  fp18_sqr(t3, t3);
74 
75  /* c_2 = (a_0 + a_2 - a_1)^2. */
76  fp18_sub(c[2], c[2], a[1]);
77  fp18_sqr(c[2], c[2]);
78 
79  /* c_2 = (c_2 + t3)/2. */
80  fp18_add(c[2], c[2], t3);
81  for (int i = 0; i < 3; i++) {
82  for (int j = 0; j < 3; j++) {
83  fp_hlv(c[2][0][i][j], c[2][0][i][j]);
84  fp_hlv(c[2][1][i][j], c[2][1][i][j]);
85  }
86  }
87 
88  /* t3 = t3 - c_2 - t1. */
89  fp18_sub(t3, t3, c[2]);
90  fp18_sub(t3, t3, t1);
91 
92  /* c_2 = c_2 - t0 - t2. */
93  fp18_sub(c[2], c[2], t0);
94  fp18_sub(c[2], c[2], t2);
95 
96  /* c_0 = t0 + t1 * E. */
97  fp18_mul_art(t4, t1);
98  fp18_add(c[0], t0, t4);
99 
100  /* c_1 = t3 + t2 * E. */
101  fp18_mul_art(t4, t2);
102  fp18_add(c[1], t3, t4);
103  } CATCH_ANY {
104  THROW(ERR_CAUGHT);
105  } FINALLY {
106  fp18_free(t0);
107  fp18_free(t1);
108  fp18_free(t2);
109  fp18_free(t3);
110  fp18_free(t4);
111  }
112 }
