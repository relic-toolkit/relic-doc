46static void bn_add_imp(bn_t c, const bn_t a, const bn_t b) {
47 int max, min;
48 dig_t carry;
49
50 max = a->used;
51 min = b->used;
52
53 if (min == 0) {
54 bn_copy(c, a);
55 return;
56 }
57
58 RLC_TRY {
59 /* Grow the result. */
60 bn_grow(c, max);
61
62 if (a->used == b->used) {
63 carry = bn_addn_low(c->dp, a->dp, b->dp, max);
64 } else {
65 carry = bn_addn_low(c->dp, a->dp, b->dp, min);
66 carry = bn_add1_low(c->dp + min, a->dp + min, carry, max - min);
67 }
68 if (carry) {
69 bn_grow(c, max + 1);
70 c->dp[max] = carry;
71 }
72 c->used = max + carry;
73 bn_trim(c);
74 } RLC_CATCH_ANY {
75 RLC_THROW(ERR_CAUGHT);
76 }
77}

86static void bn_sub_imp(bn_t c, const bn_t a, const bn_t b) {

88 dig_t carry;
89
90 max = a->used;
91 min = b->used;
92
93 if (min == 0) {
94 bn_copy(c, a);
95 return;
96 }
97
98 RLC_TRY {
99 /* Grow the destination to accomodate the result. */
100 bn_grow(c, max);
101
102 if (a->used == b->used) {
103 carry = bn_subn_low(c->dp, a->dp, b->dp, min);
104 } else {
105 carry = bn_subn_low(c->dp, a->dp, b->dp, min);
106 carry = bn_sub1_low(c->dp + min, a->dp + min, carry, max - min);
107 }
108 c->used = max;
109 bn_trim(c);
110 } RLC_CATCH_ANY {
111 RLC_THROW(ERR_CAUGHT);
112 }

119void bn_add(bn_t c, const bn_t a, const bn_t b) {
120 int sa, sb;
121
122 sa = a->sign;
123 sb = b->sign;
124
125 if (sa == sb) {
126 /* If the signs are equal, copy the sign and add. */
127 c->sign = sa;
128 if (bn_cmp_abs(a, b) == RLC_LT) {
129 bn_add_imp(c, b, a);
130 } else {
131 bn_add_imp(c, a, b);
132 }
133 } else {
134 /* If the signs are different, subtract. */
135 if (bn_cmp_abs(a, b) == RLC_LT) {
136 bn_sub_imp(c, b, a);
137 c->sign = sb;
138 } else {
139 bn_sub_imp(c, a, b);
140 c->sign = sa;
141 }
142 }

145void bn_add_dig(bn_t c, const bn_t a, dig_t b) {
146 dig_t carry;
147
148 RLC_TRY {
149 bn_grow(c, a->used);
150
151 if (a->sign == RLC_POS) {
152 carry = bn_add1_low(c->dp, a->dp, b, a->used);
153 if (carry) {
154 bn_grow(c, a->used + 1);
155 c->dp[a->used] = carry;
156 }
157 c->used = a->used + carry;
158 c->sign = RLC_POS;
159 } else {
160 /* If a < 0 && |a| >= b, compute c = -(|a| - b). */
161 if (a->used > 1 || a->dp[0] >= b) {
162 carry = bn_sub1_low(c->dp, a->dp, b, a->used);
163 c->used = a->used;
164 c->sign = RLC_NEG;
165 } else {
166 /* If a < 0 && |a| < b. */
167 if (a->used == 1) {
168 c->dp[0] = b - a->dp[0];
169 } else {
170 c->dp[0] = b;
171 }
172 c->used = 1;
173 c->sign = RLC_POS;
174 }
175 }
176 bn_trim(c);
177 } RLC_CATCH_ANY {
178 RLC_THROW(ERR_CAUGHT);
179 }

182void bn_sub(bn_t c, const bn_t a, const bn_t b) {
183 int sa, sb;
184
185 sa = a->sign;
186 sb = b->sign;
187
188 if (sa != sb) {
189 /* If the signs are different, copy the sign of the first number and
190 * add. */
191 c->sign = sa;
192 if (bn_cmp_abs(a, b) == RLC_LT) {
193 bn_add_imp(c, b, a);
194 } else {
195 bn_add_imp(c, a, b);
196 }
197 } else {
198 /* If the signs are equal, adjust the sign and subtract. */
199 if (bn_cmp_abs(a, b) != RLC_LT) {
200 bn_sub_imp(c, a, b);
201 c->sign = sa;
202 } else {
203 bn_sub_imp(c, b, a);
204 c->sign = (sa == RLC_POS) ? RLC_NEG : RLC_POS;
205 }
206 }

209void bn_sub_dig(bn_t c, const bn_t a, dig_t b) {
210 dig_t carry;
211
212 RLC_TRY {
213 bn_grow(c, a->used);
214
215 /* If a < 0, compute c = -(|a| + b). */
216 if (a->sign == RLC_NEG) {
217 carry = bn_add1_low(c->dp, a->dp, b, a->used);
218 if (carry) {
219 bn_grow(c, a->used + 1);
220 c->dp[a->used] = carry;
221 }
222 c->used = a->used + carry;
223 c->sign = RLC_NEG;
224 } else {
225 /* If a > 0 && |a| >= b, compute c = (|a| - b). */
226 if (a->used > 1 || a->dp[0] >= b) {
227 carry = bn_sub1_low(c->dp, a->dp, b, a->used);
228 c->used = a->used;
