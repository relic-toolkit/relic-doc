41 void fp_exp_basic(fp_t c, const fp_t a, const bn_t b) {
42  int i, l;
43  fp_t r;
44 
45  fp_null(r);
46 
47  if (bn_is_zero(b)) {
48  fp_set_dig(c, 1);
49  return;
50  }
51 
52  TRY {
53  fp_new(r);
54 
55  l = bn_bits(b);
56 
57  fp_copy(r, a);
58 
59  for (i = l - 2; i >= 0; i--) {
60  fp_sqr(r, r);
61  if (bn_get_bit(b, i)) {
62  fp_mul(r, r, a);
63  }
64  }
65 
66  if (bn_sign(b) == RLC_NEG) {
67  fp_inv(c, r);
68  } else {
69  fp_copy(c, r);
70  }
71  }
72  CATCH_ANY {
73  THROW(ERR_CAUGHT);
74  }
75  FINALLY {
76  fp_free(r);
77  }

84 void fp_exp_slide(fp_t c, const fp_t a, const bn_t b) {
85  fp_t t[1 << (FP_WIDTH - 1)], r;
86  int i, j, l;
87  uint8_t win[RLC_FP_BITS + 1];
88 
89  fp_null(r);
90 
91  if (bn_is_zero(b)) {
92  fp_set_dig(c, 1);
93  return;
94  }
95 
96 
97  /* Initialize table. */
98  for (i = 0; i < (1 << (FP_WIDTH - 1)); i++) {
99  fp_null(t[i]);
100  }
101 
102  TRY {
103  for (i = 0; i < (1 << (FP_WIDTH - 1)); i ++) {
104  fp_new(t[i]);
105  }
106  fp_new(r);
107 
108  fp_copy(t[0], a);
109  fp_sqr(r, a);
110 
111  /* Create table. */
112  for (i = 1; i < 1 << (FP_WIDTH - 1); i++) {
113  fp_mul(t[i], t[i - 1], r);
114  }
115 
116  fp_set_dig(r, 1);
117  l = RLC_FP_BITS + 1;
118  bn_rec_slw(win, &l, b, FP_WIDTH);
119  for (i = 0; i < l; i++) {
120  if (win[i] == 0) {
121  fp_sqr(r, r);
122  } else {
123  for (j = 0; j < util_bits_dig(win[i]); j++) {
124  fp_sqr(r, r);
125  }
126  fp_mul(r, r, t[win[i] >> 1]);
127  }
128  }
129 
130  if (bn_sign(b) == RLC_NEG) {
131  fp_inv(c, r);
132  } else {
133  fp_copy(c, r);
134  }
135  }
136  CATCH_ANY {
137  THROW(ERR_CAUGHT);
138  }
139  FINALLY {
140  for (i = 0; i < (1 << (FP_WIDTH - 1)); i++) {
141  fp_free(t[i]);
142  }
143  fp_free(r);
144  }
145 }
