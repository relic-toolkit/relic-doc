73 void fp8_sqr_unr(dv8_t c, fp8_t a) {
74  fp4_t t;
75  dv4_t u0, u1, u2;
76 
77  fp4_null(t);
78  dv4_null(u0);
79  dv4_null(u1);
80  dv4_null(u2);
81 
82  TRY {
83  fp4_new(t);
84  dv4_new(u0);
85  dv4_new(u1);
86  dv4_new(u2);
87 
88  /* t0 = a^2. */
89  fp4_sqr_unr(u0, a[0]);
90  /* t1 = b^2. */
91  fp4_sqr_unr(u1, a[1]);
92 
93  fp4_add(t, a[0], a[1]);
94 
95  /* c = a^2 + b^2 * E. */
96  dv_copy(u2[1][0], u1[0][0], 2 * RLC_FP_DIGS);
97  dv_copy(u2[1][1], u1[0][1], 2 * RLC_FP_DIGS);
98  fp2_nord_low(u2[0], u1[1]);
99  fp2_addc_low(c[0][0], u2[0], u0[0]);
100  fp2_addc_low(c[0][1], u2[1], u0[1]);
101 
102  /* d = (a + b)^2 - a^2 - b^2 = 2 * a * b. */
103  fp2_addc_low(u1[0], u1[0], u0[0]);
104  fp2_addc_low(u1[1], u1[1], u0[1]);
105 
106  fp4_sqr_unr(u0, t);
107  fp2_subc_low(c[1][0], u0[0], u1[0]);
108  fp2_subc_low(c[1][1], u0[1], u1[1]);
109  } CATCH_ANY {
110  THROW(ERR_CAUGHT);
111  } FINALLY {
112  fp4_free(t);
113  dv4_free(u0);
114  dv4_free(u1);
115  dv4_free(u2);
