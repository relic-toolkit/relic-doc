39int fp2_field_get_qnr() {
40 /* Override some of the results when cubic non-residue is also needed. */
41#if FP_PRIME == 1150
42 return 32;
43#elif FP_PRIME == 158 || FP_PRIME == 256
44 return 4;
45#elif FP_PRIME == 446 && !defined(FP_QNRES)
46 return 16;
47#else
48 return core_get()->qnr2;
49#endif
50}

52int fp3_field_get_cnr() {
53#if FP_PRIME == 638
54 if (fp_param_get() == K18_638) {
55 return 8;
56 } else {
57 return 3;
58 }
59#elif FP_PRIME == 768
60 return -4;
61#endif
62 return core_get()->cnr3;

65void fp2_field_init(void) {
66 bn_t e;
67 fp2_t t0, t1;
68 ctx_t *ctx = core_get();
69
70 bn_null(e);
71 fp2_null(t0);
72 fp2_null(t1);
73
74 RLC_TRY {
75 bn_new(e);
76 fp2_new(t0);
77 fp2_new(t1);
78
79 /* Start by finding a quadratic/cubic non-residue. */
80#ifdef FP_QNRES
81 ctx->qnr2 = 1;
82#else
83 /* First start with u as quadratic non-residue. */
84 ctx->qnr2 = 0;
85 fp_zero(t0[0]);
86 fp_set_dig(t0[1], 1);
87 /* If it does not work, attempt (u + 1), otherwise double. */
88 /* We cannot used QR test here due to Frobenius constants below. */
89 if (fp2_srt(t1, t0)) {
90 ctx->qnr2 = 1;
91 fp_set_dig(t0[0], ctx->qnr2);
92 while (fp2_srt(t1, t0) && util_bits_dig(ctx->qnr2) < RLC_DIG - 1) {
93 /* Pick a power of 2 for efficiency. */
94 ctx->qnr2 *= 2;
95 fp_set_dig(t0[0], ctx->qnr2);
96 }
97 }
98#endif /* !FP_QNRES */
99
100 /* Compute QNR^(p - 1)/6 and consecutive powers. */
101 fp2_set_dig(t1, 1);
102 fp2_mul_nor(t0, t1);
103 e->used = RLC_FP_DIGS;
104 dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
105 bn_sub_dig(e, e, 1);
106 bn_div_dig(e, e, 6);
107 fp2_exp(t0, t0, e);
108
109 fp_copy(ctx->fp2_p1[0][0], t0[0]);
110 fp_copy(ctx->fp2_p1[0][1], t0[1]);
111 fp2_sqr(t1, t0);
112 fp_copy(ctx->fp2_p1[1][0], t1[0]);
113 fp_copy(ctx->fp2_p1[1][1], t1[1]);
114 fp2_mul(t1, t1, t0);
115 fp_copy(ctx->fp2_p1[2][0], t1[0]);
116 fp_copy(ctx->fp2_p1[2][1], t1[1]);
117 fp2_sqr(t1, t0);
118 fp2_sqr(t1, t1);
119 fp_copy(ctx->fp2_p1[3][0], t1[0]);
120 fp_copy(ctx->fp2_p1[3][1], t1[1]);
121 fp2_mul(t1, t1, t0);
122 fp_copy(ctx->fp2_p1[4][0], t1[0]);
123 fp_copy(ctx->fp2_p1[4][1], t1[1]);
124
125 /* Compute QNR^(p - (p mod 4))/4. */
126 fp2_set_dig(t1, 1);
127 fp2_mul_nor(t0, t1);
128 e->used = RLC_FP_DIGS;
129 dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
130 bn_div_dig(e, e, 4);
131 fp2_exp(t0, t0, e);
132 fp_copy(ctx->fp2_p2[0][0], t0[0]);
133 fp_copy(ctx->fp2_p2[0][1], t0[1]);
134
135 /* Compute QNR^(p - (p mod 8))/8. */
136 fp2_set_dig(t1, 1);
137 fp2_mul_nor(t0, t1);
138 e->used = RLC_FP_DIGS;
139 dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
140 bn_div_dig(e, e, 8);
141 fp2_exp(t0, t0, e);
142 fp_copy(ctx->fp2_p2[1][0], t0[0]);
143 fp_copy(ctx->fp2_p2[1][1], t0[1]);
144
145 /* Compute QNR^(p - (p mod 12))/12. */
146 fp2_set_dig(t1, 1);
147 fp2_mul_nor(t0, t1);
148 e->used = RLC_FP_DIGS;
149 dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
150 bn_div_dig(e, e, 12);
151 fp2_exp(t0, t0, e);
152 fp_copy(ctx->fp2_p2[2][0], t0[0]);
153 fp_copy(ctx->fp2_p2[2][1], t0[1]);
154
155 /* Compute QNR^(p - (p mod 24))/24. */
156 fp2_set_dig(t1, 1);
157 fp2_mul_nor(t0, t1);
158 e->used = RLC_FP_DIGS;
159 dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
160 bn_div_dig(e, e, 24);
161 fp2_exp(t0, t0, e);
162 fp_copy(ctx->fp2_p2[3][0], t0[0]);
163 fp_copy(ctx->fp2_p2[3][1], t0[1]);
164 } RLC_CATCH_ANY {
165 RLC_THROW(ERR_CAUGHT);
166 } RLC_FINALLY {
167 bn_free(e);
168 fp2_free(t0);
169 fp2_free(t1);
170 }
171}

173void fp3_field_init(void) {
174 bn_t e;
175 fp3_t t0, t1;
176 ctx_t *ctx = core_get();
177
178 bn_null(e);
179 fp3_null(t0);
180 fp3_null(t1);
181
182 RLC_TRY {
183 bn_new(e);
184 fp3_new(t0);
185 fp3_new(t1);
186
187 /* Start by trying a trivial quadratic non-residue. */
188 ctx->cnr3 = 0;
189 fp_zero(t0[0]);
190 fp_set_dig(t0[1], 1);
191 fp_zero(t0[2]);
192 /* If it does not work, attempt (u + 1), otherwise double. */
193 /* This code will fail if p \neq 1 mod 8 because square root in Fp^3
194 * relies on Frobenius. Must implement explicit test for those cases. */
195 if (fp3_srt(t1, t0)) {
196 ctx->cnr3 = 1;
197 fp_set_dig(t0[0], ctx->cnr3);
198 while (fp3_srt(t1, t0) && util_bits_dig(ctx->qnr2) < RLC_DIG - 1) {
199 /* Pick a power of 2 for efficiency. */
200 ctx->cnr3 *= 2;
201 fp_set_dig(t0[0], ctx->cnr3);
202 }
203 }
204
205 /* Compute t0 = u^((p - (p mod 3))/3). */
206 if (fp_prime_get_cnr() < 0) {
207 fp_set_dig(ctx->fp3_p0[0], -fp_prime_get_cnr());
208 fp_neg(ctx->fp3_p0[0], ctx->fp3_p0[0]);
209 } else {
210 fp_set_dig(ctx->fp3_p0[0], fp_prime_get_cnr());
211 }
212 bn_read_raw(e, fp_prime_get(), RLC_FP_DIGS);
213 bn_div_dig(e, e, 3);
214 fp_exp(ctx->fp3_p0[0], ctx->fp3_p0[0], e);
215 fp_sqr(ctx->fp3_p0[1], ctx->fp3_p0[0]);
216
217 /* Compute t0 = u^((p - (p mod 6))/6). */
218 fp3_set_dig(t1, 1);
219 fp3_mul_nor(t0, t1);
220 bn_read_raw(e, fp_prime_get(), RLC_FP_DIGS);
221 bn_div_dig(e, e, 6);
222 fp3_exp(t0, t0, e);
223 if (fp3_field_get_cnr() == 0) {
224 /* Look for a non-trivial subfield element.. */
225 ctx->frb3[0] = 0;
226 while (ctx->frb3[0] < 3 && fp_is_zero(t0[ctx->frb3[0]++]));
227 /* Fill rest of table with powers of constant. */
228 fp_copy(ctx->fp3_p1[0][0], t0[--ctx->frb3[0] % 3]);
229 fp3_sqr(t1, t0);
230 fp_copy(ctx->fp3_p1[1][0], t1[(2 * ctx->frb3[0]) % 3]);
231 fp3_mul(t1, t1, t0);
232 fp_copy(ctx->fp3_p1[2][0], t1[(3 * ctx->frb3[0]) % 3]);
233 fp3_sqr(t1, t0);
234 fp3_sqr(t1, t1);
235 fp_copy(ctx->fp3_p1[3][0], t1[(4 * ctx->frb3[0]) % 3]);
236 fp3_mul(t1, t1, t0);
237 fp_copy(ctx->fp3_p1[4][0], t1[(5 * ctx->frb3[0]) % 3]);
238 } else {
239 fp_copy(ctx->fp3_p1[0][0], t0[0]);
240 fp_copy(ctx->fp3_p1[0][1], t0[1]);
241 fp_copy(ctx->fp3_p1[0][2], t0[2]);
242 fp3_sqr(t1, t0);
243 fp_copy(ctx->fp3_p1[1][0], t1[0]);
244 fp_copy(ctx->fp3_p1[1][1], t1[1]);
245 fp_copy(ctx->fp3_p1[1][2], t1[2]);
246 fp3_mul(t1, t1, t0);
247 fp_copy(ctx->fp3_p1[2][0], t1[0]);
248 fp_copy(ctx->fp3_p1[2][1], t1[1]);
249 fp_copy(ctx->fp3_p1[2][2], t1[2]);
250 fp3_sqr(t1, t0);
251 fp3_sqr(t1, t1);
252 fp_copy(ctx->fp3_p1[3][0], t1[0]);
253 fp_copy(ctx->fp3_p1[3][1], t1[1]);
254 fp_copy(ctx->fp3_p1[3][2], t1[2]);
255 fp3_mul(t1, t1, t0);
256 fp_copy(ctx->fp3_p1[4][0], t1[0]);
257 fp_copy(ctx->fp3_p1[4][1], t1[1]);
258 fp_copy(ctx->fp3_p1[4][2], t1[2]);
259 }
260
261 /* Compute t0 = u^((p - (p mod 9))/9). */
262 fp3_set_dig(t1, 1);
263 fp3_mul_nor(t0, t1);
264 bn_read_raw(e, fp_prime_get(), RLC_FP_DIGS);
265 bn_div_dig(e, e, 9);
266 fp3_exp(t0, t0, e);
267 /* Look for a non-trivial subfield element.. */
268 if (fp3_field_get_cnr() == 0) {
269 ctx->frb3[1] = 0;
270 while (ctx->frb3[1] < 3 && fp_is_zero(t0[ctx->frb3[1]++]));
271 fp_copy(ctx->fp3_p2[0][0], t0[--ctx->frb3[1]]);
272 } else {
273 fp_copy(ctx->fp3_p2[0][0], t0[0]);
274 fp_copy(ctx->fp3_p2[0][1], t0[1]);
275 fp_copy(ctx->fp3_p2[0][2], t0[2]);
276 }
277
278 /* Compute t0 = u^((p - (p mod 18))/18). */
279 fp3_set_dig(t1, 1);
280 fp3_mul_nor(t0, t1);
281 bn_read_raw(e, fp_prime_get(), RLC_FP_DIGS);
282 bn_div_dig(e, e, 18);
283 fp3_exp(t0, t0, e);
284 /* Look for a non-trivial subfield element.. */
285 if (fp3_field_get_cnr() == 0) {
286 ctx->frb3[2] = 0;
287 while (ctx->frb3[2] < 3 && fp_is_zero(t0[ctx->frb3[2]++]));
288 fp_copy(ctx->fp3_p2[1][0], t0[--ctx->frb3[2]]);
289 } else {
290 fp_copy(ctx->fp3_p2[1][0], t0[0]);
291 fp_copy(ctx->fp3_p2[1][1], t0[1]);
292 fp_copy(ctx->fp3_p2[1][2], t0[2]);
293 }
294 } RLC_CATCH_ANY {
295 RLC_THROW(ERR_CAUGHT);
296 } RLC_FINALLY {
297 bn_free(e);
298 fp3_free(t0);
299 fp3_free(t1);
300 }
301}
