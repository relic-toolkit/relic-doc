50 void fp2_field_init(void) {
51  bn_t e;
52  fp2_t t0, t1;
53  ctx_t *ctx = core_get();
54 
55  bn_null(e);
56  fp2_null(t0);
57  fp2_null(t1);
58 
59  TRY {
60  bn_new(e);
61  fp2_new(t0);
62  fp2_new(t1);
63 
64  /* Start by finding a quadratic/cubic non-residue. */
65 #ifdef FP_QNRES
66  ctx->qnr2 = 1;
67 #else
68  /* First start with u as quadratic non-residue. */
69  ctx->qnr2 = 0;
70  fp_zero(t0[0]);
71  fp_set_dig(t0[1], 1);
72  /* If it does not work, attempt (u + 2), otherwise double. */
73  if (fp2_srt(t1, t0) == 1) {
74  ctx->qnr2 = 2;
75  fp_set_dig(t0[0], ctx->qnr2);
76  while (fp2_srt(t1, t0) == 1) {
77  /* Pick a power of 2 for efficiency. */
78  ctx->qnr2 *= 2;
79  fp_set_dig(t0[0], ctx->qnr2);
80  }
81  }
82 #endif /* !FP_QNRES */
83 
84  /* Compute QNR^(p - 1)/6 and consecutive powers. */
85  fp2_set_dig(t1, 1);
86  fp2_mul_nor(t0, t1);
87  e->used = RLC_FP_DIGS;
88  dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
89  bn_sub_dig(e, e, 1);
90  bn_div_dig(e, e, 6);
91  fp2_exp(t0, t0, e);
92  fp_copy(ctx->fp2_p1[0][0], t0[0]);
93  fp_copy(ctx->fp2_p1[0][1], t0[1]);
94  fp2_sqr(t1, t0);
95  fp_copy(ctx->fp2_p1[1][0], t1[0]);
96  fp_copy(ctx->fp2_p1[1][1], t1[1]);
97  fp2_mul(t1, t1, t0);
98  fp_copy(ctx->fp2_p1[2][0], t1[0]);
99  fp_copy(ctx->fp2_p1[2][1], t1[1]);
100  fp2_sqr(t1, t0);
101  fp2_sqr(t1, t1);
102  fp_copy(ctx->fp2_p1[3][0], t1[0]);
103  fp_copy(ctx->fp2_p1[3][1], t1[1]);
104  fp2_mul(t1, t1, t0);
105  fp_copy(ctx->fp2_p1[4][0], t1[0]);
106  fp_copy(ctx->fp2_p1[4][1], t1[1]);
107 
108  /* Compute QNR^(p - (p mod 4))/4. */
109  fp2_set_dig(t1, 1);
110  fp2_mul_nor(t0, t1);
111  e->used = RLC_FP_DIGS;
112  dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
113  bn_div_dig(e, e, 4);
114  fp2_exp(t0, t0, e);
115  fp_copy(ctx->fp2_p2[0][0], t0[0]);
116  fp_copy(ctx->fp2_p2[0][1], t0[1]);
117 
118  /* Compute QNR^(p - (p mod 12))/12. */
119  fp2_set_dig(t1, 1);
120  fp2_mul_nor(t0, t1);
121  e->used = RLC_FP_DIGS;
122  dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
123  bn_div_dig(e, e, 12);
124  fp2_exp(t0, t0, e);
125  fp_copy(ctx->fp2_p2[1][0], t0[0]);
126  fp_copy(ctx->fp2_p2[1][1], t0[1]);
127 
128  /* Compute QNR^(p - (p mod 24))/24. */
129  fp2_set_dig(t1, 1);
130  fp2_mul_nor(t0, t1);
131  e->used = RLC_FP_DIGS;
132  dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
133  bn_div_dig(e, e, 24);
134  fp2_exp(t0, t0, e);
135  fp_copy(ctx->fp2_p2[2][0], t0[0]);
136  fp_copy(ctx->fp2_p2[2][1], t0[1]);
137  } CATCH_ANY {
138  THROW(ERR_CAUGHT);
139  } FINALLY {
140  bn_free(e);
141  fp2_free(t0);
142  fp2_free(t1);
143  }
144 }

146 void fp3_field_init(void) {
147  bn_t e;
148  fp3_t t0, t1, t2;
149  ctx_t *ctx = core_get();
150 
151  bn_null(e);
152  fp3_null(t0);
153  fp3_null(t1);
154  fp3_null(t2);
155 
156  TRY {
157  bn_new(e);
158  fp3_new(t0);
159  fp3_new(t1);
160  fp3_new(t2);
161 
162  /* Compute t0 = u^((p - (p mod 3))/3). */
163  fp_set_dig(ctx->fp3_p0[0], fp_prime_get_cnr());
164  if (fp_prime_get_cnr() < 0) {
165  fp_neg(ctx->fp3_p0[0], ctx->fp3_p0[0]);
166  }
167  e->used = RLC_FP_DIGS;
168  dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
169  bn_div_dig(e, e, 3);
170  fp_exp(ctx->fp3_p0[0], ctx->fp3_p0[0], e);
171  fp_sqr(ctx->fp3_p0[1], ctx->fp3_p0[0]);
172 
173  /* Compute t0 = u^((p - (p mod 6))/6). */
174  fp3_zero(t0);
175  fp_set_dig(t0[1], 1);
176  dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
177  bn_div_dig(e, e, 6);
178  fp3_exp(t0, t0, e);
179 
180  /* Look for a non-trivial subfield element.. */
181  ctx->frb3[0] = 0;
182  while (fp_is_zero(t0[ctx->frb3[0]++]));
183  /* Fill rest of table with powers of constant. */
184  fp_copy(ctx->fp3_p1[0], t0[--ctx->frb3[0] % 3]);
185  fp3_sqr(t1, t0);
186  fp_copy(ctx->fp3_p1[1], t1[(2 * ctx->frb3[0]) % 3]);
187  fp3_mul(t2, t1, t0);
188  fp_copy(ctx->fp3_p1[2], t2[(3 * ctx->frb3[0]) % 3]);
189  fp3_sqr(t2, t1);
190  fp_copy(ctx->fp3_p1[3], t2[(4 * ctx->frb3[0]) % 3]);
191  fp3_mul(t2, t2, t0);
192  fp_copy(ctx->fp3_p1[4], t2[(5 * ctx->frb3[0]) % 3]);
193 
194  /* Compute t0 = u^((p - (p mod 9))/9). */
195  fp3_zero(t0);
196  fp_set_dig(t0[1], 1);
197  dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
198  bn_div_dig(e, e, 9);
199  fp3_exp(t0, t0, e);
200  /* Look for a non-trivial subfield element.. */
201  ctx->frb3[1] = 0;
202  while (fp_is_zero(t0[ctx->frb3[1]++]));
203  fp_copy(ctx->fp3_p2[0], t0[--ctx->frb3[1]]);
204 
205  /* Compute t0 = u^((p - (p mod 18))/18). */
206  fp3_zero(t0);
207  fp_set_dig(t0[1], 1);
208  dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
209  bn_div_dig(e, e, 18);
210  fp3_exp(t0, t0, e);
211  /* Look for a non-trivial subfield element.. */
212  ctx->frb3[2] = 0;
213  while (fp_is_zero(t0[ctx->frb3[2]++]));
214  fp_copy(ctx->fp3_p2[1], t0[--ctx->frb3[2]]);
215  } CATCH_ANY {
216  THROW(ERR_CAUGHT);
217  } FINALLY {
218  bn_free(e);
219  fp3_free(t0);
220  fp3_free(t1);
221  fp3_free(t2);
222  }
223 }
