174void ep_sub(ep_t r, const ep_t p, const ep_t q) {
175 ep_t t;
176
177 ep_null(t);
178
179 if (p == q) {
180 ep_set_infty(r);
181 return;
182 }
183
184 RLC_TRY {
185 ep_new(t);
186 ep_neg(t, q);
187 ep_add(r, p, t);
188 }
189 RLC_CATCH_ANY {
190 RLC_THROW(ERR_CAUGHT);
191 }
192 RLC_FINALLY {
193 ep_free(t);
194 }
