49 static void ep_add_basic_imp(ep_t r, fp_t s, const ep_t p, const ep_t q) {
50  fp_t t0, t1, t2;
51 
52  fp_null(t0);
53  fp_null(t1);
54  fp_null(t2);
55 
56  TRY {
57  fp_new(t0);
58  fp_new(t1);
59  fp_new(t2);
60 
61  /* t0 = x2 - x1. */
62  fp_sub(t0, q->x, p->x);
63  /* t1 = y2 - y1. */
64  fp_sub(t1, q->y, p->y);
65 
66  /* If t0 is zero. */
67  if (fp_is_zero(t0)) {
68  if (fp_is_zero(t1)) {
69  /* If t1 is zero, q = p, should have doubled. */
70  ep_dbl_basic(r, p);
71  } else {
72  /* If t1 is not zero and t0 is zero, q = -p and r = infinity. */
73  ep_set_infty(r);
74  }
75  } else {
76  /* t2 = 1/(x2 - x1). */
77  fp_inv(t2, t0);
78  /* t2 = lambda = (y2 - y1)/(x2 - x1). */
79  fp_mul(t2, t1, t2);
80 
81  /* x3 = lambda^2 - x2 - x1. */
82  fp_sqr(t1, t2);
83  fp_sub(t0, t1, p->x);
84  fp_sub(t0, t0, q->x);
85 
86  /* y3 = lambda * (x1 - x3) - y1. */
87  fp_sub(t1, p->x, t0);
88  fp_mul(t1, t2, t1);
89  fp_sub(r->y, t1, p->y);
90 
91  fp_copy(r->x, t0);
92  fp_copy(r->z, p->z);
93 
94  if (s != NULL) {
95  fp_copy(s, t2);
96  }
97 
98  r->norm = 1;
99  }
100  fp_free(t0);
101  fp_free(t1);
102  fp_free(t2);
103  }
104  CATCH_ANY {
105  THROW(ERR_CAUGHT);
106  }
107  FINALLY {
108  fp_free(t0);
109  fp_free(t1);
110  fp_free(t2);
111  }
112 }

126 static void ep_add_projc_mix(ep_t r, const ep_t p, const ep_t q) {
127  fp_t t0, t1, t2, t3, t4, t5, t6;
128 
129  fp_null(t0);
130  fp_null(t1);
131  fp_null(t2);
132  fp_null(t3);
133  fp_null(t4);
134  fp_null(t5);
135  fp_null(t6);
136 
137  TRY {
138  fp_new(t0);
139  fp_new(t1);
140  fp_new(t2);
141  fp_new(t3);
142  fp_new(t4);
143  fp_new(t5);
144  fp_new(t6);
145 
146  /* madd-2007-bl formulas: 7M + 4S + 9add + 1*4 + 3*2. */
147  /* http://www.hyperelliptic.org/EFD/g1p/auto-shortw-jacobian-3.html#addition-madd-2007-bl */
148 
149  if (!p->norm) {
150  /* t0 = z1^2. */
151  fp_sqr(t0, p->z);
152 
153  /* t3 = U2 = x2 * z1^2. */
154  fp_mul(t3, q->x, t0);
155 
156  /* t1 = S2 = y2 * z1^3. */
157  fp_mul(t1, t0, p->z);
158  fp_mul(t1, t1, q->y);
159 
160  /* t3 = H = U2 - x1. */
161  fp_sub(t3, t3, p->x);
162 
163  /* t1 = R = 2 * (S2 - y1). */
164  fp_sub(t1, t1, p->y);
165  fp_dbl(t1, t1);
166  } else {
167  /* H = x2 - x1. */
168  fp_sub(t3, q->x, p->x);
169 
170  /* t1 = R = 2 * (y2 - y1). */
171  fp_sub(t1, q->y, p->y);
172  fp_dbl(t1, t1);
173  }
174 
175  /* t2 = HH = H^2. */
176  fp_sqr(t2, t3);
177 
178  /* If H is zero. */
179  if (fp_is_zero(t3)) {
180  if (fp_is_zero(t1)) {
181  /* If I is zero, p = q, should have doubled. */
182  ep_dbl_projc(r, p);
183  } else {
184  /* If I is not zero, q = -p, r = infinity. */
185  ep_set_infty(r);
186  }
187  } else {
188  /* t4 = I = 4*HH. */
189  fp_dbl(t4, t2);
190  fp_dbl(t4, t4);
191 
192  /* t5 = J = H * I. */
193  fp_mul(t5, t3, t4);
194 
195  /* t4 = V = x1 * I. */
196  fp_mul(t4, p->x, t4);
197 
198  /* x3 = R^2 - J - 2 * V. */
199  fp_sqr(r->x, t1);
200  fp_sub(r->x, r->x, t5);
201  fp_dbl(t6, t4);
202  fp_sub(r->x, r->x, t6);
203 
204  /* y3 = R * (V - x3) - 2 * Y1 * J. */
205  fp_sub(t4, t4, r->x);
206  fp_mul(t4, t4, t1);
207  fp_mul(t1, p->y, t5);
208  fp_dbl(t1, t1);
209  fp_sub(r->y, t4, t1);
210 
211  if (!p->norm) {
212  /* z3 = (z1 + H)^2 - z1^2 - HH. */
213  fp_add(r->z, p->z, t3);
214  fp_sqr(r->z, r->z);
215  fp_sub(r->z, r->z, t0);
216  fp_sub(r->z, r->z, t2);
217  } else {
218  /* z3 = 2 * H. */
219  fp_dbl(r->z, t3);
220  }
221  }
222  r->norm = 0;
223  }
224  CATCH_ANY {
225  THROW(ERR_CAUGHT);
226  }
227  FINALLY {
228  fp_free(t0);
229  fp_free(t1);
230  fp_free(t2);
231  fp_free(t3);
232  fp_free(t4);
233  fp_free(t5);
234  fp_free(t6);
235  }
236 }

246 static void ep_add_projc_imp(ep_t r, const ep_t p, const ep_t q) {
247 #if defined(EP_MIXED) && defined(STRIP)
248  /* If code size is a problem, leave only the mixed version. */
249  ep_add_projc_mix(r, p, q);
250 #else /* General addition. */
251 
252 #if defined(EP_MIXED) || !defined(STRIP)
253  /* Test if z2 = 1 only if mixed coordinates are turned on. */
254  if (q->norm) {
255  ep_add_projc_mix(r, p, q);
256  return;
257  }
258 #endif
259 
260  fp_t t0, t1, t2, t3, t4, t5, t6;
261 
262  fp_null(t0);
263  fp_null(t1);
264  fp_null(t2);
265  fp_null(t3);
266  fp_null(t4);
267  fp_null(t5);
268  fp_null(t6);
269 
270  TRY {
271  fp_new(t0);
272  fp_new(t1);
273  fp_new(t2);
274  fp_new(t3);
275  fp_new(t4);
276  fp_new(t5);
277  fp_new(t6);
278 
279  /* add-2007-bl formulas: 11M + 5S + 9add + 4*2 */
280  /* http://www.hyperelliptic.org/EFD/g1p/auto-shortw-jacobian-3.html#addition-add-2007-bl */
281 
282  /* t0 = z1^2. */
283  fp_sqr(t0, p->z);
284 
285  /* t1 = z2^2. */
286  fp_sqr(t1, q->z);
287 
288  /* t2 = U1 = x1 * z2^2. */
289  fp_mul(t2, p->x, t1);
290 
291  /* t3 = U2 = x2 * z1^2. */
292  fp_mul(t3, q->x, t0);
293 
294  /* t6 = z1^2 + z2^2. */
295  fp_add(t6, t0, t1);
296 
297  /* t0 = S2 = y2 * z1^3. */
298  fp_mul(t0, t0, p->z);
299  fp_mul(t0, t0, q->y);
300 
301  /* t1 = S1 = y1 * z2^3. */
302  fp_mul(t1, t1, q->z);
303  fp_mul(t1, t1, p->y);
304 
305  /* t3 = H = U2 - U1. */
306  fp_sub(t3, t3, t2);
307 
308  /* t0 = R = 2 * (S2 - S1). */
309  fp_sub(t0, t0, t1);
310  fp_dbl(t0, t0);
311 
312  /* If E is zero. */
313  if (fp_is_zero(t3)) {
314  if (fp_is_zero(t0)) {
315  /* If I is zero, p = q, should have doubled. */
316  ep_dbl_projc(r, p);
317  } else {
318  /* If I is not zero, q = -p, r = infinity. */
319  ep_set_infty(r);
320  }
321  } else {
322  /* t4 = I = (2*H)^2. */
323  fp_dbl(t4, t3);
324  fp_sqr(t4, t4);
325 
326  /* t5 = J = H * I. */
327  fp_mul(t5, t3, t4);
328 
329  /* t4 = V = U1 * I. */
330  fp_mul(t4, t2, t4);
331 
332  /* x3 = R^2 - J - 2 * V. */
333  fp_sqr(r->x, t0);
334  fp_sub(r->x, r->x, t5);
335  fp_dbl(t2, t4);
336  fp_sub(r->x, r->x, t2);
337 
338  /* y3 = R * (V - x3) - 2 * S1 * J. */
339  fp_sub(t4, t4, r->x);
340  fp_mul(t4, t4, t0);
341  fp_mul(t1, t1, t5);
342  fp_dbl(t1, t1);
343  fp_sub(r->y, t4, t1);
344 
345  /* z3 = ((z1 + z2)^2 - z1^2 - z2^2) * H. */
346  fp_add(r->z, p->z, q->z);
347  fp_sqr(r->z, r->z);
348  fp_sub(r->z, r->z, t6);
349  fp_mul(r->z, r->z, t3);
350  }
351  r->norm = 0;
352  }
353  CATCH_ANY {
354  THROW(ERR_CAUGHT);
355  }
356  FINALLY {
357  fp_free(t0);
358  fp_free(t1);
359  fp_free(t2);
360  fp_free(t3);
361  fp_free(t4);
362  fp_free(t5);
363  fp_free(t6);
364  }
365 #endif
