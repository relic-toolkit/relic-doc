42 void fp6_sqr_basic(fp6_t c, fp6_t a) {
43  fp2_t t0, t1, t2, t3, t4;
44 
45  fp2_null(t0);
46  fp2_null(t1);
47  fp2_null(t2);
48  fp2_null(t3);
49  fp2_null(t4);
50 
51  TRY {
52  fp2_new(t0);
53  fp2_new(t1);
54  fp2_new(t2);
55  fp2_new(t3);
56  fp2_new(t4);
57 
58  /* t0 = a_0^2 */
59  fp2_sqr(t0, a[0]);
60 
61  /* t1 = 2 * a_1 * a_2 */
62  fp2_mul(t1, a[1], a[2]);
63  fp2_dbl(t1, t1);
64 
65  /* t2 = a_2^2. */
66  fp2_sqr(t2, a[2]);
67 
68  /* c2 = a_0 + a_2. */
69  fp2_add(c[2], a[0], a[2]);
70 
71  /* t3 = (a_0 + a_2 + a_1)^2. */
72  fp2_add(t3, c[2], a[1]);
73  fp2_sqr(t3, t3);
74 
75  /* c2 = (a_0 + a_2 - a_1)^2. */
76  fp2_sub(c[2], c[2], a[1]);
77  fp2_sqr(c[2], c[2]);
78 
79  /* c2 = (c2 + t3)/2. */
80  fp2_add(c[2], c[2], t3);
81  fp_hlv(c[2][0], c[2][0]);
82  fp_hlv(c[2][1], c[2][1]);
83 
84  /* t3 = t3 - c2 - t1. */
85  fp2_sub(t3, t3, c[2]);
86  fp2_sub(t3, t3, t1);
87 
88  /* c2 = c2 - t0 - t2. */
89  fp2_sub(c[2], c[2], t0);
90  fp2_sub(c[2], c[2], t2);
91 
92  /* c0 = t0 + t1 * E. */
93  fp2_mul_nor(t4, t1);
94  fp2_add(c[0], t0, t4);
95 
96  /* c1 = t3 + t2 * E. */
97  fp2_mul_nor(t4, t2);
98  fp2_add(c[1], t3, t4);
99  } CATCH_ANY {
100  THROW(ERR_CAUGHT);
101  } FINALLY {
102  fp2_free(t0);
103  fp2_free(t1);
104  fp2_free(t2);
105  fp2_free(t3);
106  fp2_free(t4);
107  }
108 }
