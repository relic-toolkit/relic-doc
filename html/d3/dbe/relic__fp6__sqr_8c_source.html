42void fp6_sqr_basic(fp6_t c, const fp6_t a) {
43 fp2_t t0, t1, t2, t3, t4;
44
45 fp2_null(t0);
46 fp2_null(t1);
47 fp2_null(t2);
48 fp2_null(t3);
49 fp2_null(t4);
50
51 RLC_TRY {
52 fp2_new(t0);
53 fp2_new(t1);
54 fp2_new(t2);
55 fp2_new(t3);
56 fp2_new(t4);
57
58 /* t0 = a_0^2 */
59 fp2_sqr(t0, a[0]);
60
61 /* t1 = 2 * a_1 * a_2 */
62 fp2_mul(t1, a[1], a[2]);
63 fp2_dbl(t1, t1);
64
65 /* t2 = a_2^2. */
66 fp2_sqr(t2, a[2]);
67
68 /* c2 = a_0 + a_2. */
69 fp2_add(c[2], a[0], a[2]);
70
71 /* t3 = (a_0 + a_2 + a_1)^2. */
72 fp2_add(t3, c[2], a[1]);
73 fp2_sqr(t3, t3);
74
75 /* c2 = (a_0 + a_2 - a_1)^2. */
76 fp2_sub(c[2], c[2], a[1]);
77 fp2_sqr(c[2], c[2]);
78
79 /* c2 = (c2 + t3)/2. */
80 fp2_add(c[2], c[2], t3);
81 fp_hlv(c[2][0], c[2][0]);
82 fp_hlv(c[2][1], c[2][1]);
83
84 /* t3 = t3 - c2 - t1. */
85 fp2_sub(t3, t3, c[2]);
86 fp2_sub(t3, t3, t1);
87
88 /* c2 = c2 - t0 - t2. */
89 fp2_sub(c[2], c[2], t0);
90 fp2_sub(c[2], c[2], t2);
91
92 /* c0 = t0 + t1 * E. */
93 fp2_mul_nor(t4, t1);
94 fp2_add(c[0], t0, t4);
95
96 /* c1 = t3 + t2 * E. */
97 fp2_mul_nor(t4, t2);
98 fp2_add(c[1], t3, t4);
99 } RLC_CATCH_ANY {
100 RLC_THROW(ERR_CAUGHT);
101 } RLC_FINALLY {
102 fp2_free(t0);
103 fp2_free(t1);
104 fp2_free(t2);
105 fp2_free(t3);
106 fp2_free(t4);
107 }
108}

114void fp6_sqr_unr(dv6_t c, const fp6_t a) {
115 dv2_t u0, u1, u2, u3, u4, u5;
116 fp2_t t0, t1, t2, t3;
117
118 dv2_null(u0);
119 dv2_null(u1);
120 dv2_null(u2);
121 dv2_null(u3);
122 dv2_null(u4);
123 dv2_null(u5);
124 fp2_null(t0);
125 fp2_null(t1);
126 fp2_null(t2);
127 fp2_null(t3);
128
129 RLC_TRY {
130 dv2_new(u0);
131 dv2_new(u1);
132 dv2_new(u2);
133 dv2_new(u3);
134 dv2_new(u4);
135 dv2_new(u5);
136 fp2_new(t0);
137 fp2_new(t1);
138 fp2_new(t2);
139 fp2_new(t3);
140
141 /* u0 = a_0^2 */
142 fp2_sqrn_low(u0, a[0]);
143
144 /* t1 = 2 * a_1 * a_2 */
145 fp2_dblm_low(t0, a[1]);
146
147 fp2_muln_low(u1, t0, a[2]);
148
149 /* u2 = a_2^2. */
150 fp2_sqrn_low(u2, a[2]);
151
152 /* t4 = a_0 + a_2. */
153 fp2_addm_low(t3, a[0], a[2]);
154
155 /* u3 = (a_0 + a_2 + a_1)^2. */
156 fp2_addm_low(t2, t3, a[1]);
157 fp2_sqrn_low(u3, t2);
158
159 /* u4 = (a_0 + a_2 - a_1)^2. */
160 fp2_subm_low(t1, t3, a[1]);
161 fp2_sqrn_low(u4, t1);
162
163 /* u4 = (u4 + u3)/2. */
164 fp2_addc_low(u4, u4, u3);
165
166 fp_hlvd_low(u4[0], u4[0]);
167 fp_hlvd_low(u4[1], u4[1]);
168
169 /* u3 = u3 - u4 - u1. */
170 fp2_addc_low(u5, u1, u4);
171 fp2_subc_low(u3, u3, u5);
172
173 /* c2 = u4 - u0 - u2. */
174 fp2_addc_low(u5, u0, u2);
175 fp2_subc_low(c[2], u4, u5);
176
177 /* c0 = u0 + u1 * E. */
178 fp2_nord_low(u4, u1);
179 fp2_addc_low(c[0], u0, u4);
180
181 /* c1 = u3 + u2 * E. */
182 fp2_nord_low(u4, u2);
183 fp2_addc_low(c[1], u3, u4);
184 } RLC_CATCH_ANY {
185 RLC_THROW(ERR_CAUGHT);
186 } RLC_FINALLY {
187 dv2_free(u0);
188 dv2_free(u1);
189 dv2_free(u2);
190 dv2_free(u3);
191 dv2_free(u4);
192 dv2_free(u5);
193 fp2_free(t0);
194 fp2_free(t1);
195 fp2_free(t2);
196 fp2_free(t3);
197 }
198}
