42void fp16_sqr_basic(fp16_t c, const fp16_t a) {
43 fp8_t t0, t1;
44
45 fp8_null(t0);
46 fp8_null(t1);
47
48 RLC_TRY {
49 fp8_new(t0);
50 fp8_new(t1);
51
52 fp8_add(t0, a[0], a[1]);
53 fp8_mul_art(t1, a[1]);
54 fp8_add(t1, a[0], t1);
55 fp8_mul(t0, t0, t1);
56 fp8_mul(c[1], a[0], a[1]);
57 fp8_sub(c[0], t0, c[1]);
58 fp8_mul_art(t1, c[1]);
59 fp8_sub(c[0], c[0], t1);
60 fp8_dbl(c[1], c[1]);
61 } RLC_CATCH_ANY {
62 RLC_THROW(ERR_CAUGHT);
63 } RLC_FINALLY {
64 fp8_free(t0);
65 fp8_free(t1);
66 }
67}

73void fp16_sqr_unr(dv16_t c, const fp16_t a) {
74 fp8_t t;
75 dv8_t u0, u1, u2;
76
77 fp8_null(t);
78 dv8_null(u0);
79 dv8_null(u1);
80 dv8_null(u2);
81
82 RLC_TRY {
83 fp8_new(t);
84 dv8_new(u0);
85 dv8_new(u1);
86 dv8_new(u2);
87
88 /* t0 = a^2. */
89 fp8_sqr_unr(u0, a[0]);
90 /* t1 = b^2. */
91 fp8_sqr_unr(u1, a[1]);
92
93 fp8_add(t, a[0], a[1]);
94
95 /* c = a^2 + b^2 * E. */
96 dv_copy(u2[1][0][0], u1[0][0][0], 2 * RLC_FP_DIGS);
97 dv_copy(u2[1][0][1], u1[0][0][1], 2 * RLC_FP_DIGS);
98 dv_copy(u2[1][1][0], u1[0][1][0], 2 * RLC_FP_DIGS);
99 dv_copy(u2[1][1][1], u1[0][1][1], 2 * RLC_FP_DIGS);
100 fp2_nord_low(u2[0][0], u1[1][1]);
101 dv_copy(u2[0][1][0], u1[1][0][0], 2 * RLC_FP_DIGS);
102 dv_copy(u2[0][1][1], u1[1][0][1], 2 * RLC_FP_DIGS);
103 fp2_addc_low(c[0][0][0], u2[0][0], u0[0][0]);
104 fp2_addc_low(c[0][0][1], u2[0][1], u0[0][1]);
105 fp2_addc_low(c[0][1][0], u2[1][0], u0[1][0]);
106 fp2_addc_low(c[0][1][1], u2[1][1], u0[1][1]);
107
108 /* d = (a + b)^2 - a^2 - b^2 = 2 * a * b. */
109 fp2_addc_low(u1[0][0], u1[0][0], u0[0][0]);
110 fp2_addc_low(u1[0][1], u1[0][1], u0[0][1]);
111 fp2_addc_low(u1[1][0], u1[1][0], u0[1][0]);
112 fp2_addc_low(u1[1][1], u1[1][1], u0[1][1]);
113
114 fp8_sqr_unr(u0, t);
115 fp2_subc_low(c[1][0][0], u0[0][0], u1[0][0]);
116 fp2_subc_low(c[1][0][1], u0[0][1], u1[0][1]);
117 fp2_subc_low(c[1][1][0], u0[1][0], u1[1][0]);
118 fp2_subc_low(c[1][1][1], u0[1][1], u1[1][1]);
119 } RLC_CATCH_ANY {
120 RLC_THROW(ERR_CAUGHT);
121 } RLC_FINALLY {
122 fp8_free(t);
123 dv8_free(u0);
124 dv8_free(u1);
125 dv8_free(u2);
126 }
127}

170 fp8_sub(c[1], t2, c[0]);
171 fp8_dbl(c[0], c[0]);
172 fp_add_dig(c[0][0][0][0], c[0][0][0][0], 1);
173 fp_sub_dig(c[1][0][0][0], c[1][0][0][0], 1);
174 } RLC_CATCH_ANY {
