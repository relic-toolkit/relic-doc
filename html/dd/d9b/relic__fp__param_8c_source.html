549 int fp_param_set_any_dense(void) {
550  bn_t p;
551  int result = RLC_OK;
552 
553  bn_null(p);
554 
555  TRY {
556  bn_new(p);
557 #ifdef FP_QNRES
558  do {
559  bn_gen_prime(p, RLC_FP_BITS);
560  } while ((p->dp[0] & 0x7) != 3);
561 #else
562  bn_gen_prime(p, RLC_FP_BITS);
563 #endif
564  if (!bn_is_prime(p)) {
565  result = RLC_ERR;
566  } else {
567  fp_prime_set_dense(p);
568  }
569  }
570  CATCH_ANY {
571  THROW(ERR_CAUGHT);
572  }
573  FINALLY {
574  bn_free(p);
575  }
576  return result;
