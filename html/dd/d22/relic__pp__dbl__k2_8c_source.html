44void pp_dbl_k2_basic(fp2_t l, ep_t r, const ep_t p, const ep_t q) {
45 fp_t s;
46 ep_t t;
47
48 fp_null(s);
49 ep_null(t);
50
51 RLC_TRY {
52 fp_new(s);
53 ep_new(t);
54
55 ep_copy(t, p);
56 ep_dbl_slp_basic(r, s, p);
57 fp_add(l[0], t->x, q->x);
58 fp_mul(l[0], l[0], s);
59 fp_sub(l[0], t->y, l[0]);

61 } RLC_CATCH_ANY {

75void pp_dbl_k2_projc_basic(fp2_t l, ep_t r, const ep_t p, const ep_t q) {
76 fp_t t0, t1, t2, t3, t4, t5;
77
78 fp_null(t0);
79 fp_null(t1);
80 fp_null(t2);
81 fp_null(t3);
82 fp_null(t4);
83 fp_null(t5);
84
85 RLC_TRY {
86 fp_new(t0);
87 fp_new(t1);
88 fp_new(t2);
89 fp_new(t3);
90 fp_new(t4);
91 fp_new(t5);
92
93 /* For these curves, we always can choose a = -3. */
94 /* dbl-2001-b formulas: 3M + 5S + 8add + 1*4 + 2*8 + 1*3 */
95 /* http://www.hyperelliptic.org/EFD/g1p/auto-shortw-jacobian-3.html#doubling-dbl-2001-b */
96
97 /* t0 = delta = z1^2. */
98 fp_sqr(t0, p->z);
99
100 /* t1 = gamma = y1^2. */
101 fp_sqr(t1, p->y);
102
103 /* t2 = beta = x1 * y1^2. */
104 fp_mul(t2, p->x, t1);
105
106 /* t3 = alpha = 3 * (x1 - z1^2) * (x1 + z1^2). */
107 fp_sub(t3, p->x, t0);
108 fp_add(t4, p->x, t0);
109 fp_mul(t4, t3, t4);
110 fp_dbl(t3, t4);
111 fp_add(t3, t3, t4);
112
113 fp_dbl(t2, t2);
114 fp_dbl(t2, t2);
115
116 /* z3 = (y1 + z1)^2 - gamma - delta. */
117 fp_add(r->z, p->y, p->z);
118 fp_sqr(r->z, r->z);
119 fp_sub(r->z, r->z, t1);
120 fp_sub(r->z, r->z, t0);
121
122 /* l0 = 2 * gamma - alpha * (delta * xq + x1). */
123 fp_dbl(t1, t1);
124 fp_mul(t5, t0, q->x);
125 fp_add(t5, t5, p->x);
126 fp_mul(t5, t5, t3);
127 fp_sub(l[0], t1, t5);
128
129 /* x3 = alpha^2 - 8 * beta. */
130 fp_dbl(t5, t2);
131 fp_sqr(r->x, t3);
132 fp_sub(r->x, r->x, t5);
133
134 /* y3 = alpha * (4 * beta - x3) - 8 * gamma^2. */
135 fp_sqr(t4, t1);
136 fp_dbl(t4, t4);
137 fp_sub(r->y, t2, r->x);
138 fp_mul(r->y, r->y, t3);
139 fp_sub(r->y, r->y, t4);
140
141 /* l1 = - z3 * delta * yq. */
142 fp_mul(l[1], r->z, t0);
143 fp_mul(l[1], l[1], q->y);
144
145 r->coord = JACOB;
146 }
147 RLC_CATCH_ANY {
148 RLC_THROW(ERR_CAUGHT);
149 }
150 RLC_FINALLY {
151 fp_free(t0);
152 fp_free(t1);
153 fp_free(t2);
154 fp_free(t3);
155 fp_free(t4);
156 fp_free(t5);
157 }
158}
