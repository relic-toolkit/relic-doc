40int eb_cmp(const eb_t p, const eb_t q) {
41 eb_t r, s;
42 int result = RLC_NE;
43
44 if (eb_is_infty(p) && eb_is_infty(q)) {
45 return RLC_EQ;
46 }
47
48 eb_null(r);
49 eb_null(s);
50
51 RLC_TRY {
52 eb_new(r);
53 eb_new(s);
54
55 if ((p->coord == PROJC) && (q->coord == PROJC)) {
56 /* If the two points are not normalized, it is faster to compare
57 * x1 * z2 == x2 * z1 and y1 * z2^2 == y2 * z1^2. */
58 fb_mul(r->x, p->x, q->z);
59 fb_mul(s->x, q->x, p->z);
60 fb_sqr(r->z, p->z);
61 fb_sqr(s->z, q->z);
62 fb_mul(r->y, p->y, s->z);
63 fb_mul(s->y, q->y, r->z);
64 } else {
65 if (p->coord == BASIC) {
66 eb_copy(r, p);
67 } else {
68 eb_norm(r, p);
69 }
70
71 if (q->coord == BASIC) {
72 eb_copy(s, q);
73 } else {
74 eb_norm(s, q);
75 }
76 }
77
78 if ((fb_cmp(r->x, s->x) == RLC_EQ) && (fb_cmp(r->y, s->y) == RLC_EQ)) {
79 result = RLC_EQ;
80 }
81 } RLC_CATCH_ANY {
82 RLC_THROW(ERR_CAUGHT);
83 } RLC_FINALLY {
84 eb_free(r);
85 eb_free(s);
86 }
87
88 return result;
89}
