39void ep2_frb(ep2_t r, const ep2_t p, int i) {
40 if (ep2_curve_opt_a() == RLC_ZERO) {
41 ctx_t *ctx = core_get();
42
43 ep2_copy(r, p);
44 for (; i > 0; i--) {
45 fp2_frb(r->x, r->x, 1);
46 fp2_frb(r->y, r->y, 1);
47 fp2_frb(r->z, r->z, 1);
48 fp2_mul(r->x, r->x, ctx->ep2_frb[0]);
49 fp2_mul(r->y, r->y, ctx->ep2_frb[1]);
50 }
51 } else {
52 bn_t t;
53
54 bn_null(t);
55
56 RLC_TRY {
57 bn_new(t);
58
59 /* Can we do faster than this? */
60 fp_prime_get_par(t);
61 for (; i > 0; i--) {
62 ep2_mul_basic(r, p, t);
63 }
64 } RLC_CATCH_ANY {
65 RLC_THROW(ERR_NO_MEMORY);
66 } RLC_FINALLY {
67 bn_free(t);
68 }
69 }
