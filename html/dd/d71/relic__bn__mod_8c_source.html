40void bn_mod_2b(bn_t c, const bn_t a, int b) {
41 int i, first, d;
42
43 if (b <= 0) {
44 bn_zero(c);
45 return;
46 }
47
48 if (b >= (int)(a->used * RLC_DIG)) {
49 bn_copy(c, a);
50 return;
51 }
52
53 bn_copy(c, a);
54
55 RLC_RIP(b, d, b);
56
57 first = (d) + (b == 0 ? 0 : 1);
58 for (i = first; i < c->used; i++) {
59 c->dp[i] = 0;
60 }
61
62 c->dp[d] &= RLC_MASK(b);
63
64 bn_trim(c);
65}

87void bn_mod_barrt(bn_t c, const bn_t a, const bn_t m, const bn_t u) {
88 bn_t q, t;
89 int mu, neg;
90
91 bn_null(q);
92 bn_null(t);
93
94 if (bn_is_zero(m) || bn_sign(m) != RLC_POS) {
95 RLC_THROW(ERR_NO_VALID);
96 return;
97 }
98
99 if (bn_cmp_abs(a, m) == RLC_LT) {
100 bn_copy(c, a);
101 return;
102 }
103
104 if (a->used > 2 * m->used) {
105 bn_mod(c, a, m);
106 return;
107 }
108
109 RLC_TRY {
110 bn_new(q);
111 bn_new(t);
112 bn_zero(t);
113
114 neg = (bn_sign(a) == RLC_NEG);
115 bn_abs(c, a);
116
117 bn_rsh(q, c, (m->used - 1) * RLC_DIG);
118
119 if (m->used > ((dig_t)1) << (RLC_DIG - 1)) {
120 bn_mul(t, q, u);
121 } else {
122 bn_grow(t, q->used + u->used);
123 t->used = q->used + u->used;
124 mu = u->used - q->used;
125 if (q->used > u->used) {
126 bn_muld_low(t->dp, q->dp, q->used, u->dp, u->used, mu, t->used);
127 } else {
128 mu = (mu > u->used - q->used ? mu - (u->used - q->used) : 0);
129 bn_muld_low(t->dp, u->dp, u->used, q->dp, q->used, mu, t->used);
130 }
131 bn_trim(t);
132 }
133
134 bn_rsh(q, t, (m->used + 1) * RLC_DIG);
135
136 if (q->used > m->used) {
137 bn_muld_low(t->dp, q->dp, q->used, m->dp, m->used, 0, q->used + 1);
138 } else {
139 bn_muld_low(t->dp, m->dp, m->used, q->dp, q->used, 0, m->used + 1);
140 }
141 t->used = m->used + 1;
142 bn_trim(t);
143
144 bn_mod_2b(q, t, RLC_DIG * (m->used + 1));
145 bn_mod_2b(t, c, RLC_DIG * (m->used + 1));
146 bn_sub(t, t, q);
147
148 if (bn_sign(t) == RLC_NEG) {
149 bn_set_dig(q, (dig_t)1);
150 bn_lsh(q, q, (m->used + 1) * RLC_DIG);
151 bn_add(t, t, q);
152 }
153
154 while (bn_cmp(t, m) != RLC_LT) {
155 bn_sub(t, t, m);
156 }
157
158 bn_copy(c, t);
159 if (neg) {
160 bn_sub(c, m, c);
161 }
162 }
163 RLC_CATCH_ANY {
164 RLC_THROW(ERR_CAUGHT);
165 }
166 RLC_FINALLY {
167 bn_free(q);
168 bn_free(t);
169 }
170}

176void bn_mod_pre_monty(bn_t u, const bn_t m) {
177 dig_t x, b = m->dp[0];
178
179 if (bn_is_even(m) || bn_sign(m) != RLC_POS) {
180 RLC_THROW(ERR_NO_VALID);
181 return;
182 }
183
184 x = (((b + 2) & 4) << 1) + b; /* here x*a==1 mod 2**4 */
185 x *= (dig_t)2 - b * x; /* here x*a==1 mod 2**8 */
186#if WSIZE > 8
187 x *= (dig_t)2 - b * x; /* here x*a==1 mod 2**16 */
188#endif
189#if WSIZE > 16
190 x *= (dig_t)2 - b * x; /* here x*a==1 mod 2**32 */
191#endif
192#if WSIZE > 32
193 x *= (dig_t)2 - b * x; /* here x*a==1 mod 2**64 */
194#endif
195 /* u = -1/m0 (mod 2^RLC_DIG) */
196 bn_set_dig(u, -x);
197}

233void bn_mod_monty_basic(bn_t c, const bn_t a, const bn_t m, const bn_t u) {
234 int digits, i;
235 dig_t r, u0, *tmp;
236 bn_t t;
237
238 if (bn_is_even(m) || bn_sign(m) != RLC_POS) {
239 RLC_THROW(ERR_NO_VALID);
240 return;
241 }
242
243 bn_null(t);
244 digits = 2 * m->used;
245
246 RLC_TRY {
247 bn_new_size(t, digits);
248 bn_zero(t);
249 bn_copy(t, a);
250
251 u0 = u->dp[0];
252 tmp = t->dp;
253
254 for (i = 0; i < m->used; i++, tmp++) {
255 r = (dig_t)(*tmp * u0);
256 *tmp = bn_mula_low(tmp, m->dp, r, m->used);
257 }
258 if (bn_addn_low(t->dp, t->dp, tmp, m->used)) {
259 bn_subn_low(t->dp, t->dp, m->dp, m->used);
260 }
261 t->used = m->used;
262 bn_trim(t);
263
264 if (bn_cmp_abs(t, m) != RLC_LT) {
265 bn_sub(t, t, m);
266 }
267
268 bn_copy(c, t);
269 }
270 RLC_CATCH_ANY {
271 RLC_THROW(ERR_CAUGHT);
272 }
273 RLC_FINALLY {
274 bn_free(t);
275 }
276}

331void bn_mod_pmers(bn_t c, const bn_t a, const bn_t m, const bn_t u) {
332 bn_t q, t;
333 int neg = 0, bits = bn_bits(m);
334
335 if (bn_is_zero(m) || bn_sign(m) != RLC_POS) {
336 RLC_THROW(ERR_NO_VALID);
337 return;
338 }
339
340 bn_null(q);
341 bn_null(t);
342
343 RLC_TRY {
344 /* Implement algorithm 10.25 from HEHC. */
345
346 bn_new(q);
347 bn_new(t);
348
349 bn_copy(c, a);
350 if (bn_sign(c) == RLC_NEG) {
351 neg = 1;
352 bn_sub(c, m, c);
353 }
354
355 bn_rsh(q, c, bits);
356 bn_mod_2b(c, c, bits);
357
358 while (bits > 0 && !bn_is_zero(q)) {
359 if (u -> used == 1) {
360 bn_mul_dig(t, q, u->dp[0]);
361 } else {
362 bn_mul(t, q, u);
363 }
364 bn_rsh(q, t, bits);
365 bn_mod_2b(t, t, bits);
366
367 bn_add(c, c, t);
368 }
369 while (bits > 0 && bn_cmp_abs(c, m) != RLC_LT) {
370 bn_sub(c, c, m);
371 }
372
373 if (neg) {
374 bn_sub(c, m, c);
375 }
376 }
377 RLC_CATCH_ANY {
378 RLC_THROW(ERR_CAUGHT);
379 }
380 RLC_FINALLY {
381 bn_free(t);
382 bn_free(q);
383 }
384}
