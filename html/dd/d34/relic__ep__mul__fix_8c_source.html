48 static void ep_mul_fix_plain(ep_t r, const ep_t *t, const bn_t k) {
49  int l, i, n;
50  int8_t naf[RLC_FP_BITS + 1];
51 
52  if (bn_is_zero(k)) {
53  ep_set_infty(r);
54  return;
55  }
56 
57  /* Compute the w-TNAF representation of k. */
58  l = RLC_FP_BITS + 1;
59  bn_rec_naf(naf, &l, k, EP_DEPTH);
60 
61  n = naf[l - 1];
62  if (n > 0) {
63  ep_copy(r, t[n / 2]);
64  } else {
65  ep_neg(r, t[-n / 2]);
66  }
67 
68  for (i = l - 2; i >= 0; i--) {
69  ep_dbl(r, r);
70 
71  n = naf[i];
72  if (n > 0) {
73  ep_add(r, r, t[n / 2]);
74  }
75  if (n < 0) {
76  ep_sub(r, r, t[-n / 2]);
77  }
78  }
79  /* Convert r to affine coordinates. */
80  ep_norm(r, r);
81  if (bn_sign(k) == RLC_NEG) {
82  ep_neg(r, r);
83  }
84 }

100 static void ep_mul_combs_endom(ep_t r, const ep_t *t, const bn_t k) {
101  int i, j, l, w0, w1, n0, n1, p0, p1, s0, s1;
102  bn_t n, k0, k1, v1[3], v2[3];
103  ep_t u;
104 
105  if (bn_is_zero(k)) {
106  ep_set_infty(r);
107  return;
108  }
109 
110  bn_null(n);
111  bn_null(k0);
112  bn_null(k1);
113  ep_null(u);
114 
115  TRY {
116  bn_new(n);
117  bn_new(k0);
118  bn_new(k1);
119  ep_new(u);
120  for (i = 0; i < 3; i++) {
121  bn_null(v1[i]);
122  bn_null(v2[i]);
123  bn_new(v1[i]);
124  bn_new(v2[i]);
125  }
126 
127  ep_curve_get_ord(n);
128  ep_curve_get_v1(v1);
129  ep_curve_get_v2(v2);
130  l = bn_bits(n);
131  l = ((l % (2 * EP_DEPTH)) ==
132  0 ? (l / (2 * EP_DEPTH)) : (l / (2 * EP_DEPTH)) + 1);
133 
134  bn_rec_glv(k0, k1, k, n, (const bn_t *)v1, (const bn_t *)v2);
135  s0 = bn_sign(k0);
136  s1 = bn_sign(k1);
137  bn_abs(k0, k0);
138  bn_abs(k1, k1);
139 
140  n0 = bn_bits(k0);
141  n1 = bn_bits(k1);
142 
143  p0 = (EP_DEPTH) * l - 1;
144 
145  ep_set_infty(r);
146 
147  for (i = l - 1; i >= 0; i--) {
148  ep_dbl(r, r);
149 
150  w0 = 0;
151  w1 = 0;
152  p1 = p0--;
153  for (j = EP_DEPTH - 1; j >= 0; j--, p1 -= l) {
154  w0 = w0 << 1;
155  w1 = w1 << 1;
156  if (p1 < n0 && bn_get_bit(k0, p1)) {
157  w0 = w0 | 1;
158  }
159  if (p1 < n1 && bn_get_bit(k1, p1)) {
160  w1 = w1 | 1;
161  }
162  }
163  if (w0 > 0) {
164  if (s0 == RLC_POS) {
165  ep_add(r, r, t[w0]);
166  } else {
167  ep_sub(r, r, t[w0]);
168  }
169  }
170  if (w1 > 0) {
171  ep_copy(u, t[w1]);
172  if (ep_curve_opt_a() == RLC_ZERO) {
173  fp_mul(u->x, u->x, ep_curve_get_beta());
174  } else {
175  fp_neg(u->x, u->x);
176  fp_mul(u->y, u->y, ep_curve_get_beta());
177  }
178  if (s1 == RLC_NEG) {
179  ep_neg(u, u);
180  }
181  ep_add(r, r, u);
182  }
183  }
184  ep_norm(r, r);
185  if (bn_sign(k) == RLC_NEG) {
186  ep_neg(r, r);
187  }
188  }
189  CATCH_ANY {
190  THROW(ERR_CAUGHT);
191  }
192  FINALLY {
193  bn_free(n);
194  bn_free(k0);
195  bn_free(k1);
196  ep_free(u);
197  for (i = 0; i < 3; i++) {
198  bn_free(v1[i]);
199  bn_free(v2[i]);
200  }
201  }
202 }
203 
204 #endif /* EP_ENDOM */

440 void ep_mul_fix_combd(ep_t r, const ep_t *t, const bn_t k) {
441  int i, j, d, e, w0, w1, n0, p0, p1;
442  bn_t n;
443 
444  if (bn_is_zero(k)) {
445  ep_set_infty(r);
446  return;
447  }
448 
449  bn_null(n);
450 
451  TRY {
452  bn_new(n);
453 
454  ep_curve_get_ord(n);
455  d = RLC_CEIL(bn_bits(n), EP_DEPTH);
456  e = (d % 2 == 0 ? (d / 2) : (d / 2) + 1);
457 
458  ep_set_infty(r);
459  n0 = bn_bits(k);
460 
461  p1 = (e - 1) + (EP_DEPTH - 1) * d;
462  for (i = e - 1; i >= 0; i--) {
463  ep_dbl(r, r);
464 
465  w0 = 0;
466  p0 = p1;
467  for (j = EP_DEPTH - 1; j >= 0; j--, p0 -= d) {
468  w0 = w0 << 1;
469  if (p0 < n0 && bn_get_bit(k, p0)) {
470  w0 = w0 | 1;
471  }
472  }
473 
474  w1 = 0;
475  p0 = p1-- + e;
476  for (j = EP_DEPTH - 1; j >= 0; j--, p0 -= d) {
477  w1 = w1 << 1;
478  if (i + e < d && p0 < n0 && bn_get_bit(k, p0)) {
479  w1 = w1 | 1;
480  }
481  }
482 
483  ep_add(r, r, t[w0]);
484  ep_add(r, r, t[(1 << EP_DEPTH) + w1]);
485  }
486  ep_norm(r, r);
487  if (bn_sign(k) == RLC_NEG) {
488  ep_neg(r, r);
489  }
490  }
491  CATCH_ANY {

493  }
494  FINALLY {
