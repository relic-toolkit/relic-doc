48static void ep_mul_fix_plain(ep_t r, const ep_t *t, const bn_t k) {
49 int i, n;
50 int8_t naf[RLC_FP_BITS + 1];
51 size_t l;
52
53 /* Compute the w-TNAF representation of k. */
54 l = RLC_FP_BITS + 1;
55 bn_rec_naf(naf, &l, k, RLC_DEPTH);
56
57 n = naf[l - 1];
58 if (n > 0) {
59 ep_copy(r, t[n / 2]);
60 } else {
61 ep_neg(r, t[-n / 2]);
62 }
63
64 for (i = l - 2; i >= 0; i--) {
65 ep_dbl(r, r);
66
67 n = naf[i];
68 if (n > 0) {
69 ep_add(r, r, t[n / 2]);
70 }
71 if (n < 0) {
72 ep_sub(r, r, t[-n / 2]);
73 }
74 }
75 /* Convert r to affine coordinates. */
76 ep_norm(r, r);
77 if (bn_sign(k) == RLC_NEG) {
78 ep_neg(r, r);
79 }
80}

96static void ep_mul_combs_endom(ep_t r, const ep_t *t, const bn_t k) {
97 int i, j, l, w0, w1, n0, n1, p0, p1, s0, s1;
98 bn_t n, _k, k0, k1, v1[3], v2[3];
99 ep_t u;
100
101 bn_null(n);
102 bn_null(_k);
103 bn_null(k0);
104 bn_null(k1);
105 ep_null(u);
106
107 RLC_TRY {
108 bn_new(n);
109 bn_new(_k);
110 bn_new(k0);
111 bn_new(k1);
112 ep_new(u);
113 for (i = 0; i < 3; i++) {
114 bn_null(v1[i]);
115 bn_null(v2[i]);
116 bn_new(v1[i]);
117 bn_new(v2[i]);
118 }
119
120 ep_curve_get_ord(n);
121 ep_curve_get_v1(v1);
122 ep_curve_get_v2(v2);
123 l = RLC_CEIL(bn_bits(n), (2 * RLC_DEPTH));
124
125 bn_mod(_k, k, n);
126 bn_rec_glv(k0, k1, _k, n, (const bn_t *)v1, (const bn_t *)v2);
127 s0 = bn_sign(k0);
128 s1 = bn_sign(k1);
129 bn_abs(k0, k0);
130 bn_abs(k1, k1);
131
132 n0 = bn_bits(k0);
133 n1 = bn_bits(k1);
134
135 p0 = (RLC_DEPTH) * l - 1;
136
137 ep_set_infty(r);
138 if (n0 > p0 + 1) {
139 ep_copy(r, t[1 << (RLC_DEPTH-1)]);
140 }
141 if (n1 > p0 + 1) {
142 ep_psi(u, t[1 << (RLC_DEPTH-1)]);
143 ep_add(r, r, u);
144 }
145
146 for (i = l - 1; i >= 0; i--) {
147 ep_dbl(r, r);
148
149 w0 = w1 = 0;
150 p1 = p0--;
151 for (j = RLC_DEPTH - 1; j >= 0; j--, p1 -= l) {
152 w0 = w0 << 1;
153 w1 = w1 << 1;
154 if (p1 < n0 && bn_get_bit(k0, p1)) {
155 w0 = w0 | 1;
156 }
157 if (p1 < n1 && bn_get_bit(k1, p1)) {
158 w1 = w1 | 1;
159 }
160 }
161 if (w0 > 0) {
162 if (s0 == RLC_POS) {
163 ep_add(r, r, t[w0]);
164 } else {
165 ep_sub(r, r, t[w0]);
166 }
167 }
168 if (w1 > 0) {
169 ep_psi(u, t[w1]);
170 if (s1 == RLC_POS) {
171 ep_add(r, r, u);
172 } else {
173 ep_sub(r, r, u);
174 }
175 }
176 }
177 ep_norm(r, r);
178 }
179 RLC_CATCH_ANY {
180 RLC_THROW(ERR_CAUGHT);
181 }
182 RLC_FINALLY {
183 bn_free(n);
184 bn_free(_k);
185 bn_free(k0);
186 bn_free(k1);
187 ep_free(u);
188 for (i = 0; i < 3; i++) {
189 bn_free(v1[i]);
190 bn_free(v2[i]);
191 }
192 }
193}
194
195#endif /* EP_ENDOM */

206static void ep_mul_combs_plain(ep_t r, const ep_t *t, const bn_t k) {
207 int i, j, l, w, n0, p0, p1;
208 bn_t n, _k;
209
210 bn_null(n);
211 bn_null(_k);
212
213 RLC_TRY {
214 bn_new(n);
215 bn_new(_k);
216
217 ep_curve_get_ord(n);
218 l = RLC_CEIL(bn_bits(n), RLC_DEPTH);
219
220 bn_mod(_k, k, n);
221 n0 = bn_bits(_k);
222 p0 = (RLC_DEPTH) * l - 1;
223
224 w = 0;
225 p1 = p0--;
226 for (j = RLC_DEPTH - 1; j >= 0; j--, p1 -= l) {
227 w = w << 1;
228 if (p1 < n0 && bn_get_bit(_k, p1)) {
229 w = w | 1;
230 }
231 }
232
233 ep_copy(r, t[w]);
234 for (i = l - 2; i >= 0; i--) {
235 ep_dbl(r, r);
236
237 w = 0;
238 p1 = p0--;
239 for (j = RLC_DEPTH - 1; j >= 0; j--, p1 -= l) {
240 w = w << 1;
241 if (p1 < n0 && bn_get_bit(_k, p1)) {
242 w = w | 1;
243 }
244 }
245 if (w > 0) {
246 ep_add(r, r, t[w]);
247 }
248 }
249 ep_norm(r, r);
250 }
251 RLC_CATCH_ANY {
252 RLC_THROW(ERR_CAUGHT);
253 }
254 RLC_FINALLY {
255 bn_free(n);
256 bn_free(_k);
257 }
258}

397void ep_mul_pre_combd(ep_t *t, const ep_t p) {
398 int i, j, d, e;
399 bn_t n;
400
401 bn_null(n);
402
403 RLC_TRY {
404 bn_new(n);
405
406 ep_curve_get_ord(n);
407 d = RLC_CEIL(bn_bits(n), RLC_DEPTH);
408 e = (d % 2 == 0 ? (d / 2) : (d / 2) + 1);
409
410 ep_set_infty(t[0]);
411 ep_copy(t[1], p);
412 for (j = 1; j < RLC_DEPTH; j++) {
413 ep_dbl(t[1 << j], t[1 << (j - 1)]);
414 for (i = 1; i < d; i++) {
415 ep_dbl(t[1 << j], t[1 << j]);
416 }
417#if defined(EP_MIXED)
418 ep_norm(t[1 << j], t[1 << j]);
419#endif
420 for (i = 1; i < (1 << j); i++) {
421 ep_add(t[(1 << j) + i], t[i], t[1 << j]);
422 }
423 }
424 ep_set_infty(t[1 << RLC_DEPTH]);
425 for (j = 1; j < (1 << RLC_DEPTH); j++) {
426 ep_dbl(t[(1 << RLC_DEPTH) + j], t[j]);
427 for (i = 1; i < e; i++) {
428 ep_dbl(t[(1 << RLC_DEPTH) + j], t[(1 << RLC_DEPTH) + j]);
429 }
430 }
431
432 ep_norm_sim(t + 2, (const ep_t *)t + 2, (1 << RLC_DEPTH) - 2);
433 ep_norm_sim(t + (1 << RLC_DEPTH) + 1,
434 (const ep_t *)t + (1 << RLC_DEPTH) + 1, (1 << RLC_DEPTH) - 1);
435 }
436 RLC_CATCH_ANY {
437 RLC_THROW(ERR_CAUGHT);
438 }
439 RLC_FINALLY {
440 bn_free(n);
441 }

444void ep_mul_fix_combd(ep_t r, const ep_t *t, const bn_t k) {
445 int i, j, d, e, w0, w1, n0, p0, p1;
446 bn_t n, _k;
447
448 if (bn_is_zero(k)) {
449 ep_set_infty(r);
450 return;
451 }
452
453 bn_null(n);
454 bn_null(_k);
455
456 RLC_TRY {
457 bn_new(n);
458 bn_new(_k);
459
460 ep_curve_get_ord(n);
461 d = RLC_CEIL(bn_bits(n), RLC_DEPTH);
462 e = (d % 2 == 0 ? (d / 2) : (d / 2) + 1);
463
464 bn_mod(_k, k, n);
465 ep_set_infty(r);
466 n0 = bn_bits(_k);
467
468 p1 = (e - 1) + (RLC_DEPTH - 1) * d;
469 for (i = e - 1; i >= 0; i--) {
470 ep_dbl(r, r);
471
472 w0 = 0;
473 p0 = p1;
474 for (j = RLC_DEPTH - 1; j >= 0; j--, p0 -= d) {
475 w0 = w0 << 1;
476 if (p0 < n0 && bn_get_bit(_k, p0)) {
477 w0 = w0 | 1;
478 }
479 }
480
481 w1 = 0;
482 p0 = p1-- + e;
483 for (j = RLC_DEPTH - 1; j >= 0; j--, p0 -= d) {
484 w1 = w1 << 1;
485 if (i + e < d && p0 < n0 && bn_get_bit(_k, p0)) {
486 w1 = w1 | 1;
487 }
488 }
489
490 ep_add(r, r, t[w0]);
491 ep_add(r, r, t[(1 << RLC_DEPTH) + w1]);
492 }
493 ep_norm(r, r);
494 }
495 RLC_CATCH_ANY {
496 RLC_THROW(ERR_CAUGHT);
497 }
498 RLC_FINALLY {
499 bn_free(n);
500 bn_free(_k);
501 }
502}
