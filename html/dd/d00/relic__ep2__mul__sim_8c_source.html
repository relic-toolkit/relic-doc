41 static void ep2_mul_sim_plain(ep2_t r, ep2_t p, bn_t k, ep2_t q, bn_t m,
42  ep2_t *t) {
43  int i, l, l0, l1, n0, n1, w, gen;
44  int8_t naf0[2 * RLC_FP_BITS + 1], naf1[2 * RLC_FP_BITS + 1], *_k, *_m;
45  ep2_t t0[1 << (EP_WIDTH - 2)];
46  ep2_t t1[1 << (EP_WIDTH - 2)];
47 
48  TRY {
49  gen = (t == NULL ? 0 : 1);
50  if (!gen) {
51  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
52  ep2_null(t0[i]);
53  ep2_new(t0[i]);
54  }
55  ep2_tab(t0, p, EP_WIDTH);
56  t = (ep2_t *)t0;
57  }
58 
59  /* Prepare the precomputation table. */
60  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
61  ep2_null(t1[i]);
62  ep2_new(t1[i]);
63  }
64  /* Compute the precomputation table. */
65  ep2_tab(t1, q, EP_WIDTH);
66 
67  /* Compute the w-TNAF representation of k. */
68  if (gen) {
69  w = EP_DEPTH;
70  } else {
71  w = EP_WIDTH;
72  }
73  l0 = l1 = 2 * RLC_FP_BITS + 1;
74  bn_rec_naf(naf0, &l0, k, w);
75  bn_rec_naf(naf1, &l1, m, EP_WIDTH);
76 
77  l = RLC_MAX(l0, l1);
78  _k = naf0 + l - 1;
79  _m = naf1 + l - 1;
80  for (i = l0; i < l; i++) {
81  naf0[i] = 0;
82  }
83  for (i = l1; i < l; i++) {
84  naf1[i] = 0;
85  }
86 
87  if (bn_sign(k) == RLC_NEG) {
88  for (i = 0; i < l0; i++) {
89  naf0[i] = -naf0[i];
90  }
91  }
92  if (bn_sign(m) == RLC_NEG) {
93  for (i = 0; i < l1; i++) {
94  naf1[i] = -naf1[i];
95  }
96  }
97 
98  ep2_set_infty(r);
99  for (i = l - 1; i >= 0; i--, _k--, _m--) {
100  ep2_dbl(r, r);
101 
102  n0 = *_k;
103  n1 = *_m;
104  if (n0 > 0) {
105  ep2_add(r, r, t[n0 / 2]);
106  }
107  if (n0 < 0) {
108  ep2_sub(r, r, t[-n0 / 2]);
109  }
110  if (n1 > 0) {
111  ep2_add(r, r, t1[n1 / 2]);
112  }
113  if (n1 < 0) {
114  ep2_sub(r, r, t1[-n1 / 2]);
115  }
116  }
117  /* Convert r to affine coordinates. */
118  ep2_norm(r, r);
119  }
120  CATCH_ANY {
121  THROW(ERR_CAUGHT);
122  }
123  FINALLY {
124  /* Free the precomputation tables. */
125  if (!gen) {
126  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
127  ep2_free(t0[i]);
128  }
129  }
130  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
131  ep2_free(t1[i]);
132  }
133  }
134 }

168 void ep2_mul_sim_trick(ep2_t r, ep2_t p, bn_t k, ep2_t q, bn_t m) {
169  ep2_t t0[1 << (EP_WIDTH / 2)];
170  ep2_t t1[1 << (EP_WIDTH / 2)];
171  ep2_t t[1 << EP_WIDTH];
172  bn_t n;
173  int l0, l1, w = EP_WIDTH / 2;
174  uint8_t w0[2 * RLC_FP_BITS], w1[2 * RLC_FP_BITS];
175 
176  bn_null(n);
177 
178  if (bn_is_zero(k) || ep2_is_infty(p)) {
179  ep2_mul(r, q, m);
180  return;
181  }
182  if (bn_is_zero(m) || ep2_is_infty(q)) {
183  ep2_mul(r, p, k);
184  return;
185  }
186 
187  TRY {
188  bn_new(n);
189 
190  ep2_curve_get_ord(n);
191 
192  for (int i = 0; i < (1 << w); i++) {
193  ep2_null(t0[i]);
194  ep2_null(t1[i]);
195  ep2_new(t0[i]);
196  ep2_new(t1[i]);
197  }
198  for (int i = 0; i < (1 << EP_WIDTH); i++) {
199  ep2_null(t[i]);
200  ep2_new(t[i]);
201  }
202 
203  ep2_set_infty(t0[0]);
204  ep2_copy(t0[1], p);
205  if (bn_sign(k) == RLC_NEG) {
206  ep2_neg(t0[1], t0[1]);
207  }
208  for (int i = 2; i < (1 << w); i++) {
209  ep2_add(t0[i], t0[i - 1], t0[1]);
210  }
211 
212  ep2_set_infty(t1[0]);
213  ep2_copy(t1[1], q);
214  if (bn_sign(m) == RLC_NEG) {
215  ep2_neg(t1[1], t1[1]);
216  }
217  for (int i = 1; i < (1 << w); i++) {
218  ep2_add(t1[i], t1[i - 1], t1[1]);
219  }
220 
221  for (int i = 0; i < (1 << w); i++) {
222  for (int j = 0; j < (1 << w); j++) {
223  ep2_add(t[(i << w) + j], t0[i], t1[j]);
224  }
225  }
226 
227 #if defined(EP_MIXED)
228  ep2_norm_sim(t + 1, t + 1, (1 << (EP_WIDTH)) - 1);
229 #endif
230 
231  l0 = l1 = RLC_CEIL(2 * RLC_FP_BITS, w);
232  bn_rec_win(w0, &l0, k, w);
233  bn_rec_win(w1, &l1, m, w);
234 
235  for (int i = l0; i < l1; i++) {
236  w0[i] = 0;
237  }
238  for (int i = l1; i < l0; i++) {
239  w1[i] = 0;
240  }
241 
242  ep2_set_infty(r);
243  for (int i = RLC_MAX(l0, l1) - 1; i >= 0; i--) {
244  for (int j = 0; j < w; j++) {
245  ep2_dbl(r, r);
246  }
247  ep2_add(r, r, t[(w0[i] << w) + w1[i]]);
248  }
249  ep2_norm(r, r);
250  } CATCH_ANY {
251  THROW(ERR_CAUGHT);
252  }
253  FINALLY {
254  bn_free(n);
255  for (int i = 0; i < (1 << w); i++) {
256  ep2_free(t0[i]);
257  ep2_free(t1[i]);
258  }
259  for (int i = 0; i < (1 << EP_WIDTH); i++) {
260  ep2_free(t[i]);
261  }
262  }
263 }

295 void ep2_mul_sim_joint(ep2_t r, ep2_t p, bn_t k, ep2_t q, bn_t m) {
296  ep2_t t[5];
297  int i, l, u_i, offset;
298  int8_t jsf[4 * (RLC_FP_BITS + 1)];
299 
300  if (bn_is_zero(k) || ep2_is_infty(p)) {
301  ep2_mul(r, q, m);
302  return;
303  }
304  if (bn_is_zero(m) || ep2_is_infty(q)) {
305  ep2_mul(r, p, k);
306  return;
307  }
308 
309  TRY {
310  for (i = 0; i < 5; i++) {
311  ep2_null(t[i]);
312  ep2_new(t[i]);
313  }
314 
315  ep2_set_infty(t[0]);
316  ep2_copy(t[1], q);
317  if (bn_sign(m) == RLC_NEG) {
318  ep2_neg(t[1], t[1]);
319  }
320  ep2_copy(t[2], p);
321  if (bn_sign(k) == RLC_NEG) {
322  ep2_neg(t[2], t[2]);
323  }
324  ep2_add(t[3], t[2], t[1]);
325  ep2_sub(t[4], t[2], t[1]);
326 #if defined(EP_MIXED)
327  ep2_norm_sim(t + 3, t + 3, 2);
328 #endif
329 
330  l = 4 * (RLC_FP_BITS + 1);
331  bn_rec_jsf(jsf, &l, k, m);
332 
333  ep2_set_infty(r);
334 
335  offset = RLC_MAX(bn_bits(k), bn_bits(m)) + 1;
336  for (i = l - 1; i >= 0; i--) {
337  ep2_dbl(r, r);
338  if (jsf[i] != 0 && jsf[i] == -jsf[i + offset]) {
339  u_i = jsf[i] * 2 + jsf[i + offset];
340  if (u_i < 0) {
341  ep2_sub(r, r, t[4]);
342  } else {
343  ep2_add(r, r, t[4]);
344  }
345  } else {
346  u_i = jsf[i] * 2 + jsf[i + offset];
347  if (u_i < 0) {
348  ep2_sub(r, r, t[-u_i]);
349  } else {
350  ep2_add(r, r, t[u_i]);
351  }
352  }
353  }
354  ep2_norm(r, r);
355  }
356  CATCH_ANY {
357  THROW(ERR_CAUGHT);
358  }
359  FINALLY {
360  for (i = 0; i < 5; i++) {
361  ep2_free(t[i]);
362  }
363  }
364 }
