55static void ep2_mul_sim_endom(ep2_t r, const ep2_t p, const bn_t k,
56 const ep2_t q, const bn_t m) {
57 size_t l, _l[4];
58 bn_t _k[4], _m[4], n, u;
59 int8_t naf0[4][RLC_FP_BITS + 1], naf1[4][RLC_FP_BITS + 1];
60 ep2_t _p[4], _q[4];
61
62 bn_null(n);
63 bn_null(u);
64
65 RLC_TRY {
66 bn_new(n);
67 bn_new(u);
68 for (int i = 0; i < 4; i++) {
69 bn_null(_k[i]);
70 bn_new(_k[i]);
71 bn_null(_m[i]);
72 bn_new(_m[i]);
73 ep2_null(_p[i]);
74 ep2_null(_q[i]);
75 ep2_new(_p[i]);
76 ep2_new(_q[i]);
77 }
78
79 ep2_norm(_p[0], p);
80 ep2_frb(_p[1], _p[0], 1);
81 ep2_frb(_p[2], _p[1], 1);
82 ep2_frb(_p[3], _p[2], 1);
83 ep2_norm(_q[0], q);
84 ep2_frb(_q[1], _q[0], 1);
85 ep2_frb(_q[2], _q[1], 1);
86 ep2_frb(_q[3], _q[2], 1);
87
88 ep2_curve_get_ord(n);
89 fp_prime_get_par(u);
90 bn_mod(_k[0], k, n);
91 bn_rec_frb(_k, 4, _k[0], u, n, ep_curve_is_pairf() == EP_BN);
92 bn_mod(_m[0], m, n);
93 bn_rec_frb(_m, 4, _m[0], u, n, ep_curve_is_pairf() == EP_BN);
94
95 l = 0;
96 for (int i = 0; i < 4; i++) {
97 _l[i] = RLC_FP_BITS + 1;
98 bn_rec_naf(naf0[i], &_l[i], _k[i], 2);
99 if (bn_sign(_k[i]) == RLC_NEG) {
100 ep2_neg(_p[i], _p[i]);
101 }
102 l = RLC_MAX(l, _l[i]);
103 _l[i] = RLC_FP_BITS + 1;
104 bn_rec_naf(naf1[i], &_l[i], _m[i], 2);
105 if (bn_sign(_m[i]) == RLC_NEG) {
106 ep2_neg(_q[i], _q[i]);
107 }
108 l = RLC_MAX(l, _l[i]);
109 }
110
111 ep2_set_infty(r);
112 for (int i = l - 1; i >= 0; i--) {
113 ep2_dbl(r, r);
114 for (int j = 0; j < 4; j++) {
115 if (naf0[j][i] > 0) {
116 ep2_add(r, r, _p[j]);
117 }
118 if (naf0[j][i] < 0) {
119 ep2_sub(r, r, _p[j]);
120 }
121 if (naf1[j][i] > 0) {
122 ep2_add(r, r, _q[j]);
123 }
124 if (naf1[j][i] < 0) {
125 ep2_sub(r, r, _q[j]);
126 }
127 }
128 }
129
130 /* Convert r to affine coordinates. */
131 ep2_norm(r, r);
132 } RLC_CATCH_ANY {
133 RLC_THROW(ERR_CAUGHT);
134 } RLC_FINALLY {
135 bn_free(n);
136 bn_free(u);
137 for (int i = 0; i < 4; i++) {
138 bn_free(_k[i]);
139 bn_free(_m[i]);
140 ep2_free(_p[i]);
141 ep2_free(_q[i]);
142 }
143 }
144}

162static void ep2_mul_sim_plain(ep2_t r, const ep2_t p, const bn_t k,
163 const ep2_t q, const bn_t m, const ep2_t *t) {
164 int i, w, gen = (t == NULL ? 0 : 1);
165 int8_t n0, n1, naf0[2 * RLC_FP_BITS + 1], naf1[2 * RLC_FP_BITS + 1];
166 ep2_t t0[1 << (RLC_WIDTH - 2)], t1[1 << (RLC_WIDTH - 2)];
167 size_t l, l0, l1;
168
169 RLC_TRY {
170 if (!gen) {
171 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
172 ep2_null(t0[i]);
173 ep2_new(t0[i]);
174 }
175 ep2_tab(t0, p, RLC_WIDTH);
176 t = (ep2_t *)t0;
177 }
178
179 /* Prepare the precomputation table. */
180 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
181 ep2_null(t1[i]);
182 ep2_new(t1[i]);
183 }
184 /* Compute the precomputation table. */
185 ep2_tab(t1, q, RLC_WIDTH);
186
187 /* Compute the w-TNAF representation of k. */
188 if (gen) {
189 w = RLC_DEPTH;
190 } else {
191 w = RLC_WIDTH;
192 }
193 l0 = l1 = 2 * RLC_FP_BITS + 1;
194 bn_rec_naf(naf0, &l0, k, w);
195 bn_rec_naf(naf1, &l1, m, RLC_WIDTH);
196
197 l = RLC_MAX(l0, l1);
198 if (bn_sign(k) == RLC_NEG) {
199 for (i = 0; i < l0; i++) {
200 naf0[i] = -naf0[i];
201 }
202 }
203 if (bn_sign(m) == RLC_NEG) {
204 for (i = 0; i < l1; i++) {
205 naf1[i] = -naf1[i];
206 }
207 }
208
209 ep2_set_infty(r);
210 for (i = l - 1; i >= 0; i--) {
211 ep2_dbl(r, r);
212
213 n0 = naf0[i];
214 n1 = naf1[i];
215 if (n0 > 0) {
216 ep2_add(r, r, t[n0 / 2]);
217 }
218 if (n0 < 0) {
219 ep2_sub(r, r, t[-n0 / 2]);
220 }
221 if (n1 > 0) {
222 ep2_add(r, r, t1[n1 / 2]);
223 }
224 if (n1 < 0) {
225 ep2_sub(r, r, t1[-n1 / 2]);
226 }
227 }
228 /* Convert r to affine coordinates. */
229 ep2_norm(r, r);
230 }
231 RLC_CATCH_ANY {
232 RLC_THROW(ERR_CAUGHT);
233 }
234 RLC_FINALLY {
235 /* Free the precomputation tables. */
236 if (!gen) {
237 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
238 ep2_free(t0[i]);
239 }
240 }
241 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
242 ep2_free(t1[i]);
243 }
244 }
245}

281void ep2_mul_sim_trick(ep2_t r, const ep2_t p, const bn_t k, const ep2_t q,
282 const bn_t m) {
283 ep2_t t0[1 << (RLC_WIDTH / 2)], t1[1 << (RLC_WIDTH / 2)], t[1 << RLC_WIDTH];
284 bn_t n, _k, _m;
285 size_t l0, l1, w = RLC_WIDTH / 2;
286 uint8_t w0[2 * RLC_FP_BITS], w1[2 * RLC_FP_BITS];
287
288 bn_null(n);
289 bn_null(_k);
290 bn_null(_m);
291
292 if (bn_is_zero(k) || ep2_is_infty(p)) {
293 ep2_mul(r, q, m);
294 return;
295 }
296 if (bn_is_zero(m) || ep2_is_infty(q)) {
297 ep2_mul(r, p, k);
298 return;
299 }
300
301 RLC_TRY {
302 bn_new(n);
303 bn_new(_k);
304 bn_new(_m);
305
306 for (int i = 0; i < (1 << w); i++) {
307 ep2_null(t0[i]);
308 ep2_null(t1[i]);
309 ep2_new(t0[i]);
310 ep2_new(t1[i]);
311 }
312 for (int i = 0; i < (1 << RLC_WIDTH); i++) {
313 ep2_null(t[i]);
314 ep2_new(t[i]);
315 }
316
317 ep2_curve_get_ord(n);
318 bn_mod(_k, k, n);
319 bn_mod(_m, m, n);
320
321 ep2_set_infty(t0[0]);
322 ep2_copy(t0[1], p);
323 for (int i = 2; i < (1 << w); i++) {
324 ep2_add(t0[i], t0[i - 1], t0[1]);
325 }
326
327 ep2_set_infty(t1[0]);
328 ep2_copy(t1[1], q);
329 for (int i = 2; i < (1 << w); i++) {
330 ep2_add(t1[i], t1[i - 1], t1[1]);
331 }
332
333 for (int i = 0; i < (1 << w); i++) {
334 for (int j = 0; j < (1 << w); j++) {
335 ep2_add(t[(i << w) + j], t0[i], t1[j]);
336 }
337 }
338
339#if defined(EP_MIXED)
340 ep2_norm_sim(t + 2, (const ep2_t *)(t + 2), (1 << (w + w)) - 2);
341#endif
342
343 l0 = l1 = RLC_CEIL(2 * RLC_FP_BITS, w);
344 bn_rec_win(w0, &l0, _k, w);
345 bn_rec_win(w1, &l1, _m, w);
346
347 ep2_set_infty(r);
348 for (int i = RLC_MAX(l0, l1) - 1; i >= 0; i--) {
349 for (int j = 0; j < w; j++) {
350 ep2_dbl(r, r);
351 }
352 ep2_add(r, r, t[(w0[i] << w) + w1[i]]);
353 }
354 ep2_norm(r, r);
355 } RLC_CATCH_ANY {
356 RLC_THROW(ERR_CAUGHT);
357 }
358 RLC_FINALLY {
359 bn_free(n);
360 bn_free(_k);
361 bn_free(_m);
362 for (int i = 0; i < (1 << w); i++) {
363 ep2_free(t0[i]);
364 ep2_free(t1[i]);
365 }
366 for (int i = 0; i < (1 << RLC_WIDTH); i++) {
367 ep2_free(t[i]);
368 }
369 }
370}

402void ep2_mul_sim_joint(ep2_t r, const ep2_t p, const bn_t k, const ep2_t q,
403 const bn_t m) {
404 bn_t n, _k, _m;
405 ep2_t t[5];
406 int i, u_i, offset;
407 int8_t jsf[2 * (RLC_FP_BITS + 1)];
408 size_t l;
409
410 if (bn_is_zero(k) || ep2_is_infty(p)) {
411 ep2_mul(r, q, m);
412 return;
413 }
414 if (bn_is_zero(m) || ep2_is_infty(q)) {
415 ep2_mul(r, p, k);
416 return;
417 }
418
419 bn_null(n);
420 bn_null(_k);
421 bn_null(_m);
422
423 RLC_TRY {
424 bn_new(n);
425 bn_new(_k);
426 bn_new(_m);
427 for (i = 0; i < 5; i++) {
428 ep2_null(t[i]);
429 ep2_new(t[i]);
430 }
431
432 ep2_curve_get_ord(n);
433 bn_mod(_k, k, n);
434 bn_mod(_m, m, n);
435
436 ep2_set_infty(t[0]);
437 ep2_copy(t[1], q);
438 if (bn_sign(_m) == RLC_NEG) {
439 ep2_neg(t[1], t[1]);
440 }
441 ep2_copy(t[2], p);
442 if (bn_sign(_k) == RLC_NEG) {
443 ep2_neg(t[2], t[2]);
444 }
445 ep2_add(t[3], t[2], t[1]);
446 ep2_sub(t[4], t[2], t[1]);
447#if defined(EP_MIXED)
448 ep2_norm_sim(t + 3, t + 3, 2);
449#endif
450
451 l = 2 * (RLC_FP_BITS + 1);
452 bn_rec_jsf(jsf, &l, _k, _m);
453
454 ep2_set_infty(r);
455
456 offset = RLC_MAX(bn_bits(_k), bn_bits(_m)) + 1;
457 for (i = l - 1; i >= 0; i--) {
458 ep2_dbl(r, r);
459 if (jsf[i] != 0 && jsf[i] == -jsf[i + offset]) {
460 u_i = jsf[i] * 2 + jsf[i + offset];
461 if (u_i < 0) {
462 ep2_sub(r, r, t[4]);
463 } else {
464 ep2_add(r, r, t[4]);
465 }
466 } else {
467 u_i = jsf[i] * 2 + jsf[i + offset];
468 if (u_i < 0) {
469 ep2_sub(r, r, t[-u_i]);
470 } else {
471 ep2_add(r, r, t[u_i]);
472 }
473 }
474 }
475 ep2_norm(r, r);
476 }
477 RLC_CATCH_ANY {
478 RLC_THROW(ERR_CAUGHT);
479 }
480 RLC_FINALLY {
481 bn_free(n);
482 bn_free(_k);
483 bn_free(_m);
484 for (i = 0; i < 5; i++) {
485 ep2_free(t[i]);
486 }
487 }
488}

492void ep2_mul_sim_gen(ep2_t r, const bn_t k, const ep2_t q, const bn_t m) {
493 ep2_t g;
494 bn_t n, _k, _m;
495
496 if (bn_is_zero(k)) {
497 ep2_mul(r, q, m);
498 return;
499 }
500 if (bn_is_zero(m) || ep2_is_infty(q)) {
501 ep2_mul_gen(r, k);
502 return;
503 }
504
505 ep2_null(g);
506 bn_null(n);
507 bn_null(_k);
508 bn_null(_m);
509
510 RLC_TRY {
511 ep2_new(g);
512 bn_new(n);
513 bn_new(_k);
514 bn_new(_m);
515
516 ep2_curve_get_gen(g);
517 ep2_curve_get_ord(n);
518
519 bn_mod(_k, k, n);
520 bn_mod(_m, m, n);
521
522#if defined(EP_ENDOM)
523#if EP_SIM == INTER && EP_FIX == LWNAF && defined(EP_PRECO)
524 if (ep_curve_is_endom()) {
525 ep2_mul_sim_endom(r, g, _k, q, _m, ep2_curve_get_tab());
526 }
527#else
528 if (ep_curve_is_endom()) {
529 ep2_mul_sim(r, g, _k, q, _m);
530 }
531#endif
532#endif
533
534#if defined(EP_PLAIN) || defined(EP_SUPER)
535#if EP_SIM == INTER && EP_FIX == LWNAF && defined(EP_PRECO)
536 if (!ep_curve_is_endom()) {
537 ep2_mul_sim_plain(r, g, _k, q, _m, ep2_curve_get_tab());
538 }
539#else
540 if (!ep_curve_is_endom()) {
541 ep2_mul_sim(r, g, _k, q, _m);
542 }
543#endif
544#endif
545 }
546 RLC_CATCH_ANY {
547 RLC_THROW(ERR_CAUGHT);
548 }
549 RLC_FINALLY {
550 ep2_free(g);
551 bn_free(n);
552 bn_free(_k);
553 bn_free(_m);
554 }
555}

557void ep2_mul_sim_dig(ep2_t r, const ep2_t p[], const dig_t k[], size_t len) {
558 ep2_t t;
559 int max;
560
561 ep2_null(t);
562
563 max = util_bits_dig(k[0]);
564 for (int i = 1; i < len; i++) {
565 max = RLC_MAX(max, util_bits_dig(k[i]));
566 }
567
568 RLC_TRY {
569 ep2_new(t);
570
571 ep2_set_infty(t);
572 for (int i = max - 1; i >= 0; i--) {
573 ep2_dbl(t, t);
574 for (int j = 0; j < len; j++) {
575 if (k[j] & ((dig_t)1 << i)) {
576 ep2_add(t, t, p[j]);
577 }
578 }
579 }
580
581 ep2_norm(r, t);
582 }
583 RLC_CATCH_ANY {
584 RLC_THROW(ERR_CAUGHT);
585 }
586 RLC_FINALLY {
587 ep2_free(t);
588 }

591void ep2_mul_sim_lot(ep2_t r, const ep2_t p[], const bn_t k[], size_t n) {
592 const size_t len = RLC_FP_BITS + 1;
593 int i, j, m;
594 bn_t _k[4], q, x;
595 int8_t ptr, *naf = RLC_ALLOCA(int8_t, 4 * n * len);
596 size_t l, _l[4];
597
598 if (n == 0) {
599 ep2_set_infty(r);
600 return;
601 }
602
603 bn_null(q);
604 bn_null(x);
605
606 if (n <= 10) {
607 ep2_t *_p = RLC_ALLOCA(ep2_t, 4 * n);
608
609 RLC_TRY {
610 if (naf == NULL || _p == NULL) {
611 RLC_THROW(ERR_NO_MEMORY);
612 }
613 bn_new(q);
614 bn_new(x);
615 for (j = 0; j < 4; j++) {
616 bn_null(_k[j]);
617 bn_new(_k[j]);
618 for (i = 0; i < n; i++) {
619 ep2_null(_p[4*i + j]);
620 ep2_new(_p[4*i + j]);
621 }
622 }
623
624 l = 0;
625 ep2_curve_get_ord(q);
626 fp_prime_get_par(x);
627 for (i = 0; i < n; i++) {
628 ep2_norm(_p[4*i], p[i]);
629 ep2_frb(_p[4*i + 1], _p[4*i], 1);
630 ep2_frb(_p[4*i + 2], _p[4*i + 1], 1);
631 ep2_frb(_p[4*i + 3], _p[4*i + 2], 1);
632
633 bn_mod(_k[0], k[i], q);
634 bn_rec_frb(_k, 4, _k[0], x, q, ep_curve_is_pairf() == EP_BN);
635 for (j = 0; j < 4; j++) {
636 _l[j] = len;
637 bn_rec_naf(&naf[(4*i + j)*len], &_l[j], _k[j], 2);
638 if (bn_sign(_k[j]) == RLC_NEG) {
639 ep2_neg(_p[4*i + j], _p[4*i + j]);
640 }
641 l = RLC_MAX(l, _l[j]);
642 }
643 }
644
645 ep2_set_infty(r);
646 for (i = l - 1; i >= 0; i--) {
647 ep2_dbl(r, r);
648 for (j = 0; j < n; j++) {
649 for (m = 0; m < 4; m++) {
650 if (naf[(4*j + m)*len + i] > 0) {
651 ep2_add(r, r, _p[4*j + m]);
652 }
653 if (naf[(4*j + m)*len + i] < 0) {
654 ep2_sub(r, r, _p[4*j + m]);
655 }
656 }
657 }
658 }
659
660 /* Convert r to affine coordinates. */
661 ep2_norm(r, r);
662 } RLC_CATCH_ANY {
663 RLC_THROW(ERR_CAUGHT);
664 } RLC_FINALLY {
665 bn_free(q);
666 bn_free(x);
667 for (j = 0; j < 4; j++) {
668 bn_free(_k[j]);
669 for (i = 0; i < n; i++) {
670 ep2_free(_p[4*i + j]);
671 }
672 }
673 RLC_FREE(_p);
674 RLC_FREE(naf);
675 }
676 } else {
677 const int w = RLC_MAX(2, util_bits_dig(n) - 2), c = (1 << (w - 2));
678 ep2_t s, t, u, v, *_p = RLC_ALLOCA(ep2_t, 4 * c);
679
680 ep2_null(s);
681 ep2_null(t);
682 ep2_null(u);
683 ep2_null(v);
684
685 RLC_TRY {
686 if (naf == NULL || _p == NULL) {
687 RLC_THROW(ERR_NO_MEMORY);
688 }
689 bn_new(q);
690 bn_new(x);
691 ep2_new(s);
692 ep2_new(t);
693 ep2_new(u);
694 ep2_new(v);
695 for (i = 0; i < 4; i++) {
696 bn_null(_k[i]);
697 bn_new(_k[i]);
698 for (j = 0; j < c; j++) {
699 ep2_null(_p[i*c + j]);
700 ep2_new(_p[i*c + j]);
701 ep2_set_infty(_p[i*c + j]);
702 }
703 }
704
705 l = 0;
706 ep2_curve_get_ord(q);
707 fp_prime_get_par(x);
708 for (i = 0; i < n; i++) {
709 bn_mod(_k[0], k[i], q);
