44  fp2_t t0, t1;
45 
46  dv2_null(u0);
47  dv2_null(u1);
48  dv2_null(u2);

51  fp2_null(t1);
52 
53  TRY {
54  dv2_new(u0);
55  dv2_new(u1);
56  dv2_new(u2);
57  dv2_new(u3);
58  fp2_new(t0);
59  fp2_new(t1);
60

62  fp2_mulc_low(u0, a[0], b[0]);
63  fp2_mulc_low(u1, a[1], b[1]);
64  fp2_addn_low(t0, a[0], a[1]);
65  fp2_addn_low(t1, b[0], b[1]);
66

68  fp2_muln_low(u2, t0, t1);

73  fp2_mulc_low(u2, a[2], b[1]);
74  fp2_norh_low(c[0], u2);
75  fp2_addc_low(c[0], u0, c[0]);
76 
77  /* c_2 = a_0b_2 + a_1b_1 */
78  fp2_mulc_low(u2, a[2], b[0]);
79  fp2_addc_low(c[2], u1, u2);
80 #else

85

87  fp2_muln_low(u2, t0, t1);

89  fp2_subc_low(c[1], u2, u1);

92  fp2_muln_low(u2, a[2], b[1]);

94  fp2_addc_low(c[0], u0, c[0]);
95 
96  /* c_2 = a_0b_2 + a_1b_1 */
97  fp2_muln_low(u2, a[2], b[0]);
98  fp2_addc_low(c[2], u1, u2);
99 #endif
100  } CATCH_ANY {
101  THROW(ERR_CAUGHT);
102  } FINALLY {
103  dv2_free(u0);

120 void fp12_mul_basic(fp12_t c, fp12_t a, fp12_t b) {
121  fp6_t t0, t1, t2;
122 
123  fp6_null(t0);
124  fp6_null(t1);
125  fp6_null(t2);
126 
127  TRY {
128  fp6_new(t0);
129  fp6_new(t1);
130  fp6_new(t2);
131 
132  /* Karatsuba algorithm. */
133 
134  /* t0 = a_0 * b_0. */
135  fp6_mul(t0, a[0], b[0]);
136  /* t1 = a_1 * b_1. */
137  fp6_mul(t1, a[1], b[1]);
138  /* t2 = b_0 + b_1. */
139  fp6_add(t2, b[0], b[1]);
140 
141  /* c_1 = a_0 + a_1. */
142  fp6_add(c[1], a[0], a[1]);
143 
144  /* c_1 = (a_0 + a_1) * (b_0 + b_1) */
145  fp6_mul(c[1], c[1], t2);
146  fp6_sub(c[1], c[1], t0);
147  fp6_sub(c[1], c[1], t1);
148 
149  /* c_0 = a_0b_0 + v * a_1b_1. */
150  fp6_mul_art(t1, t1);
151  fp6_add(c[0], t0, t1);
152  } CATCH_ANY {
153  THROW(ERR_CAUGHT);
154  } FINALLY {
155  fp6_free(t0);
156  fp6_free(t1);
157  fp6_free(t2);
158  }
159 }

161 void fp12_mul_dxs_basic(fp12_t c, fp12_t a, fp12_t b) {
162  fp6_t t0, t1, t2;
163 
164  fp6_null(t0);
165  fp6_null(t1);
166  fp6_null(t2);
167 
168  TRY {
169  fp6_new(t0);
170  fp6_new(t1);
171  fp6_new(t2);
172 
173  if (ep2_curve_is_twist() == EP_DTYPE) {
174 #if EP_ADD == BASIC
175  /* t0 = a_0 * b_0 */
176  fp_mul(t0[0][0], a[0][0][0], b[0][0][0]);
177  fp_mul(t0[0][1], a[0][0][1], b[0][0][0]);
178  fp_mul(t0[1][0], a[0][1][0], b[0][0][0]);
179  fp_mul(t0[1][1], a[0][1][1], b[0][0][0]);
180  fp_mul(t0[2][0], a[0][2][0], b[0][0][0]);
181  fp_mul(t0[2][1], a[0][2][1], b[0][0][0]);
182  /* t2 = b_0 + b_1. */
183  fp_add(t2[0][0], b[0][0][0], b[1][0][0]);
184  fp_copy(t2[0][1], b[1][0][1]);
185  fp2_copy(t2[1], b[1][1]);
186 #elif EP_ADD == PROJC
187  /* t0 = a_0 * b_0 */
188  fp2_mul(t0[0], a[0][0], b[0][0]);
189  fp2_mul(t0[1], a[0][1], b[0][0]);
190  fp2_mul(t0[2], a[0][2], b[0][0]);
191  /* t2 = b_0 + b_1. */
192  fp2_add(t2[0], b[0][0], b[1][0]);
193  fp2_copy(t2[1], b[1][1]);
194 #endif
195  /* t1 = a_1 * b_1. */
196  fp6_mul_dxs(t1, a[1], b[1]);
197  } else {
198  /* t0 = a_0 * b_0. */
199  fp6_mul_dxs(t0, a[0], b[0]);
200 #if EP_ADD == BASIC
201  /* t1 = a_1 * b_1. */
202  fp_mul(t2[0][0], a[1][2][0], b[1][1][0]);
203  fp_mul(t2[0][1], a[1][2][1], b[1][1][0]);
204  fp2_mul_nor(t1[0], t2[0]);
205  fp_mul(t1[1][0], a[1][0][0], b[1][1][0]);
206  fp_mul(t1[1][1], a[1][0][1], b[1][1][0]);
207  fp_mul(t1[2][0], a[1][1][0], b[1][1][0]);
208  fp_mul(t1[2][1], a[1][1][1], b[1][1][0]);
209  /* t2 = b_0 + b_1. */
210  fp2_copy(t2[0], b[0][0]);
211  fp_add(t2[1][0], b[0][1][0], b[1][1][0]);
212  fp_copy(t2[1][1], b[0][1][1]);
213 #elif EP_ADD == PROJC
214  /* t1 = a_1 * b_1. */
215  fp2_mul(t2[0], a[1][2], b[1][1]);
216  fp2_mul_nor(t1[0], t2[0]);
217  fp2_mul(t1[1], a[1][0], b[1][1]);
218  fp2_mul(t1[2], a[1][1], b[1][1]);
219  /* t2 = b_0 + b_1. */
220  fp2_copy(t2[0], b[0][0]);
221  fp2_add(t2[1], b[0][1], b[1][1]);
222 #endif
223  }
224  /* c_1 = a_0 + a_1. */
225  fp6_add(c[1], a[0], a[1]);
226  /* c_1 = (a_0 + a_1) * (b_0 + b_1) - a_0 * b_0 - a_1 * b_1. */
227  fp6_mul_dxs(c[1], c[1], t2);
228  fp6_sub(c[1], c[1], t0);
229  fp6_sub(c[1], c[1], t1);
230  /* c_0 = a_0 * b_0 + v * a_1 * b_1. */
231  fp6_mul_art(t1, t1);
232  fp6_add(c[0], t0, t1);
233  }
234  CATCH_ANY {
235  THROW(ERR_CAUGHT);
236  }
237  FINALLY {
238  fp6_free(t0);
239  fp6_free(t1);
240  fp6_free(t2);
241  }
242 }

248 void fp12_mul_unr(dv12_t c, fp12_t a, fp12_t b) {
249  fp6_t t0, t1;
250  dv6_t u0, u1, u2, u3;
251 
252  dv6_null(u0);
253  dv6_null(u1);
254  dv6_null(u2);
255  dv6_null(u3);
256  fp6_null(t0);
257  fp6_null(t1);
258 
259  TRY {
260  dv6_new(u0);
261  dv6_new(u1);
262  dv6_new(u2);
263  dv6_new(u3);
264  fp6_new(t0);
265  fp6_new(t1);
266 
267  /* Karatsuba algorithm. */
268 
269  /* u0 = a_0 * b_0. */
270  fp6_mul_unr(u0, a[0], b[0]);
271  /* u1 = a_1 * b_1. */
272  fp6_mul_unr(u1, a[1], b[1]);
273  /* t1 = a_0 + a_1. */
274  fp6_add(t0, a[0], a[1]);
275  /* t0 = b_0 + b_1. */
276  fp6_add(t1, b[0], b[1]);
277  /* u2 = (a_0 + a_1) * (b_0 + b_1) */
278  fp6_mul_unr(u2, t0, t1);
279  /* c_1 = u2 - a_0b_0 - a_1b_1. */
280  for (int i = 0; i < 3; i++) {
281  fp2_addc_low(u3[i], u0[i], u1[i]);
282  fp2_subc_low(c[1][i], u2[i], u3[i]);
283  }
284  /* c_0 = a_0b_0 + v * a_1b_1. */
285  fp2_nord_low(u2[0], u1[2]);
286  fp2_addc_low(c[0][0], u0[0], u2[0]);
287  fp2_addc_low(c[0][1], u0[1], u1[0]);
288  fp2_addc_low(c[0][2], u0[2], u1[1]);
289  } CATCH_ANY {
290  THROW(ERR_CAUGHT);
291  } FINALLY {
292  dv6_free(u0);
293  dv6_free(u1);
294  dv6_free(u2);
295  dv6_free(u3);
296  fp6_free(t0);
297  fp6_free(t1);
298  }
299 }
