42inline static void fp6_mul_dxs_unr_lazyr(dv6_t c, const fp6_t a, const fp6_t b) {
43 dv2_t u0, u1, u2, u3;
44 fp2_t t0, t1;
45
46 dv2_null(u0);
47 dv2_null(u1);
48 dv2_null(u2);
49 dv2_null(u3);
50 fp2_null(t0);
51 fp2_null(t1);
52
53 RLC_TRY {
54 dv2_new(u0);
55 dv2_new(u1);
56 dv2_new(u2);
57 dv2_new(u3);
58 fp2_new(t0);
59 fp2_new(t1);
60
61 fp2_muln_low(u0, a[0], b[0]);
62 fp2_muln_low(u1, a[1], b[1]);
63 fp2_addm_low(t0, a[0], a[1]);
64 fp2_addm_low(t1, b[0], b[1]);
65
66 /* c_1 = (a_0 + a_1)(b_0 + b_1) - a_0b_0 - a_1b_1 */
67 fp2_muln_low(u2, t0, t1);
68 fp2_subc_low(u2, u2, u0);
69 fp2_subc_low(c[1], u2, u1);
70
71 /* c_0 = a_0b_0 + E a_2b_1 */
72 fp2_muln_low(u2, a[2], b[1]);
73 fp2_nord_low(c[0], u2);
74 fp2_addc_low(c[0], u0, c[0]);
75
76 /* c_2 = a_0b_2 + a_1b_1 */
77 fp2_muln_low(u2, a[2], b[0]);
78 fp2_addc_low(c[2], u1, u2);
79 } RLC_CATCH_ANY {
80 RLC_THROW(ERR_CAUGHT);
81 } RLC_FINALLY {
82 dv2_free(u0);
83 dv2_free(u1);
84 dv2_free(u2);
85 dv2_free(u3);
86 fp2_free(t0);
87 fp2_free(t1);
88 }
89}

99void fp12_mul_basic(fp12_t c, const fp12_t a, const fp12_t b) {
100 fp6_t t0, t1, t2;
101
102 fp6_null(t0);
103 fp6_null(t1);
104 fp6_null(t2);
105
106 RLC_TRY {
107 fp6_new(t0);
108 fp6_new(t1);
109 fp6_new(t2);
110
111 /* Karatsuba algorithm. */
112
113 /* t0 = a_0 * b_0. */
114 fp6_mul(t0, a[0], b[0]);
115 /* t1 = a_1 * b_1. */
116 fp6_mul(t1, a[1], b[1]);
117 /* t2 = b_0 + b_1. */
118 fp6_add(t2, b[0], b[1]);
119
120 /* c_1 = a_0 + a_1. */
121 fp6_add(c[1], a[0], a[1]);
122
123 /* c_1 = (a_0 + a_1) * (b_0 + b_1) */
124 fp6_mul(c[1], c[1], t2);
125 fp6_sub(c[1], c[1], t0);
126 fp6_sub(c[1], c[1], t1);
127
128 /* c_0 = a_0b_0 + v * a_1b_1. */
129 fp6_mul_art(t1, t1);
130 fp6_add(c[0], t0, t1);
131 } RLC_CATCH_ANY {
132 RLC_THROW(ERR_CAUGHT);
133 } RLC_FINALLY {
134 fp6_free(t0);
135 fp6_free(t1);
136 fp6_free(t2);
137 }
138}

140void fp12_mul_dxs_basic(fp12_t c, const fp12_t a, const fp12_t b) {
141 fp6_t t0, t1, t2;
142
143 fp6_null(t0);
144 fp6_null(t1);
145 fp6_null(t2);
146
147 RLC_TRY {
148 fp6_new(t0);
149 fp6_new(t1);
150 fp6_new(t2);
151
152 if (ep2_curve_is_twist() == RLC_EP_DTYPE) {
153#if EP_ADD == BASIC
154 /* t0 = a_0 * b_0 */
155 fp_mul(t0[0][0], a[0][0][0], b[0][0][0]);
156 fp_mul(t0[0][1], a[0][0][1], b[0][0][0]);
157 fp_mul(t0[1][0], a[0][1][0], b[0][0][0]);
158 fp_mul(t0[1][1], a[0][1][1], b[0][0][0]);
159 fp_mul(t0[2][0], a[0][2][0], b[0][0][0]);
160 fp_mul(t0[2][1], a[0][2][1], b[0][0][0]);
161 /* t2 = b_0 + b_1. */
162 fp_add(t2[0][0], b[0][0][0], b[1][0][0]);
163 fp_copy(t2[0][1], b[1][0][1]);
164 fp2_copy(t2[1], b[1][1]);
165#elif EP_ADD == PROJC || EP_ADD == JACOB
166 /* t0 = a_0 * b_0 */
167 fp2_mul(t0[0], a[0][0], b[0][0]);
168 fp2_mul(t0[1], a[0][1], b[0][0]);
169 fp2_mul(t0[2], a[0][2], b[0][0]);
170 /* t2 = b_0 + b_1. */
171 fp2_add(t2[0], b[0][0], b[1][0]);
172 fp2_copy(t2[1], b[1][1]);
173#endif
174 /* t1 = a_1 * b_1. */
175 fp6_mul_dxs(t1, a[1], b[1]);
176 } else {
177 /* t0 = a_0 * b_0. */
178 fp6_mul_dxs(t0, a[0], b[0]);
179#if EP_ADD == BASIC
180 /* t1 = a_1 * b_1. */
181 fp_mul(t2[0][0], a[1][2][0], b[1][1][0]);
182 fp_mul(t2[0][1], a[1][2][1], b[1][1][0]);
183 fp2_mul_nor(t1[0], t2[0]);
184 fp_mul(t1[1][0], a[1][0][0], b[1][1][0]);
185 fp_mul(t1[1][1], a[1][0][1], b[1][1][0]);
186 fp_mul(t1[2][0], a[1][1][0], b[1][1][0]);
187 fp_mul(t1[2][1], a[1][1][1], b[1][1][0]);
188 /* t2 = b_0 + b_1. */
189 fp2_copy(t2[0], b[0][0]);
190 fp_add(t2[1][0], b[0][1][0], b[1][1][0]);
191 fp_copy(t2[1][1], b[0][1][1]);
192#elif EP_ADD == PROJC || EP_ADD == JACOB
193 /* t1 = a_1 * b_1. */
194 fp2_mul(t2[0], a[1][2], b[1][1]);
195 fp2_mul_nor(t1[0], t2[0]);
196 fp2_mul(t1[1], a[1][0], b[1][1]);
197 fp2_mul(t1[2], a[1][1], b[1][1]);
198 /* t2 = b_0 + b_1. */
199 fp2_copy(t2[0], b[0][0]);
200 fp2_add(t2[1], b[0][1], b[1][1]);
201#endif
202 }
203 /* c_1 = a_0 + a_1. */
204 fp6_add(c[1], a[0], a[1]);
205 /* c_1 = (a_0 + a_1) * (b_0 + b_1) - a_0 * b_0 - a_1 * b_1. */
206 fp6_mul_dxs(c[1], c[1], t2);
207 fp6_sub(c[1], c[1], t0);
208 fp6_sub(c[1], c[1], t1);
209 /* c_0 = a_0 * b_0 + v * a_1 * b_1. */
210 fp6_mul_art(t1, t1);
211 fp6_add(c[0], t0, t1);
212 }
213 RLC_CATCH_ANY {
214 RLC_THROW(ERR_CAUGHT);
215 }
216 RLC_FINALLY {
217 fp6_free(t0);
218 fp6_free(t1);
219 fp6_free(t2);
220 }
221}

227void fp12_mul_unr(dv12_t c, const fp12_t a, const fp12_t b) {
228 fp6_t t0, t1;
229 dv6_t u0, u1, u2, u3;
230
231 dv6_null(u0);
232 dv6_null(u1);
233 dv6_null(u2);
234 dv6_null(u3);
235 fp6_null(t0);
236 fp6_null(t1);
237
238 RLC_TRY {
239 dv6_new(u0);
240 dv6_new(u1);
241 dv6_new(u2);
242 dv6_new(u3);
243 fp6_new(t0);
244 fp6_new(t1);
245
246 /* Karatsuba algorithm. */
247
248 /* u0 = a_0 * b_0. */
249 fp6_mul_unr(u0, a[0], b[0]);
250 /* u1 = a_1 * b_1. */
251 fp6_mul_unr(u1, a[1], b[1]);
252 /* t1 = a_0 + a_1. */
253 fp6_add(t0, a[0], a[1]);
254 /* t0 = b_0 + b_1. */
255 fp6_add(t1, b[0], b[1]);
256 /* u2 = (a_0 + a_1) * (b_0 + b_1) */
257 fp6_mul_unr(u2, t0, t1);
258 /* c_1 = u2 - a_0b_0 - a_1b_1. */
259 for (int i = 0; i < 3; i++) {
260 fp2_addc_low(u3[i], u0[i], u1[i]);
261 fp2_subc_low(c[1][i], u2[i], u3[i]);
262 }
263 /* c_0 = a_0b_0 + v * a_1b_1. */
264 fp2_nord_low(u2[0], u1[2]);
265 fp2_addc_low(c[0][0], u0[0], u2[0]);
266 fp2_addc_low(c[0][1], u0[1], u1[0]);
267 fp2_addc_low(c[0][2], u0[2], u1[1]);
268 } RLC_CATCH_ANY {
269 RLC_THROW(ERR_CAUGHT);
270 } RLC_FINALLY {
271 dv6_free(u0);
272 dv6_free(u1);
273 dv6_free(u2);
274 dv6_free(u3);
275 fp6_free(t0);
276 fp6_free(t1);
277 }
278}

299void fp12_mul_dxs_lazyr(fp12_t c, const fp12_t a, const fp12_t b) {
300 fp6_t t0;
301 dv6_t u0, u1, u2;
302
303 fp6_null(t0);
304 dv6_null(u0);
305 dv6_null(u1);
306 dv6_null(u2);
307
308 RLC_TRY {
309 fp6_new(t0);
310 dv6_new(u0);
311 dv6_new(u1);
312 dv6_new(u2);
313
314 if (ep2_curve_is_twist() == RLC_EP_DTYPE) {
315#if EP_ADD == BASIC
316 /* t0 = a_0 * b_0. */
317 fp_muln_low(u0[0][0], a[0][0][0], b[0][0][0]);
318 fp_muln_low(u0[0][1], a[0][0][1], b[0][0][0]);
319 fp_muln_low(u0[1][0], a[0][1][0], b[0][0][0]);
320 fp_muln_low(u0[1][1], a[0][1][1], b[0][0][0]);
321 fp_muln_low(u0[2][0], a[0][2][0], b[0][0][0]);
322 fp_muln_low(u0[2][1], a[0][2][1], b[0][0][0]);
323 /* t2 = b_0 + b_1. */
324 fp_add(t0[0][0], b[0][0][0], b[1][0][0]);
325 fp_copy(t0[0][1], b[1][0][1]);
326 fp2_copy(t0[1], b[1][1]);
327#elif EP_ADD == PROJC || EP_ADD == JACOB
328 /* t0 = a_0 * b_0. */
329 fp2_muln_low(u0[0], a[0][0], b[0][0]);
330 fp2_muln_low(u0[1], a[0][1], b[0][0]);
331 fp2_muln_low(u0[2], a[0][2], b[0][0]);
332
333 /* t2 = b_0 + b_1. */
334 fp2_add(t0[0], b[0][0], b[1][0]);
335 fp2_copy(t0[1], b[1][1]);
336#endif
337 /* t1 = a_1 * b_1. */
338 fp6_mul_dxs_unr_lazyr(u1, a[1], b[1]);
339 } else {
340 /* t0 = a_0 * b_0. */
341 fp6_mul_dxs_unr_lazyr(u0, a[0], b[0]);
342#if EP_ADD == BASIC
343 /* t0 = a_0 * b_0. */
344 fp_muln_low(u1[1][0], a[1][2][0], b[1][1][0]);
345 fp_muln_low(u1[1][1], a[1][2][1], b[1][1][0]);
346 fp2_nord_low(u1[0], u1[1]);
347 fp_muln_low(u1[1][0], a[1][0][0], b[1][1][0]);
348 fp_muln_low(u1[1][1], a[1][0][1], b[1][1][0]);
349 fp_muln_low(u1[2][0], a[1][1][0], b[1][1][0]);
350 fp_muln_low(u1[2][1], a[1][1][1], b[1][1][0]);
351 /* t2 = b_0 + b_1. */
352 fp2_copy(t0[0], b[0][0]);
353 fp_add(t0[1][0], b[0][1][0], b[1][1][0]);
354 fp_copy(t0[1][1], b[0][1][1]);
355#elif EP_ADD == PROJC || EP_ADD == JACOB
356 /* t1 = a_1 * b_1. */
357 fp2_muln_low(u1[1], a[1][2], b[1][1]);
358 fp2_nord_low(u1[0], u1[1]);
359 fp2_muln_low(u1[1], a[1][0], b[1][1]);
360 fp2_muln_low(u1[2], a[1][1], b[1][1]);
361 /* t2 = b_0 + b_1. */
362 fp2_copy(t0[0], b[0][0]);
363 fp2_add(t0[1], b[0][1], b[1][1]);
364#endif
365 }
366 /* c_1 = a_0 + a_1. */
367 fp6_add(c[1], a[0], a[1]);
368 /* c_1 = (a_0 + a_1) * (b_0 + b_1) */
369 fp6_mul_dxs_unr_lazyr(u2, c[1], t0);
370 for (int i = 0; i < 3; i++) {
371 fp2_subc_low(u2[i], u2[i], u0[i]);
372 fp2_subc_low(u2[i], u2[i], u1[i]);
373 }
374 fp2_rdcn_low(c[1][0], u2[0]);
375 fp2_rdcn_low(c[1][1], u2[1]);
376 fp2_rdcn_low(c[1][2], u2[2]);
377
378 fp2_nord_low(u2[0], u1[2]);
379 fp2_addc_low(u0[0], u0[0], u2[0]);
380 fp2_addc_low(u0[1], u0[1], u1[0]);
381 fp2_addc_low(u0[2], u0[2], u1[1]);
382 /* c_0 = a_0b_0 + v * a_1b_1. */
383 fp2_rdcn_low(c[0][0], u0[0]);
384 fp2_rdcn_low(c[0][1], u0[1]);
385 fp2_rdcn_low(c[0][2], u0[2]);
386 } RLC_CATCH_ANY {
387 RLC_THROW(ERR_CAUGHT);
388 } RLC_FINALLY {
389 fp6_free(t0);
390 dv6_free(u0);
391 dv6_free(u1);
392 dv6_free(u2);
393 }
394}
