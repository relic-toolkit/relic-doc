48static void bn_div_imp(bn_t c, bn_t d, const bn_t a, const bn_t b) {
49 bn_t q, x, y, r;
50 int sign;
51
52 bn_null(q);
53 bn_null(x);
54 bn_null(y);
55 bn_null(r);
56
57 /* If |a| < |b|, we're done. */
58 if (bn_cmp_abs(a, b) == RLC_LT) {
59 if (bn_sign(a) == bn_sign(b)) {
60 if (c != NULL) {
61 bn_zero(c);
62 }
63 if (d != NULL) {
64 bn_copy(d, a);
65 }
66 } else {
67 if (c != NULL) {
68 bn_set_dig(c, 1);
69 bn_neg(c, c);
70 }
71 if (d != NULL) {
72 bn_add(d, a, b);
73 }
74 }
75 return;
76 }
77
78 RLC_TRY {
79 /* Be conservative about space for scratch memory, many attempts to
80 * optimize these had invalid reads. */
81 bn_new_size(x, a->used + 1);
82 bn_new_size(q, a->used + 1);
83 bn_new_size(y, a->used + 1);
84 bn_new_size(r, a->used + 1);
85 bn_zero(q);
86 bn_zero(r);
87 bn_abs(x, a);
88 bn_abs(y, b);
89
90 /* Find the sign. */
91 sign = (a->sign == b->sign ? RLC_POS : RLC_NEG);
92
93 bn_divn_low(q->dp, r->dp, x->dp, a->used, y->dp, b->used);
94
95 q->used = a->used - b->used + 1;
96 q->sign = sign;
97 bn_trim(q);
98
99 r->used = b->used;
100 r->sign = b->sign;
101 bn_trim(r);
102
103 /* We have the quotient in q and the remainder in r. */
104 if (c != NULL) {
105 if ((bn_is_zero(r)) || (bn_sign(a) == bn_sign(b))) {
106 bn_copy(c, q);
107 } else {
108 bn_sub_dig(c, q, 1);
109 }
110 }
111
112 if (d != NULL) {
113 if ((bn_is_zero(r)) || (bn_sign(a) == bn_sign(b))) {
114 bn_copy(d, r);
115 } else {
116 bn_sub(d, b, r);
117 }
118 }
119 }
120 RLC_CATCH_ANY {
121 RLC_THROW(ERR_CAUGHT);
122 }
123 RLC_FINALLY {
124 bn_free(r);
125 bn_free(q);
126 bn_free(x);
127 bn_free(y);
128 }
129}

135void bn_div(bn_t c, const bn_t a, const bn_t b) {
136 if (bn_is_zero(b)) {

140 bn_div_imp(c, NULL, a, b);

151void bn_div_dig(bn_t c, const bn_t a, dig_t b) {
152 bn_t q;
153 dig_t r;
154
155 bn_null(q);
156
157 if (b == 0) {
158 RLC_THROW(ERR_NO_VALID);
159 return;
160 }
161
162 if (b == 1 || bn_is_zero(a) == 1) {
163 if (c != NULL) {
164 bn_copy(c, a);
165 }
166 return;
167 }
168
169 RLC_TRY {
170 bn_new(q);
171
172 bn_copy(q, a);
173 bn_div1_low(q->dp, &r, (const dig_t *)a->dp, b, a->used);
174 if (c != NULL) {
175 bn_copy(c, q);
176 }
177 }
178 RLC_CATCH_ANY {
179 RLC_THROW(ERR_CAUGHT);
180 }
181 RLC_FINALLY {
182 bn_free(q);
183 }
184}

186void bn_div_rem_dig(bn_t c, dig_t *d, const bn_t a, dig_t b) {
187 bn_t q;
188 dig_t r;
189
190 bn_null(q);
191
192 if (b == 0) {
193 RLC_THROW(ERR_NO_VALID);
194 return;
195 }
196
197 if (b == 1 || bn_is_zero(a) == 1) {
198 if (d != NULL) {
199 *d = 0;
200 }
201 if (c != NULL) {
202 bn_copy(c, a);
203 }
204 return;
205 }
206
207 RLC_TRY {
208 bn_new(q);
209
210 bn_copy(q, a);
211 bn_div1_low(q->dp, &r, (const dig_t *)a->dp, b, a->used);
212
213 if (c != NULL) {
214 bn_copy(c, q);
215 }
216
217 if (d != NULL) {
218 if (bn_sign(a) == RLC_NEG) {
219 *d = b - r;
220 } else {
221 *d = r;
222 }
223 }
224 }
225 RLC_CATCH_ANY {
226 RLC_THROW(ERR_CAUGHT);
227 }
228 RLC_FINALLY {
229 bn_free(q);
230 }
