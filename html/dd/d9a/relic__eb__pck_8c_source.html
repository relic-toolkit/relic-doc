41void eb_pck(eb_t r, const eb_t p) {
42 /* z3 = y1/x1. */
43 fb_inv(r->z, p->x);
44 fb_mul(r->z, r->z, p->y);
45 /* x3 = x1. */
46 fb_copy(r->x, p->x);
47 /* y3 = b(y1/x1). */
48 fb_set_dig(r->y, fb_get_bit(r->z, 0));
49 /* z3 = 1. */
50 fb_set_dig(r->z, 1);
51
52 r->coord = BASIC;
53}

55int eb_upk(eb_t r, const eb_t p) {
56 fb_t t0, t1;
57 int result = 0;
58
59 fb_null(t0);
60 fb_null(t1);
61
62 RLC_TRY {
63 fb_new(t0);
64 fb_new(t1);
65
66 eb_rhs(t1, p->x);
67
68 fb_sqr(t0, p->x);
69 /* t0 = 1/x1^2. */
70 fb_inv(t0, t0);
71 /* t0 = t1/x1^2. */
72 fb_mul(t0, t0, t1);
73 result = (fb_trc(t0) == 0);
74 if (result) {
75 /* Solve t1^2 + t1 = t0. */
76 fb_slv(t1, t0);
77 /* If this is not the correct solution, try the other. */
78 if (fb_get_bit(t1, 0) != fb_get_bit(p->y, 0)) {
79 fb_add_dig(t1, t1, 1);
80 }
81 /* x3 = x1, y3 = t1 * x1, z3 = 1. */
82 fb_mul(r->y, t1, p->x);
83
84 fb_copy(r->x, p->x);
85 fb_set_dig(r->z, 1);
86 r->coord = BASIC;
87 }
88 }
89 RLC_CATCH_ANY {
90 RLC_THROW(ERR_CAUGHT);
91 }
92 RLC_FINALLY {
93 fb_free(t0);
94 fb_free(t1);
95 }
96 return result;
97}
