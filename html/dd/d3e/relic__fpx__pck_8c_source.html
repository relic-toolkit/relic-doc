38void fp2_pck(fp2_t c, const fp2_t a) {
39 int b = fp_get_bit(a[1], 0);
40 fp2_copy(c, a);
41 if (fp2_test_cyc(a)) {
42 fp_copy(c[0], a[0]);
43 fp_zero(c[1]);
44 fp_set_bit(c[1], 0, b);
45 }
46}

48int fp2_upk(fp2_t c, const fp2_t a) {
49 if (fp_bits(a[1]) <= 1) {
50 int result, b = fp_get_bit(a[1], 0);
51 fp_t t;
52
53 fp_null(t);
54
55 RLC_TRY {
56 fp_new(t);
57
58 /* a_0^2 + a_1^2 = 1, thus a_1^2 = 1 - a_0^2. */
59 fp_sqr(t, a[0]);
60 fp_sub_dig(t, t, 1);
61 fp_neg(t, t);
62
63 /* a1 = sqrt(a_0^2). */
64 result = fp_srt(t, t);
65
66 if (result) {
67 /* Verify if least significant bit of the result matches the
68 * compressed second coordinate. */
69 if (fp_get_bit(t, 0) != b) {
70 fp_neg(t, t);
71 }
72 fp_copy(c[0], a[0]);
73 fp_copy(c[1], t);
74 }
75 } RLC_CATCH_ANY {
76 RLC_THROW(ERR_CAUGHT);
77 } RLC_FINALLY {
78 fp_free(t);
79 }
80 return result;
81 } else {
82 fp2_copy(c, a);
83 return 1;
84 }
85}

109void fp12_pck_max(fp12_t c, const fp12_t a) {
110 fp12_copy(c, a);
111 if (fp12_test_cyc(c)) {
112 /* Use torus-based compression from Section 4.1 in
113 * "On Compressible Pairings and Their Computation" by Naehrig et al.
114 */
115 fp2_add_dig(c[0][0], a[0][0], 1);
116 fp6_inv(c[1], a[1]);
117 fp6_mul(c[0], c[0], c[1]);
118 fp6_zero(c[1]);
119 }
120}

122int fp12_upk_max(fp12_t c, const fp12_t a) {
123 if (fp6_is_zero(a[1])) {
124 fp12_t t;
125
126 fp12_null(t);
127
128 RLC_TRY {
129 fp12_new(t);
130 /* Formula for decompression for the odd q case from Section 2 in
131 * "Compression in finite fields and torus-based cryptography" by
132 * Rubin-Silverberg.
133 */
134 fp6_copy(t[0], a[0]);
135 fp6_zero(t[1]);
136 fp_set_dig(t[1][0][0], 1);
137 fp_neg(t[1][0][0], t[1][0][0]);
138 fp12_inv(t, t);
139 fp6_copy(c[0], a[0]);
140 fp6_set_dig(c[1], 1);
141 fp12_mul(c, c, t);
142 } RLC_CATCH_ANY {
143 RLC_THROW(ERR_CAUGHT);
144 } RLC_FINALLY {
145 fp12_free(t);
146 }
147 if (fp12_test_cyc(c)) {
148 return 1;
149 } else {
150 return 0;
151 }
152 } else {
153 fp12_copy(c, a);
154 return 1;
155 }
