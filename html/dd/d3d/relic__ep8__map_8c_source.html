40void ep8_map(ep8_t p, const uint8_t *msg, size_t len) {
41 /* enough space for two field elements plus extra bytes for uniformity */
42 const size_t elm = (FP_PRIME + ep_param_level() + 7) / 8;
43 uint8_t t0z, t0, t1, s[2][2], sign, *r = RLC_ALLOCA(uint8_t, 16 * elm + 1);
44 fp8_t t, u, v, w, y, x1, y1, z1;
45 ctx_t *ctx = core_get();
46 dig_t c2, c3;
47 bn_t k;
48
49 bn_null(k);
50 fp8_null(t);
51 fp8_null(u);
52 fp8_null(v);
53 fp8_null(w);
54 fp8_null(y);
55 fp8_null(x1);
56 fp8_null(y1);
57 fp8_null(z1);
58
59 RLC_TRY {
60 bn_new(k);
61 fp8_new(t);
62 fp8_new(u);
63 fp8_new(v);
64 fp8_new(w);
65 fp8_new(y);
66 fp8_new(x1);
67 fp8_new(y1);
68 fp8_new(z1);
69
70 md_xmd(r, 16 * elm + 1, msg, len, (const uint8_t *)"RELIC", 5);
71
72 for (int i = 0; i < 2; i++) {
73 for (int j = 0; j < 2; j++) {
74 for (int l = 0; l < 2; l++) {
75 bn_read_bin(k, r, elm);
76 fp_prime_conv(u[i][j][l], k);
77 r += elm;
78 bn_read_bin(k, r, elm);
79 fp_prime_conv(t[i][j][l], k);
80 r += elm;
81 }
82 }
83 }
84 sign = r[0] & 1;
85 r -= 16*elm;
86
87 /* Assume that a = 0. */
88 fp8_sqr(x1, u);
89 fp8_mul(x1, x1, u);
90 fp8_sqr(y1, t);
91 fp8_add(x1, x1, ctx->ep8_b);
92 fp8_sub(x1, x1, y1);
93 fp8_dbl(y1, y1);
94 fp8_add(y1, y1, x1);
95 fp8_copy(z1, u);
96 for (int i = 0; i < 2; i++) {
97 for (int j = 0; j < 2; j++) {
98 for (int l = 0; l < 2; l++) {
99 fp_mul(z1[i][j][l], z1[i][j][l], ctx->ep_map_c[4]);
100 }
101 }
102 }
103
104 fp8_mul(x1, x1, z1);
105 fp8_mul(z1, z1, t);
106 fp8_dbl(z1, z1);
107
108 fp8_dbl(y, y1);
109 fp8_sqr(y, y);
110 fp8_mul(v, y1, u);
111 fp8_sub(v, x1, v);
112 fp8_mul(v, v, z1);
113 fp8_mul(w, y1, z1);
114 fp8_dbl(w, w);
115
116 if (fp8_is_zero(w)) {
117 ep8_set_infty(p);
118 } else {
119 fp8_inv(w, w);
120 fp8_mul(x1, v, w);
121 fp8_add(y1, u, x1);
122 fp8_neg(y1, y1);
123 fp8_mul(z1, y, w);
124 fp8_sqr(z1, z1);
125 fp8_add(z1, z1, u);
126
127 ep8_rhs(t, x1);
128 ep8_rhs(u, y1);
129 ep8_rhs(v, z1);
130
131 c2 = fp8_is_sqr(u);
132 c3 = fp8_is_sqr(v);
133
134 fp8_copy_sec(t, u, c2);
135 fp8_copy_sec(x1, y1, c2);
136 fp8_copy_sec(t, v, c3);
137 fp8_copy_sec(x1, z1, c3);
138
139 if (!fp8_srt(t, t)) {
140 RLC_THROW(ERR_NO_VALID);
141 }
142
143 for (int i = 0; i < 2; i++) {
144 for (int j = 0; j < 2; j++) {
145 t0z = fp_is_zero(t[i][j][0]);
146 fp_prime_back(k, t[i][j][0]);
147 t0 = bn_get_bit(k, 0);
148 fp_prime_back(k, t[i][j][1]);
149 t1 = bn_get_bit(k, 0);
150 /* t[0] == 0 ? sgn0(t[1]) : sgn0(t[0]) */
151 s[i][j] = t0 | (t0z & t1);
152 }
153
154 t0z = fp4_is_zero(t[i]);
155 sign ^= (s[i][0] | (t0z & s[i][1]));
156 }
157
158 fp8_neg(u, t);
159 fp8_copy_sec(t, u, sign);
160
161 fp8_copy(p->x, x1);
162 fp8_copy(p->y, t);
163 fp8_set_dig(p->z, 1);
164 p->coord = BASIC;
165
166 ep8_mul_cof(p, p);
167 }
168 }
169 RLC_CATCH_ANY {
170 RLC_THROW(ERR_CAUGHT);
171 }
172 RLC_FINALLY {
173 bn_free(k);
174 fp8_free(t);
175 fp8_free(u);
176 fp8_free(v);
177 fp8_free(w);
178 fp8_free(y);
179 fp8_free(x1);
180 fp8_free(y1);
181 fp8_free(z1);
182 RLC_FREE(r);
183 }
184}
