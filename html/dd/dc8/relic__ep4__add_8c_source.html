175void ep4_sub(ep4_t r, const ep4_t p, const ep4_t q) {
176 ep4_t t;
177
178 ep4_null(t);
179
180 if (p == q) {
181 ep4_set_infty(r);
182 return;
183 }
184
185 RLC_TRY {
186 ep4_new(t);
187 ep4_neg(t, q);
188 ep4_add(r, p, t);
189 }
190 RLC_CATCH_ANY {
191 RLC_THROW(ERR_CAUGHT);
192 }
193 RLC_FINALLY {
194 ep4_free(t);
195 }
