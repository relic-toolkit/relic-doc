57 int ep_cmp(const ep_t p, const ep_t q) {
58  ep_t r, s;
59  int result = RLC_EQ;
60 
61  ep_null(r);
62  ep_null(s);
63 
64  TRY {
65  ep_new(r);
66  ep_new(s);
67 
68  if ((!p->norm) && (!q->norm)) {
69  /* If the two points are not normalized, it is faster to compare
70  * x1 * z2^2 == x2 * z1^2 and y1 * z2^3 == y2 * z1^3. */
71  fp_sqr(r->z, p->z);
72  fp_sqr(s->z, q->z);
73  fp_mul(r->x, p->x, s->z);
74  fp_mul(s->x, q->x, r->z);
75  fp_mul(r->z, r->z, p->z);
76  fp_mul(s->z, s->z, q->z);
77  fp_mul(r->y, p->y, s->z);
78  fp_mul(s->y, q->y, r->z);
79  } else {
80  ep_copy(r, p);
81  ep_copy(s, q);
82  if (!p->norm) {
83  ep_norm(r, p);
84  }
85  if (!q->norm) {
86  ep_norm(s, q);
87  }
88  }
89 
90  if (fp_cmp(r->x, s->x) != RLC_EQ) {
91  result = RLC_NE;
92  }
93  if (fp_cmp(r->y, s->y) != RLC_EQ) {
94  result = RLC_NE;
95  }
96  } CATCH_ANY {
97  THROW(ERR_CAUGHT);
98  } FINALLY {
99  ep_free(r);
100  ep_free(s);
101  }
102 
103  return result;
104 }

128 void ep_rhs(fp_t rhs, const ep_t p) {
129  fp_t t0;
130  fp_t t1;
131 
132  fp_null(t0);
133  fp_null(t1);
134 
135  TRY {
136  fp_new(t0);
137  fp_new(t1);
138 
139  /* t0 = x1^2. */
140  fp_sqr(t0, p->x);
141  /* t1 = x1^3. */
142  fp_mul(t1, t0, p->x);
143 
144  /* t1 = x1^3 + a * x1 + b. */
145  switch (ep_curve_opt_a()) {
146  case RLC_ZERO:
147  break;
148  case RLC_ONE:
149  fp_add(t1, t1, p->x);
150  break;
151 #if FP_RDC != MONTY
152  case RLC_TINY:
153  fp_mul_dig(t0, p->x, ep_curve_get_a()[0]);
154  fp_add(t1, t1, t0);
155  break;
156 #endif
157  default:
158  fp_mul(t0, p->x, ep_curve_get_a());
159  fp_add(t1, t1, t0);
160  break;
161  }
162 
163  switch (ep_curve_opt_b()) {
164  case RLC_ZERO:
165  break;
166  case RLC_ONE:
167  fp_add_dig(t1, t1, 1);
168  break;
169 #if FP_RDC != MONTY
170  case RLC_TINY:
171  fp_add_dig(t1, t1, ep_curve_get_b()[0]);
172  break;
173 #endif
174  default:
175  fp_add(t1, t1, ep_curve_get_b());
176  break;
177  }
178 
179  fp_copy(rhs, t1);
180  } CATCH_ANY {
181  THROW(ERR_CAUGHT);
182  } FINALLY {
183  fp_free(t0);
184  fp_free(t1);
185  }
186 }

188 int ep_is_valid(const ep_t p) {
189  ep_t t;
190  int r = 0;
191 
192  ep_null(t);
193 
194  TRY {
195  ep_new(t);
196 
197  ep_norm(t, p);
198  ep_rhs(t->x, t);
199  fp_sqr(t->y, t->y);
200  r = (fp_cmp(t->x, t->y) == RLC_EQ) || ep_is_infty(p);
201  } CATCH_ANY {
202  THROW(ERR_CAUGHT);
203  } FINALLY {
204  ep_free(t);
205  }
206  return r;
207 }

209 void ep_tab(ep_t *t, const ep_t p, int w) {
210  if (w > 2) {
211  ep_dbl(t[0], p);
212 #if defined(EP_MIXED)
213  ep_norm(t[0], t[0]);
214 #endif
215  ep_add(t[1], t[0], p);
216  for (int i = 2; i < (1 << (w - 2)); i++) {
217  ep_add(t[i], t[i - 1], t[0]);
218  }
219 #if defined(EP_MIXED)
220  ep_norm_sim(t + 1, (const ep_t *)t + 1, (1 << (w - 2)) - 1);
221 #endif
222  }
223  ep_copy(t[0], p);
224 }

247 void ep_read_bin(ep_t a, const uint8_t *bin, int len) {
248  if (len == 1) {
249  if (bin[0] == 0) {
250  ep_set_infty(a);
251  return;
252  } else {
253  THROW(ERR_NO_BUFFER);
254  return;
255  }
256  }
257 
258  if (len != (RLC_FP_BYTES + 1) && len != (2 * RLC_FP_BYTES + 1)) {
259  THROW(ERR_NO_BUFFER);
260  return;
261  }
262 
263  a->norm = 1;
264  fp_set_dig(a->z, 1);
265  fp_read_bin(a->x, bin + 1, RLC_FP_BYTES);
266  if (len == RLC_FP_BYTES + 1) {
267  switch(bin[0]) {
268  case 2:
269  fp_zero(a->y);
270  break;
271  case 3:
272  fp_zero(a->y);
273  fp_set_bit(a->y, 0, 1);
274  break;
275  default:
276  THROW(ERR_NO_VALID);
277  break;
278  }
279  ep_upk(a, a);
280  }
281 
282  if (len == 2 * RLC_FP_BYTES + 1) {
283  if (bin[0] == 4) {
284  fp_read_bin(a->y, bin + RLC_FP_BYTES + 1, RLC_FP_BYTES);
285  } else {
286  THROW(ERR_NO_VALID);
287  }
288  }
289 }

291 void ep_write_bin(uint8_t *bin, int len, const ep_t a, int pack) {
292  ep_t t;
293 
294  ep_null(t);
295 
296  if (ep_is_infty(a)) {
297  if (len < 1) {
298  THROW(ERR_NO_BUFFER);
299  } else {
300  bin[0] = 0;
301  return;
302  }
303  }
304 
305  TRY {
306  ep_new(t);
307 
308  ep_norm(t, a);
309 
310  if (pack) {
311  if (len < RLC_FP_BYTES + 1) {
312  THROW(ERR_NO_BUFFER);
313  } else {
314  ep_pck(t, t);
315  bin[0] = 2 | fp_get_bit(t->y, 0);
316  fp_write_bin(bin + 1, RLC_FP_BYTES, t->x);
317  }
318  } else {
319  if (len < 2 * RLC_FP_BYTES + 1) {
320  THROW(ERR_NO_BUFFER);
321  } else {
322  bin[0] = 4;
323  fp_write_bin(bin + 1, RLC_FP_BYTES, t->x);
324  fp_write_bin(bin + RLC_FP_BYTES + 1, RLC_FP_BYTES, t->y);
325  }
326  }
327  } CATCH_ANY {
328  THROW(ERR_CAUGHT);
329  }
330  FINALLY {
331  ep_free(t);
332  }
