38 void bn_smb_leg(bn_t c, const bn_t a, const bn_t b) {
39  bn_t t;
40 
41  bn_null(t);
42 
43  if (bn_cmp(a, b) == RLC_EQ) {
44  bn_zero(c);
45  return;
46  }
47 
48  TRY {
49  bn_new(t);
50 
51  if (bn_sign(b) == RLC_NEG) {
52  THROW(ERR_NO_VALID);
53  }
54 
55  /* t = (b - 1)/2. */
56  bn_sub_dig(t, b, 1);
57  bn_rsh(t, t, 1);
58  bn_mxp(c, a, t, b);
59  bn_sub_dig(t, b, 1);
60  if (bn_cmp(c, t) == RLC_EQ) {
61  bn_set_dig(c, 1);
62  bn_neg(c, c);
63  }
64  }
65  CATCH_ANY {
66  THROW(ERR_CAUGHT);
67  }
68  FINALLY {
69  bn_free(t);
70  }
71 }

73 void bn_smb_jac(bn_t c, const bn_t a, const bn_t b) {
74  bn_t t0, t1, r;
75  int t, h;
76 
77  bn_null(t0);
78  bn_null(t1);
79  bn_null(r);
80 
81  TRY {
82  bn_new(t0);
83  bn_new(t1);
84  bn_new(r);
85  t = 1;
86 
87  /* Argument b must be odd. */
88  if (bn_is_even(b) || bn_sign(b) == RLC_NEG) {
89  THROW(ERR_NO_VALID);
90  }
91 
92  if (bn_sign(a) == RLC_NEG) {
93  bn_add(t0, a, b);
94  } else {
95  bn_copy(t0, a);
96  }
97  bn_copy(t1, b);
98 
99  while (1) {
100  /* t0 = a mod b. */
101  bn_mod(t0, t0, t1);
102  /* If a = 0 then if n = 1 return t else return 0. */
103  if (bn_is_zero(t0)) {
104  if (bn_cmp_dig(t1, 1) == RLC_EQ) {
105  bn_set_dig(c, 1);
106  if (t == -1) {
107  bn_neg(c, c);
108  }
109  break;
110  } else {
111  bn_zero(c);
112  break;
113  }
114  }
115  /* Write t0 as 2^h * t0. */
116  h = 0;
117  while (bn_is_even(t0)) {
118  h++;
119  bn_rsh(t0, t0, 1);
120  }
121  /* If h != 0 (mod 2) and n != +-1 (mod 8) then t = -t. */
122  bn_mod_2b(r, t1, 3);
123  if ((h % 2 != 0) && (bn_cmp_dig(r, 1) != RLC_EQ) &&
124  (bn_cmp_dig(r, 7) != RLC_EQ)) {
125  t = -t;
126  }
127  /* If t0 != 1 (mod 4) and n != 1 (mod 4) then t = -t. */
128  bn_mod_2b(r, t0, 2);
129  if (bn_cmp_dig(r, 1) != RLC_EQ) {
130  bn_mod_2b(r, t1, 2);
131  if (bn_cmp_dig(r, 1) != RLC_EQ) {
132  t = -t;
133  }
134  }
135  bn_copy(r, t0);
136  bn_copy(t0, t1);
137  bn_copy(t1, r);
138  }
139  }
140  CATCH_ANY {
141  THROW(ERR_CAUGHT);
142  }
143  FINALLY {
144  bn_free(t0);
145  bn_free(t1);
146  bn_free(r);
147  }
148 }
