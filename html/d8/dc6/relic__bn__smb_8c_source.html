38int bn_smb_leg(const bn_t a, const bn_t b) {
39 bn_t t;
40 int res;
41
42 bn_null(t);
43
44 if (bn_sign(b) == RLC_NEG) {
45 RLC_THROW(ERR_NO_VALID);
46 return 0;
47 }
48
49 if (bn_cmp(a, b) == RLC_EQ) {
50 return 0;
51 }
52
53 RLC_TRY {
54 bn_new(t);
55
56 /* t = (b - 1)/2. */
57 bn_sub_dig(t, b, 1);
58 bn_rsh(t, t, 1);
59 bn_mxp(t, a, t, b);
60 res = 0;
61 if (bn_cmp_dig(t, 1) == RLC_EQ) {
62 res = 1;
63 }
64 bn_sub(t, b, t);
65 if (bn_cmp_dig(t, 1) == RLC_EQ) {
66 res = -1;
67 }
68 }
69 RLC_CATCH_ANY {
70 RLC_THROW(ERR_CAUGHT);
71 }
72 RLC_FINALLY {
73 bn_free(t);
74 }
75
76 return res;
77}

79int bn_smb_jac(const bn_t a, const bn_t b) {
80 dis_t ai, bi, ci, di;
81 dig_t n, d, t;
82 bn_t t0, t1, t2, t3;
83 uint_t z, i, s = (RLC_DIG >> 1) - 2;
84 int r;
85
86 bn_null(t0);
87 bn_null(t1);
88 bn_null(t2);
89 bn_null(t3);
90
91 /* Optimized Pornin's Algorithm by Aleksei Vambol from
92 * https://github.com/privacy-scaling-explorations/halo2curves/pull/95 */
93
94 /* Argument b must be odd for Jacobi symbol. */
95 if (bn_is_even(b) || bn_sign(b) == RLC_NEG) {
96 RLC_THROW(ERR_NO_VALID);
97 return 0;
98 }
99
100 RLC_TRY {
101 bn_new(t0);
102 bn_new(t1);
103 bn_new(t2);
104 bn_new(t3);
105
106 bn_mod(t0, a, b);
107 bn_copy(t1, b);
108 t = 0;
109
110 while (1) {
111 ai = di = 1;
112 bi = ci = 0;
113
114 i = RLC_MAX(t0->used, t1->used);
115 bn_grow(t0, i);
116 bn_grow(t1, i);
117 dv_zero(t0->dp + t0->used, i - t0->used);
118 dv_zero(t1->dp + t1->used, i - t1->used);
119 if (i == 1) {
120 n = t0->dp[0];
121 d = t1->dp[0];
122 while (n != 0) {
123 if (n & 1) {
124 if (n < d) {
125 RLC_SWAP(n, d);
126 t ^= (n & d);
127 }
128 n = (n - d) >> 1;
129 t ^= d ^ (d >> 1);
130 } else {
131 z = arch_tzcnt(n);
132 t ^= (d ^ (d >> 1)) & (z << 1);
133 n >>= z;
134 }
135 }
136 r = (d == 1 ? 1 - (t & 2) : 0);
137 break;
138 }
139
140 z = RLC_MIN(arch_lzcnt(t0->dp[i - 1]), arch_lzcnt(t1->dp[i - 1]));
141 n = t0->dp[i - 1] << z;
142 d = t1->dp[i - 1] << z;
143 if (z > (RLC_DIG >> 1)) {
144 n |= t0->dp[i - 2] >> z;
145 d |= t1->dp[i - 2] >> z;
146 }
147 n = (n & RLC_HMASK) | (t0->dp[0] & RLC_LMASK);
148 d = (d & RLC_HMASK) | (t1->dp[0] & RLC_LMASK);
149
150 i = s;
151 while (i > 0) {
152 if (n & 1) {
153 if (n < d) {
154 RLC_SWAP(ai, ci);
155 RLC_SWAP(bi, di);
156 RLC_SWAP(n, d);
157 t ^= (n & d);
158 }
159 n = (n - d) >> 1;
160 ai = ai - ci;
161 bi = bi - di;
162 ci += ci;
163 di += di;
164 t ^= d ^ (d >> 1);
165 i -= 1;
166 } else {
167 z = RLC_MIN(i, arch_tzcnt(n));
168 t ^= (d ^ (d >> 1)) & (z << 1);
169 ci = (dig_t)ci << z;
170 di = (dig_t)di << z;
171 n >>= z;
172 i -= z;
173 }
174 }
175
176 if (ai < 0) {
177 bn_mul_dig(t2, t0, -ai);
178 bn_neg(t2, t2);
179 } else {
180 bn_mul_dig(t2, t0, ai);
181 }
182 if (bi < 0) {
183 bn_mul_dig(t3, t1, -bi);
184 bn_neg(t3, t3);
185 } else {
186 bn_mul_dig(t3, t1, bi);
187 }
188 bn_add(t3, t3, t2);
189
190 if (ci < 0) {
191 bn_mul_dig(t2, t0, -ci);
192 bn_neg(t2, t2);
193 } else {
194 bn_mul_dig(t2, t0, ci);
195 }
196 if (di < 0) {
197 bn_mul_dig(t1, t1, -di);
198 bn_neg(t1, t1);
199 } else {
200 bn_mul_dig(t1, t1, di);
201 }
202 bn_add(t1, t1, t2);
203 bn_rsh(t1, t1, s);
204 bn_rsh(t0, t3, s);
205
206 if (bn_is_zero(t0)) {
207 r = (bn_cmp_dig(t1, 1) == RLC_EQ ? 1 - (t & 2) : 0);
208 break;
209 }
210
211 if (bn_sign(t0) == RLC_NEG) {
212 t ^= t1->dp[0];
213 bn_neg(t0, t0);
214 }
215 if (bn_sign(t1) == RLC_NEG) {
216 bn_neg(t1, t1);
217 }
218 }
219 }
220 RLC_CATCH_ANY {
221 RLC_THROW(ERR_CAUGHT);
222 }
223 RLC_FINALLY {
224 bn_free(t0);
225 bn_free(t1);
226 bn_free(t2);
227 bn_free(t3);
228 }
229
230 return r;
231}
