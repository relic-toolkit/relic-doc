48 int cp_bgn_gen(bgn_t pub, bgn_t prv) {

53

56 
57  g1_get_ord(n);
58 
59  bn_rand_mod(prv->x, n);

123 int cp_bgn_dec1(dig_t *out, g1_t in[2], bgn_t prv) {
124  bn_t r, n;
125  g1_t s, t, u;
126  int i, result = RLC_ERR;
127 
128  bn_null(n);
129  bn_null(r);
130  g1_null(s);
131  g1_null(t);
132  g1_null(u);
133 
134  TRY {
135  bn_new(n);
136  bn_new(r);
137  g1_new(s);
138  g1_new(t);
139  g1_new(u);
140 
141  g1_get_ord(n);
142  /* Compute T = x(ym + r)G - (zm + xr)G = m(xy - z)G. */
143  g1_mul(t, in[0], prv->x);
144  g1_sub(t, t, in[1]);
145  g1_norm(t, t);
146  /* Compute U = (xy - z)G and find m. */
147  bn_mul(r, prv->x, prv->y);
148  bn_sub(r, r, prv->z);
149  bn_mod(r, r, n);
150  g1_mul_gen(s, r);
151  g1_copy(u, s);
152 
153  if (g1_is_infty(t) == 1){
154  *out = 0;
155  result = RLC_OK;
156  } else {
157  for (i = 0; i < INT_MAX; i++) {
158  if (g1_cmp(t, u) == RLC_EQ) {
159  *out = i + 1;
160  result = RLC_OK;
161  break;
162  }
163  g1_add(u, u, s);
164  g1_norm(u, u);
165  }
166  }
167  } CATCH_ANY {
168  result = RLC_ERR;
169  }
170  FINALLY {
171  bn_free(n);
172  bn_free(r);
173  g1_free(s);
174  g1_free(t);
175  g1_free(u);
176  }
177 
178  return result;
179 }
