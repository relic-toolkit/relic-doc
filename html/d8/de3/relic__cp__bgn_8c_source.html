42int cp_bgn_gen(bgn_t pub, bgn_t prv) {
43 bn_t n;
44 int result = RLC_OK;
45
46 bn_null(n);
47
48 RLC_TRY {
49 bn_new(n);
50
51 pc_get_ord(n);
52
53 bn_rand_mod(prv->x, n);
54 bn_rand_mod(prv->y, n);
55 bn_rand_mod(prv->z, n);
56
57 g1_mul_gen(pub->gx, prv->x);
58 g1_mul_gen(pub->gy, prv->y);
59 g1_mul_gen(pub->gz, prv->z);
60
61 g2_mul_gen(pub->hx, prv->x);
62 g2_mul_gen(pub->hy, prv->y);
63 g2_mul_gen(pub->hz, prv->z);
64 }
65 RLC_CATCH_ANY {
66 result = RLC_ERR;
67 }
68 RLC_FINALLY {
69 bn_free(n);
70 }
71
72 return result;
73}

75int cp_bgn_enc1(g1_t out[2], const dig_t in, const bgn_t pub) {
76 bn_t r, n;
77 g1_t t;
78 int result = RLC_OK;
79
80 bn_null(n);
81 bn_null(r);
82 g1_null(t);
83
84 RLC_TRY {
85 bn_new(n);
86 bn_new(r);
87 g1_new(t);
88
89 pc_get_ord(n);
90 bn_rand_mod(r, n);
91
92 /* Compute c0 = (ym + r)G. */
93 g1_mul_dig(out[0], pub->gy, in);
94
95 g1_mul_gen(t, r);
96 g1_add(out[0], out[0], t);
97 g1_norm(out[0], out[0]);
98
99 /* Compute c1 = (zm + xr)G. */
100 g1_mul_dig(out[1], pub->gz, in);
101 g1_mul(t, pub->gx, r);
102 g1_add(out[1], out[1], t);
103 g1_norm(out[1], out[1]);
104 }
105 RLC_CATCH_ANY {
106 result = RLC_ERR;
107 }
108 RLC_FINALLY {
109 bn_free(n);
110 bn_free(r);
111 g1_free(t);
112 }
113
114 return result;
115}

117int cp_bgn_dec1(dig_t *out, const g1_t in[2], const bgn_t prv) {
118 bn_t r, n;
119 g1_t s, t, u;
120 int i, result = RLC_ERR;
121
122 bn_null(n);
123 bn_null(r);
124 g1_null(s);
125 g1_null(t);
126 g1_null(u);
127
128 RLC_TRY {
129 bn_new(n);
130 bn_new(r);
131 g1_new(s);
132 g1_new(t);
133 g1_new(u);
134
135 pc_get_ord(n);
136 /* Compute T = x(ym + r)G - (zm + xr)G = m(xy - z)G. */
137 g1_mul(t, in[0], prv->x);
138 g1_sub(t, t, in[1]);
139 g1_norm(t, t);
140 /* Compute U = (xy - z)G and find m. */
141 bn_mul(r, prv->x, prv->y);
142 bn_sub(r, r, prv->z);
143 bn_mod(r, r, n);
144 g1_mul_gen(s, r);
145 g1_copy(u, s);
146
147 if (g1_is_infty(t) == 1){
148 *out = 0;
149 result = RLC_OK;
150 } else {
151 for (i = 0; i < INT_MAX; i++) {
152 if (g1_cmp(t, u) == RLC_EQ) {
153 *out = i + 1;
154 result = RLC_OK;
155 break;
156 }
157 g1_add(u, u, s);
158 g1_norm(u, u);
159 }
160 }
161 } RLC_CATCH_ANY {
162 result = RLC_ERR;
163 }
164 RLC_FINALLY {
165 bn_free(n);
166 bn_free(r);
167 g1_free(s);
168 g1_free(t);
169 g1_free(u);
170 }
171
172 return result;
173}

290int cp_bgn_dec(dig_t *out, const gt_t in[4], const bgn_t prv) {
291 int i, result = RLC_ERR;
292 g1_t g;
293 g2_t h;
294 gt_t t[4];
295 bn_t n, r, s;
296
297 bn_null(n);
298 bn_null(r);
299 bn_null(s);
300 g1_null(g);
301 g2_null(h);
302
303 RLC_TRY {
304 bn_new(n);
305 bn_new(r);
306 bn_new(s);
307 g1_new(g);
308 g2_new(h);
309 for (i = 0; i < 4; i++) {
310 gt_null(t[i]);
311 gt_new(t[i]);
312 }
313
314 gt_exp(t[0], in[0], prv->x);
315 gt_exp(t[0], t[0], prv->x);
316
317 gt_mul(t[1], in[1], in[2]);
318 gt_exp(t[1], t[1], prv->x);
319 gt_inv(t[1], t[1]);
320
321 gt_mul(t[3], in[3], t[1]);
322 gt_mul(t[3], t[3], t[0]);
323
324 pc_get_ord(n);
325 g1_get_gen(g);
326 g2_get_gen(h);
327
328 bn_mul(r, prv->x, prv->y);
329 bn_sqr(r, r);
330
331 bn_mul(s, prv->x, prv->y);
332 bn_mul(s, s, prv->z);
333 bn_sub(r, r, s);
334 bn_sub(r, r, s);
335
336 bn_sqr(s, prv->z);
337 bn_add(r, r, s);
338 bn_mod(r, r, n);
339 pc_map(t[1], g, h);
340 gt_exp(t[1], t[1], r);
341
342 gt_copy(t[2], t[1]);
343
344 if (gt_is_unity(t[3]) == 1) {
345 *out = 0;
346 result = RLC_OK;
347 } else {
348 for (i = 0; i < INT_MAX; i++) {
349 if (gt_cmp(t[2], t[3]) == RLC_EQ) {
350 *out = i + 1;
351 result = RLC_OK;
352 break;
353 }
354 gt_mul(t[2], t[2], t[1]);
355 }
356 }
357 } RLC_CATCH_ANY {
358 result = RLC_ERR;
359 } RLC_FINALLY {
360 bn_free(n);
361 bn_free(r);
362 bn_free(s);
363 g1_free(g);
364 g2_free(h);
365 for (i = 0; i < 4; i++) {
366 gt_free(t[i]);
367 }
368 }
369
370 return result;
371}
