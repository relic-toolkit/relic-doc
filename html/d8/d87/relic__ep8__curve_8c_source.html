134static void detect_opt(int *opt, fp8_t a) {
135 fp8_t t;
136 fp8_null(t);
137
138 RLC_TRY {
139 fp8_new(t);
140 fp8_set_dig(t, 3);
141 fp8_neg(t, t);
142
143 if (fp8_cmp(a, t) == RLC_EQ) {
144 *opt = RLC_MIN3;
145 } else if (fp8_is_zero(a)) {
146 *opt = RLC_ZERO;
147 } else if (fp8_cmp_dig(a, 1) == RLC_EQ) {
148 *opt = RLC_ONE;
149 } else if (fp8_cmp_dig(a, 2) == RLC_EQ) {
150 *opt = RLC_TWO;
151 } else if ((fp_bits(a[0][0][0]) <= RLC_DIG) && fp_is_zero(a[0][0][1]) &&
152 fp2_is_zero(a[0][1]) && fp4_is_zero(a[1])) {
153 *opt = RLC_TINY;
154 } else {
155 *opt = RLC_HUGE;
156 }
157 }
158 RLC_CATCH_ANY {
159 RLC_THROW(ERR_CAUGHT);
160 }
161 RLC_FINALLY {
162 fp8_free(t);
163 }
164}

224void ep8_curve_mul_a(fp8_t c, const fp8_t a) {
225 ctx_t *ctx = core_get();
226 switch (ctx->ep8_opt_a) {
227 case RLC_ZERO:
228 fp8_zero(c);
229 break;
230 case RLC_ONE:
231 fp8_copy(c, a);
232 break;
233 case RLC_TWO:
234 fp8_dbl(c, a);
235 break;
236#if FP_RDC != MONTY
237 case RLC_TINY:
238 fp8_mul_dig(c, a, ctx->ep8_a[0][0][0][0]);
239 break;
240#endif
241 default:
242 fp8_mul(c, a, ctx->ep8_a);
243 break;
244 }
245}

308void ep8_curve_set_twist(int type) {
309 char str[16 * RLC_FP_BYTES + 1];
310 ctx_t *ctx = core_get();
311 ep8_t g;
312 fp8_t a, b;
313 bn_t r, h;
314
315 ep8_null(g);
316 fp8_null(a);
317 fp8_null(b);
318 bn_null(r);
319 bn_null(h);
320
321 ctx->ep8_is_twist = 0;
322 if (type == RLC_EP_MTYPE || type == RLC_EP_DTYPE) {
323 ctx->ep8_is_twist = type;
324 } else {
325 return;
326 }
327
328 RLC_TRY {
329 ep8_new(g);
330 fp8_new(a);
331 fp8_new(b);
332 bn_new(r);
333 bn_new(h);
334
335 switch (ep_param_get()) {
336#if FP_PRIME == 575
337 case B48_P575:
338 fp8_zero(a);
339 ASSIGN(B48_P575);
340 break;
341#endif
342 default:
343 (void)str;
344 RLC_THROW(ERR_NO_VALID);
345 break;
346 }
347
348 fp8_zero(g->z);
349 fp8_set_dig(g->z, 1);
350 g->coord = BASIC;
351
352 ep8_copy(ctx->ep8_g, g);
353 fp8_copy(ctx->ep8_a, a);
354 fp8_copy(ctx->ep8_b, b);
355
356 detect_opt(&(ctx->ep8_opt_a), ctx->ep8_a);
357 detect_opt(&(ctx->ep8_opt_b), ctx->ep8_b);
358
359 bn_copy(&(ctx->ep8_r), r);
360 bn_copy(&(ctx->ep8_h), h);
361
362 if (type == RLC_EP_MTYPE) {
363 fp8_zero(a);
364 fp_copy(a[1][1][0], ctx->fp8_p1[0]);
365 fp_copy(a[1][1][1], ctx->fp8_p1[1]);
366 fp8_inv(a, a);
367 fp_copy(ctx->fp8_p1[0], a[1][0][0]);
368 fp_copy(ctx->fp8_p1[1], a[1][0][1]);
369 }
370
371#if defined(WITH_PC)
372 /* Compute pairing generator. */
373 pc_core_calc();
374#endif
375
376#if defined(EP_PRECO)
377 ep8_mul_pre((ep8_t *)ep8_curve_get_tab(), ctx->ep8_g);
378#endif
379 }
380 RLC_CATCH_ANY {
381 RLC_THROW(ERR_CAUGHT);
382 }
383 RLC_FINALLY {
384 ep8_free(g);
385 fp8_free(a);
386 fp8_free(b);
387 bn_free(r);
388 bn_free(h);
389 }
390}
