51 static void eb_mul_ltnaf_imp(eb_t r, const eb_t p, const bn_t k) {
52  int i, l, n;
53  int8_t tnaf[RLC_FB_BITS + 8], u;
54  eb_t t[1 << (EB_WIDTH - 2)];
55 
56  if (eb_curve_opt_a() == RLC_ZERO) {
57  u = -1;
58  } else {
59  u = 1;
60  }
61 
62  TRY {
63  /* Prepare the precomputation table. */
64  for (i = 0; i < (1 << (EB_WIDTH - 2)); i++) {
65  eb_null(t[i]);
66  eb_new(t[i]);
67  }
68  /* Compute the precomputation table. */
69  eb_tab(t, p, EB_WIDTH);
70 
71  /* Compute the w-TNAF representation of k. */
72  l = sizeof(tnaf);
73  bn_rec_tnaf(tnaf, &l, k, u, RLC_FB_BITS, EB_WIDTH);
74 
75  n = tnaf[l - 1];
76  if (n > 0) {
77  eb_copy(r, t[n / 2]);
78  } else {
79  eb_neg(r, t[-n / 2]);
80  }
81 
82  for (i = l - 2; i >= 0; i--) {
83  eb_frb(r, r);
84 
85  n = tnaf[i];
86  if (n > 0) {
87  eb_add(r, r, t[n / 2]);
88  }
89  if (n < 0) {
90  eb_sub(r, r, t[-n / 2]);
91  }
92  }
93  /* Convert r to affine coordinates. */
94  eb_norm(r, r);
95  if (bn_sign(k) == RLC_NEG) {
96  eb_neg(r, r);
97  }
98  }
99  CATCH_ANY {
100  THROW(ERR_CAUGHT);
101  }
102  FINALLY {
103  /* Free the precomputation table. */
104  for (i = 0; i < (1 << (EB_WIDTH - 2)); i++) {
105  eb_free(t[i]);
106  }
107  }
108 }

614 void eb_mul_basic(eb_t r, const eb_t p, const bn_t k) {
615  eb_t t;
616 
617  if (bn_is_zero(k) || eb_is_infty(p)) {
618  eb_set_infty(r);
619  return;
620  }
621 
622  eb_null(t);
623 
624  TRY {
625  eb_new(t);
626 
627  eb_copy(t, p);
628  for (int i = bn_bits(k) - 2; i >= 0; i--) {
629  eb_dbl(t, t);
630  if (bn_get_bit(k, i)) {
631  eb_add(t, t, p);
632  }
633  }
634 
635  eb_norm(r, t);
636  if (bn_sign(k) == RLC_NEG) {
637  eb_neg(r, r);
638  }
639  }
640  CATCH_ANY {
641  THROW(ERR_CAUGHT);
642  }
643  FINALLY {
644  eb_free(t);

706  break;
707  case RLC_TINY:
708  fb_add_dig(x2, x2, b[0]);
709  break;

711  fb_addn_low(x2, x2, b);
712  break;

719  fb_muln_low(r4, r1, r2);

721  dv_swap_cond(x1, x2, RLC_FB_DIGS, j ^ 1);

723  fb_sqr(z1, r3);

747  fb_sqrl_low(x2, r2);
748  fb_muln_low(r5, r1, b);
749  fb_addd_low(x2, x2, r5, 2 * RLC_FB_DIGS);
750  fb_rdcn_low(x2, x2);
751  break;
752  }
753  dv_swap_cond(x1, x2, RLC_FB_DIGS, j ^ 1);
754  dv_swap_cond(z1, z2, RLC_FB_DIGS, j ^ 1);
755  }

758  /* The point q is at infinity. */
759  eb_set_infty(r);
760  } else {

764  fb_set_dig(r->z, 1);
765  } else {

768  /* z1 = (x1 + x * z1). */
769  fb_mul(z1, z1, p->x);
770  fb_add(z1, z1, x1);
771  /* z2 = x * z2. */
772  fb_mul(z2, z2, p->x);
773  /* x1 = x1 * z2. */

775  /* z2 = (x2 + x * z2)(x1 + x * z1). */
776  fb_add(z2, z2, x2);
777  fb_mul(z2, z2, z1);

780  fb_sqr(r4, p->x);
781  fb_add(r4, r4, p->y);
782  fb_mul(r4, r4, r3);
783  fb_add(r4, r4, z2);

791  fb_mul(x2, x1, r3);

793  fb_add(z2, x2, p->x);
794

796  fb_mul(z2, z2, r4);
797  fb_add(z2, z2, p->y);
798 
799  fb_copy(r->x, x2);
800  fb_copy(r->y, z2);
801  fb_set_dig(r->z, 1);
802  }
803  }
804

807  eb_neg(r, r);
808  }
809  }
810  CATCH_ANY {
811  THROW(ERR_CAUGHT);
812  }
813  FINALLY {

815  bn_free(t);
816  dv_free(x1);
817  dv_free(z1);
818  dv_free(x2);
819  dv_free(z2);

821  dv_free(r2);
822  dv_free(r3);
823  dv_free(r4);
824  dv_free(r5);

882 void eb_mul_halve(eb_t r, const eb_t p, const bn_t k) {
883  int i, j, l, trc, cof;
884  int8_t naf[RLC_FB_BITS + 1] = { 0 }, *_k;
885  eb_t q, s, t[1 << (EB_WIDTH - 2)];
886  bn_t n, m;
887  fb_t u, v, w, z;
888 
889  if (bn_is_zero(k) || eb_is_infty(p)) {
890  eb_set_infty(r);
891  return;
892  }
893 
894  bn_null(m);
895  bn_null(n);
896  eb_null(q);
897  eb_null(s);
898  for (i = 0; i < (1 << (EB_WIDTH - 2)); i++) {
899  eb_null(t[i]);
900  }
901  fb_null(u);
902  fb_null(v);
903  fb_null(w);
904  fb_null(z);
905 
906  TRY {
907  bn_new(n);
908  bn_new(m);
909  eb_new(q);
910  eb_new(s);
911  fb_new(u);
912  fb_new(v);
913  fb_new(w);
914  fb_new(z);
915 
916  /* Prepare the precomputation table. */
917  for (i = 0; i < (1 << (EB_WIDTH - 2)); i++) {
918  eb_new(t[i]);
919  eb_set_infty(t[i]);
920  }
921 
922  /* Convert k to alternate representation k' = (2^{t-1}k mod n). */
923  eb_curve_get_ord(n);
924  bn_lsh(m, k, bn_bits(n) - 1);
925  bn_mod(m, m, n);
926 
927  /* Compute the w-NAF representation of k'. */
928  l = sizeof(naf);
929  bn_rec_naf(naf, &l, m, EB_WIDTH);
930 
931  for (i = l; i <= bn_bits(n); i++) {
932  naf[i] = 0;
933  }
934  if (naf[bn_bits(n)] == 1) {
935  eb_dbl(t[0], p);
936  }
937  l = bn_bits(n);
938  _k = naf + l - 1;
939 
940  eb_copy(q, p);
941  eb_curve_get_cof(n);
942 
943  /* Test if curve has a cofactor bigger than 2. */
944  if (bn_cmp_dig(n, 2) == RLC_GT) {
945  cof = 1;
946  } else {
947  cof = 0;
948  }
949 
950  trc = fb_trc(eb_curve_get_a());
951 
952  if (cof) {
953  /* Curves with cofactor > 2, u = sqrt(a), v = Solve(u). */
954  fb_srt(u, eb_curve_get_a());
955  fb_slv(v, u);
956 
957  bn_rand(n, RLC_POS, l);
958 
959  for (i = l - 1; i >= 0; i--, _k--) {
960  j = *_k;
961  if (j > 0) {
962  eb_norm(s, q);
963  eb_add(t[j / 2], t[j / 2], s);
964  }
965  if (j < 0) {
966  eb_norm(s, q);
967  eb_sub(t[-j / 2], t[-j / 2], s);
968  }
969 
970  /* T = 1/2(Q). */
971  eb_hlv(s, q);
972 
973  /* If Tr(x_T) != Tr(a). */
974  if (fb_trc(s->x) != 0) {
975  /* z = l_t, w = sqrt(l_Q), l_T = l_T + sqrt(l_Q) + v. */
976  fb_copy(z, s->y);
977  fb_srt(w, q->y);
978  fb_add(s->y, s->y, w);
979  fb_add(s->y, s->y, v);
980  /* z = (z + x_Q + v + sqrt(a)). */
981  fb_add(z, z, q->x);
982  fb_add(z, z, v);
983  fb_add(z, z, u);
984  /* w = sqrt(w + x_Q + l_Q + sqrt(a)). */
985  fb_add(w, w, q->x);
986  fb_add(w, w, q->y);
987  fb_add(w, w, u);
988  /* x_T = sqrt(w * z), . */
989  fb_mul(w, w, z);
990  fb_srt(s->x, w);
991  fb_set_dig(s->z, 1);
992  s->norm = 2;
993  }
994  eb_copy(q, s);
995  }
996  } else {
997  for (i = l - 1; i >= 0; i--, _k--) {
998  j = *_k;
999  if (j > 0) {
1000  eb_norm(q, q);
1001  eb_add(t[j / 2], t[j / 2], q);
1002  }
1003  if (j < 0) {
1004  eb_norm(q, q);
1005  eb_sub(t[-j / 2], t[-j / 2], q);
1006  }
1007  eb_hlv(q, q);
1008  }
1009  }
1010 
1011 #if EB_WIDTH == 2
1012  eb_norm(r, t[0]);
1013 #else
1014  /* Compute Q_i = Q_i + Q_{i+2} for i from 2^{w-1}-3 to 1. */
1015  for (i = (1 << (EB_WIDTH - 1)) - 3; i >= 1; i -= 2) {
1016  eb_add(t[i / 2], t[i / 2], t[(i + 2) / 2]);
1017  }
1018  /* Compute R = Q_1 + 2 * sum_{i != 1}Q_i. */
1019  eb_copy(r, t[1]);
1020  for (i = 2; i < (1 << (EB_WIDTH - 2)); i++) {
1021  eb_add(r, r, t[i]);
1022  }
1023  eb_dbl(r, r);
1024  eb_add(r, r, t[0]);
1025  eb_norm(r, r);
1026 #endif
1027 
1028  /* We may need to fix an error of a 2-torsion point if the curve has a
1029  * 4-cofactor. */
1030  if (cof) {
1031  eb_hlv(s, r);
1032  if (fb_trc(s->x) != trc) {
1033  fb_zero(s->x);
1034  fb_srt(s->y, eb_curve_get_b());
1035  fb_set_dig(s->z, 1);
1036  eb_add(r, r, s);
1037  eb_norm(r, r);
1038  }
1039  }
1040  }
1041  CATCH_ANY {
1042  THROW(ERR_CAUGHT);
1043  }
1044  FINALLY {
1045  /* Free the precomputation table. */
1046  for (i = 0; i < (1 << (EB_WIDTH - 2)); i++) {
1047  eb_free(t[i]);
1048  }
1049  bn_free(n);
1050  bn_free(m);
1051  eb_free(q);
1052  eb_free(s);
1053  fb_free(u);
1054  fb_free(v);
1055  fb_free(w);
1056  fb_free(z);
1057  }
1058 }
