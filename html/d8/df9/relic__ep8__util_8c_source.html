81void ep8_blind(ep8_t r, const ep8_t p) {
82 fp8_t rand;
83
84 fp8_null(rand);
85
86 RLC_TRY {
87 fp8_new(rand);
88 fp8_rand(rand);
89#if EP_ADD == BASIC
90 (void)rand;
91 ep8_copy(r, p);
92#elif EP_ADD == PROJC
93 fp8_mul(r->x, p->x, rand);
94 fp8_mul(r->y, p->y, rand);
95 fp8_mul(r->z, p->z, rand);
96 r->coord = PROJC;
97#elif EP_ADD == JACOB
98 fp8_mul(r->z, p->z, rand);
99 fp8_mul(r->y, p->y, rand);
100 fp8_sqr(rand, rand);
101 fp8_mul(r->x, r->x, rand);
102 fp8_mul(r->y, r->y, rand);
103 r->coord = JACOB;
104#endif
105 } RLC_CATCH_ANY {
106 RLC_THROW(ERR_CAUGHT);
107 } RLC_FINALLY {
108 fp8_free(rand);
109 }
110}

112void ep8_rhs(fp8_t rhs, const fp8_t x) {
113 fp8_t t0;
114
115 fp8_null(t0);
116
117 RLC_TRY {
118 fp8_new(t0);
119
120 fp8_sqr(t0, x);
121
122 switch (ep8_curve_opt_a()) {
123 case RLC_ZERO:
124 break;
125#if FP_RDC != MONTY
126 case RLC_MIN3:
127 fp_sub_dig(t0[0][0][0], t0[0][0][0], 3);
128 break;
129 case RLC_ONE:
130 fp_add_dig(t0[0][0][0], t0[0][0][0], 1);
131 break;
132 case RLC_TWO:
133 fp_add_dig(t0[0][0][0], t0[0][0][0], 2);
134 break;
135 case RLC_TINY:
136 fp_add_dig(t0[0][0][0], t0[0][0][0],
137 ep8_curve_get_a()[0][0][0][0]);
138 break;
139#endif
140 default:
141 fp8_add(t0, t0, ep8_curve_get_a());
142 break;
143 }
144
145 fp8_mul(t0, t0, x);
146
147 switch (ep8_curve_opt_b()) {
148 case RLC_ZERO:
149 break;
150#if FP_RDC != MONTY
151 case RLC_MIN3:
152 fp_sub_dig(t0[0][0][0], t0[0][0][0], 3);
153 break;
154 case RLC_ONE:
155 fp_add_dig(t0[0][0][0], t0[0][0][0], 1);
156 break;
157 case RLC_TWO:
158 fp_add_dig(t0[0][0][0], t0[0][0][0], 2);
159 break;
160 case RLC_TINY:
161 fp_add_dig(t0[0][0][0], t0[0][0][0],
162 ep8_curve_get_b()[0][0][0][0]);
163 break;
164#endif
165 default:
166 fp8_add(t0, t0, ep8_curve_get_b());
167 break;
168 }
169
170 fp8_copy(rhs, t0);
171 } RLC_CATCH_ANY {
172 RLC_THROW(ERR_CAUGHT);
173 } RLC_FINALLY {
174 fp8_free(t0);
175 }
176}

178int ep8_on_curve(const ep8_t p) {
179 ep8_t t;
180 int r = 0;
181
182 ep8_null(t);
183
184 RLC_TRY {
185 ep8_new(t);
186
187 ep8_norm(t, p);
188
189 ep8_rhs(t->x, t->x);
190 fp8_sqr(t->y, t->y);
191
192 r = (fp8_cmp(t->x, t->y) == RLC_EQ) || ep8_is_infty(p);
193 } RLC_CATCH_ANY {
194 RLC_THROW(ERR_CAUGHT);
195 } RLC_FINALLY {
196 ep8_free(t);
197 }
198 return r;
199}

201void ep8_tab(ep8_t *t, const ep8_t p, int w) {
202 if (w > 2) {
203 ep8_dbl(t[0], p);
204#if defined(EP_MIXED)
205 ep8_norm(t[0], t[0]);
206#endif
207 ep8_add(t[1], t[0], p);
208 for (int i = 2; i < (1 << (w - 2)); i++) {
209 ep8_add(t[i], t[i - 1], t[0]);
210 }
211#if defined(EP_MIXED)
212 ep8_norm_sim(t + 1, t + 1, (1 << (w - 2)) - 1);
213#endif
214 }
215#if defined(EP_MIXED)
216 ep8_norm(t[0], p);
217#else
218 ep8_copy(t[0], p);
219#endif
220}

254void ep8_read_bin(ep8_t a, const uint8_t *bin, size_t len) {
255 if (len == 1) {
256 if (bin[0] == 0) {
257 ep8_set_infty(a);
258 return;
259 } else {
260 RLC_THROW(ERR_NO_BUFFER);
261 return;
262 }
263 }
264
265 if (len != (16 * RLC_FP_BYTES + 1)) {
266 RLC_THROW(ERR_NO_BUFFER);
267 return;
268 }
269
270 a->coord = BASIC;
271 fp8_set_dig(a->z, 1);
272 fp8_read_bin(a->x, bin + 1, 8 * RLC_FP_BYTES);
273
274 if (len == 16 * RLC_FP_BYTES + 1) {
275 if (bin[0] == 4) {
276 fp8_read_bin(a->y, bin + 8 * RLC_FP_BYTES + 1, 8 * RLC_FP_BYTES);
277 } else {
278 RLC_THROW(ERR_NO_VALID);
279 return;
280 }
281 }
282
283 if (!ep8_on_curve(a)) {
284 RLC_THROW(ERR_NO_VALID);
285 }
286}

288void ep8_write_bin(uint8_t *bin, size_t len, const ep8_t a, int pack) {
289 ep8_t t;
290
291 ep8_null(t);
292
293 memset(bin, 0, len);
294
295 if (ep8_is_infty(a)) {
296 if (len < 1) {
297 RLC_THROW(ERR_NO_BUFFER);
298 return;
299 } else {
300 return;
301 }
302 }
303
304 RLC_TRY {
305 ep8_new(t);
306
307 ep8_norm(t, a);
308
309 if (len < 16 * RLC_FP_BYTES + 1) {
310 RLC_THROW(ERR_NO_BUFFER);
311 } else {
312 bin[0] = 4;
313 fp8_write_bin(bin + 1, 8 * RLC_FP_BYTES, t->x, 0);
314 fp8_write_bin(bin + 8 * RLC_FP_BYTES + 1, 8 * RLC_FP_BYTES, t->y, 0);
315 }
316 } RLC_CATCH_ANY {
317 RLC_THROW(ERR_CAUGHT);
318 } RLC_FINALLY {
319 ep8_free(t);
320 }
