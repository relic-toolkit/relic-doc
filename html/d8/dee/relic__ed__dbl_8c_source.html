42 void ed_dbl_basic(ed_t r, const ed_t p) {
43  fp_t t0, t1, t2;
44 
45  fp_null(t0);
46  fp_null(t1);
47  fp_null(t2);
48 
49  TRY {
50  fp_new(t0);
51  fp_new(t1);
52  fp_new(t2);
53 
54  /* x3 = (x1*y1+y1*x1)/(1+d*x1*x1*y1*y1)
55  * y3 = (y1*y1-a*x1*x1)/(1-d*x1*x1*y1*y1) */
56 
57  fp_mul(t0, p->x, p->y);
58  fp_sqr(t1, t0);
59 
60  fp_mul(t1, t1, core_get()->ed_d);
61  fp_add_dig(t2, t1, 1);
62  fp_inv(t2, t2);
63  fp_sub_dig(t1, t1, 1);
64  fp_neg(t1, t1);
65  fp_inv(t1, t1);
66 
67  fp_dbl(t0, t0);
68  fp_mul(t0, t0, t2);
69 
70  fp_sqr(t2, p->x);
71  fp_mul(t2, t2, core_get()->ed_a);
72  fp_sqr(r->y, p->y);
73  fp_sub(r->y, r->y, t2);
74  fp_mul(r->y, r->y, t1);
75 
76  fp_copy(r->x, t0);
77  fp_copy(r->z, p->z);
78 
79  r->norm = 1;
80  }
81  CATCH_ANY {
82  THROW(ERR_CAUGHT);
83  }
84  FINALLY {
85  fp_free(t0);
86  fp_free(t1);
87  fp_free(t2);
88  }
89 }

95 void ed_dbl_projc(ed_t r, const ed_t p) {

106  TRY {

108  fp_new(t1);

111  fp_new(t4);
112  fp_new(t5);
113  fp_new(t6);
114 
115  // 3M + 4S + 1D + 7add
116 
117  /* B = (x1 + y1)^2 */
118  fp_add(t0, p->x, p->y);
119  fp_sqr(t0, t0);
120 
121  /* C = x1^2 */
122  fp_sqr(t1, p->x);
123 
124  /* D = y1^2 */
125  fp_sqr(t2, p->y);
126 
127  /* E = a * C */
128  fp_mul(t3, core_get()->ed_a, t1);
129 
130  /* F = E + D, H = Z^2 */
131  fp_add(t4, t3, t2);
132  fp_sqr(t5, p->z);
133 
134  // J = F - 2H
135  fp_dbl(t6, t5);
136  fp_sub(t6, t4, t6);
137 
138  // x3 = (B - C - D) * J
139  fp_sub(r->x, t0, t1);
140  fp_sub(r->x, r->x, t2);
141  fp_mul(r->x, r->x, t6);
142 
143  // x3 = F * (E - D)
144  fp_sub(r->y, t3, t2);
145  fp_mul(r->y, t4, r->y);
146 
147  /* x3 = F * J */
148  fp_mul(r->z, t4, t6);
149 
150  r->norm = 0;
151  } CATCH_ANY {

153  } FINALLY {

185  // 4M + 4S + 1D + 7add
186 
187  /* A = X^2, B = Y^2 */
188  fp_sqr(t0, p->x);
189  fp_sqr(t1, p->y);
190

194

197

199  fp_add(t2, p->x, p->y);
200  fp_sqr(t2, t2);
201  fp_sub(t2, t2, t0);
202  fp_sub(t2, t2, t1);
203 
204  /* G = D + B, F = G - C, H = D - B */
205  fp_add(t4, r->t, t1);
206  fp_sub(t3, t4, r->z);
207  fp_sub(r->z, r->t, t1);
208 
209  /* X = E * F */
210  fp_mul(r->x, t2, t3);
211  /* Y = G * H */
212  fp_mul(r->y, t4, r->z);
213  if (r->norm != 2) {
214  /* T = E * H */
215  fp_mul(r->t, t2, r->z);
216  }
217  /* Z = F * G */
218  fp_mul(r->z, t3, t4);
219 
220  r->norm = 0;
221  } CATCH_ANY {
222  THROW(ERR_CAUGHT);
223  } FINALLY {
224  fp_free(t0);
225  fp_free(t1);
226  fp_free(t2);
227  fp_free(t3);
228  fp_free(t4);
229  }
230 }
