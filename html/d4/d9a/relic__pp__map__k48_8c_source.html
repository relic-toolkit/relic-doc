40 static void pp_mil_k48(fp48_t r, fp8_t qx, fp8_t qy, ep_t p, bn_t a) {
41  fp48_t l;
42  ep_t _p;
43  fp8_t rx, ry, rz;
44  int i, len = bn_bits(a) + 1;
45  int8_t s[RLC_FP_BITS + 1];
46 
47  fp48_null(l);
48  ep_null(_p);
49  fp8_null(rx);
50  fp8_null(ry);
51  fp8_null(rz);
52 
53  TRY {
54  fp48_new(l);
55  ep_new(_p);
56  fp8_new(rx);
57  fp8_new(ry);
58  fp8_new(rz);
59 
60  fp48_zero(l);
61  fp8_copy(rx, qx);
62  fp8_copy(ry, qy);
63  fp8_set_dig(rz, 1);
64 #if EP_ADD == BASIC
65  ep_neg(_p, p);
66 #else
67  fp_add(_p->x, p->x, p->x);
68  fp_add(_p->x, _p->x, p->x);
69  fp_neg(_p->y, p->y);
70 #endif
71 
72  bn_rec_naf(s, &len, a, 2);
73  for (i = len - 2; i >= 0; i--) {
74  fp48_sqr(r, r);
75  pp_dbl_k48(l, rx, ry, rz, _p);
76  fp48_mul_dxs(r, r, l);
77  if (s[i] > 0) {
78  pp_add_k48(l, rx, ry, rz, qx, qy, p);
79  fp48_mul_dxs(r, r, l);
80  }
81  if (s[i] < 0) {
82  fp8_neg(qy, qy);
83  pp_add_k48(l, rx, ry, rz, qx, qy, p);
84  fp8_neg(qy, qy);
85  fp48_mul_dxs(r, r, l);
86  }
87  }
88  }
89  CATCH_ANY {
90  THROW(ERR_CAUGHT);
91  }
92  FINALLY {
93  fp48_free(l);
94  ep_free(_p);
95  fp8_free(rx);
96  fp8_free(ry);
97  fp8_free(rz);
98  }
99 }

105 void pp_map_k48(fp48_t r, ep_t p, fp8_t qx, fp8_t qy) {
106  bn_t a;
107 
108  bn_null(a);
109 
110  TRY {
111  bn_new(a);
112 
113  fp_prime_get_par(a);
114  fp48_set_dig(r, 1);
115 
116  if (!ep_is_infty(p) && !(fp8_is_zero(qx) && fp8_is_zero(qy))) {
117  switch (ep_curve_is_pairf()) {
118  case EP_B48:
119  /* r = f_{|a|,Q}(P). */
120  pp_mil_k48(r, qx, qy, p, a);
121  if (bn_sign(a) == RLC_NEG) {
122  fp48_inv_cyc(r, r);
123  }
124  pp_exp_k48(r, r);
125  break;
126  }
127  }
128  }
129  CATCH_ANY {
130  THROW(ERR_CAUGHT);
131  }
132  FINALLY {
133  bn_free(a);
134  }
