44 void pp_add_k12_basic(fp12_t l, ep2_t r, ep2_t q, ep_t p) {
45  int one = 1, zero = 0;
46  fp2_t s;
47  ep2_t t;
48 
49  fp2_null(s);
50  ep2_null(t);
51 
52  TRY {
53  fp2_new(s);
54  ep2_new(t);
55 
56  ep2_copy(t, r);
57  ep2_add_slp_basic(r, s, r, q);
58 
59  if (ep2_curve_is_twist() == EP_MTYPE) {
60  one ^= 1;
61  zero ^= 1;
62  }
63 
64  fp_mul(l[one][zero][0], s[0], p->x);
65  fp_mul(l[one][zero][1], s[1], p->x);
66  fp2_mul(l[one][one], s, t->x);
67  fp2_sub(l[one][one], t->y, l[one][one]);
68  fp_neg(l[zero][zero][0], p->y);
69  }
70  CATCH_ANY {
71  THROW(ERR_CAUGHT);
72  }
73  FINALLY {
74  fp2_free(s);
75  ep2_free(t);
76  }
77 }

85 void pp_add_k12_projc_basic(fp12_t l, ep2_t r, ep2_t q, ep_t p) {
86  fp2_t t0, t1, t2, t3, t4;
87  int one = 1, zero = 0;
88 
89  fp2_null(t0);
90  fp2_null(t1);
91  fp2_null(t2);
92  fp2_null(t3);
93  fp2_null(t4);
94 
95  TRY {
96  fp2_new(t0);
97  fp2_new(t1);
98  fp2_new(t2);
99  fp2_new(t3);
100  fp2_new(t4);
101 
102  /* B = t0 = x1 - x2 * z1. */
103  fp2_mul(t0, r->z, q->x);
104  fp2_sub(t0, r->x, t0);
105  /* A = t1 = y1 - y2 * z1. */
106  fp2_mul(t1, r->z, q->y);
107  fp2_sub(t1, r->y, t1);
108 
109  /* D = B^2. */
110  fp2_sqr(t2, t0);
111  /* G = x1 * D. */
112  fp2_mul(r->x, r->x, t2);
113  /* E = B^3. */
114  fp2_mul(t2, t2, t0);
115  /* C = A^2. */
116  fp2_sqr(t3, t1);
117  /* F = E + z1 * C. */
118  fp2_mul(t3, t3, r->z);
119  fp2_add(t3, t2, t3);
120 
121  if (ep2_curve_is_twist() == EP_MTYPE) {
122  one ^= 1;
123  zero ^= 1;
124  }
125 
126  /* l10 = - (A * xp). */
127  fp_mul(l[one][zero][0], t1[0], p->x);
128  fp_mul(l[one][zero][1], t1[1], p->x);
129  fp2_neg(l[one][zero], l[one][zero]);
130 
131  /* t4 = B * x2. */
132  fp2_mul(t4, q->x, t1);
133 
134  /* H = E + F - 2 * G. */
135  fp2_sub(t3, t3, r->x);
136  fp2_sub(t3, t3, r->x);
137  /* y3 = A * (G - H) - y1 * E. */
138  fp2_sub(r->x, r->x, t3);
139  fp2_mul(t1, t1, r->x);
140  fp2_mul(r->y, t2, r->y);
141  fp2_sub(r->y, t1, r->y);
142  /* x3 = B * H. */
143  fp2_mul(r->x, t0, t3);
144  /* z3 = z1 * E. */
145  fp2_mul(r->z, r->z, t2);
146 
147  /* l11 = J = B * x2 - A * y2. */
148  fp2_mul(t2, q->y, t0);
149  fp2_sub(l[one][one], t4, t2);
150 
151  /* l00 = B * yp. */
152  fp_mul(l[zero][zero][0], t0[0], p->y);
153  fp_mul(l[zero][zero][1], t0[1], p->y);
154 
155  r->norm = 0;
156  }
157  CATCH_ANY {
158  THROW(ERR_CAUGHT);
159  }
160  FINALLY {
161  fp2_free(t0);
162  fp2_free(t1);
163  fp2_free(t2);
164  fp2_free(t3);
165  fp2_free(t4);
166  }
167 }
