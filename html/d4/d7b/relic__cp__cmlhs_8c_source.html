43 int cp_cmlhs_gen(bn_t x[], gt_t hs[], int len, uint8_t prf[], int plen,
44  bn_t sk, g2_t pk, bn_t d, g2_t y) {
45  g1_t g1;
46  g2_t g2;
47  gt_t gt;
48  bn_t n;
49  int result = RLC_OK;
50 
51  g1_null(g1);
52  g2_null(g2);
53  gt_null(gt);
54  bn_null(n);
55 
56  TRY {
57  bn_new(n);
58  g1_new(g1);
59  g2_new(g2);
60  gt_new(gt);
61 
62  g1_get_ord(n);
63  g1_get_gen(g1);
64  g2_get_gen(g2);
65  pc_map(gt, g1, g2);
66 
67  rand_bytes(prf, plen);
68  cp_bls_gen(sk, pk);
69 
70  g1_get_ord(n);
71  /* Generate elements for n tags. */
72  for (int i = 0; i < len; i++) {
73  bn_rand_mod(x[i], n);
74  gt_exp(hs[i], gt, x[i]);
75  }
76 
77  bn_rand_mod(d, n);
78  g2_mul_gen(y, d);
79  }
80  CATCH_ANY {
81  result = RLC_ERR;
82  }
83  FINALLY {
84  g1_free(g1);
85  g2_free(g2);
86  gt_free(gt);
87  bn_free(n);
88  }
89  return result;
90 }

92 int cp_cmlhs_sig(g1_t sig, g2_t z, g1_t a, g1_t c, g1_t r, g2_t s, bn_t msg,
93  char *data, int dlen, int label, bn_t x, g1_t h,
94  uint8_t prf[], int plen, bn_t d, bn_t sk) {
95  bn_t k, m, n;
96  g1_t t;
97  uint8_t mac[RLC_MD_LEN];
98  uint8_t *buf = RLC_ALLOCA(uint8_t, 1 + 4 * RLC_FP_BYTES + dlen);
99  int result = RLC_OK;
100 
101  bn_null(k);
102  bn_null(m);
103  bn_null(n);
104  g1_null(t);
105 
106  TRY {
107  bn_new(k);
108  bn_new(m);
109  bn_new(n);
110  g1_new(t);
111  if (buf == NULL) {
112  THROW(ERR_NO_MEMORY);
113  }
114 
115  g1_get_ord(n);
116  /* Generate r and s. */
117  bn_rand_mod(k, n);
118  bn_rand_mod(m, n);
119 
120  /* Compute S = -g2^s, C = g1^s. */
121  g2_mul_gen(s, m);
122  g2_neg(s, s);
123  g1_mul_gen(c, m);
124  /* Compute R = g1^(r - ys). */
125  bn_mul(m, d, m);
126  bn_mod(m, m, n);
127  bn_sub(m, k, m);
128  bn_mod(m, m, n);
129  g1_mul_gen(r, m);
130 
131  /* Compute A = g1^(x + r) * \prod H_j^(y * m_j). */
132  bn_add(k, x, k);
133  bn_mod(k, k, n);
134  g1_mul_gen(a, k);
135  bn_mul(k, d, msg);
136  bn_mod(k, k, n);
137  g1_mul(t, h, k);
138  g1_add(a, a, t);
139  g1_norm(a, a);
140  /* Compute z = F_K(delta), Z = g2^z, A = A^(1/z). */
141  md_hmac(mac, (const uint8_t *)data, dlen, prf, plen);
142  bn_read_bin(k, mac, sizeof(mac));
143  bn_mod(k, k, n);
144  g2_mul_gen(z, k);
145  bn_gcd_ext(m, k, NULL, k, n);
146  if (bn_sign(k) == RLC_NEG) {
147  bn_add(k, k, n);
148  }
149  g1_mul(a, a, k);
150 
151  /* Compute C = C * sum H_j^m_j. */
152  bn_mod(k, msg, n);
153  g1_mul(t, h, k);
154  g1_add(c, c, t);
155  g1_norm(c, c);
156 
157  g2_write_bin(buf, 1 + 4 * RLC_FP_BYTES, z, 0);
158  memcpy(buf + 4 * RLC_FP_BYTES + 1, data, dlen);
159  cp_bls_sig(sig, buf, 4 * RLC_FP_BYTES + 1 + dlen, sk);
160  }
161  CATCH_ANY {
162  result = RLC_ERR;
163  }
164  FINALLY {
165  bn_free(k);
166  bn_free(m);
167  bn_free(n);
168  g1_free(t);
169  RLC_FREE(buf);
170  }
171  return result;
172 }
