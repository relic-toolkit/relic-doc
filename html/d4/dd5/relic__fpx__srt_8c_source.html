39 int fp2_srt(fp2_t c, fp2_t a) {
40  int r = 0;
41  fp_t t1;
42  fp_t t2;
43  fp_t t3;
44 
45  fp_null(t1);
46  fp_null(t2);
47  fp_null(t3);
48 
49  TRY {
50  fp_new(t1);
51  fp_new(t2);
52  fp_new(t3);
53 
54  /* t1 = a[0]^2 - u^2 * a[1]^2 */
55  fp_sqr(t1, a[0]);
56  fp_sqr(t2, a[1]);
57  for (int i = -1; i > fp_prime_get_qnr(); i--) {
58  fp_add(t1, t1, t2);
59  }
60  fp_add(t1, t1, t2);
61 
62  if (fp_srt(t2, t1)) {
63  /* t1 = (a_0 + sqrt(t1)) / 2 */
64  fp_add(t1, a[0], t2);
65  fp_hlv(t1, t1);
66 
67  if (!fp_srt(t3, t1)) {
68  /* t1 = (a_0 - sqrt(t1)) / 2 */
69  fp_sub(t1, a[0], t2);
70  fp_hlv(t1, t1);
71  fp_srt(t3, t1);
72  }
73  /* c_0 = sqrt(t1) */
74  fp_copy(c[0], t3);
75  /* c_1 = a_1 / (2 * sqrt(t1)) */
76  fp_dbl(t3, t3);
77  fp_inv(t3, t3);
78  fp_mul(c[1], a[1], t3);
79  r = 1;
80  }
81  }
82  CATCH_ANY {
83  THROW(ERR_CAUGHT);
84  }
85  FINALLY {
86  fp_free(t1);
87  fp_free(t2);
88  fp_free(t3);
89  }
90  return r;
91 }
