39int fp2_is_sqr(const fp2_t a) {
40 fp2_t t;
41 int r = 0;
42
43 fp2_null(t);
44
45 /* QR testing in extension fields from "Square root computation over
46 * even extension fields", by Gora Adj and Francisco Rodríguez-Henríquez.
47 * https://eprint.iacr.org/2012/685 */
48
49 RLC_TRY {
50 fp2_new(t);
51
52 fp2_frb(t, a, 1);
53 fp2_mul(t, t, a);
54 r = fp_is_sqr(t[0]);
55 } RLC_CATCH_ANY {
56 RLC_THROW(ERR_CAUGHT);
57 } RLC_FINALLY {
58 fp2_free(t);
59 }
60
61 return r;
62}

64int fp2_srt(fp2_t c, const fp2_t a) {
65 int r = 0;
66 bn_t e;
67 fp2_t t, u;
68
69 bn_null(e);
70 fp2_null(t);
71 fp2_null(u);
72
73 if (fp2_is_zero(a)) {
74 fp2_zero(c);
75 return 1;
76 }
77
78 RLC_TRY {
79 bn_new(e);
80 fp2_new(t);
81 fp2_new(u);
82
83 if (fp_prime_get_mod8() % 4 == 3) {
84 /* "From Optimized One-Dimensional SQIsign Verification on Intel and
85 * Cortex-M4" by Aardal et al.: https://eprint.iacr.org/2024/1563 */
86 fp_sqr(t[0], a[0]);
87 fp_sqr(t[1], a[1]);
88 fp_add(t[0], t[0], t[1]);
89
90 e->used = RLC_FP_DIGS;
91 dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
92 bn_add_dig(e, e, 1);
93 bn_rsh(e, e, 2);
94
95 fp_exp(t[0], t[0], e);
96 fp_copy_sec(t[0], a[0], fp_is_zero(a[1]));
97 fp_add(t[0], t[0], a[0]);
98 fp_dbl(u[0], t[0]);
99
100 bn_sub_dig(e, e, 1);
101 fp_exp(t[1], u[0], e);
102 fp_mul(t[0], t[0], t[1]);
103 fp_mul(t[1], t[1], a[1]);
104 fp_dbl(u[1], t[0]);
105 fp_sqr(u[1], u[1]);
106 fp_sub(u[0], u[0], u[1]);
107 int f = fp_is_zero(u[0]);
108 fp_neg(u[1], t[0]);
109 fp_copy(u[0], t[1]);
110 fp2_copy_sec(u, t, f);
111 fp2_sqr(t, u);
112 r = (fp2_cmp(a, t) == RLC_EQ);
113 fp2_copy(c, u);
114 } else {
115 if (fp_is_zero(a[1])) {
116 /* special case: either a[0] is square and sqrt is purely 'real'
117 * or a[0] is non-square and sqrt is purely 'imaginary' */
118 r = 1;
119 if (fp_is_sqr(a[0])) {
120 fp_srt(t[0], a[0]);
121 fp_copy(c[0], t[0]);
122 fp_zero(c[1]);
123 } else {
124 /* Compute a[0]/i^2. */
125 #ifdef FP_QNRES
126 fp_copy(t[0], a[0]);
127 #else
128 if (fp_prime_get_qnr() == -2) {
129 fp_hlv(t[0], a[0]);
130 } else {
131 fp_set_dig(t[0], -fp_prime_get_qnr());
132 fp_inv(t[0], t[0]);
133 fp_mul(t[0], t[0], a[0]);
134 }
135 #endif
136 fp_neg(t[0], t[0]);
137 fp_zero(c[0]);
138 if (!fp_srt(c[1], t[0])) {
139 /* should never happen! */
140 RLC_THROW(ERR_NO_VALID);
141 }
142 }
143 } else {
144 /* t[0] = a[0]^2 - i^2 * a[1]^2 */
145 fp_sqr(t[0], a[0]);
146 fp_sqr(t[1], a[1]);
147 for (int i = -1; i > fp_prime_get_qnr(); i--) {
148 fp_add(t[0], t[0], t[1]);
149 }
150 fp_add(t[0], t[0], t[1]);
151
152 if (fp_is_sqr(t[0])) {
153 fp_srt(t[1], t[0]);
154 /* t[0] = (a_0 + sqrt(t[0])) / 2 */
155 fp_add(t[0], a[0], t[1]);
156 fp_hlv(t[0], t[0]);
157 /* t[1] = (a_0 - sqrt(t[0])) / 2 */
158 fp_sub(t[1], a[0], t[1]);
159 fp_hlv(t[1], t[1]);
160 fp_copy_sec(t[0], t[1], !fp_is_sqr(t[0]));
161
162 /* Should always be a quadratic residue. */
163 fp_srt(t[1], t[0]);
164 /* c_0 = sqrt(t1) */
165 fp_copy(c[0], t[1]);
166 /* c_1 = a_1 / (2 * sqrt(t[0])) */
167 fp_dbl(t[1], t[1]);
168 fp_inv(t[1], t[1]);
169 fp_mul(c[1], a[1], t[1]);
170 r = 1;
171 }
172 }
173 }
174 }
175 RLC_CATCH_ANY {
176 RLC_THROW(ERR_CAUGHT);
177 }
178 RLC_FINALLY {
179 bn_free(e);
180 fp2_free(t);
181 fp2_free(u);
182 }
183 return r;
184}

212int fp3_srt(fp3_t c, const fp3_t a) {
213 int f = 0, r = 0;
214 fp_t root;
215 fp3_t t0, t1, t2, t3;
216 bn_t d, e;
217
218 fp_null(root);
219 fp3_null(t0);
220 fp3_null(t1);
221 fp3_null(t2);
222 fp3_null(t3);
223 bn_null(d);
224 bn_null(e);
225
226 if (fp3_is_zero(a)) {
227 fp3_zero(c);
228 return 1;
229 }
230
231 RLC_TRY {
232 fp_new(root);
233 fp3_new(t0);
234 fp3_new(t1);
235 fp3_new(t2);
236 fp3_new(t3);
237 bn_new(d);
238 bn_new(e);
239
240 e->used = RLC_FP_DIGS;
241 dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
242
243 switch (fp_prime_get_mod8()) {
244 case 1:
245 /* Implement constant-time version of Tonelli-Shanks algorithm
246 * as per https://eprint.iacr.org/2020/1497.pdf */
247
248 /* Compute progenitor as x^(p^3-1-2^f)/2^(f+1) for 2^f|(p-1).
249 * Let q = (p-1)/2^f. We will write the exponent in p and q.
250 * Write (p^3-1-2^f)/2^(f+1) as (q*(p^2+p))/2 + (q - 1)/2 */
251 bn_sqr(d, e);
252 bn_add(d, d, e);
253 bn_rsh(d, d, 1);
254 /* Compute (q - 1)/2 = (p-1)/2^(f+1).*/
255 f = fp_prime_get_2ad();
256 bn_sub_dig(e, e, 1);
257 bn_rsh(e, e, f + 1);
258 fp3_exp(t1, a, e);
259 /* Now compute the power (q*(p^2+p))/2. */
260 fp3_sqr(t0, t1);
261 fp3_mul(t0, t0, a);
262 fp3_exp(t0, t0, d);
263 fp3_mul(t0, t0, t1);
264
265 /* Generate root of unity, and continue algorithm. */
266 dv_copy(root, fp_prime_get_srt(), RLC_FP_DIGS);
267
268 fp3_sqr(t1, t0);
269 fp3_mul(t1, t1, a);
270 fp3_mul(t3, t0, a);
271 fp3_copy(t2, t1);
272 for (int j = f; j > 1; j--) {
273 for (int i = 1; i < j - 1; i++) {
274 fp3_sqr(t2, t2);
275 }
276 fp_mul(t0[0], t3[0], root);
277 fp_mul(t0[1], t3[1], root);
278 fp_mul(t0[2], t3[2], root);
279 fp3_copy_sec(t3, t0, fp3_cmp_dig(t2, 1) != RLC_EQ);
280 fp_sqr(root, root);
281 fp_mul(t0[0], t1[0], root);
282 fp_mul(t0[1], t1[1], root);
283 fp_mul(t0[2], t1[2], root);
284 fp3_copy_sec(t1, t0, fp3_cmp_dig(t2, 1) != RLC_EQ);
285 fp3_copy(t2, t1);
286 }
287 break;
288 case 5:
289 fp3_dbl(t3, a);
290 fp3_frb(t0, t3, 1);
291
292 fp3_sqr(t1, t0);
293 fp3_mul(t2, t1, t0);
294 fp3_mul(t1, t1, t2);
295
296 fp3_frb(t0, t0, 1);
297 fp3_mul(t3, t3, t1);
298 fp3_mul(t0, t0, t3);
299
300 bn_div_dig(e, e, 8);
301 fp3_exp(t0, t0, e);
302
303 fp3_mul(t0, t0, t2);
304 fp3_sqr(t1, t0);
305 fp3_mul(t1, t1, a);
306 fp3_dbl(t1, t1);
307
308 fp3_mul(t0, t0, a);
309 fp_sub_dig(t1[0], t1[0], 1);
310 fp3_mul(t3, t0, t1);
311 break;
312 case 3:
313 case 7:
314 fp3_frb(t0, a, 1);
315 fp3_sqr(t1, t0);
316 fp3_mul(t2, t1, t0);
317 fp3_frb(t0, t0, 1);
318 fp3_mul(t3, t2, a);
319 fp3_mul(t0, t0, t3);
320
321 bn_div_dig(e, e, 4);
322 fp3_exp(t0, t0, e);
323
324 fp3_mul(t0, t0, a);
325 fp3_mul(t3, t0, t1);
326 break;
327 default:
328 fp3_zero(c);
329 break;
330 }
331 /* Assume it is a square and test at the end. */
332 /* We cannot use QR test because it depends on Frobenius constants. */
333 fp3_sqr(t0, t3);
334 r = (fp3_cmp(t0, a) == RLC_EQ ? 1 : 0);
335 fp3_zero(c);
336 fp3_copy_sec(c, t3, r);
337 } RLC_CATCH_ANY {
338 RLC_THROW(ERR_CAUGHT);
339 } RLC_FINALLY {
340 fp_free(root);
341 fp3_free(t0);
342 fp3_free(t1);
343 fp3_free(t2);
344 fp3_free(t3);
345 bn_free(d);
346 bn_free(e);
347 }
348
349 return r;
350}

380int fp4_srt(fp4_t c, const fp4_t a) {
381 int c0, r = 0;
382 fp2_t t0, t1, t2;
383
384 fp2_null(t0);
385 fp2_null(t1);
386 fp2_null(t2);
387
388 if (fp4_is_zero(a)) {
389 fp4_zero(c);
390 return 1;
391 }
392
393 RLC_TRY {
394 fp2_new(t0);
395 fp2_new(t1);
396 fp2_new(t2);
397
398 if (fp2_is_zero(a[1])) {
399 /* special case: either a[0] is square and sqrt is purely 'real'
400 * or a[0] is non-square and sqrt is purely 'imaginary' */
401 r = 1;
402 if (fp2_is_sqr(a[0])) {
403 fp2_srt(c[0], a[0]);
404 fp2_zero(c[1]);
405 } else {
406 /* Compute a[0]/s^2. */
407 fp2_set_dig(t0, 1);
408 fp2_mul_nor(t0, t0);
409 fp2_inv(t0, t0);
410 fp2_mul(t0, a[0], t0);
411 fp2_zero(c[0]);
412 if (!fp2_srt(c[1], t0)) {
413 /* should never happen! */
414 RLC_THROW(ERR_NO_VALID);
415 }
416 }
417 } else {
418 /* t0 = a[0]^2 - s^2 * a[1]^2 */
419 fp2_sqr(t0, a[0]);
420 fp2_sqr(t1, a[1]);
421 fp2_mul_nor(t2, t1);
422 fp2_sub(t0, t0, t2);
423
424 if (fp2_is_sqr(t0)) {
425 fp2_srt(t1, t0);
426 /* t0 = (a_0 + sqrt(t0)) / 2 */
427 fp2_add(t0, a[0], t1);
428 fp_hlv(t0[0], t0[0]);
429 fp_hlv(t0[1], t0[1]);
430 c0 = fp2_is_sqr(t0);
431 /* t0 = (a_0 - sqrt(t0)) / 2 */
432 fp2_sub(t1, a[0], t1);
433 fp_hlv(t1[0], t1[0]);
434 fp_hlv(t1[1], t1[1]);
435 fp2_copy_sec(t0, t1, !c0);
436 /* Should always be a quadratic residue. */
437 fp2_srt(t2, t0);
438 /* c_0 = sqrt(t0) */
439 fp2_copy(c[0], t2);
440
441 /* c_1 = a_1 / (2 * sqrt(t0)) */
442 fp2_dbl(t2, t2);
443 fp2_inv(t2, t2);
444 fp2_mul(c[1], a[1], t2);
445 r = 1;
446 }
447 }
448 } RLC_CATCH_ANY {
449 RLC_THROW(ERR_CAUGHT);
450 } RLC_FINALLY {
451 fp2_free(t0);
452 fp2_free(t1);
453 fp2_free(t2);
454 }
455 return r;
456}

610 fp8_new(t0);

631 }
632 }
633 } else {

635 fp8_sqr(t0, a[0]);

637 fp8_mul_art(t2, t1);

641 fp8_srt(t1, t0);
642 /* t0 = (a_0 + sqrt(t0)) / 2 */
643 fp8_add(t0, a[0], t1);
644 fp_hlv(t0[0][0][0], t0[0][0][0]);
645 fp_hlv(t0[0][0][1], t0[0][0][1]);
646 fp_hlv(t0[0][1][0], t0[0][1][0]);
647 fp_hlv(t0[0][1][1], t0[0][1][1]);
648 fp_hlv(t0[1][0][0], t0[1][0][0]);
649 fp_hlv(t0[1][0][1], t0[1][0][1]);
650 fp_hlv(t0[1][1][0], t0[1][1][0]);

652 c0 = fp8_is_sqr(t0);
653 /* t0 = (a_0 - sqrt(t0)) / 2 */

655 for (int i = 0; i < 2; i++) {
656 for (int j = 0; j < 2; j++) {
657 for (int k = 0; k < 2; k++) {
658 fp_hlv(t1[i][j][k], t1[i][j][k]);
659 fp_copy_sec(t0[i][j][k], t1[i][j][k], !c0);
