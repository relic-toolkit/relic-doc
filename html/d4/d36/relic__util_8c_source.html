85uint32_t util_conv_endian(uint32_t i) {
86 uint32_t i1, i2, i3, i4;
87 i1 = i & 0xFF;
88 i2 = (i >> 8) & 0xFF;
89 i3 = (i >> 16) & 0xFF;
90 i4 = (i >> 24) & 0xFF;
91
92 return ((uint32_t) i1 << 24) | ((uint32_t) i2 << 16) | ((uint32_t) i3 << 8)
93 | i4;
94}

112char util_conv_char(dig_t i) {
113#if WSIZE == 8 || WSIZE == 16
114 /* Avoid tables to save up some memory. This is not performance-critical. */
115 if (i < 10) {
116 return i + '0';
117 }
118 if (i < 36) {
119 return (i - 10) + 'A';
120 }
121 if (i < 62) {
122 return (i - 36) + 'a';
123 }
124 if (i == 62) {
125 return '+';
126 } else {
127 return '/';
128 }
129#else
130 /* Use a table. */
131 static const char conv_table[] =
132 "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz+/";
133 return conv_table[i];
134#endif
135}

141int util_cmp_sec(const void *a, const void *b, size_t size) {
142 const uint8_t *_a = (const uint8_t *)a;
143 const uint8_t *_b = (const uint8_t *)b;
144 uint8_t result = 0;
145 int i;
146
147 for (i = 0; i < size; i++) {
148 result |= _a[i] ^ _b[i];
149 }
150
151 return (result == 0 ? RLC_EQ : RLC_NE);
152}

154void util_perm(uint_t p[], size_t n) {
155 size_t i, j, k;
156
157 for (i = 0; i < n; i++) {
158 rand_bytes((uint8_t *)&k, sizeof(size_t));
159 j = k % (i+1);
160 p[i] = p[j];
161 p[j] = i;
162 }
163}
