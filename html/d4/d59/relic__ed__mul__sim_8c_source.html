54static void ed_mul_sim_plain(ed_t r, const ed_t p, const bn_t k, const ed_t q,
55 const bn_t m, const ed_t *t) {
56 int i, n0, n1, w, gen = (t == NULL ? 0 : 1);
57 int8_t naf0[RLC_FP_BITS + 1], naf1[RLC_FP_BITS + 1], *_k, *_m;
58 ed_t t0[1 << (RLC_WIDTH - 2)];
59 ed_t t1[1 << (RLC_WIDTH - 2)];
60 size_t l, l0, l1;
61
62 RLC_TRY {
63 if (!gen) {
64 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
65 ed_null(t0[i]);
66 ed_new(t0[i]);
67 }
68 ed_tab(t0, p, RLC_WIDTH);
69 t = (const ed_t *)t0;
70 }
71
72 /* Prepare the precomputation table. */
73 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
74 ed_null(t1[i]);
75 ed_new(t1[i]);
76 }
77 /* Compute the precomputation table. */
78 ed_tab(t1, q, RLC_WIDTH);
79
80 /* Compute the w-TNAF representation of k. */
81 if (gen) {
82 w = RLC_DEPTH;
83 } else {
84 w = RLC_WIDTH;
85 }
86 l0 = l1 = RLC_FP_BITS + 1;
87 bn_rec_naf(naf0, &l0, k, w);
88 bn_rec_naf(naf1, &l1, m, RLC_WIDTH);
89
90 l = RLC_MAX(l0, l1);
91 if (bn_sign(k) == RLC_NEG) {
92 for (i = 0; i < l0; i++) {
93 naf0[i] = -naf0[i];
94 }
95 }
96 if (bn_sign(m) == RLC_NEG) {
97 for (i = 0; i < l1; i++) {
98 naf1[i] = -naf1[i];
99 }
100 }
101
102 _k = naf0 + l - 1;
103 _m = naf1 + l - 1;
104 ed_set_infty(r);
105 for (i = l - 1; i >= 0; i--, _k--, _m--) {
106 ed_dbl(r, r);
107
108 n0 = *_k;
109 n1 = *_m;
110 if (n0 > 0) {
111 ed_add(r, r, t[n0 / 2]);
112 }
113 if (n0 < 0) {
114 ed_sub(r, r, t[-n0 / 2]);
115 }
116 if (n1 > 0) {
117 ed_add(r, r, t1[n1 / 2]);
118 }
119 if (n1 < 0) {
120 ed_sub(r, r, t1[-n1 / 2]);
121 }
122 }
123 /* Convert r to affine coordinates. */
124 ed_norm(r, r);
125 }
126 RLC_CATCH_ANY {
127 RLC_THROW(ERR_CAUGHT);
128 }
129 RLC_FINALLY {
130 /* Free the precomputation tables. */
131 if (!gen) {
132 for (i = 0; i < 1 << (RLC_WIDTH - 2); i++) {
133 ed_free(t0[i]);
134 }
135 }
136 for (i = 0; i < 1 << (RLC_WIDTH - 2); i++) {
137 ed_free(t1[i]);
138 }
139 }
140}

175void ed_mul_sim_trick(ed_t r, const ed_t p, const bn_t k, const ed_t q,
176 const bn_t m) {
177 ed_t t0[1 << (RLC_WIDTH / 2)], t1[1 << (RLC_WIDTH / 2)], t[1 << RLC_WIDTH];
178 bn_t n;
179 size_t l0, l1, w = RLC_WIDTH / 2;
180 uint8_t w0[RLC_FP_BITS + 1], w1[RLC_FP_BITS + 1];
181
182 bn_null(n);
183
184 if (bn_is_zero(k) || ed_is_infty(p)) {
185 ed_mul(r, q, m);
186 return;
187 }
188 if (bn_is_zero(m) || ed_is_infty(q)) {
189 ed_mul(r, p, k);
190 return;
191 }
192
193 RLC_TRY {
194 bn_new(n);
195
196 ed_curve_get_ord(n);
197
198 for (int i = 0; i < (1 << w); i++) {
199 ed_null(t0[i]);
200 ed_null(t1[i]);
201 ed_new(t0[i]);
202 ed_new(t1[i]);
203 }
204 for (int i = 0; i < (1 << RLC_WIDTH); i++) {
205 ed_null(t[i]);
206 ed_new(t[i]);
207 }
208
209 ed_set_infty(t0[0]);
210 ed_copy(t0[1], p);
211 if (bn_sign(k) == RLC_NEG) {
212 ed_neg(t0[1], t0[1]);
213 }
214 for (int i = 2; i < (1 << w); i++) {
215 ed_add(t0[i], t0[i - 1], t0[1]);
216 }
217
218 ed_set_infty(t1[0]);
219 ed_copy(t1[1], q);
220 if (bn_sign(m) == RLC_NEG) {
221 ed_neg(t1[1], t1[1]);
222 }
223 for (int i = 1; i < (1 << w); i++) {
224 ed_add(t1[i], t1[i - 1], t1[1]);
225 }
226
227 for (int i = 0; i < (1 << w); i++) {
228 for (int j = 0; j < (1 << w); j++) {
229 ed_add(t[(i << w) + j], t0[i], t1[j]);
230 }
231 }
232
233#if defined(ED_MIXED)
234 ed_norm_sim(t + 2, (const ed_t *)t + 2, (1 << ((w + w))) - 2);
235#endif
236
237 l0 = l1 = RLC_CEIL(RLC_FP_BITS, w);
238 bn_rec_win(w0, &l0, k, w);
239 bn_rec_win(w1, &l1, m, w);
240
241 ed_set_infty(r);
242 for (int i = RLC_MAX(l0, l1) - 1; i >= 0; i--) {
243 for (int j = 0; j < w; j++) {
244 ed_dbl(r, r);
245 }
246 ed_add(r, r, t[(w0[i] << w) + w1[i]]);
247 }
248 ed_norm(r, r);
249 } RLC_CATCH_ANY {
250 RLC_THROW(ERR_CAUGHT);
251 }
252 RLC_FINALLY {
253 bn_free(n);
254 for (int i = 0; i < (1 << w); i++) {
255 ed_free(t0[i]);
256 ed_free(t1[i]);
257 }
258 for (int i = 0; i < (1 << RLC_WIDTH); i++) {
259 ed_free(t[i]);
260 }
261 }
262}

286void ed_mul_sim_joint(ed_t r, const ed_t p, const bn_t k, const ed_t q,
287 const bn_t m) {
288 ed_t t[5];
289 int i, u_i, offset;
290 int8_t jsf[2 * (RLC_FP_BITS + 1)];
291 size_t l;
292
293 if (bn_is_zero(k) || ed_is_infty(p)) {
294 ed_mul(r, q, m);
295 return;
296 }
297 if (bn_is_zero(m) || ed_is_infty(q)) {
298 ed_mul(r, p, k);
299 return;
300 }
301
302 RLC_TRY {
303 for (i = 0; i < 5; i++) {
304 ed_null(t[i]);
305 ed_new(t[i]);
306 }
307
308 ed_set_infty(t[0]);
309 ed_copy(t[1], q);
310 if (bn_sign(m) == RLC_NEG) {
311 ed_neg(t[1], t[1]);
312 }
313 ed_copy(t[2], p);
314 if (bn_sign(k) == RLC_NEG) {
315 ed_neg(t[2], t[2]);
316 }
317 ed_add(t[3], t[2], t[1]);
318 ed_sub(t[4], t[2], t[1]);
319#if defined(ED_MIXED)
320 ed_norm_sim(t + 3, (const ed_t *)t + 3, 2);
321#endif
322
323 l = 2 * (RLC_FP_BITS + 1);
324 bn_rec_jsf(jsf, &l, k, m);
325
326 ed_set_infty(r);
327
328 offset = RLC_MAX(bn_bits(k), bn_bits(m)) + 1;
329 for (i = l - 1; i >= 0; i--) {
330 ed_dbl(r, r);
331 if (jsf[i] != 0 && jsf[i] == -jsf[i + offset]) {
332 u_i = jsf[i] * 2 + jsf[i + offset];
333 if (u_i < 0) {
334 ed_sub(r, r, t[4]);
335 } else {
336 ed_add(r, r, t[4]);
337 }
338 } else {
339 u_i = jsf[i] * 2 + jsf[i + offset];
340 if (u_i < 0) {
341 ed_sub(r, r, t[-u_i]);
342 } else {
343 ed_add(r, r, t[u_i]);
344 }
345 }
346 }
347 ed_norm(r, r);
348 }
349 RLC_CATCH_ANY {
350 RLC_THROW(ERR_CAUGHT);
351 }
352 RLC_FINALLY {
353 for (i = 0; i < 5; i++) {
354 ed_free(t[i]);
355 }
356 }
357}

394void ed_mul_sim_lot(ed_t r, const ed_t p[], const bn_t k[], int n) {
395 int i, j;
396 int8_t *naf = NULL;
397 ed_t *_p = RLC_ALLOCA(ed_t, n);
398 size_t l, *_l = RLC_ALLOCA(size_t, n);
399
400 RLC_TRY {
401 l = 0;
402 for (i = 0; i < n; i++) {
403 l = RLC_MAX(l, bn_bits(k[i]) + 1);
404 }
405 naf = RLC_ALLOCA(int8_t, n * l);
406 if (naf == NULL || _p == NULL || _l == NULL) {
407 RLC_THROW(ERR_NO_MEMORY);
408 }
409
410 for (i = 0; i < n; i++) {
411 ed_null(_p[i]);
412 ed_new(_p[i]);
413 }
414
415 for (i = 0; i < n; i++) {
416 _l[i] = l;
417 ed_norm(_p[i], p[i]);
418 bn_rec_naf(&naf[i*l], &_l[i], k[i], 2);
419 if (bn_sign(k[i]) == RLC_NEG) {
420 ed_neg(_p[i], _p[i]);
421 }
422 }
423
424 ed_set_infty(r);
425 for (i = l - 1; i >= 0; i--) {
426 ed_dbl(r, r);
427 for (j = 0; j < n; j++) {
428 if (naf[j*l + i] > 0) {
429 ed_add(r, r, _p[j]);
430 }
431 if (naf[j*l + i] < 0) {
432 ed_sub(r, r, _p[j]);
433 }
434 }
435 }
436
437 /* Convert r to affine coordinates. */
438 ed_norm(r, r);
439 } RLC_CATCH_ANY {
440 RLC_THROW(ERR_CAUGHT);
441 } RLC_FINALLY {
442 for (i = 0; i < n; i++) {
443 ed_free(_p[i]);
444 }
445 RLC_FREE(_l);
446 RLC_FREE(_p);
447 RLC_FREE(naf);
448 }
