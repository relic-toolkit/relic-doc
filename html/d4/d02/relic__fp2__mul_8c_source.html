42void fp2_mul_basic(fp2_t c, const fp2_t a, const fp2_t b) {
43 dv_t t0, t1, t2, t3, t4;
44
45 dv_null(t0);
46 dv_null(t1);
47 dv_null(t2);
48 dv_null(t3);
49 dv_null(t4);
50
51 RLC_TRY {
52 dv_new(t0);
53 dv_new(t1);
54 dv_new(t2);
55 dv_new(t3);
56 dv_new(t4);
57
58 /* Karatsuba algorithm. */
59
60 /* t2 = a_0 + a_1, t1 = b_0 + b_1. */
61 fp_add(t2, a[0], a[1]);
62 fp_add(t1, b[0], b[1]);
63
64 /* t3 = (a_0 + a_1) * (b_0 + b_1). */
65 fp_muln_low(t3, t2, t1);
66
67 /* t0 = a_0 * b_0, t4 = a_1 * b_1. */
68 fp_muln_low(t0, a[0], b[0]);
69 fp_muln_low(t4, a[1], b[1]);
70
71 /* t2 = (a_0 * b_0) + (a_1 * b_1). */
72 fp_addc_low(t2, t0, t4);
73
74 /* t1 = (a_0 * b_0) + i^2 * (a_1 * b_1). */
75 fp_subc_low(t1, t0, t4);
76 for (int i = -1; i > fp_prime_get_qnr(); i--) {
77 fp_subc_low(t1, t1, t4);
78 }
79 for (int i = 1; i < fp_prime_get_qnr(); i++) {
80 fp_addc_low(t1, t1, t4);
81 }
82
83 /* c_0 = t1 mod p. */
84 fp_rdc(c[0], t1);
85
86 /* t4 = t3 - t2. */
87 fp_subc_low(t4, t3, t2);
88
89 /* c_1 = t4 mod p. */
90 fp_rdc(c[1], t4);
91 }
92 RLC_CATCH_ANY {
93 RLC_THROW(ERR_CAUGHT);
94 }
95 RLC_FINALLY {
96 dv_free(t0);
97 dv_free(t1);
98 dv_free(t2);
99 dv_free(t3);
100 dv_free(t4);
101 }
102}

104void fp2_mul_nor_basic(fp2_t c, const fp2_t a) {
105 fp2_t t;
106 bn_t b;
107
108 fp2_null(t);
109 bn_null(b);
110
111 RLC_TRY {
112 fp2_new(t);
113 bn_new(b);
114
115#ifdef FP_QNRES
116 /* If p = 3 mod 8, (1 + i) is a QNR/CNR. */
117 fp_neg(t[0], a[1]);
118 fp_add(c[1], a[0], a[1]);
119 fp_add(c[0], t[0], a[0]);
120#else
121 int qnr = fp2_field_get_qnr();
122
123 switch (fp_prime_get_mod8()) {
124 case 1:
125 case 5:
126 /* If p = 1,5 mod 8, (i) is a QNR/CNR. */
127 fp2_mul_art(c, a);
128 break;
129 case 3:
130 if (qnr == 1) {
131 /* If p = 3 mod 8, (1 + i) is a QNR/CNR. */
132 fp_neg(t[0], a[1]);
133 fp_add(c[1], a[0], a[1]);
134 fp_add(c[0], t[0], a[0]);
135 break;
136 }
137 /* Otherwise, fall back to next one. */
138 case 7:
139 /* If p = 7 mod 8, we choose (2^k + i) as a QNR/CNR. */
140 fp2_mul_art(t, a);
141 fp2_copy(c, a);
142 while (qnr > 1) {
143 fp2_dbl(c, c);
144 qnr = qnr >> 1;
145 }
146 fp2_add(c, c, t);
147 break;
148 default:
149 RLC_THROW(ERR_NO_VALID);
150 break;
151 }
152#endif
153 }
154 RLC_CATCH_ANY {
155 RLC_THROW(ERR_CAUGHT);
156 }
157 RLC_FINALLY {
158 fp2_free(t);
159 bn_free(b);
160 }
161}

177void fp2_mul_art(fp2_t c, const fp2_t a) {
178 fp_t t;
179
180 fp_null(t);
181
182 RLC_TRY {
183 fp_new(t);
184
185#ifdef FP_QNRES
186 /* (a_0 + a_1 * i) * i = -a_1 + a_0 * i. */
187 fp_copy(t, a[0]);
188 fp_neg(c[0], a[1]);
189 fp_copy(c[1], t);
190#else
191 /* (a_0 + a_1 * i) * i = (a_1 * i^2) + a_0 * i. */
192 fp_copy(t, a[0]);
193 fp_neg(c[0], a[1]);
194 for (int i = -1; i > fp_prime_get_qnr(); i--) {
195 fp_sub(c[0], c[0], a[1]);
196 }
197 for (int i = 0; i <= fp_prime_get_qnr(); i++) {
198 fp_add(c[0], c[0], a[1]);
199 }
200 fp_copy(c[1], t);
201#endif
202 }
203 RLC_CATCH_ANY {
204 RLC_THROW(ERR_CAUGHT);
205 }
206 RLC_FINALLY {
207 fp_free(t);
208 }
209}

211void fp2_mul_frb(fp2_t c, const fp2_t a, int i, int j) {
212 ctx_t *ctx = core_get();
213
214 fp2_copy(c, a);
215#if ALLOC == AUTO
216 switch(i) {
217 case 1:
218 fp2_mul(c, c, ctx->fp2_p1[j - 1]);
219 break;
220 case 2:
221 fp2_mul(c, c, ctx->fp2_p2[j - 1]);
222 break;
223 }
224#else
225 fp2_t t;
226
227 fp2_null(t);
228
229 RLC_TRY {
230 fp2_new(t);
231
232 switch(i) {
233 case 1:
234 fp_copy(t[0], ctx->fp2_p1[j - 1][0]);
235 fp_copy(t[1], ctx->fp2_p1[j - 1][1]);
236 break;
237 case 2:
238 fp_copy(t[0], ctx->fp2_p2[j - 1][0]);
239 fp_copy(t[1], ctx->fp2_p2[j - 1][1]);
240 break;
241 }
242
243 fp2_mul(c, c, t);
244 }
245 RLC_CATCH_ANY {
246 RLC_THROW(ERR_CAUGHT);
247 }
248 RLC_FINALLY {
249 fp2_free(t);
250 }
251#endif
