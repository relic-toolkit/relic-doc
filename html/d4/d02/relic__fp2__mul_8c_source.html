42 void fp2_mul_basic(fp2_t c, fp2_t a, fp2_t b) {
43  dv_t t0, t1, t2, t3, t4;
44 
45  dv_null(t0);
46  dv_null(t1);
47  dv_null(t2);
48  dv_null(t3);
49  dv_null(t4);
50 
51  TRY {
52  dv_new(t0);
53  dv_new(t1);
54  dv_new(t2);
55  dv_new(t3);
56  dv_new(t4);
57 
58  /* Karatsuba algorithm. */
59 
60  /* t2 = a_0 + a_1, t1 = b_0 + b_1. */
61  fp_add(t2, a[0], a[1]);
62  fp_add(t1, b[0], b[1]);
63 
64  /* t3 = (a_0 + a_1) * (b_0 + b_1). */
65  fp_muln_low(t3, t2, t1);
66 
67  /* t0 = a_0 * b_0, t4 = a_1 * b_1. */
68  fp_muln_low(t0, a[0], b[0]);
69  fp_muln_low(t4, a[1], b[1]);
70 
71  /* t2 = (a_0 * b_0) + (a_1 * b_1). */
72  fp_addc_low(t2, t0, t4);
73 
74  /* t1 = (a_0 * b_0) + i^2 * (a_1 * b_1). */
75  fp_subc_low(t1, t0, t4);
76 
77  /* t1 = u^2 * (a_1 * b_1). */
78  for (int i = -1; i > fp_prime_get_qnr(); i--) {
79  fp_subc_low(t1, t1, t4);
80  }
81  for (int i = 1; i < fp_prime_get_qnr(); i++) {
82  fp_addc_low(t1, t1, t4);
83  }
84 
85  /* c_0 = t1 mod p. */
86  fp_rdc(c[0], t1);
87 
88  /* t4 = t3 - t2. */
89  fp_subc_low(t4, t3, t2);
90 
91  /* c_1 = t4 mod p. */
92  fp_rdc(c[1], t4);
93  }
94  CATCH_ANY {
95  THROW(ERR_CAUGHT);
96  }
97  FINALLY {
98  dv_free(t0);
99  dv_free(t1);
100  dv_free(t2);
101  dv_free(t3);
102  dv_free(t4);
103  }
104 }

106 void fp2_mul_nor_basic(fp2_t c, fp2_t a) {
107  fp2_t t;
108  bn_t b;
109 
110  fp2_null(t);
111  bn_null(b);
112 
113  TRY {
114  fp2_new(t);
115  bn_new(b);
116 
117 #ifdef FP_QNRES
118  /* If p = 3 mod 8, (1 + i) is a QNR/CNR. */
119  fp_neg(t[0], a[1]);
120  fp_add(c[1], a[0], a[1]);
121  fp_add(c[0], t[0], a[0]);
122 #else
123  int qnr = fp2_field_get_qnr();
124 
125  switch (fp_prime_get_mod8()) {
126  case 3:
127  /* If p = 3 mod 8, (1 + i) is a QNR/CNR. */
128  fp_neg(t[0], a[1]);
129  fp_add(c[1], a[0], a[1]);
130  fp_add(c[0], t[0], a[0]);
131  break;
132  case 1:
133  case 5:
134  /* If p = 1,5 mod 8, (i) is a QNR/CNR. */
135  fp2_mul_art(c, a);
136  break;
137  case 7:
138  /* If p = 7 mod 8, we choose (2^k + i) as a QNR/CNR. */
139  fp2_mul_art(t, a);
140  fp2_copy(c, a);
141  while (qnr > 1) {
142  fp2_dbl(c, c);
143  qnr = qnr >> 1;
144  }
145  fp2_add(c, c, t);
146  break;
147  default:
148  THROW(ERR_NO_VALID);
149  break;
150  }
151 #endif
152  }
153  CATCH_ANY {
154  THROW(ERR_CAUGHT);
155  }
156  FINALLY {
157  fp2_free(t);
158  bn_free(b);
159  }
160 }

176 void fp2_mul_art(fp2_t c, fp2_t a) {
177  fp_t t;
178 
179  fp_null(t);
180 
181  TRY {
182  fp_new(t);
183 
184 #ifdef FP_QNRES
185  /* (a_0 + a_1 * i) * i = -a_1 + a_0 * i. */
186  fp_copy(t, a[0]);
187  fp_neg(c[0], a[1]);
188  fp_copy(c[1], t);
189 #else
190  /* (a_0 + a_1 * i) * i = (a_1 * i^2) + a_0 * i. */
191  fp_copy(t, a[0]);
192  fp_neg(c[0], a[1]);
193  for (int i = -1; i > fp_prime_get_qnr(); i--) {
194  fp_sub(c[0], c[0], a[1]);
195  }
196  for (int i = 0; i <= fp_prime_get_qnr(); i++) {
197  fp_add(c[0], c[0], a[1]);
198  }
199  fp_copy(c[1], t);
200 #endif
201  }
202  CATCH_ANY {
203  THROW(ERR_CAUGHT);
204  }
205  FINALLY {
206  fp_free(t);
207  }
208 }
