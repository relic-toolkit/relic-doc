60void fp_add_dig(fp_t c, const fp_t a, dig_t b) {
61#if FP_RDC == MONTY
62 if (b == 1) {
63 fp_add(c, a, core_get()->one.dp);
64 } else {
65 fp_t t;
66
67 fp_null(t);
68
69 RLC_TRY {
70 fp_new(t);
71
72 fp_set_dig(t, b);
73 fp_add(c, a, t);
74 } RLC_CATCH_ANY {
75 RLC_THROW(ERR_CAUGHT);
76 } RLC_FINALLY {
77 fp_free(t);
78 }
79 }
80#else
81 dig_t carry;
82
83 carry = fp_add1_low(c, a, b);
84 if (carry || dv_cmp(c, fp_prime_get(), RLC_FP_DIGS) != RLC_LT) {
85 carry = fp_subn_low(c, c, fp_prime_get());
86 }
87#endif
88}
