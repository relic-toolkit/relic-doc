64#define _make_md_xmd(HName, HBlockSize, HHashSize, HContext, HReset, HInput, HResult) \
65 void HName(uint8_t *buf, int buf_len, const uint8_t *in, int in_len, const uint8_t *dst, int dst_len) { \
66 const unsigned ell = (buf_len + HHashSize - 1) / HHashSize; \
67 if (buf_len < 0 || ell > 255 || dst_len > 255) { \
68 RLC_THROW(ERR_NO_VALID); \
69 return; \
70 } \
71 \
72 /* info needed for hashing: zero padding and some lengths */ \
73 const uint8_t Z_pad[HBlockSize] = { \
74 0, \
75 }; \
76 const uint8_t l_i_b_0_str[] = {buf_len >> 8, buf_len & 0xff, 0, dst_len}; \
77 const uint8_t *dstlen_str = l_i_b_0_str + 3; \
78 \
79 /* now compute b_0 */ \
80 uint8_t b_0[HHashSize]; \
81 HContext ctx; \
82 _check_md(HReset(&ctx)); \
83 _check_md(HInput(&ctx, Z_pad, HBlockSize)); /* Z_pad */ \
84 _check_md(HInput(&ctx, in, in_len)); /* msg */ \
85 _check_md(HInput(&ctx, l_i_b_0_str, 3)); /* l_i_b_str || I2OSP(0, 1) */ \
86 _check_md(HInput(&ctx, dst, dst_len)); /* DST */ \
87 _check_md(HInput(&ctx, dstlen_str, 1)); /* I2OSP(len(dst), 1) */ \
88 _check_md(HResult(&ctx, b_0)); /* finalize computation */ \
89 \
90 /* now compute b_i */ \
91 uint8_t b_i[HHashSize + 1] = { \
92 0, \
93 }; \
94 for (unsigned i = 1; i <= ell; ++i) { \
95 /* compute b_0 XOR b_(i-1) */ \
96 for (unsigned j = 0; j < HHashSize; ++j) { \
97 b_i[j] = b_0[j] ^ b_i[j]; \
98 } \
99 b_i[HHashSize] = i; \
100 \
101 _check_md(HReset(&ctx)); \
102 _check_md(HInput(&ctx, b_i, HHashSize + 1)); /* b_0 ^ b_(i-1) || I2OSP(i, 1) */ \
103 _check_md(HInput(&ctx, dst, dst_len)); /* DST */ \
104 _check_md(HInput(&ctx, dstlen_str, 1)); /* I2OSP(len(dst), 1) */ \
105 _check_md(HResult(&ctx, b_i)); /* finalize computation */ \
106 \
107 /* copy into output buffer */ \
108 const int rem_after = buf_len - i * HHashSize; \
109 const int copy_len = HHashSize + (rem_after < 0 ? rem_after : 0); \
110 memcpy(buf + (i - 1) * HHashSize, b_i, copy_len); \
111 } \
112 }
