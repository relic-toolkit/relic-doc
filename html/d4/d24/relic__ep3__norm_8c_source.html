48static void ep3_norm_imp(ep3_t r, const ep3_t p, int inv) {
49 if (p->coord != BASIC) {
50 fp3_t t;
51
52 fp3_null(t);
53
54 RLC_TRY {
55 fp3_new(t);
56
57 if (inv) {
58 fp3_copy(r->z, p->z);
59 } else {
60 fp3_inv(r->z, p->z);
61 }
62
63 switch (p->coord) {
64 case PROJC:
65 fp3_mul(r->x, p->x, r->z);
66 fp3_mul(r->y, p->y, r->z);
67 break;
68 case JACOB:
69 fp3_sqr(t, r->z);
70 fp3_mul(r->x, p->x, t);
71 fp3_mul(t, t, r->z);
72 fp3_mul(r->y, p->y, t);
73 break;
74 default:
75 ep3_copy(r, p);
76 break;
77 }
78 fp3_set_dig(r->z, 1);
79 }
80 RLC_CATCH_ANY {
81 RLC_THROW(ERR_CAUGHT);
82 }
83 RLC_FINALLY {
84 fp3_free(t);
85 }
86 }
87
88 r->coord = BASIC;
89}

91#endif /* EP_ADD == PROJC */

97void ep3_norm(ep3_t r, const ep3_t p) {
98 if (ep3_is_infty(p)) {
99 ep3_set_infty(r);
100 return;
101 }
102
103 if (p->coord == BASIC) {
104 /* If the point is represented in affine coordinates, just copy it. */
105 ep3_copy(r, p);
106 return;
107 }
108#if EP_ADD == PROJC || EP_ADD == JACOB || !defined(STRIP)
109 ep3_norm_imp(r, p, 0);
110#endif /* EP_ADD == PROJC */
111}

113void ep3_norm_sim(ep3_t *r, const ep3_t *t, int n) {
114 int i;
115 fp3_t* a = RLC_ALLOCA(fp3_t, n);
116
117 RLC_TRY {
118 if (a == NULL) {
119 RLC_THROW(ERR_NO_MEMORY);
120 }
121 for (i = 0; i < n; i++) {
122 fp3_null(a[i]);
123 fp3_new(a[i]);
124 fp3_copy(a[i], t[i]->z);
125 }
126
127 fp3_inv_sim(a, (const fp3_t *)a, n);
128
129 for (i = 0; i < n; i++) {
130 fp3_copy(r[i]->x, t[i]->x);
131 fp3_copy(r[i]->y, t[i]->y);
132 if (!ep3_is_infty(t[i])) {
133 fp3_copy(r[i]->z, a[i]);
134 }
135 }
136#if EP_ADD == PROJC || EP_ADD == JACOB || !defined(STRIP)
137 for (i = 0; i < n; i++) {
138 ep3_norm_imp(r[i], r[i], 1);
139 }
140#endif /* EP_ADD == PROJC */
141 }
142 RLC_CATCH_ANY {
143 RLC_THROW(ERR_CAUGHT);
144 }
145 RLC_FINALLY {
146 for (i = 0; i < n; i++) {
147 fp3_free(a[i]);
148 }
149 RLC_FREE(a);
150 }
151}
