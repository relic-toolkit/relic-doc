3/*
4 * Description:
5 * This file implements the Secure Hash Signature Standard
6 * algorithms as defined in the National Institute of Standards
7 * and Technology Federal Information Processing Standards
8 * Publication (FIPS PUB) 180-1 published on April 17, 1995, 180-2
9 * published on August 1, 2002, and the FIPS PUB 180-2 Change
10 * Notice published on February 28, 2004.
11 *
12 * A combined document showing all algorithms is available at
13 * http://csrc.nist.gov/publications/fips/
14 * fips180-2/fips180-2withchangenotice.pdf
15 *
16 * The SHA-224 and SHA-256 algorithms produce 224-bit and 256-bit
17 * message digests for a given data stream. It should take about
18 * 2**n steps to find a message with the same digest as a given
19 * message and 2**(n/2) to find any two messages with the same
20 * digest, when n is the digest size in bits. Therefore, this
21 * algorithm can serve as a means of providing a
22 * "fingerprint" for a message.
23 *
24 * Portability Issues:
25 * SHA-224 and SHA-256 are defined in terms of 32-bit "words".
26 * This code uses <stdint.h> (included via "sha.h") to define 32
27 * and 8 bit unsigned integer types. If your C compiler does not
28 * support 32 bit unsigned integers, this code is not
29 * appropriate.
30 *
31 * Caveats:
32 * SHA-224 and SHA-256 are designed to work with messages less
33 * than 2^64 bits long. This implementation uses SHA224/256Input()
34 * to hash the bits that are a multiple of the size of an 8-bit
35 * character, and then uses SHA224/256FinalBits() to hash the
36 * final few bits of the input.
37 */

221{

223 return shaSuccess;
224
225 if (!context || !message_array)

230 return shaStateError;

252/*
253 * SHA256FinalBits
254 *
255 * Description:
256 * This function will add in any final bits of the message.
257 *
258 * Parameters:
259 * context: [in/out]
260 * The SHA context to update
261 * message_bits: [in]
262 * The final bits of the message, in the upper portion of the
263 * byte. (Use 0b###00000 instead of 0b00000### to input the
264 * three bits ###.)
265 * length: [in]
266 * The number of bits in message_bits, between 1 and 7.
267 *
268 * Returns:
269 * sha Error Code.
270 */
271int SHA256FinalBits(SHA256Context *context,
272 const uint8_t message_bits, unsigned int length)
273{
274 uint8_t masks[8] = {
275 /* 0 0b00000000 */ 0x00, /* 1 0b10000000 */ 0x80,
276 /* 2 0b11000000 */ 0xC0, /* 3 0b11100000 */ 0xE0,
277 /* 4 0b11110000 */ 0xF0, /* 5 0b11111000 */ 0xF8,
278 /* 6 0b11111100 */ 0xFC, /* 7 0b11111110 */ 0xFE
279 };
280
281 uint8_t markbit[8] = {
282 /* 0 0b10000000 */ 0x80, /* 1 0b01000000 */ 0x40,
283 /* 2 0b00100000 */ 0x20, /* 3 0b00010000 */ 0x10,
284 /* 4 0b00001000 */ 0x08, /* 5 0b00000100 */ 0x04,
285 /* 6 0b00000010 */ 0x02, /* 7 0b00000001 */ 0x01
286 };
287
288 if (!length)
289 return shaSuccess;
290
291 if (!context)
292 return shaNull;
293
294 if ((context->Computed) || (length >= 8) || (length == 0)) {
295 context->Corrupted = shaStateError;
296 return shaStateError;
297 }
298
299 if (context->Corrupted)
300 return context->Corrupted;
301
302 uint32_t addTemp;
303 SHA224_256AddLength(context, length);
304 SHA224_256Finalize(context, (uint8_t)
305 ((message_bits & masks[length]) | markbit[length]));
306
307 return shaSuccess;
308}

335 *
336 * Description:
337 * This helper function finishes off the digest calculations.

339 * Parameters:

364/*
365 * SHA224_256PadMessage
366 *
367 * Description:
368 * According to the standard, the message must be padded to an
369 * even 512 bits. The first padding bit must be a '1'. The
370 * last 64 bits represent the length of the original message.
371 * All bits in between should be 0. This helper function will pad
372 * the message according to those rules by filling the
373 * Message_Block array accordingly. When it returns, it can be
374 * assumed that the message digest has been computed.
375 *
376 * Parameters:
377 * context: [in/out]
378 * The context to pad
379 * Pad_Byte: [in]
380 * The last byte to add to the digest before the 0-padding
381 * and length. This will contain the last bits of the message
382 * followed by another single bit. If the message was an
383 * exact multiple of 8-bits long, Pad_Byte will be 0x80.
384 *
385 * Returns:
386 * Nothing.
387 */
388static void SHA224_256PadMessage(SHA256Context *context,
389 uint8_t Pad_Byte)
390{
391 /*
392 * Check to see if the current message block is too small to hold
393 * the initial padding bits and length. If so, we will pad the
394 * block, process it, and then continue padding into a second
395 * block.
396 */
397 if (context->Message_Block_Index >= (SHA256_Message_Block_Size-8)) {
398 context->Message_Block[context->Message_Block_Index++] = Pad_Byte;
399 while (context->Message_Block_Index < SHA256_Message_Block_Size)
400 context->Message_Block[context->Message_Block_Index++] = 0;
401 SHA224_256ProcessMessageBlock(context);
402 } else
403 context->Message_Block[context->Message_Block_Index++] = Pad_Byte;
404
405 while (context->Message_Block_Index < (SHA256_Message_Block_Size-8))
406 context->Message_Block[context->Message_Block_Index++] = 0;
407
408 /*
409 * Store the message length as the last 8 octets
410 */
411 context->Message_Block[56] = (uint8_t)(context->Length_High >> 24);
412 context->Message_Block[57] = (uint8_t)(context->Length_High >> 16);
413 context->Message_Block[58] = (uint8_t)(context->Length_High >> 8);
414 context->Message_Block[59] = (uint8_t)(context->Length_High);
415 context->Message_Block[60] = (uint8_t)(context->Length_Low >> 24);
416 context->Message_Block[61] = (uint8_t)(context->Length_Low >> 16);
417 context->Message_Block[62] = (uint8_t)(context->Length_Low >> 8);
418 context->Message_Block[63] = (uint8_t)(context->Length_Low);
419
420 SHA224_256ProcessMessageBlock(context);
421}

423/*
424 * SHA224_256ProcessMessageBlock
425 *
426 * Description:
427 * This function will process the next 512 bits of the message
428 * stored in the Message_Block array.
429 *
430 * Parameters:
431 * context: [in/out]
432 * The SHA context to update
433 *
434 * Returns:

437 * Comments:
438 * Many of the variable names in this code, especially the
439 * single character names, were used because those were the
440 * names used in the publication.
441 */
442static void SHA224_256ProcessMessageBlock(SHA256Context *context)
443{
444 /* Constants defined in FIPS-180-2, section 4.2.2 */
445 static const uint32_t K[64] = {
446 0x428a2f98, 0x71374491, 0xb5c0fbcf, 0xe9b5dba5, 0x3956c25b,
447 0x59f111f1, 0x923f82a4, 0xab1c5ed5, 0xd807aa98, 0x12835b01,
448 0x243185be, 0x550c7dc3, 0x72be5d74, 0x80deb1fe, 0x9bdc06a7,
449 0xc19bf174, 0xe49b69c1, 0xefbe4786, 0x0fc19dc6, 0x240ca1cc,
450 0x2de92c6f, 0x4a7484aa, 0x5cb0a9dc, 0x76f988da, 0x983e5152,
451 0xa831c66d, 0xb00327c8, 0xbf597fc7, 0xc6e00bf3, 0xd5a79147,
452 0x06ca6351, 0x14292967, 0x27b70a85, 0x2e1b2138, 0x4d2c6dfc,
453 0x53380d13, 0x650a7354, 0x766a0abb, 0x81c2c92e, 0x92722c85,
454 0xa2bfe8a1, 0xa81a664b, 0xc24b8b70, 0xc76c51a3, 0xd192e819,
455 0xd6990624, 0xf40e3585, 0x106aa070, 0x19a4c116, 0x1e376c08,
456 0x2748774c, 0x34b0bcb5, 0x391c0cb3, 0x4ed8aa4a, 0x5b9cca4f,
457 0x682e6ff3, 0x748f82ee, 0x78a5636f, 0x84c87814, 0x8cc70208,
458 0x90befffa, 0xa4506ceb, 0xbef9a3f7, 0xc67178f2
459 };
460 int t, t4; /* Loop counter */
461 uint32_t temp1, temp2; /* Temporary word value */
462 uint32_t W[64]; /* Word sequence */
463 uint32_t A, B, C, D, E, F, G, H; /* Word buffers */
464
465 /*
466 * Initialize the first 16 words in the array W
467 */
468 for (t = t4 = 0; t < 16; t++, t4 += 4)
469 W[t] = (((uint32_t)context->Message_Block[t4]) << 24) |
470 (((uint32_t)context->Message_Block[t4 + 1]) << 16) |
471 (((uint32_t)context->Message_Block[t4 + 2]) << 8) |
472 (((uint32_t)context->Message_Block[t4 + 3]));
473
474 for (t = 16; t < 64; t++)
475 W[t] = SHA256_sigma1(W[t-2]) + W[t-7] +
476 SHA256_sigma0(W[t-15]) + W[t-16];
477
478 A = context->Intermediate_Hash[0];
479 B = context->Intermediate_Hash[1];
480 C = context->Intermediate_Hash[2];
481 D = context->Intermediate_Hash[3];
482 E = context->Intermediate_Hash[4];
483 F = context->Intermediate_Hash[5];
484 G = context->Intermediate_Hash[6];
485 H = context->Intermediate_Hash[7];
486
487 for (t = 0; t < 64; t++) {
488 temp1 = H + SHA256_SIGMA1(E) + SHA_Ch(E,F,G) + K[t] + W[t];
489 temp2 = SHA256_SIGMA0(A) + SHA_Maj(A,B,C);
490 H = G;
491 G = F;
492 F = E;
493 E = D + temp1;
494 D = C;
495 C = B;
496 B = A;
497 A = temp1 + temp2;
498 }
499
500 context->Intermediate_Hash[0] += A;
501 context->Intermediate_Hash[1] += B;
502 context->Intermediate_Hash[2] += C;
503 context->Intermediate_Hash[3] += D;
504 context->Intermediate_Hash[4] += E;
505 context->Intermediate_Hash[5] += F;
506 context->Intermediate_Hash[6] += G;
507 context->Intermediate_Hash[7] += H;
508
509 context->Message_Block_Index = 0;
510}

555 * Description:

574{

576
577 if (!context || !Message_Digest)
