61 int cp_ecss_sig(bn_t e, bn_t s, uint8_t *msg, int len, bn_t d) {

63  ec_t p;
64  uint8_t hash[RLC_MD_LEN];
65  uint8_t *m = RLC_ALLOCA(uint8_t, len + RLC_FC_BYTES);

84  ec_curve_get_ord(n);

86  bn_rand_mod(k, n);
87  ec_mul_gen(p, k);

91 
92  memcpy(m, msg, len);
93  bn_write_bin(m + len, RLC_FC_BYTES, r);
94  md_map(hash, m, len + RLC_FC_BYTES);
95 
96  if (8 * RLC_MD_LEN > bn_bits(n)) {
97  len = RLC_CEIL(bn_bits(n), 8);
98  bn_read_bin(e, hash, len);
99  bn_rsh(e, e, 8 * RLC_MD_LEN - bn_bits(n));
100  } else {
101  bn_read_bin(e, hash, RLC_MD_LEN);
102  }
103 
104  bn_mod(e, e, n);
105 
106  bn_mul(s, d, e);
107  bn_mod(s, s, n);
108  bn_sub(s, n, s);
109  bn_add(s, s, k);
110  bn_mod(s, s, n);
111  }
112  CATCH_ANY {
113  result = RLC_ERR;
114  }
115  FINALLY {
116  bn_free(n);
117  bn_free(k);
118  bn_free(x);
119  bn_free(r);
120  ec_free(p);
121  RLC_FREE(m);
122  }
