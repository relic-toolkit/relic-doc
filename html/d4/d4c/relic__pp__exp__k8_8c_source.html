40void pp_exp_k8(fp8_t c, fp8_t a) {
41 fp8_t t0, t1, t2, t3, t4;
42 bn_t x, t;
43
44 fp8_null(t0);
45 fp8_null(t1);
46 fp8_null(t2);
47 fp8_null(t3);
48 bn_null(x);
49 bn_null(t);
50
51 RLC_TRY {
52 fp8_new(t0);
53 fp8_new(t1);
54 fp8_new(t2);
55 fp8_new(t3);
56 fp8_new(t4);
57 bn_new(x);
58 bn_new(t);
59
60 fp_prime_get_par(x);
61
62 /* First, compute m = f^(p^4 - 1). */
63 fp8_conv_cyc(c, a);
64
65 /* Now compute d = m^(p + T)(p^2 + T^2). */
66 fp8_frb(t0, c, 1);
67 fp8_exp_cyc(t1, c, x);
68 fp8_mul(t0, t0, t1);
69 fp8_exp_cyc(t1, t0, x);
70 fp8_exp_cyc(t1, t1, x);
71 fp8_frb(t0, t0, 2);
72 fp8_mul(t0, t0, t1);
73
74 /* Now compute d^(p + T)/r. */
75 bn_set_dig(t, 0xdc);
76 bn_lsh(t, t, 8);
77 bn_add_dig(t, t, 0x4);
78 /* t1 = a_y = a^y, t2 = a_u = a^u */
79 fp8_exp_cyc(t1, t0, t);
80 fp8_copy(t2, t0);
81 /* t3 = a_q = a_y^y * a_u^u, b = a_q * !a_u. */
82 fp8_exp_cyc(t3, t1, t);
83 fp8_copy(t4, t2);
84 fp8_mul(t3, t3, t4);
85 fp8_inv_cyc(t2, t2);
86 fp8_mul(t4, t3, t2);
87
88 bn_rsh(x, x, 2);
89 /* b = (b^4 * a)^(T/4) * a_y. */
90 fp8_sqr_cyc(t4, t4);
91 fp8_sqr_cyc(t4, t4);
92 fp8_mul(t4, t4, t0);
93 fp8_exp_cyc(t4, t4, x);
94 fp8_mul(t4, t4, t1);
95
96 /* b = (b^4)^U * !a_y. */
97 fp8_sqr_cyc(t4, t4);
98 fp8_sqr_cyc(t4, t4);
99 fp8_exp_cyc(t4, t4, x);
100 fp8_inv_cyc(t1, t1);
101 fp8_mul(t4, t4, t1);
102
103 /* b = (b^4 * a)^(T/4) * !a_u * a. */
104 fp8_sqr_cyc(t4, t4);
105 fp8_sqr_cyc(t4, t4);
106 fp8_mul(t4, t4, t0);
107 fp8_exp_cyc(t4, t4, x);
108 fp8_mul(t4, t4, t2);
109 fp8_mul(t4, t4, t0);
110
111 /* b = (b^4)^U * a_q. */
112 fp8_sqr_cyc(t4, t4);
113 fp8_sqr_cyc(t4, t4);
114 fp8_exp_cyc(t4, t4, x);
115 fp8_mul(t4, t4, t3);
116
117 /* return f^(p^4 - 1)*(1 + (p + T)(p^2 + T^2)(p + T)/r) */
118 fp8_mul(c, c, t4);
119 }
120 RLC_CATCH_ANY {
121 RLC_THROW(ERR_CAUGHT);
122 }
123 RLC_FINALLY {
124 fp8_free(t0);
125 fp8_free(t1);
126 fp8_free(t2);
127 fp8_free(t3);
128 fp8_free(t4);
129 bn_free(x);
130 bn_free(t);
131 }
