51static void pp_mil_k24(fp24_t r, ep4_t *t, ep4_t *q, ep_t *p, int m, bn_t a) {
52 fp24_t l;
53 ep_t *_p = RLC_ALLOCA(ep_t, m);
54 ep4_t *_q = RLC_ALLOCA(ep4_t, m);
55 size_t len = bn_bits(a) + 1;
56 int i, j;
57 int8_t s[RLC_FP_BITS + 1];
58
59 if (m == 0) {
60 return;
61 }
62
63 fp24_null(l);
64
65 RLC_TRY {
66 fp24_new(l);
67 if (_p == NULL || _q == NULL) {
68 RLC_THROW(ERR_NO_MEMORY);
69 }
70 for (j = 0; j < m; j++) {
71 ep_null(_p[j]);
72 ep4_null(_q[j]);
73 ep_new(_p[j]);
74 ep4_new(_q[j]);
75 ep4_copy(t[j], q[j]);
76 ep4_neg(_q[j], q[j]);
77#if EP_ADD == BASIC
78 ep_neg(_p[j], p[j]);
79#else
80 fp_add(_p[j]->x, p[j]->x, p[j]->x);
81 fp_add(_p[j]->x, _p[j]->x, p[j]->x);
82 fp_neg(_p[j]->y, p[j]->y);
83#endif
84 }
85
86 fp24_zero(l);
87 bn_rec_naf(s, &len, a, 2);
88 pp_dbl_k24(r, t[0], t[0], _p[0]);
89 for (j = 1; j < m; j++) {
90 pp_dbl_k24(l, t[j], t[j], _p[j]);
91 fp24_mul_dxs(r, r, l);
92 }
93 if (s[len - 2] > 0) {
94 for (j = 0; j < m; j++) {
95 pp_add_k24(l, t[j], q[j], p[j]);
96 fp24_mul_dxs(r, r, l);
97 }
98 }
99 if (s[len - 2] < 0) {
100 for (j = 0; j < m; j++) {
101 pp_add_k24(l, t[j], _q[j], p[j]);
102 fp24_mul_dxs(r, r, l);
103 }
104 }
105
106 for (i = len - 3; i >= 0; i--) {
107 fp24_sqr(r, r);
108 for (j = 0; j < m; j++) {
109 pp_dbl_k24(l, t[j], t[j], _p[j]);
110 fp24_mul_dxs(r, r, l);
111 if (s[i] > 0) {
112 pp_add_k24(l, t[j], q[j], p[j]);
113 fp24_mul_dxs(r, r, l);
114 }
115 if (s[i] < 0) {
116 pp_add_k24(l, t[j], _q[j], p[j]);
117 fp24_mul_dxs(r, r, l);
118 }
119 }
120 }
121 }
122 RLC_CATCH_ANY {
123 RLC_THROW(ERR_CAUGHT);
124 }
125 RLC_FINALLY {
126 fp24_free(l);
127 for (j = 0; j < m; j++) {
128 ep_free(_p[j]);
129 ep4_free(_q[j]);
130 }
131 RLC_FREE(_p);
132 RLC_FREE(_q);
133 }
134}

142void pp_map_k24(fp24_t r, const ep_t p, const ep4_t q) {
143 ep_t _p[1];
144 ep4_t t[1], _q[1];
145 bn_t a;
146
147 ep_null(_p[0]);
148 ep4_null(_q[0]);
149 ep4_null(t[0]);
150 bn_null(a);
151
152 RLC_TRY {
153 ep_new(_p[0]);
154 ep4_new(_q[0]);
155 ep4_new(t[0]);
156 bn_new(a);
157
158 fp_prime_get_par(a);
159 fp24_set_dig(r, 1);
160
161 ep_norm(_p[0], p);
162 ep4_norm(_q[0], q);
163
164 if (!ep_is_infty(_p[0]) && !ep4_is_infty(_q[0])) {
165 switch (ep_curve_is_pairf()) {
166 case EP_B24:
167 /* r = f_{|a|,Q}(P). */
168 pp_mil_k24(r, t, _q, _p, 1, a);
169 if (bn_sign(a) == RLC_NEG) {
170 fp24_inv_cyc(r, r);
171 }
172 pp_exp_k24(r, r);
173 break;
174 }
175 }
176 }
177 RLC_CATCH_ANY {
178 RLC_THROW(ERR_CAUGHT);
179 }
180 RLC_FINALLY {
181 ep_free(_p[0]);
182 ep4_free(_q[0]);
183 ep4_free(t[0]);
184 bn_free(a);
185 }
186}

188void pp_map_sim_k24(fp24_t r, const ep_t *p, const ep4_t *q, int m) {
189 ep_t *_p = RLC_ALLOCA(ep_t, m);

191 bn_t a;

193
194 RLC_TRY {
195 bn_null(a);
196 bn_new(a);
197 if (_p == NULL || _q == NULL || t == NULL) {
198 RLC_THROW(ERR_NO_MEMORY);

202 ep4_null(_q[i]);

206 ep4_new(t[i]);
207 }
208
209 j = 0;
210 for (i = 0; i < m; i++) {
211 if (!ep_is_infty(p[i]) && !ep4_is_infty(q[i])) {
212 ep_norm(_p[j], p[i]);
213 ep4_norm(_q[j++], q[i]);
214 }
215 }
216
217 fp_prime_get_par(a);
218 fp24_set_dig(r, 1);
219
220 if (j > 0) {
221 switch (ep_curve_is_pairf()) {
222 case EP_B24:
223 /* r = f_{|a|,Q}(P). */
224 pp_mil_k24(r, t, _q, _p, j, a);
225 if (bn_sign(a) == RLC_NEG) {
226 fp24_inv_cyc(r, r);
227 }
228 pp_exp_k24(r, r);
229 break;
230 }
231 }
232 }
233 RLC_CATCH_ANY {
234 RLC_THROW(ERR_CAUGHT);
235 }
236 RLC_FINALLY {
237 bn_free(a);
238 for (i = 0; i < m; i++) {
239 ep_free(_p[i]);
240 ep4_free(_q[i]);
241 ep4_free(t[i]);
242 }
243 RLC_FREE(_p);
244 RLC_FREE(_q);
245 RLC_FREE(t);
246 }
247}
