38int cp_phpe_gen(bn_t pub, phpe_t prv, size_t bits) {
39 int result = RLC_OK;
40
41 /* Generate primes p and q of equivalent length. */
42 do {
43 bn_gen_prime(prv->p, bits / 2);
44 bn_gen_prime(prv->q, bits / 2);
45 } while (bn_cmp(prv->p, prv->q) == RLC_EQ);
46
47 /* Compute n = pq. */
48 bn_mul(prv->n, prv->p, prv->q);
49
50#ifdef CP_CRT
51 /* Fix g = n + 1. */
52
53 /* Precompute dp = 1/(pow(g, p-1, p^2)//p mod p.
54 * with g=1+n, this is also 1/((p-1)q) mod p.
55 */
56 bn_sub_dig(prv->dp, prv->p, 1); //p-1
57 bn_mul(prv->dp, prv->dp, prv->q); //(p-1)q
58 bn_mod(prv->dp, prv->dp, prv->p); //(p-1)q mod p
59 bn_mod_inv(prv->dp, prv->dp, prv->p); //((p-1)q)^{-1} mod p
60
61 /* Precompute dq = 1/(pow(g, q-1, q^2)//q mod q.
62 * with g=1+n, this is also 1/((q-1)p) mod q.
63 */
64 bn_sub_dig(prv->dq, prv->q, 1); //q-1
65 bn_mul(prv->dq, prv->dq, prv->p); //(q-1)p
66 bn_mod(prv->dq, prv->dq, prv->q); //(q-1)p mod q
67 bn_mod_inv(prv->dq, prv->dq, prv->q); //((q-1)p)^{-1} mod q
68
69 /* qInv = q^(-1) mod p. */
70 bn_mod_inv(prv->qi, prv->q, prv->p);
71#endif
72
73 bn_copy(pub, prv->n);
74 return result;
75}

78 int result = RLC_OK;

80
81 bn_null(s);

83 RLC_TRY {

100int cp_phpe_enc(bn_t c, const bn_t m, const bn_t pub) {
101 bn_t g, r, s;
102 int result = RLC_OK;
103
104 bn_null(g);
105 bn_null(r);
106 bn_null(s);
107
108 if (pub == NULL || bn_bits(m) > bn_bits(pub)) {
109 return RLC_ERR;
110 }
111
112 RLC_TRY {
113 bn_new(g);
114 bn_new(r);
115 bn_new(s);
116
117 /* Generate r in Z_n^*. */
118 bn_rand_mod(r, pub);
119 /* Compute c = (g^m)(r^n) mod n^2.
120 * With g=1+n, this is also (1 + n*m) * r^n mod n^2.
121 */
122 bn_add_dig(g, pub, 1);
123 bn_sqr(s, pub);
124 bn_mul(c, pub, m);
125 bn_add_dig(c, c, 1);
126 bn_mod(c, c, s);
127 bn_mxp(r, r, pub, s);
128 bn_mul(c, c, r);
129 bn_mod(c, c, s);
130 }
131 RLC_CATCH_ANY {
132 result = RLC_ERR;
133 }
134 RLC_FINALLY {
135 bn_free(g);
136 bn_free(r);
137 bn_free(s);
138 }
139
140 return result;
141}

143int cp_phpe_dec(bn_t m, const bn_t c, const phpe_t prv) {
144 bn_t t, u;
145 int result = RLC_OK;
146
147 if (prv == NULL || bn_bits(c) > 2 * bn_bits(prv->n)) {
148 return RLC_ERR;
149 }
150
151 bn_null(t);
152 bn_null(u);
153
154 RLC_TRY {
155 bn_new(t);
156 bn_new(u);
157
158 bn_sub_dig(t, prv->p, 1);
159 bn_sub_dig(u, prv->q, 1);
160
161#if !defined(CP_CRT)
162 bn_mul(t, t, u);
163 /* Compute (c^l mod n^2) * u mod n. */
164 bn_sqr(u, prv->n);
165 bn_mxp(m, c, t, u);
166
167 bn_sub_dig(m, m, 1);
168 bn_div(m, m, prv->n);
169 bn_mod_inv(u, t, prv->n);
170 bn_mul(m, m, u);
171 bn_mod(m, m, prv->n);
172#else
173 bn_mxp_crt(m, c, t, u, prv, 1);
174#endif /* CP_CRT */
175 }
176 RLC_CATCH_ANY {
177 result = RLC_ERR;
178 }
179 RLC_FINALLY {
180 bn_free(t);
181 bn_free(u);
182 }
183
184 return result;
185}
