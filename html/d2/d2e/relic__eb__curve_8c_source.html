46static void detect_opt(int *opt, fb_t a) {
47 if (fb_is_zero(a)) {
48 *opt = RLC_ZERO;
49 } else {
50 if (fb_cmp_dig(a, 1) == RLC_EQ) {
51 *opt = RLC_ONE;
52 } else {
53 if (fb_bits(a) <= RLC_DIG) {
54 *opt = RLC_TINY;
55 } else {
56 *opt = RLC_HUGE;
57 }
58 }
59 }
60}

66void eb_curve_init(void) {
67 ctx_t *ctx = core_get();
68#ifdef EB_PRECO
69 for (int i = 0; i < RLC_EB_TABLE; i++) {
70 ctx->eb_ptr[i] = &(ctx->eb_pre[i]);
71 }
72#endif
73 fb_zero(ctx->eb_g.x);
74 fb_zero(ctx->eb_g.y);
75 fb_zero(ctx->eb_g.z);
76 bn_make(&(ctx->eb_r), RLC_FB_DIGS);
77 bn_make(&(ctx->eb_h), RLC_FB_DIGS);

136void eb_curve_set(const fb_t a, const fb_t b, const eb_t g, const bn_t r,
137 const bn_t h) {
138 ctx_t *ctx = core_get();
139 fb_copy(ctx->eb_a, a);
140 fb_copy(ctx->eb_b, b);
141
142 detect_opt(&(ctx->eb_opt_a), ctx->eb_a);
143 detect_opt(&(ctx->eb_opt_b), ctx->eb_b);
144
145 if (fb_cmp_dig(ctx->eb_b, 1) == RLC_EQ) {
146 ctx->eb_is_kbltz = 1;
147 } else {
148 ctx->eb_is_kbltz = 0;
149 }
150
151 eb_norm(&(ctx->eb_g), g);
152 bn_copy(&(ctx->eb_r), r);
153 bn_copy(&(ctx->eb_h), h);
154
155#if defined(EB_PRECO)
156 eb_mul_pre((eb_t *)eb_curve_get_tab(), &(ctx->eb_g));
157#endif
158}
