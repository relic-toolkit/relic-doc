46 static void detect_opt(int *opt, fb_t a) {
47  if (fb_is_zero(a)) {
48  *opt = RLC_ZERO;
49  } else {
50  if (fb_cmp_dig(a, 1) == RLC_EQ) {
51  *opt = RLC_ONE;
52  } else {
53  if (fb_bits(a) <= RLC_DIG) {
54  *opt = RLC_TINY;
55  } else {
56  *opt = RLC_HUGE;
57  }
58  }
59  }
60 }
