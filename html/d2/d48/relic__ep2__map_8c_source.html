46 void ep2_mul_cof_bn(ep2_t r, ep2_t p) {
47  bn_t x;
48  ep2_t t0, t1, t2;
49 
50  ep2_null(t0);
51  ep2_null(t1);
52  ep2_null(t2);
53  bn_null(x);
54 
55  TRY {
56  ep2_new(t0);
57  ep2_new(t1);
58  ep2_new(t2);
59  bn_new(x);
60 
61  fp_prime_get_par(x);
62 
63  /* Compute t0 = xP. */
64  ep2_mul_basic(t0, p, x);
65 
66  /* Compute t1 = \psi(3xP). */
67  ep2_dbl(t1, t0);
68  ep2_add(t1, t1, t0);
69  ep2_norm(t1, t1);
70  ep2_frb(t1, t1, 1);
71 
72  /* Compute t2 = \psi^3(P) + t0 + t1 + \psi^2(xP). */
73  ep2_frb(t2, p, 2);
74  ep2_frb(t2, t2, 1);
75  ep2_add(t2, t2, t0);
76  ep2_add(t2, t2, t1);
77  ep2_frb(t1, t0, 2);
78  ep2_add(t2, t2, t1);
79 
80  ep2_norm(r, t2);
81  }
82  CATCH_ANY {
83  THROW(ERR_CAUGHT);
84  }
85  FINALLY {
86  ep2_free(t0);
87  ep2_free(t1);
88  ep2_free(t2);
89  bn_free(x);
90  }
91 }

99 void ep2_mul_cof_b12(ep2_t r, ep2_t p) {
100  bn_t x;
101  ep2_t t0, t1, t2, t3;
102 
103  ep2_null(t0);
104  ep2_null(t1);
105  ep2_null(t2);
106  ep2_null(t3);
107  bn_null(x);
108 
109  TRY {
110  ep2_new(t0);
111  ep2_new(t1);
112  ep2_new(t2);
113  ep2_new(t3);
114  bn_new(x);
115 
116  fp_prime_get_par(x);
117 
118  /* Compute t0 = xP. */
119  ep2_mul_basic(t0, p, x);
120  /* Compute t1 = [x^2]P. */
121  ep2_mul_basic(t1, t0, x);
122 
123  /* t2 = (x^2 - x - 1)P = x^2P - x*P - P. */
124  ep2_sub(t2, t1, t0);
125  ep2_sub(t2, t2, p);
126  /* t3 = \psi(x - 1)P. */
127  ep2_sub(t3, t0, p);
128  ep2_frb(t3, t3, 1);
129  ep2_add(t2, t2, t3);
130  /* t3 = \psi^2(2P). */
131  ep2_dbl(t3, p);
132  ep2_frb(t3, t3, 2);
133  ep2_add(t2, t2, t3);
134  ep2_norm(r, t2);
135  }
136  CATCH_ANY {
137  THROW(ERR_CAUGHT);
138  }
139  FINALLY {
140  ep2_free(t0);
141  ep2_free(t1);
142  ep2_free(t2);
143  ep2_free(t3);
144  bn_free(x);
145  }

152 void ep2_map(ep2_t p, const uint8_t *msg, int len) {
153  bn_t x;
154  fp2_t t0;
155  uint8_t digest[RLC_MD_LEN];
156 
157  bn_null(x);
158  fp2_null(t0);
159 
160  TRY {
161  bn_new(x);
162  fp2_new(t0);
163 
164  md_map(digest, msg, len);
165  bn_read_bin(x, digest, RLC_MIN(RLC_FP_BYTES, RLC_MD_LEN));
166 
167  fp_prime_conv(p->x[0], x);
168  fp_zero(p->x[1]);
169  fp_set_dig(p->z[0], 1);
170  fp_zero(p->z[1]);
171 
172  while (1) {
173  ep2_rhs(t0, p);
174 
175  if (fp2_srt(p->y, t0)) {
176  p->norm = 1;
177  break;
178  }
179 
180  fp_add_dig(p->x[0], p->x[0], 1);
181  }
182 
183  switch (ep_curve_is_pairf()) {
184  case EP_BN:
185  ep2_mul_cof_bn(p, p);
186  break;
187  case EP_B12:
188  ep2_mul_cof_b12(p, p);
189  break;
190  default:
191  /* Now, multiply by cofactor to get the correct group. */
192  ep2_curve_get_cof(x);
193  if (bn_bits(x) < RLC_DIG) {
194  ep2_mul_dig(p, p, x->dp[0]);
195  } else {
196  ep2_mul_basic(p, p, x);
197  }
198  break;
199  }
200  }
201  CATCH_ANY {
202  THROW(ERR_CAUGHT);
203  }
204  FINALLY {
205  bn_free(x);
206  fp2_free(t0);
207  }
208 }
