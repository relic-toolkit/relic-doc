68/* caution: this function overwrites k, which it uses as an auxiliary variable */
69static inline int fp2_sgn0(const fp2_t t, bn_t k) {
70 const int t_0_zero = fp_is_zero(t[0]);
71
72 fp_prime_back(k, t[0]);
73 const int t_0_neg = bn_get_bit(k, 0);
74
75 fp_prime_back(k, t[1]);
76 const int t_1_neg = bn_get_bit(k, 0);
77
78 /* t[0] == 0 ? sgn0(t[1]) : sgn0(t[0]) */
79 return t_0_neg | (t_0_zero & t_1_neg);
80}

92static void ep2_map_from_field(ep2_t p, const uint8_t *r, size_t len) {
93 bn_t k;
94 fp2_t t;
95 ep2_t q;
96 int neg;
97 /* enough space for two extension field elements plus extra bytes for uniformity */
98 const int lpe = (FP_PRIME + ep_param_level() + 7) / 8;
99
100 bn_null(k);
101 fp2_null(t);
102 ep2_null(q);
103
104 RLC_TRY {
105 if (len != 2 * lpe) {
106 RLC_THROW(ERR_NO_VALID);
107 }
108
109 bn_new(k);
110 fp2_new(t);
111 ep2_new(q);
112
113 /* which hash function should we use? */
114 const int abNeq0 = (ep2_curve_opt_a() != RLC_ZERO) &&
115 (ep2_curve_opt_b() != RLC_ZERO);
116 void (*const map_fn)(ep2_t, fp2_t) = (void (*const))
117 (ep2_curve_is_ctmap() || abNeq0 ? ep2_map_sswu : ep2_map_svdw);
118
119#define EP2_MAP_CONVERT_BYTES(IDX) \
120 do { \
121 bn_read_bin(k, r + 2 * IDX * lpe, lpe); \
122 fp_prime_conv(t[0], k); \
123 bn_read_bin(k, r + (2 * IDX + 1) * lpe, lpe); \
124 fp_prime_conv(t[1], k); \
125 } while (0)
126
127#define EP2_MAP_APPLY_MAP(PT) \
128 do { \
129 /* sign of t */ \
130 neg = fp2_sgn0(t, k); \
131 /* convert */ \
132 map_fn(PT, t); \
133 /* compare sign of y to sign of t; fix if necessary */ \
134 neg = neg != fp2_sgn0(PT->y, k); \
135 fp2_neg(t, PT->y); \
136 fp2_copy_sec(PT->y, t, neg); \
137 } while (0)
138
139 /* first map invocation */
140 EP2_MAP_CONVERT_BYTES(0);
141 EP2_MAP_APPLY_MAP(p);
142 TMPL_MAP_CALL_ISOMAP(ep2, p);
143
144 /* second map invocation */
145 EP2_MAP_CONVERT_BYTES(1);
146 EP2_MAP_APPLY_MAP(q);
147 TMPL_MAP_CALL_ISOMAP(ep2, q);
148
149 /* XXX(rsw) could add p and q and then apply isomap,
150 * but need ep_add to support addition on isogeny curves */
151
152#undef EP2_MAP_CONVERT_BYTES
153#undef EP2_MAP_APPLY_MAP
154
155 /* sum the result */
156 ep2_add(p, p, q);
157 ep2_norm(p, p);
158 ep2_mul_cof(p, p);
159 }
160 RLC_CATCH_ANY {
161 RLC_THROW(ERR_CAUGHT);
162 }
163 RLC_FINALLY {
164 bn_free(k);
165 fp2_free(t);
166 ep2_free(q);
167 }
168}

176void ep2_map_basic(ep2_t p, const uint8_t *msg, size_t len) {
177 bn_t x;
178 fp2_t t0;
179 uint8_t digest[RLC_MD_LEN];
180
181 bn_null(x);
182 fp2_null(t0);
183
184 RLC_TRY {
185 bn_new(x);
186 fp2_new(t0);
187
188 md_map(digest, msg, len);
189 bn_read_bin(x, digest, RLC_MIN(RLC_FP_BYTES, RLC_MD_LEN));
190
191 fp2_zero(p->x);
192 fp_prime_conv(p->x[0], x);
193 fp2_set_dig(p->z, 1);
194
195 while (1) {
196 ep2_rhs(t0, p->x);
197
198 if (fp2_is_sqr(t0) == 1) {
199 fp2_srt(p->y, t0);
200 p->coord = BASIC;
201 break;
202 }
203
204 fp2_add_dig(p->x, p->x, 1);
205 }
206
207 ep2_mul_cof(p, p);
208 }
209 RLC_CATCH_ANY {
210 RLC_THROW(ERR_CAUGHT);
211 }
212 RLC_FINALLY {
213 bn_free(x);
214 fp2_free(t0);
215 }
216}

235 }
236 RLC_CATCH_ANY {
237 RLC_THROW(ERR_CAUGHT);
238 }
239 RLC_FINALLY {
240 RLC_FREE(r);

462 fp2_sub(v, x1, v);
463 fp2_mul(v, v, z1);
464 fp2_mul(w, y1, z1);
465 fp2_dbl(w, w);

469 } else {
