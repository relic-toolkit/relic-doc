55static void eb_mul_sim_kbltz(eb_t r, const eb_t p, const bn_t k, const eb_t q,
56 const bn_t m, const eb_t *t) {
57 int i, n0, n1, w, g;
58 int8_t u, tnaf0[RLC_FB_BITS + 8], tnaf1[RLC_FB_BITS + 8], *_k, *_m;
59 eb_t t0[1 << (RLC_WIDTH - 2)];
60 eb_t t1[1 << (RLC_WIDTH - 2)];
61 size_t l, l0, l1;
62
63 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
64 eb_null(t0[i]);
65 eb_null(t1[i]);
66 }
67
68 RLC_TRY {
69 /* Compute the w-TNAF representation of k. */
70 if (eb_curve_opt_a() == RLC_ZERO) {
71 u = -1;
72 } else {
73 u = 1;
74 }
75
76 g = (t == NULL ? 0 : 1);
77 if (!g) {
78 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
79 eb_new(t0[i]);
80 eb_set_infty(t0[i]);
81 fb_set_bit(t0[i]->z, 0, 1);
82 t0[i]->coord = BASIC;
83 }
84 eb_tab(t0, p, RLC_WIDTH);
85 t = (const eb_t *)t0;
86 }
87
88 /* Prepare the precomputation table. */
89 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
90 eb_new(t1[i]);
91 eb_set_infty(t1[i]);
92 fb_set_bit(t1[i]->z, 0, 1);
93 t1[i]->coord = BASIC;
94 }
95 /* Compute the precomputation table. */
96 eb_tab(t1, q, RLC_WIDTH);
97
98 /* Compute the w-TNAF representation of k. */
99 if (g) {
100 w = RLC_DEPTH;
101 } else {
102 w = RLC_WIDTH;
103 }
104
105 l0 = l1 = RLC_FB_BITS + 8;
106 bn_rec_tnaf(tnaf0, &l0, k, u, RLC_FB_BITS, w);
107 bn_rec_tnaf(tnaf1, &l1, m, u, RLC_FB_BITS, RLC_WIDTH);
108
109 l = RLC_MAX(l0, l1);
110 _k = tnaf0 + l - 1;
111 _m = tnaf1 + l - 1;
112
113 if (bn_sign(k) == RLC_NEG) {
114 for (i = 0; i < l0; i++) {
115 tnaf0[i] = -tnaf0[i];
116 }
117 }
118 if (bn_sign(m) == RLC_NEG) {
119 for (i = 0; i < l1; i++) {
120 tnaf1[i] = -tnaf1[i];
121 }
122 }
123
124 _k = tnaf0 + l - 1;
125 _m = tnaf1 + l - 1;
126 eb_set_infty(r);
127 for (i = l - 1; i >= 0; i--, _k--, _m--) {
128 eb_frb(r, r);
129
130 n0 = *_k;
131 n1 = *_m;
132 if (n0 > 0) {
133 eb_add(r, r, t[n0 / 2]);
134 }
135 if (n0 < 0) {
136 eb_sub(r, r, t[-n0 / 2]);
137 }
138 if (n1 > 0) {
139 eb_add(r, r, t1[n1 / 2]);
140 }
141 if (n1 < 0) {
142 eb_sub(r, r, t1[-n1 / 2]);
143 }
144 }
145 /* Convert r to affine coordinates. */
146 eb_norm(r, r);
147 }
148 RLC_CATCH_ANY {
149 RLC_THROW(ERR_CAUGHT);
150 }
151 RLC_FINALLY {
152 if (!g) {
153 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
154 eb_free(t0[i]);
155 }
156 }
157 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
158 eb_free(t1[i]);
159 }
160 }
161}

163#endif /* EB_KBLTZ */

179static void eb_mul_sim_plain(eb_t r, const eb_t p, const bn_t k, const eb_t q,
180 const bn_t m, const eb_t *t) {
181 int i, n0, n1, w, g = (t == NULL ? 0 : 1);
182 int8_t naf0[RLC_FB_BITS + 1], naf1[RLC_FB_BITS + 1], *_k, *_m;
183 eb_t t0[1 << (RLC_WIDTH - 2)];
184 eb_t t1[1 << (RLC_WIDTH - 2)];
185 size_t l, l0, l1;
186
187 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
188 eb_null(t0[i]);
189 eb_null(t1[i]);
190 }
191
192 RLC_TRY {
193 if (!g) {
194 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
195 eb_new(t0[i]);
196 }
197 eb_tab(t0, p, RLC_WIDTH);
198 t = (const eb_t *)t0;
199 }
200
201 /* Prepare the precomputation table. */
202 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
203 eb_new(t1[i]);
204 }
205 /* Compute the precomputation table. */
206 eb_tab(t1, q, RLC_WIDTH);
207
208 /* Compute the w-NAF representation of k. */
209 if (g) {
210 w = RLC_DEPTH;
211 } else {
212 w = RLC_WIDTH;
213 }
214
215 l0 = l1 = RLC_FB_BITS + 1;
216 bn_rec_naf(naf0, &l0, k, w);
217 bn_rec_naf(naf1, &l1, m, RLC_WIDTH);
218
219 l = RLC_MAX(l0, l1);
220 if (bn_sign(k) == RLC_NEG) {
221 for (i = 0; i < l0; i++) {
222 naf0[i] = -naf0[i];
223 }
224 }
225 if (bn_sign(m) == RLC_NEG) {
226 for (i = 0; i < l1; i++) {
227 naf1[i] = -naf1[i];
228 }
229 }
230
231 _k = naf0 + l - 1;
232 _m = naf1 + l - 1;
233 eb_set_infty(r);
234 for (i = l - 1; i >= 0; i--, _k--, _m--) {
235 eb_dbl(r, r);
236
237 n0 = *_k;
238 n1 = *_m;
239 if (n0 > 0) {
240 eb_add(r, r, t[n0 / 2]);
241 }
242 if (n0 < 0) {
243 eb_sub(r, r, t[-n0 / 2]);
244 }
245 if (n1 > 0) {
246 eb_add(r, r, t1[n1 / 2]);
247 }
248 if (n1 < 0) {
249 eb_sub(r, r, t1[-n1 / 2]);
250 }
251 }
252 /* Convert r to affine coordinates. */
253 eb_norm(r, r);
254 }
255 RLC_CATCH_ANY {
256 RLC_THROW(ERR_CAUGHT);
257 }
258 RLC_FINALLY {
259 /* Free the precomputation tables. */
260 if (!g) {
261 for (i = 0; i < 1 << (RLC_WIDTH - 2); i++) {
262 eb_free(t0[i]);
263 }
264 }
265 for (i = 0; i < 1 << (RLC_WIDTH - 2); i++) {
266 eb_free(t1[i]);
267 }
268 }
269}

305void eb_mul_sim_trick(eb_t r, const eb_t p, const bn_t k, const eb_t q,
306 const bn_t m) {
307 eb_t t0[1 << (RLC_WIDTH / 2)], t1[1 << (RLC_WIDTH / 2)], t[1 << RLC_WIDTH];
308 size_t l0, l1, w = RLC_WIDTH / 2;
309 uint8_t w0[RLC_FB_BITS], w1[RLC_FB_BITS];
310 bn_t n;
311
312 bn_null(n);
313
314 if (bn_is_zero(k) || eb_is_infty(p)) {
315 eb_mul(r, q, m);
316 return;
317 }
318 if (bn_is_zero(m) || eb_is_infty(q)) {
319 eb_mul(r, p, k);
320 return;
321 }
322
323 RLC_TRY {
324 bn_new(n);
325
326 eb_curve_get_ord(n);
327
328 for (int i = 0; i < (1 << w); i++) {
329 eb_null(t0[i]);
330 eb_null(t1[i]);
331 eb_new(t0[i]);
332 eb_new(t1[i]);
333 }
334 for (int i = 0; i < (1 << RLC_WIDTH); i++) {
335 eb_null(t[i]);
336 eb_new(t[i]);
337 }
338
339 eb_set_infty(t0[0]);
340 eb_copy(t0[1], p);
341 if (bn_sign(k) == RLC_NEG) {
342 eb_neg(t0[1], t0[1]);
343 }
344 for (int i = 2; i < (1 << w); i++) {
345 eb_add(t0[i], t0[i - 1], t0[1]);
346 }
347
348 eb_set_infty(t1[0]);
349 eb_copy(t1[1], q);
350 if (bn_sign(m) == RLC_NEG) {
351 eb_neg(t1[1], t1[1]);
352 }
353 for (int i = 2; i < (1 << w); i++) {
354 eb_add(t1[i], t1[i - 1], t1[1]);
355 }
356
357 for (int i = 0; i < (1 << w); i++) {
358 for (int j = 0; j < (1 << w); j++) {
359 eb_add(t[(i << w) + j], t0[i], t1[j]);
360 }
361 }
362
363#if RLC_WIDTH > 2 && defined(EB_MIXED)
364 eb_norm_sim(t + 2, (const eb_t *)(t + 2), (1 << (w + w)) - 2);
365#endif
366
367 l0 = l1 = RLC_CEIL(RLC_FB_BITS + 1, w);
368 bn_rec_win(w0, &l0, k, w);
369 bn_rec_win(w1, &l1, m, w);
370
371 eb_set_infty(r);
372 for (int i = RLC_MAX(l0, l1) - 1; i >= 0; i--) {
373 for (int j = 0; j < w; j++) {
374 eb_dbl(r, r);
375 }
376 eb_add(r, r, t[(w0[i] << w) + w1[i]]);
377 }
378 eb_norm(r, r);
379 } RLC_CATCH_ANY {
380 RLC_THROW(ERR_CAUGHT);
381 }
382 RLC_FINALLY {
383 bn_free(n);
384 for (int i = 0; i < (1 << w); i++) {
385 eb_free(t0[i]);
386 eb_free(t1[i]);
387 }
388 for (int i = 0; i < (1 << RLC_WIDTH); i++) {
389 eb_free(t[i]);
390 }
391 }
392}

425void eb_mul_sim_joint(eb_t r, const eb_t p, const bn_t k, const eb_t q,
426 const bn_t m) {
427 eb_t t[5];
428 int i, u_i, offset;
429 int8_t jsf[2 * (RLC_FB_BITS + 1)];
430 size_t len;
431
432 if (bn_is_zero(k) || eb_is_infty(p)) {
433 eb_mul(r, q, m);
434 return;
435 }
436 if (bn_is_zero(m) || eb_is_infty(q)) {
437 eb_mul(r, p, k);
438 return;
439 }
440
441 RLC_TRY {
442 for (i = 0; i < 5; i++) {
443 eb_null(t[i]);
444 eb_new(t[i]);
445 }
446
447 eb_set_infty(t[0]);
448 eb_copy(t[1], q);
449 if (bn_sign(m) == RLC_NEG) {
450 eb_neg(t[1], t[1]);
451 }
452 eb_copy(t[2], p);
453 if (bn_sign(k) == RLC_NEG) {
454 eb_neg(t[2], t[2]);
455 }
456 eb_add(t[3], t[2], t[1]);
457 eb_sub(t[4], t[2], t[1]);
458#if defined(EB_MIXED)
459 eb_norm_sim(t + 3, (const eb_t*)(t + 3), 2);
460#endif
461
462 len = 2 * (RLC_FB_BITS + 1);
463 bn_rec_jsf(jsf, &len, k, m);
464
465 eb_set_infty(r);
466 offset = RLC_MAX(bn_bits(k), bn_bits(m)) + 1;
467 for (i = len - 1; i >= 0; i--) {
468 eb_dbl(r, r);
469 if (jsf[i] != 0 && jsf[i] == -jsf[i + offset]) {
470 u_i = jsf[i] * 2 + jsf[i + offset];
471 if (u_i < 0) {
472 eb_sub(r, r, t[4]);
473 } else {
474 eb_add(r, r, t[4]);
475 }
476 } else {
477 u_i = jsf[i] * 2 + jsf[i + offset];
478 if (u_i < 0) {
479 eb_sub(r, r, t[-u_i]);
480 } else {
481 eb_add(r, r, t[u_i]);
482 }
483 }
484 }
485 eb_norm(r, r);
486 }
487 RLC_CATCH_ANY {
488 RLC_THROW(ERR_CAUGHT);
489 }
490 RLC_FINALLY {
491 for (i = 0; i < 5; i++) {
492 eb_free(t[i]);
493 }
494 }
495}
