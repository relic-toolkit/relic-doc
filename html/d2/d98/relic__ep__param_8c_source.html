40 * A note on the MAP_U values below.
41 *
42 * These values were generated using the methods recommended in
43 * draft-irtf-cfrg-hash-to-curve-06, Appendix E. In particular,
44 * for any curve with j-invariant 0 or 1728 (i.e., a = 0 or b = 0),
45 * the MAP_U value is the output of the find_z_svdw function.
46 * For all other curves, the MAP_U value was calculated using the
47 * find_z_sswu function.
48 *
49 * Note that for the BLS12-381 curve (B12_P381) when using an
50 * isogeny map (defined(EP_CTMAP)), the MAP_U value is from the
51 * find_z_sswu function for the isogenous curve. Similarly, when
52 * specifying isogeny maps for other curves one should specify
53 * the MAP_U value for the isogenous curve.

846static int ep_param_get_coeffs(fp_st *coeffs, const char *str) {
847 if (str[0] == '\0') {
848 /* need nonzero strlen */
849 RLC_THROW(ERR_NO_VALID);
850 }
851 int degree = 0;
852 unsigned offset = 0;
853 for (; degree < RLC_EP_CTMAP_MAX; ++degree) {
854 const char *end = strchr(str + offset, ';');
855 if (end == NULL) {
856 /* last one */
857 fp_read_str(coeffs[degree], str + offset, strlen(str + offset), 16);
858 break;
859 }
860 unsigned len = end - str - offset;
861 fp_read_str(coeffs[degree], str + offset, len, 16);
862 offset += len + 1; /* move to after ';' */
863 }
864 if (degree == RLC_EP_CTMAP_MAX) {
865 /* ran out of space before converting all coeffs */
866 RLC_THROW(ERR_NO_VALID);
867 }
868 return degree;
869}
