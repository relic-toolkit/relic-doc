46static void ep2_mul_cof_bn(ep2_t r, const ep2_t p) {
47 bn_t x;
48 ep2_t t0, t1, t2;
49
50 ep2_null(t0);
51 ep2_null(t1);
52 ep2_null(t2);
53 bn_null(x);
54
55 RLC_TRY {
56 ep2_new(t0);
57 ep2_new(t1);
58 ep2_new(t2);
59 bn_new(x);
60
61 fp_prime_get_par(x);
62
63 /* Compute t0 = xP. */
64 ep2_mul_basic(t0, p, x);
65
66 /* Compute t1 = \psi(3xP). */
67 ep2_dbl(t1, t0);
68 ep2_add(t1, t1, t0);
69 ep2_norm(t1, t1);
70 ep2_frb(t1, t1, 1);
71
72 /* Compute t2 = \psi^3(P) + t0 + t1 + \psi^2(xP). */
73 ep2_frb(t2, p, 2);
74 ep2_frb(t2, t2, 1);
75 ep2_add(t2, t2, t0);
76 ep2_add(t2, t2, t1);
77 ep2_frb(t1, t0, 2);
78 ep2_add(t2, t2, t1);
79
80 ep2_norm(r, t2);
81 }
82 RLC_CATCH_ANY {
83 RLC_THROW(ERR_CAUGHT);
84 }
85 RLC_FINALLY {
86 ep2_free(t0);
87 ep2_free(t1);
88 ep2_free(t2);
89 bn_free(x);
90 }
91}

99static void ep2_mul_cof_b12(ep2_t r, const ep2_t p) {
100 bn_t x;
101 ep2_t t0, t1, t2, t3;
102
103 ep2_null(t0);
104 ep2_null(t1);
105 ep2_null(t2);
106 ep2_null(t3);
107 bn_null(x);
108
109 RLC_TRY {
110 ep2_new(t0);
111 ep2_new(t1);
112 ep2_new(t2);
113 ep2_new(t3);
114 bn_new(x);
115
116 fp_prime_get_par(x);
117
118 /* Compute t0 = xP. */
119 ep2_mul_basic(t0, p, x);
120 /* Compute t1 = [x^2]P. */
121 ep2_mul_basic(t1, t0, x);
122
123 /* t2 = (x^2 - x - 1)P = x^2P - x*P - P. */
124 ep2_sub(t2, t1, t0);
125 ep2_sub(t2, t2, p);
126 /* t3 = \psi(x - 1)P. */
127 ep2_sub(t3, t0, p);
128 ep2_frb(t3, t3, 1);
129 ep2_add(t2, t2, t3);
130 /* t3 = \psi^2(2P). */
131 ep2_dbl(t3, p);
132 ep2_frb(t3, t3, 2);
133 ep2_add(t2, t2, t3);
134 ep2_norm(r, t2);
135 }
136 RLC_CATCH_ANY {
137 RLC_THROW(ERR_CAUGHT);
138 }
139 RLC_FINALLY {
140 ep2_free(t0);
141 ep2_free(t1);
142 ep2_free(t2);
143 ep2_free(t3);
144 bn_free(x);
145 }
146}

152void ep2_mul_cof(ep2_t r, const ep2_t p) {
153 bn_t k;
154
155 bn_null(k);
156
157 RLC_TRY {
158 switch (ep_curve_is_pairf()) {
159 case EP_BN:
160 ep2_mul_cof_bn(r, p);

165 default:
166 /* Now, multiply by cofactor to get the correct group. */
167 ep2_curve_get_cof(k);
168 ep2_mul_big(r, p, k);
169 break;
170 }
171 } RLC_CATCH_ANY {
172 RLC_THROW(ERR_CAUGHT);
173 } RLC_FINALLY {
174 bn_free(k);
