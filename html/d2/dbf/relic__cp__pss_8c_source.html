91int cp_pss_ver(const g1_t a, const g1_t b, const bn_t m, const g2_t g,
92 const g2_t x, const g2_t y) {
93 g1_t p[2];
94 g2_t r[2];
95 gt_t e;
96 int result = 0;
97
98 g1_null(p[0]);
99 g1_null(p[1]);
100 g2_null(r[0]);
101 g2_null(r[1]);
102 gt_null(e);
103
104 RLC_TRY {
105 g1_new(p[0]);
106 g1_new(p[1]);
107 g2_new(r[0]);
108 g2_new(r[1]);
109 gt_new(e);
110
111 g1_copy(p[0], a);
112 g1_copy(p[1], b);
113 g2_copy(r[1], g);
114 g2_neg(r[1], r[1]);
115
116 g2_mul(r[0], y, m);
117 g2_add(r[0], r[0], x);
118 g2_norm(r[0], r[0]);
119
120 pc_map_sim(e, p, r, 2);
121 if (gt_is_unity(e) && !g1_is_infty(a)) {
122 result = 1;
123 }
124 }
125 RLC_CATCH_ANY {
126 RLC_THROW(ERR_CAUGHT);
127 }
128 RLC_FINALLY {
129 g1_free(p[0]);
130 g1_free(p[1]);
131 g2_free(r[0]);
132 g2_free(r[1]);
133 gt_free(e);
134 }
135 return result;
136}

165int cp_psb_sig(g1_t a, g1_t b, const bn_t ms[], const bn_t r, const bn_t s[],
166 size_t l) {
167 bn_t m, n, t;
168 int i, result = RLC_OK;
169
170 bn_null(m);
171 bn_null(n);
172 bn_null(t);
173
174 RLC_TRY {
175 bn_new(m);
176 bn_new(n);
177 bn_new(t);
178
179 /* Choose random a in G1. */
180 g1_rand(a);
181 /* Compute b = a^x+\sum y_im_i. */
182 pc_get_ord(n);
183
184 bn_copy(t, r);
185 for (i = 0; i < l; i++) {
186 bn_mod(m, ms[i], n);
187 bn_mul(m, m, s[i]);
188 bn_mod(m, m, n);
189 bn_add(t, t, m);
190 bn_mod(t, t, n);
191 }
192 g1_mul(b, a, t);
193 }
194 RLC_CATCH_ANY {
195 result = RLC_ERR;
196 }
197 RLC_FINALLY {
198 bn_free(m);
199 bn_free(n);
200 bn_free(t);
201 }
202 return result;
203}

205int cp_psb_ver(const g1_t a, const g1_t b, const bn_t ms[], const g2_t g,
206 const g2_t x, const g2_t y[], size_t l) {
207 g1_t p[2];
208 g2_t q[2];
209 gt_t e;
210 bn_t m, n;
211 int result = 0;
212
213 g1_null(p[0]);
214 g1_null(p[1]);
215 g2_null(q[0]);
216 g2_null(q[1]);
217 gt_null(e);
218 bn_null(m);
219 bn_null(n);
220
221 RLC_TRY {
222 g1_new(p[0]);
223 g1_new(p[1]);
224 g2_new(q[0]);
225 g2_new(q[1]);
226 gt_new(e);
227 bn_new(m);
228 bn_new(n);
229
230 /* Check that e(a, x \prod y_i^m_i) = e(b, g). */
231 g1_copy(p[0], a);
232 g1_copy(p[1], b);
233
234 pc_get_ord(n);
235 g2_mul_sim_lot(q[0], y, ms, l);
236 g2_add(q[0], q[0], x);
237 g2_norm(q[0], q[0]);
238 g2_copy(q[1], g);
239 g2_neg(q[1], q[1]);
240 pc_map_sim(e, p, q, 2);
241 if (!g1_is_infty(a) && gt_is_unity(e)) {
242 result = 1;
243 }
244 }
245 RLC_CATCH_ANY {
246 RLC_THROW(ERR_CAUGHT);
247 }
248 RLC_FINALLY {
249 g1_free(p[0]);
250 g1_free(p[1]);
251 g2_free(q[0]);
252 g2_free(q[1]);
253 gt_free(e);
254 bn_free(m);
255 bn_free(n);
256 }
257 return result;
258}
