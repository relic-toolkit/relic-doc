39void bn_mod_inv(bn_t c, const bn_t a, const bn_t b) {
40 bn_t t, u;
41
42 bn_null(t);
43 bn_null(u);
44
45 RLC_TRY {
46 bn_new(t);
47 bn_new(u);
48
49 bn_copy(u, b);
50 bn_gcd_ext(t, c, NULL, a, b);
51
52 if (bn_sign(c) == RLC_NEG) {
53 bn_add(c, c, u);
54 }
55
56 if (bn_cmp_dig(t, 1) != RLC_EQ) {
57 RLC_THROW(ERR_NO_VALID);
58 }
59 } RLC_CATCH_ANY {
60 RLC_THROW(ERR_CAUGHT);
61 } RLC_FINALLY {
62 bn_free(t);
63 bn_free(u);
64 }
65}

67void bn_mod_inv_sim(bn_t *c, const bn_t *a, const bn_t b, int n) {

76 }

78 bn_null(t[i]);

86 for (i = 1; i < n; i++) {
87 bn_copy(t[i], a[i]);
88 bn_mul(c[i], c[i - 1], a[i]);
89 bn_mod(c[i], c[i], b);

91
92 bn_mod_inv(u, c[n - 1], b);
93
94 for (i = n - 1; i > 0; i--) {
95 bn_mul(c[i], u, c[i - 1]);
96 bn_mod(c[i], c[i], b);
97 bn_mul(u, u, t[i]);
98 bn_mod(u, u, b);
99 }
100 bn_copy(c[0], u);
101 }
102 RLC_CATCH_ANY {
103 RLC_THROW(ERR_CAUGHT);
104 }
105 RLC_FINALLY {
106 for (i = 0; i < n; i++) {
107 bn_free(t[i]);
108 }
109 bn_free(u);
110 RLC_FREE(t);
