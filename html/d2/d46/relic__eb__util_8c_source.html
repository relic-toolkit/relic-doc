58 int eb_cmp(const eb_t p, const eb_t q) {
59  eb_t r, s;
60  int result = RLC_EQ;
61 
62  eb_null(r);
63  eb_null(s);
64 
65  TRY {
66  eb_new(r);
67  eb_new(s);
68 
69  if ((p->norm == 0) && (q->norm == 0)) {
70  /* If the two points are not normalized, it is faster to compare
71  * x1 * z2 == x2 * z1 and y1 * z2^2 == y2 * z1^2. */
72  fb_mul(r->x, p->x, q->z);
73  fb_mul(s->x, q->x, p->z);
74  fb_sqr(r->z, p->z);
75  fb_sqr(s->z, q->z);
76  fb_mul(r->y, p->y, s->z);
77  fb_mul(s->y, q->y, r->z);
78  } else {
79  if (p->norm == 1) {
80  eb_copy(r, p);
81  } else {
82  eb_norm(r, p);
83  }
84 
85  if (q->norm == 1) {
86  eb_copy(s, q);
87  } else {
88  eb_norm(s, q);
89  }
90  }
91 
92  if (fb_cmp(r->x, s->x) != RLC_EQ) {
93  result = RLC_NE;
94  }
95 
96  if (fb_cmp(r->y, s->y) != RLC_EQ) {
97  result = RLC_NE;
98  }
99  } CATCH_ANY {
100  THROW(ERR_CAUGHT);
101  } FINALLY {
102  eb_free(r);
103  eb_free(s);
104  }
105 
106  return result;
107 }

132 void eb_rhs(fb_t rhs, const eb_t p) {
133  fb_t t0, t1;
134 
135  fb_null(t0);
136  fb_null(t1);
137 
138  TRY {
139  fb_new(t0);
140  fb_new(t1);
141 
142  /* t0 = x1^2. */
143  fb_sqr(t0, p->x);
144  /* t1 = x1^3. */
145  fb_mul(t1, t0, p->x);
146 
147  /* t1 = x1^3 + a * x1^2 + b. */
148  switch (eb_curve_opt_a()) {
149  case RLC_ZERO:
150  break;
151  case RLC_ONE:
152  fb_add(t1, t1, t0);
153  break;
154  case RLC_TINY:
155  fb_mul_dig(t0, t0, eb_curve_get_a()[0]);
156  fb_add(t1, t1, t0);
157  break;
158  default:
159  fb_mul(t0, t0, eb_curve_get_a());
160  fb_add(t1, t1, t0);
161  break;
162  }
163 
164  switch (eb_curve_opt_b()) {
165  case RLC_ZERO:
166  break;
167  case RLC_ONE:
168  fb_add_dig(t1, t1, 1);
169  break;
170  case RLC_TINY:
171  fb_add_dig(t1, t1, eb_curve_get_b()[0]);
172  break;
173  default:
174  fb_add(t1, t1, eb_curve_get_b());
175  break;
176  }
177 
178  fb_copy(rhs, t1);
179  }
180  CATCH_ANY {
181  THROW(ERR_CAUGHT);
182  }
183  FINALLY {
184  fb_free(t0);
185  fb_free(t1);
186  }
187 }

197  TRY {

212  FINALLY {
