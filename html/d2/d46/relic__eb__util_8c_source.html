81void eb_rhs(fb_t rhs, const fb_t x) {
82 fb_t t0, t1;
83
84 fb_null(t0);
85 fb_null(t1);
86
87 RLC_TRY {
88 fb_new(t0);
89 fb_new(t1);
90
91 /* t0 = x1^2. */
92 fb_sqr(t0, x);
93 /* t1 = x1^3. */
94 fb_mul(t1, t0, x);
95
96 /* t1 = x1^3 + a * x1^2 + b. */
97 switch (eb_curve_opt_a()) {
98 case RLC_ZERO:
99 break;
100 case RLC_ONE:
101 fb_add(t1, t1, t0);
102 break;
103 case RLC_TINY:
104 fb_mul_dig(t0, t0, eb_curve_get_a()[0]);
105 fb_add(t1, t1, t0);
106 break;
107 default:
108 fb_mul(t0, t0, eb_curve_get_a());
109 fb_add(t1, t1, t0);
110 break;
111 }
112
113 switch (eb_curve_opt_b()) {
114 case RLC_ZERO:
115 break;
116 case RLC_ONE:
117 fb_add_dig(t1, t1, 1);
118 break;
119 case RLC_TINY:
120 fb_add_dig(t1, t1, eb_curve_get_b()[0]);
121 break;
122 default:
123 fb_add(t1, t1, eb_curve_get_b());
124 break;
125 }
126
127 fb_copy(rhs, t1);
128 }
129 RLC_CATCH_ANY {
130 RLC_THROW(ERR_CAUGHT);
131 }
132 RLC_FINALLY {
133 fb_free(t0);
134 fb_free(t1);
135 }
136}

138void eb_blind(eb_t r, const eb_t p) {
139 fb_t rand;
140
141 fb_null(rand);
142
143 RLC_TRY {
144 fb_new(rand);
145
146 fb_rand(rand);
147 fb_mul(r->z, p->z, rand);

152 } RLC_CATCH_ANY {
153 RLC_THROW(ERR_CAUGHT);
154 } RLC_FINALLY {

159int eb_on_curve(const eb_t p) {
160 eb_t t;
161 fb_t lhs;
162 int r = 0;
163
164 eb_null(t);
165 fb_null(lhs);
166
167 RLC_TRY {
168 eb_new(t);
169 fb_new(lhs);
170
171 eb_norm(t, p);
172
173 fb_mul(lhs, t->x, t->y);
174 eb_rhs(t->x, t->x);
175 fb_sqr(t->y, t->y);
176 fb_add(lhs, lhs, t->y);
177 r = (fb_cmp(lhs, t->x) == RLC_EQ) || eb_is_infty(p);
178 }
179 RLC_CATCH_ANY {
180 RLC_THROW(ERR_CAUGHT);
181 }
182 RLC_FINALLY {
183 eb_free(t);
184 fb_free(lhs);
185 }
186 return r;
187}

492size_t eb_size_bin(const eb_t a, int pack) {

507void eb_read_bin(eb_t a, const uint8_t *bin, size_t len) {
508 if (len == 1) {
509 if (bin[0] == 0) {
510 eb_set_infty(a);
511 return;
512 } else {
513 RLC_THROW(ERR_NO_BUFFER);
514 return;
515 }
516 }
517
518 if (len != (RLC_FB_BYTES + 1) && len != (2 * RLC_FB_BYTES + 1)) {
519 RLC_THROW(ERR_NO_BUFFER);
520 return;
521 }
522
523 a->coord = BASIC;
524 fb_set_dig(a->z, 1);
525 fb_read_bin(a->x, bin + 1, RLC_FB_BYTES);
526 if (len == RLC_FB_BYTES + 1) {
527 switch(bin[0]) {
528 case 2:
529 fb_zero(a->y);
530 break;
531 case 3:
532 fb_zero(a->y);
533 fb_set_bit(a->y, 0, 1);
534 break;
535 default:
536 RLC_THROW(ERR_NO_VALID);
537 break;
538 }
539 eb_upk(a, a);
540 }
541
542 if (len == 2 * RLC_FB_BYTES + 1) {
543 if (bin[0] == 4) {
544 fb_read_bin(a->y, bin + RLC_FB_BYTES + 1, RLC_FB_BYTES);
545 } else {
546 RLC_THROW(ERR_NO_VALID);
547 return;
548 }
549 }
550
551 if (!eb_on_curve(a)) {
552 RLC_THROW(ERR_NO_VALID);
553 return;
554 }
555}

557void eb_write_bin(uint8_t *bin, size_t len, const eb_t a, int pack) {
558 eb_t t;
559
560 eb_null(t);
561
562 memset(bin, 0, len);
563
564 if (eb_is_infty(a)) {
565 if (len < 1) {
566 RLC_THROW(ERR_NO_BUFFER);
567 return;
568 } else {
569 return;
570 }
571 }
572
573 RLC_TRY {
574 eb_new(t);
575
576 eb_norm(t, a);
577
578 if (pack) {
579 if (len < RLC_FB_BYTES + 1) {
580 RLC_THROW(ERR_NO_BUFFER);
581 } else {
582 eb_pck(t, t);
583 bin[0] = 2 | fb_get_bit(t->y, 0);
584 fb_write_bin(bin + 1, RLC_FB_BYTES, t->x);
585 }
586 } else {
587 if (len < 2 * RLC_FB_BYTES + 1) {
588 RLC_THROW(ERR_NO_BUFFER);
589 } else {
590 bin[0] = 4;
591 fb_write_bin(bin + 1, RLC_FB_BYTES, t->x);
592 fb_write_bin(bin + RLC_FB_BYTES + 1, RLC_FB_BYTES, t->y);
593 }
594 }
595 } RLC_CATCH_ANY {
596 RLC_THROW(ERR_CAUGHT);
597 }
598 RLC_FINALLY {
599 eb_free(t);
600 }
