45 void fb_inv_basic(fb_t c, const fb_t a) {

47  int i, x;
48 
49  fb_null(t);
50  fb_null(u);
51  fb_null(v);
52 
53  if (fb_is_zero(a)) {
54  THROW(ERR_NO_VALID);
55  }
56 
57  TRY {
58  fb_new(t);
59  fb_new(u);
60  fb_new(v);
61 
62 #if (FB_POLYN % 2) == 0
63  fb_sqr(v, a);
64  for (i = 2; i < RLC_FB_BITS; i++) {
65  fb_sqr(u, a);
66  for (int j = 1; j < i; j++) {
67  fb_sqr(u, u);
68  }
69  fb_mul(v, v, u);
70  }
71  fb_copy(c, v);
72 #else
73  /* u = a^2, v = 1, x = (m - 1)/2. */
74  fb_sqr(u, a);
75  fb_set_dig(v, 1);
76  x = (RLC_FB_BITS - 1) >> 1;
77 
78  while (x != 0) {
79  /* u = u * a^{2x}. */
80  fb_copy(t, u);
81  for (i = 0; i < x; i++) {
82  fb_sqr(t, t);
83  }
84  fb_mul(u, u, t);
85  if ((x & 0x01) == 0) {
86  x = x >> 1;
87  } else {
88  /* v = v * u, u = u^2, x = (x - 1)/2. */
89  fb_mul(v, v, u);
90  fb_sqr(u, u);
91  x = (x - 1) >> 1;
92  }
93  }
94 #endif
95  fb_copy(c, v);
96  }
97  CATCH_ANY {
98  THROW(ERR_CAUGHT);
99  }
100  FINALLY {
101  fb_free(t);
102  fb_free(u);
103  fb_free(v);
104  }
105 }

111 void fb_inv_binar(fb_t c, const fb_t a) {
112  int lu, lv;
113  dv_t u, v, g1, g2;
114 
115  dv_null(u);
116  dv_null(v);
117  dv_null(g1);
118  dv_null(g2);
119 
120  if (fb_is_zero(a)) {
121  THROW(ERR_NO_VALID);
122  }
123 
124  TRY {
125  dv_new(u);
126  dv_new(v);
127  dv_new(g1);
128  dv_new(g2);
129 
130  /* u = a, v = f, g1 = 1, g2 = 0. */
131  fb_copy(u, a);
132  fb_copy(v, fb_poly_get());
133  if (RLC_FB_BITS % RLC_DIG == 0) {
134  v[RLC_FB_DIGS] = 1;
135  }
136  dv_zero(g1, 2 * RLC_FB_DIGS);
137  g1[0] = 1;
138  dv_zero(g2, 2 * RLC_FB_DIGS);
139 
140  lu = RLC_FB_DIGS;
141  lv = RLC_FB_DIGS + (RLC_FB_BITS % RLC_DIG == 0);
142 
143  /* While (u != 1 && v != 1. */
144  while (1) {
145  /* While z divides u do. */
146  while ((u[0] & 0x01) == 0) {
147  /* u = u/z. */
148  bn_rsh1_low(u, u, lu);
149  /* If z divides g1 then g1 = g1/z; else g1 = (g1 + f)/z. */
150  if ((g1[0] & 0x01) == 1) {
151  fb_poly_add(g1, g1);
152  if (RLC_FB_BITS % RLC_DIG == 0) {
153  g1[RLC_FB_DIGS] ^= 1;
154  }
155  }
156  bn_rsh1_low(g1, g1, RLC_FB_DIGS + 1);
157  }
158 
159  while (u[lu - 1] == 0)
160  lu--;
161  if (lu == 1 && u[0] == 1)
162  break;
163 
164  /* While z divides v do. */
165  while ((v[0] & 0x01) == 0) {
166  /* v = v/z. */
167  bn_rsh1_low(v, v, lv);
168  /* If z divides g2 then g2 = g2/z; else (g2 = g2 + f)/z. */
169  if ((g2[0] & 0x01) == 1) {
170  fb_poly_add(g2, g2);
171  if (RLC_FB_BITS % RLC_DIG == 0) {
172  g2[RLC_FB_DIGS] ^= 1;
173  }
174  }
175  bn_rsh1_low(g2, g2, RLC_FB_DIGS + 1);
176  }
177 
178  while (v[lv - 1] == 0)
179  lv--;
180  if (lv == 1 && v[0] == 1)
181  break;
182 
183  /* If deg(u) > deg(v) then u = u + v, g1 = g1 + g2. */
184  if (lu > lv || (lu == lv && u[lu - 1] > v[lv - 1])) {
185  fb_addd_low(u, u, v, lv);
186  fb_add(g1, g1, g2);
187  } else {
188  /* Else v = v + u, g2 = g2 + g1. */
189  fb_addd_low(v, v, u, lu);
190  fb_add(g2, g2, g1);
191  }
192  }
193 
194  /* If u == 1 then return g1; else return g2. */
195  if (lu == 1 && u[0] == 1) {
196  fb_copy(c, g1);
197  } else {
198  fb_copy(c, g2);
199  }
200  }
201  CATCH_ANY {
202  THROW(ERR_CAUGHT);
203  }
204  FINALLY {
205  dv_free(u);
206  dv_free(v);
207  dv_free(g1);
208  dv_free(g2);
209  }
210 }

216 void fb_inv_exgcd(fb_t c, const fb_t a) {
217  int j, d, lu, lv, lt, l1, l2, bu, bv;
218  dv_t _u, _v, _g1, _g2;
219  dig_t *t = NULL, *u = NULL, *v = NULL, *g1 = NULL, *g2 = NULL, carry;
220 
221  fb_null(_u);
222  fb_null(_v);
223  fb_null(_g1);
224  fb_null(_g2);
225 
226  if (fb_is_zero(a)) {
227  THROW(ERR_NO_VALID);
228  }
229 
230  TRY {
231  dv_new(_u);
232  dv_new(_v);
233  dv_new(_g1);
234  dv_new(_g2);
235 
236  dv_zero(_g1, RLC_FB_DIGS + 1);
237  dv_zero(_g2, RLC_FB_DIGS + 1);
238 
239  u = _u;
240  v = _v;
241  g1 = _g1;
242  g2 = _g2;
243 
244  /* u = a, v = f, g1 = 1, g2 = 0. */
245  fb_copy(u, a);
246  fb_copy(v, fb_poly_get());
247  g1[0] = 1;
248 
249  lu = lv = RLC_FB_DIGS;
250  l1 = l2 = 1;
251 
252  bu = fb_bits(u);
253  bv = RLC_FB_BITS + 1;
254  j = bu - bv;
255 
256  /* While (u != 1). */
257  while (1) {
258  /* If j < 0 then swap(u, v), swap(g1, g2), j = -j. */
259  if (j < 0) {
260  t = u;
261  u = v;
262  v = t;
263 
264  lt = lu;
265  lu = lv;
266  lv = lt;
267 
268  t = g1;
269  g1 = g2;
270  g2 = t;
271 
272  lt = l1;
273  l1 = l2;
274  l2 = lt;
275 
276  j = -j;
277  }
278 
279  RLC_RIP(j, d, j);
280 
281  /* u = u + v * z^j. */
282  if (j > 0) {
283  carry = fb_lsha_low(u + d, v, j, lv);
284  u[d + lv] ^= carry;
285  } else {
286  fb_addd_low(u + d, u + d, v, lv);
287  }
288 
289  /* g1 = g1 + g2 * z^j. */
290  if (j > 0) {
291  carry = fb_lsha_low(g1 + d, g2, j, l2);
292  l1 = (l2 + d >= l1 ? l2 + d : l1);
293  if (carry) {
294  g1[d + l2] ^= carry;
295  l1 = (l2 + d >= l1 ? l1 + 1 : l1);
296  }
297  } else {
298  fb_addd_low(g1 + d, g1 + d, g2, l2);
299  l1 = (l2 + d > l1 ? l2 + d : l1);
300  }
301 
302  while (u[lu - 1] == 0)
303  lu--;
304  while (v[lv - 1] == 0)
305  lv--;
306 
307  if (lu == 1 && u[0] == 1)
308  break;
309 
310  /* j = deg(u) - deg(v). */
311  lt = util_bits_dig(u[lu - 1]) - util_bits_dig(v[lv - 1]);
312  j = ((lu - lv) << RLC_DIG_LOG) + lt;
313  }
314  /* Return g1. */
315  fb_copy(c, g1);
316  }
317  CATCH_ANY {
318  THROW(ERR_CAUGHT);
319  }
320  FINALLY {
321  dv_free(_u);
322  dv_free(_v);
323  dv_free(_g1);
324  dv_free(_g2);
325  }
326 }

332 void fb_inv_almos(fb_t c, const fb_t a) {
333  int lu, lv, lt;
334  dv_t _b, _d, _u, _v;
335  dig_t *t = NULL, *u = NULL, *v = NULL, *b = NULL, *d = NULL;
336 
337  dv_null(_b);
338  dv_null(_d);
339  dv_null(_u);
340  dv_null(_v);
341 
342  if (fb_is_zero(a)) {
343  THROW(ERR_NO_VALID);
344  }
345 
346  TRY {
347  dv_new(_b);
348  dv_new(_d);
349  dv_new(_u);
350  dv_new(_v);
351 
352  b = _b;
353  d = _d;
354  u = _u;
355  v = _v;
356 
357  /* b = 1, d = 0, u = a, v = f. */
358  dv_zero(b, 2 * RLC_FB_DIGS);
359  fb_set_dig(b, 1);
360  dv_zero(d, 2 * RLC_FB_DIGS);
361  fb_copy(u, a);
362  fb_copy(v, fb_poly_get());
363 
364  if (RLC_FB_BITS % RLC_DIG == 0) {
365  v[RLC_FB_DIGS] = 1;
366  }
367 
368  lu = RLC_FB_DIGS;
369  lv = RLC_FB_DIGS + (RLC_FB_BITS % RLC_DIG == 0);
370 
371  while (1) {
372  /* While z divides u do. */
373  while ((u[0] & 0x01) == 0) {
374  /* u = u/z. */
375  bn_rsh1_low(u, u, lu);
376  /* If z divide v then b = b/z; else b = (b + f)/z. */
377  if ((b[0] & 0x01) == 1) {
378  fb_poly_add(b, b);
379  if (RLC_FB_BITS % RLC_DIG == 0) {
380  b[RLC_FB_DIGS] ^= 1;
381  }
382  }
383  /* b often has RLC_FB_DIGS digits. */
384  bn_rsh1_low(b, b, RLC_FB_DIGS + 1);
385  }
386  /* If u = 1, return b. */
387  while (u[lu - 1] == 0)
388  lu--;
389  if (lu == 1 && u[0] == 1) {
390  break;
391  }
392  /* If deg(u) < deg(v) then swap(u, v), swap(b, d). */
393  if ((lu < lv) || ((lu == lv) && (u[lu - 1] < v[lv - 1]))) {
394  t = u;
395  u = v;
396  v = t;
397 
398  /* Swap lu and lv too. */
399  lt = lu;
400  lu = lv;
401  lv = lt;
402 
403  t = b;
404  b = d;
405  d = t;
406  }
407  /* u = u + v, b = b + d. */
408  fb_addd_low(u, u, v, lv);
409  fb_addn_low(b, b, d);
410  }
411  }
412  CATCH_ANY {
413  THROW(ERR_CAUGHT);
414  }
415  FINALLY {
416  fb_copy(c, b);
417  dv_free(_b);
418  dv_free(_d);
419  dv_free(_u);
420  dv_free(_v);
421  }
422 }

475  fb_mul(table[i], table[i], table[y]);
476  }

478 #endif
479  }
