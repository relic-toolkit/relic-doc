45void fb_inv_basic(fb_t c, const fb_t a) {
46 fb_t t, u, v;
47 int i, x;
48
49 fb_null(t);
50 fb_null(u);
51 fb_null(v);
52
53 if (fb_is_zero(a)) {
54 RLC_THROW(ERR_NO_VALID);
55 return;
56 }
57
58 RLC_TRY {
59 fb_new(t);
60 fb_new(u);
61 fb_new(v);
62
63#if (FB_POLYN % 2) == 0
64 fb_sqr(v, a);
65 for (i = 2; i < RLC_FB_BITS; i++) {
66 fb_sqr(u, a);
67 for (int j = 1; j < i; j++) {
68 fb_sqr(u, u);
69 }
70 fb_mul(v, v, u);
71 }
72 fb_copy(c, v);
73#else
74 /* u = a^2, v = 1, x = (m - 1)/2. */
75 fb_sqr(u, a);
76 fb_set_dig(v, 1);
77 x = (RLC_FB_BITS - 1) >> 1;
78
79 while (x != 0) {
80 /* u = u * a^{2x}. */
81 fb_copy(t, u);
82 for (i = 0; i < x; i++) {
83 fb_sqr(t, t);
84 }
85 fb_mul(u, u, t);
86 if ((x & 0x01) == 0) {
87 x = x >> 1;
88 } else {
89 /* v = v * u, u = u^2, x = (x - 1)/2. */
90 fb_mul(v, v, u);
91 fb_sqr(u, u);
92 x = (x - 1) >> 1;
93 }
94 }
95#endif
96 fb_copy(c, v);
97 }
98 RLC_CATCH_ANY {
99 RLC_THROW(ERR_CAUGHT);
100 }
101 RLC_FINALLY {
102 fb_free(t);
103 fb_free(u);
104 fb_free(v);
105 }
106}

112void fb_inv_binar(fb_t c, const fb_t a) {
113 int lu, lv;
114 dv_t u, v, g1, g2;
115
116 dv_null(u);
117 dv_null(v);
118 dv_null(g1);
119 dv_null(g2);
120
121 if (fb_is_zero(a)) {
122 RLC_THROW(ERR_NO_VALID);
123 return;
124 }
125
126 RLC_TRY {
127 dv_new(u);
128 dv_new(v);
129 dv_new(g1);
130 dv_new(g2);
131
132 /* u = a, v = f, g1 = 1, g2 = 0. */
133 fb_copy(u, a);
134 fb_copy(v, fb_poly_get());
135 if (RLC_FB_BITS % RLC_DIG == 0) {
136 v[RLC_FB_DIGS] = 1;
137 }
138 dv_zero(g1, 2 * RLC_FB_DIGS);
139 g1[0] = 1;
140 dv_zero(g2, 2 * RLC_FB_DIGS);
141
142 lu = RLC_FB_DIGS;
143 lv = RLC_FB_DIGS + (RLC_FB_BITS % RLC_DIG == 0);
144
145 /* While (u != 1 && v != 1. */
146 while (1) {
147 /* While z divides u do. */
148 while ((u[0] & 0x01) == 0) {
149 /* u = u/z. */
150 bn_rsh1_low(u, u, lu);
151 /* If z divides g1 then g1 = g1/z; else g1 = (g1 + f)/z. */
152 if ((g1[0] & 0x01) == 1) {
153 fb_poly_add(g1, g1);
154 if (RLC_FB_BITS % RLC_DIG == 0) {
155 g1[RLC_FB_DIGS] ^= 1;
156 }
157 }
158 bn_rsh1_low(g1, g1, RLC_FB_DIGS + 1);
159 }
160
161 while (u[lu - 1] == 0)
162 lu--;
163 if (lu == 1 && u[0] == 1)
164 break;
165
166 /* While z divides v do. */
167 while ((v[0] & 0x01) == 0) {
168 /* v = v/z. */
169 bn_rsh1_low(v, v, lv);
170 /* If z divides g2 then g2 = g2/z; else (g2 = g2 + f)/z. */
171 if ((g2[0] & 0x01) == 1) {
172 fb_poly_add(g2, g2);
173 if (RLC_FB_BITS % RLC_DIG == 0) {
174 g2[RLC_FB_DIGS] ^= 1;
175 }
176 }
177 bn_rsh1_low(g2, g2, RLC_FB_DIGS + 1);
178 }
179
180 while (v[lv - 1] == 0)
181 lv--;
182 if (lv == 1 && v[0] == 1)
183 break;
184
185 /* If deg(u) > deg(v) then u = u + v, g1 = g1 + g2. */
186 if (lu > lv || (lu == lv && u[lu - 1] > v[lv - 1])) {
187 fb_addd_low(u, u, v, lv);
188 fb_add(g1, g1, g2);
189 } else {
190 /* Else v = v + u, g2 = g2 + g1. */
191 fb_addd_low(v, v, u, lu);
192 fb_add(g2, g2, g1);
193 }
194 }
195
196 /* If u == 1 then return g1; else return g2. */
197 if (lu == 1 && u[0] == 1) {
198 fb_copy(c, g1);
199 } else {
200 fb_copy(c, g2);
201 }
202 }
203 RLC_CATCH_ANY {
204 RLC_THROW(ERR_CAUGHT);
205 }
206 RLC_FINALLY {
207 dv_free(u);
208 dv_free(v);
209 dv_free(g1);
210 dv_free(g2);
211 }
212}

218void fb_inv_exgcd(fb_t c, const fb_t a) {
219 int j, d, lu, lv, lt, l1, l2, bu, bv;
220 dv_t _u, _v, _g1, _g2;
221 dig_t *t = NULL, *u = NULL, *v = NULL, *g1 = NULL, *g2 = NULL, carry;
222
223 fb_null(_u);
224 fb_null(_v);
225 fb_null(_g1);
226 fb_null(_g2);
227
228 if (fb_is_zero(a)) {
229 RLC_THROW(ERR_NO_VALID);
230 return;
231 }
232
233 RLC_TRY {
234 dv_new(_u);
235 dv_new(_v);
236 dv_new(_g1);
237 dv_new(_g2);
238
239 dv_zero(_g1, RLC_FB_DIGS + 1);
240 dv_zero(_g2, RLC_FB_DIGS + 1);
241
242 u = _u;
243 v = _v;
244 g1 = _g1;
245 g2 = _g2;
246
247 /* u = a, v = f, g1 = 1, g2 = 0. */
248 fb_copy(u, a);
249 fb_copy(v, fb_poly_get());
250 g1[0] = 1;
251
252 lu = lv = RLC_FB_DIGS;
253 l1 = l2 = 1;
254
255 bu = fb_bits(u);
256 bv = RLC_FB_BITS + 1;
257 j = bu - bv;
258
259 /* While (u != 1). */
260 while (1) {
261 /* If j < 0 then swap(u, v), swap(g1, g2), j = -j. */
262 if (j < 0) {
263 t = u;
264 u = v;
265 v = t;
266
267 lt = lu;
268 lu = lv;
269 lv = lt;
270
271 t = g1;
272 g1 = g2;
273 g2 = t;
274
275 lt = l1;
276 l1 = l2;
277 l2 = lt;
278
279 j = -j;
280 }
281
282 RLC_RIP(j, d, j);
283
284 /* u = u + v * z^j. */
285 if (j > 0) {
286 carry = fb_lsha_low(u + d, v, j, lv);
287 u[d + lv] ^= carry;
288 } else {
289 fb_addd_low(u + d, u + d, v, lv);
290 }
291
292 /* g1 = g1 + g2 * z^j. */
293 if (j > 0) {
294 carry = fb_lsha_low(g1 + d, g2, j, l2);
295 l1 = (l2 + d >= l1 ? l2 + d : l1);
296 if (carry) {
297 g1[d + l2] ^= carry;
298 l1 = (l2 + d >= l1 ? l1 + 1 : l1);
299 }
300 } else {
301 fb_addd_low(g1 + d, g1 + d, g2, l2);
302 l1 = (l2 + d > l1 ? l2 + d : l1);
303 }
304
305 while (u[lu - 1] == 0)
306 lu--;
307 while (v[lv - 1] == 0)
308 lv--;
309
310 if (lu == 1 && u[0] == 1)
311 break;
312
313 /* j = deg(u) - deg(v). */
314 lt = util_bits_dig(u[lu - 1]) - util_bits_dig(v[lv - 1]);
315 if (lv > lu) {
316 j = -((lv - lu) << RLC_DIG_LOG) + lt;
317 } else {
318 j = ((lu - lv) << RLC_DIG_LOG) + lt;
319 }
320 }
321 /* Return g1. */
322 fb_copy(c, g1);
323 }
324 RLC_CATCH_ANY {
325 RLC_THROW(ERR_CAUGHT);
326 }
327 RLC_FINALLY {
328 dv_free(_u);
329 dv_free(_v);
330 dv_free(_g1);
331 dv_free(_g2);
332 }
333}

339void fb_inv_almos(fb_t c, const fb_t a) {
340 int lu, lv, lt;
341 dv_t _b, _d, _u, _v;
342 dig_t *t = NULL, *u = NULL, *v = NULL, *b = NULL, *d = NULL;
343
344 dv_null(_b);
345 dv_null(_d);
346 dv_null(_u);
347 dv_null(_v);
348
349 if (fb_is_zero(a)) {
350 RLC_THROW(ERR_NO_VALID);
351 return;
352 }
353
354 /* This is actually the binary version of the Extended Euclidean algorithm
355 * discussed in the Almost Inverse paper, so a rename is needed. */
356
357 RLC_TRY {
358 dv_new(_b);
359 dv_new(_d);
360 dv_new(_u);
361 dv_new(_v);
362
363 b = _b;
364 d = _d;
365 u = _u;
366 v = _v;
367
368 /* b = 1, d = 0, u = a, v = f. */
369 dv_zero(b, 2 * RLC_FB_DIGS);
370 fb_set_dig(b, 1);
371 dv_zero(d, 2 * RLC_FB_DIGS);
372 fb_copy(u, a);
373 fb_copy(v, fb_poly_get());
374
375 if (RLC_FB_BITS % RLC_DIG == 0) {
376 v[RLC_FB_DIGS] = 1;
377 }
378
379 lu = RLC_FB_DIGS;
380 lv = RLC_FB_DIGS + (RLC_FB_BITS % RLC_DIG == 0);
381
382 while (1) {
383 /* While z divides u do. */
384 while ((u[0] & 0x01) == 0) {
385 /* u = u/z. */
386 bn_rsh1_low(u, u, lu);
387 /* If z divide v then b = b/z; else b = (b + f)/z. */
388 if ((b[0] & 0x01) == 1) {
389 fb_poly_add(b, b);
390 if (RLC_FB_BITS % RLC_DIG == 0) {
391 b[RLC_FB_DIGS] ^= 1;
392 }
393 }
394 /* b often has RLC_FB_DIGS digits. */
395 bn_rsh1_low(b, b, RLC_FB_DIGS + 1);
396 }
397 /* If u = 1, return b. */
398 while (u[lu - 1] == 0)
399 lu--;
400 if (lu == 1 && u[0] == 1) {
401 break;
402 }
403 /* If deg(u) < deg(v) then swap(u, v), swap(b, d). */
404 if ((lu < lv) || ((lu == lv) && (u[lu - 1] < v[lv - 1]))) {
405 t = u;
406 u = v;
407 v = t;
408
409 /* Swap lu and lv too. */
410 lt = lu;
411 lu = lv;
412 lv = lt;
413
414 t = b;
415 b = d;
416 d = t;
417 }
418 /* u = u + v, b = b + d. */
419 fb_addd_low(u, u, v, lv);
420 fb_addn_low(b, b, d);
421 }
422 }
423 RLC_CATCH_ANY {
424 RLC_THROW(ERR_CAUGHT);
425 }
426 RLC_FINALLY {
427 fb_copy(c, b);
428 dv_free(_b);
429 dv_free(_d);
430 dv_free(_u);
431 dv_free(_v);
432 }
433}

439void fb_inv_itoht(fb_t c, const fb_t a) {
440 int i, x, y, len;
441 const int *chain = fb_poly_get_chain(&len);
442 int *u = RLC_ALLOCA(int, len + 1);
443 fb_t *table = RLC_ALLOCA(fb_t, len + 1);
444
445 for (i = 0; i <= len; i++) {
446 fb_null(table[i]);
447 }
448
449 if (fb_is_zero(a)) {
450 RLC_THROW(ERR_NO_VALID);
451 return;
452 }
453
454 RLC_TRY {
455 if (u == NULL || table == NULL) {
456 RLC_THROW(ERR_NO_MEMORY);
457 }
458 for (i = 0; i <= len; i++) {
459 fb_new(table[i]);
460 }
461
462#if (FB_POLYN % 2) == 0
463 fb_sqr(table[0], a);
464 for (i = 2; i < RLC_FB_BITS; i++) {
465 fb_sqr(table[1], a);
466 for (int j = 1; j < i; j++) {
467 fb_sqr(table[1], table[1]);
468 }
469 fb_mul(table[0], table[0], table[1]);
470 }
471 fb_copy(c, table[0]);
472#else
473 u[0] = 1;
474 u[1] = 2;
475 fb_copy(table[0], a);
476 fb_sqr(table[1], table[0]);
477 fb_mul(table[1], table[1], table[0]);
478 for (i = 2; i <= len; i++) {
479 x = chain[i - 1] >> 8;
480 y = chain[i - 1] - (x << 8);
481 if (x == y) {
482 u[i] = 2 * u[i - 1];
483 } else {
484 u[i] = u[x] + u[y];
485 }
486 fb_itr(table[i], table[x], u[y], fb_poly_tab_sqr(y));
487 fb_mul(table[i], table[i], table[y]);
488 }
489 fb_sqr(c, table[len]);
490#endif
491 }
492 RLC_CATCH_ANY {
493 RLC_THROW(ERR_CAUGHT);
494 }
495 RLC_FINALLY {
496 for (i = 0; i <= len; i++) {
497 fb_free(table[i]);
498 }
499 RLC_FREE(u);
500 RLC_FREE(table);
501 }
502}

508void fb_inv_bruch(fb_t c, const fb_t a) {
509 fb_t _r, _s, _u, _v;
510 dig_t *r = NULL, *s = NULL, *t = NULL, *u = NULL, *v = NULL;
511 int delta = 0;
512
513 fb_null(_r);
514 fb_null(_s);
515 fb_null(_u);
516 fb_null(_v);
517
518 if (fb_is_zero(a)) {
519 RLC_THROW(ERR_NO_VALID);
520 return;
521 }
522
523 RLC_TRY {
524 fb_new(_r);
525 fb_new(_s);
526 fb_new(_u);
527 fb_new(_v);
528
529 fb_copy(_r, a);
530 fb_copy(_s, fb_poly_get());
531 fb_zero(_v);
532 fb_set_dig(_u, 1);
533
534 r = _r;
535 s = _s;
536 u = _u;
537 v = _v;
538
539 for (int i = 1; i <= 2 * RLC_FB_BITS; i++) {
540 if ((r[RLC_FB_DIGS - 1] & ((dig_t)1 << (RLC_FB_BITS % RLC_DIG))) == 0) {
541 fb_lsh(r, r, 1);
542 fb_lsh(u, u, 1);
543 delta++;
544 } else {
545 if ((s[RLC_FB_DIGS - 1] & ((dig_t)1 << (RLC_FB_BITS % RLC_DIG)))) {
546 fb_add(s, s, r);
547 fb_add(v, v, u);
548 }
549 fb_lsh(s, s, 1);
550 if (delta == 0) {
551 t = r;
552 r = s;
553 s = t;
554 t = u;
555 u = v;
556 v = t;
557 fb_lsh(u, u, 1);
558 delta++;
559 } else {
560 fb_rsh(u, u, 1);
561 delta--;
562 }
563 }
564 }
565 fb_copy(c, u);
566 } RLC_CATCH_ANY {
567 RLC_THROW(ERR_CAUGHT);
568 } RLC_FINALLY {
569 fb_free(_r);
570 fb_free(_s);
571 fb_free(_u);
572 fb_free(_v);
573 }
574}

580void fb_inv_ctaia(fb_t c, const fb_t a) {
581 fb_t r, s, t, u, v;
582 int i, k, d, r0, d0;
583
584 fb_null(r);
585 fb_null(s);
586 fb_null(t);
587 fb_null(u);
588 fb_null(v);
589
590 if (fb_is_zero(a)) {
591 RLC_THROW(ERR_NO_VALID);
592 return;
593 }
594
595 RLC_TRY {
596 fb_new(r);
597 fb_new(s);
598 fb_new(t);
599 fb_new(u);
600 fb_new(v);
601
602 fb_copy(r, a);
603 fb_copy(t, fb_poly_get());
604 fb_copy(s, t);
605 fb_set_dig(u, 1);
606 fb_zero(v);
607 d = -1;
608
609 for (k = 1; k < 2 * RLC_FB_BITS; k++) {
610 r0 = r[0] & 1;
611 d0 = d >> (8 * sizeof(int) - 1);
612
613 for (i = 0; i < RLC_FB_DIGS; i++) {
614 r[i] ^= (s[i] & -r0);
615 u[i] ^= (v[i] & -r0);
616 s[i] ^= (r[i] & d0);
617 v[i] ^= (u[i] & d0);
618 }
619
620 d = RLC_SEL(d, -d, r0 & -d0);
621
622 fb_rsh(r, r, 1);
623
624 r0 = u[0] & 1;
625 fb_poly_add(t, u);
626 for (i = 0; i < RLC_FB_DIGS; i++) {
627 u[i] = RLC_SEL(u[i], t[i], r0);
628 }
629 fb_rsh(u, u, 1);
630 d--;
631 }
632
633 fb_copy(c, v);
634 } RLC_CATCH_ANY {
635 RLC_THROW(ERR_CAUGHT);
636 } RLC_FINALLY {
637 fb_free(r);
638 fb_free(s);
639 fb_free(t);
640 fb_free(u);
641 fb_free(v);
642 }
643}

659void fb_inv_sim(fb_t *c, const fb_t *a, int n) {
660 int i;
661 fb_t u, *t = RLC_ALLOCA(fb_t, n);
662
663 if (t == NULL) {
664 RLC_THROW(ERR_NO_MEMORY);
665 }
666
667 fb_null(u);
668
669 RLC_TRY {
670 for (i = 0; i < n; i++) {
671 fb_null(t[i]);
672 fb_new(t[i]);
673 }
674 fb_new(u);
675
676 fb_copy(c[0], a[0]);
677 fb_copy(t[0], a[0]);
678
679 for (i = 1; i < n; i++) {
680 fb_copy(t[i], a[i]);
681 fb_mul(c[i], c[i - 1], a[i]);
682 }
683
684 fb_inv(u, c[n - 1]);
685
686 for (i = n - 1; i > 0; i--) {
687 fb_mul(c[i], u, c[i - 1]);
688 fb_mul(u, u, t[i]);
689 }
690 fb_copy(c[0], u);
691 }
692 RLC_CATCH_ANY {
693 RLC_THROW(ERR_CAUGHT);
694 }
695 RLC_FINALLY {
696 for (i = 0; i < n; i++) {
697 fb_free(t[i]);
698 }
699 fb_free(u);
700 RLC_FREE(t);
701 }
