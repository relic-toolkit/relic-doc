271static void detect_opt(int *opt, fp4_t a) {
272 fp4_t t;
273 fp4_null(t);
274
275 RLC_TRY {
276 fp4_new(t);
277 fp4_set_dig(t, 3);
278 fp4_neg(t, t);
279
280 if (fp4_cmp(a, t) == RLC_EQ) {
281 *opt = RLC_MIN3;
282 } else if (fp4_is_zero(a)) {
283 *opt = RLC_ZERO;
284 } else if (fp4_cmp_dig(a, 1) == RLC_EQ) {
285 *opt = RLC_ONE;
286 } else if (fp4_cmp_dig(a, 2) == RLC_EQ) {
287 *opt = RLC_TWO;
288 } else if ((fp_bits(a[0][0]) <= RLC_DIG) && fp_is_zero(a[0][1]) &&
289 fp2_is_zero(a[1])) {
290 *opt = RLC_TINY;
291 } else {
292 *opt = RLC_HUGE;
293 }
294 }
295 RLC_CATCH_ANY {
296 RLC_THROW(ERR_CAUGHT);
297 }
298 RLC_FINALLY {
299 fp4_free(t);
300 }
301}
