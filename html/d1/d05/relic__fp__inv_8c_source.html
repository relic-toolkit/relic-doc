72 void fp_inv_binar(fp_t c, const fp_t a) {
73  bn_t u, v, g1, g2, p;
74 
75  bn_null(u);
76  bn_null(v);
77  bn_null(g1);
78  bn_null(g2);
79  bn_null(p);
80 
81  if (fp_is_zero(a)) {
82  THROW(ERR_NO_VALID);
83  }
84 
85  TRY {
86  bn_new(u);
87  bn_new(v);
88  bn_new(g1);
89  bn_new(g2);
90  bn_new(p);
91 
92  /* u = a, v = p, g1 = 1, g2 = 0. */
93  fp_prime_back(u, a);
94  p->used = RLC_FP_DIGS;
95  dv_copy(p->dp, fp_prime_get(), RLC_FP_DIGS);
96  bn_copy(v, p);
97  bn_set_dig(g1, 1);
98  bn_zero(g2);
99 
100  /* While (u != 1 && v != 1). */
101  while (1) {
102  /* While u is even do. */
103  while (!(u->dp[0] & 1)) {
104  /* u = u/2. */
105  fp_rsh1_low(u->dp, u->dp);
106  /* If g1 is even then g1 = g1/2; else g1 = (g1 + p)/2. */
107  if (g1->dp[0] & 1) {
108  bn_add(g1, g1, p);
109  }
110  bn_hlv(g1, g1);
111  }
112 
113  while (u->dp[u->used - 1] == 0) {
114  u->used--;
115  }
116  if (u->used == 1 && u->dp[0] == 1)
117  break;
118 
119  /* While z divides v do. */
120  while (!(v->dp[0] & 1)) {
121  /* v = v/2. */
122  fp_rsh1_low(v->dp, v->dp);
123  /* If g2 is even then g2 = g2/2; else (g2 = g2 + p)/2. */
124  if (g2->dp[0] & 1) {
125  bn_add(g2, g2, p);
126  }
127  bn_hlv(g2, g2);
128  }
129 
130  while (v->dp[v->used - 1] == 0) {
131  v->used--;
132  }
133  if (v->used == 1 && v->dp[0] == 1)
134  break;
135 
136  /* If u > v then u = u - v, g1 = g1 - g2. */
137  if (bn_cmp(u, v) == RLC_GT) {
138  bn_sub(u, u, v);
139  bn_sub(g1, g1, g2);
140  } else {
141  bn_sub(v, v, u);
142  bn_sub(g2, g2, g1);
143  }
144  }
145  /* If u == 1 then return g1; else return g2. */
146  if (bn_cmp_dig(u, 1) == RLC_EQ) {
147  while (bn_sign(g1) == RLC_NEG) {
148  bn_add(g1, g1, p);
149  }
150  while (bn_cmp(g1, p) != RLC_LT) {
151  bn_sub(g1, g1, p);
152  }
153 #if FP_RDC == MONTY
154  fp_prime_conv(c, g1);
155 #else
156  dv_copy(c, g1->dp, RLC_FP_DIGS);
157 #endif
158  } else {
159  while (bn_sign(g2) == RLC_NEG) {
160  bn_add(g2, g2, p);
161  }
162  while (bn_cmp(g2, p) != RLC_LT) {
163  bn_sub(g2, g2, p);
164  }
165 #if FP_RDC == MONTY
166  fp_prime_conv(c, g2);
167 #else
168  dv_copy(c, g2->dp, RLC_FP_DIGS);
169 #endif
170  }
171  }
172  CATCH_ANY {
173  THROW(ERR_CAUGHT);
174  }
175  FINALLY {
176  bn_free(u);
177  bn_free(v);
178  bn_free(g1);
179  bn_free(g2);
180  bn_free(p);
181  }
182 }

188 void fp_inv_monty(fp_t c, const fp_t a) {
189  bn_t _a, _p, u, v, x1, x2;
190  const dig_t *p = NULL;
191  dig_t carry;
192  int i, k, flag = 0;
193 
194  bn_null(_a);
195  bn_null(_p);
196  bn_null(u);
197  bn_null(v);
198  bn_null(x1);
199  bn_null(x2);
200 
201  if (fp_is_zero(a)) {
202  THROW(ERR_NO_VALID);
203  }
204 
205  TRY {
206  bn_new(_a);
207  bn_new(_p);
208  bn_new(u);
209  bn_new(v);
210  bn_new(x1);
211  bn_new(x2);
212 
213  p = fp_prime_get();
214 
215  /* u = a, v = p, x1 = 1, x2 = 0, k = 0. */
216  k = 0;
217  bn_set_dig(x1, 1);
218  bn_zero(x2);
219 
220 #if FP_RDC != MONTY
221  bn_read_raw(_a, a, RLC_FP_DIGS);
222  bn_read_raw(_p, p, RLC_FP_DIGS);
223  bn_mod_monty_conv(u, _a, _p);
224 #else
225  bn_read_raw(u, a, RLC_FP_DIGS);
226 #endif
227  bn_read_raw(v, p, RLC_FP_DIGS);
228 
229  while (!bn_is_zero(v)) {
230  /* If v is even then v = v/2, x1 = 2 * x1. */
231  if (!(v->dp[0] & 1)) {
232  fp_rsh1_low(v->dp, v->dp);
233  bn_dbl(x1, x1);
234  } else {
235  /* If u is even then u = u/2, x2 = 2 * x2. */
236  if (!(u->dp[0] & 1)) {
237  fp_rsh1_low(u->dp, u->dp);
238  bn_dbl(x2, x2);
239  /* If v >= u,then v = (v - u)/2, x2 += x1, x1 = 2 * x1. */
240  } else {
241  if (bn_cmp(v, u) != RLC_LT) {
242  fp_subn_low(v->dp, v->dp, u->dp);
243  fp_rsh1_low(v->dp, v->dp);
244  bn_add(x2, x2, x1);
245  bn_dbl(x1, x1);
246  } else {
247  /* u = (u - v)/2, x1 += x2, x2 = 2 * x2. */
248  fp_subn_low(u->dp, u->dp, v->dp);
249  fp_rsh1_low(u->dp, u->dp);
250  bn_add(x1, x1, x2);
251  bn_dbl(x2, x2);
252  }
253  }
254  }
255  bn_trim(u);
256  bn_trim(v);
257  k++;
258  }
259 
260  /* If x1 > p then x1 = x1 - p. */
261  for (i = x1->used; i < RLC_FP_DIGS; i++) {
262  x1->dp[i] = 0;
263  }
264 
265  while (x1->used > RLC_FP_DIGS) {
266  carry = bn_subn_low(x1->dp, x1->dp, fp_prime_get(), RLC_FP_DIGS);
267  bn_sub1_low(x1->dp + RLC_FP_DIGS, x1->dp + RLC_FP_DIGS, carry,
268  x1->used - RLC_FP_DIGS);
269  bn_trim(x1);
270  }
271  if (dv_cmp(x1->dp, fp_prime_get(), RLC_FP_DIGS) == RLC_GT) {
272  fp_subn_low(x1->dp, x1->dp, fp_prime_get());
273  }
274 
275  dv_copy(x2->dp, fp_prime_get_conv(), RLC_FP_DIGS);
276 
277  /* If k < Wt then x1 = x1 * R^2 * R^{-1} mod p. */
278  if (k <= RLC_FP_DIGS * RLC_DIG) {
279  flag = 1;
280  fp_mul(x1->dp, x1->dp, x2->dp);
281  k = k + RLC_FP_DIGS * RLC_DIG;
282  }
283 
284  /* x1 = x1 * R^2 * R^{-1} mod p. */
285  fp_mul(x1->dp, x1->dp, x2->dp);
286 
287  /* c = x1 * 2^(2Wt - k) * R^{-1} mod p. */
288  fp_copy(c, x1->dp);
289  dv_zero(x1->dp, RLC_FP_DIGS);
290  bn_set_2b(x1, 2 * RLC_FP_DIGS * RLC_DIG - k);
291  fp_mul(c, c, x1->dp);
292 
293 #if FP_RDC != MONTY
294  /*
295  * If we do not use Montgomery reduction, the result of inversion is
296  * a^{-1}R^3 mod p or a^{-1}R^4 mod p, depending on flag.
297  * Hence we must reduce the result three or four times.
298  */
299  _a->used = RLC_FP_DIGS;
300  dv_copy(_a->dp, c, RLC_FP_DIGS);
301  bn_mod_monty_back(_a, _a, _p);
302  bn_mod_monty_back(_a, _a, _p);
303  bn_mod_monty_back(_a, _a, _p);
304 
305  if (flag) {
306  bn_mod_monty_back(_a, _a, _p);
307  }
308  fp_zero(c);
309  dv_copy(c, _a->dp, _a->used);
310 #endif
311  (void)flag;
312  }
313  CATCH_ANY {
314  THROW(ERR_CAUGHT);
315  }
316  FINALLY {
317  bn_free(_a);
318  bn_free(_p);
319  bn_free(u);
320  bn_free(v);
321  bn_free(x1);
322  bn_free(x2);
323  }
324 }

330 void fp_inv_exgcd(fp_t c, const fp_t a) {
331  bn_t u, v, g1, g2, p, q, r;
332 
333  bn_null(u);
334  bn_null(v);
335  bn_null(g1);
336  bn_null(g2);
337  bn_null(p);
338  bn_null(q);
339  bn_null(r);
340 
341  if (fp_is_zero(a)) {
342  THROW(ERR_NO_VALID);
343  }
344 
345  TRY {
346  bn_new(u);
347  bn_new(v);
348  bn_new(g1);
349  bn_new(g2);
350  bn_new(p);
351  bn_new(q);
352  bn_new(r);
353 
354  /* u = a, v = p, g1 = 1, g2 = 0. */
355  fp_prime_back(u, a);
356  p->used = RLC_FP_DIGS;
357  dv_copy(p->dp, fp_prime_get(), RLC_FP_DIGS);
358  bn_copy(v, p);
359  bn_set_dig(g1, 1);
360  bn_zero(g2);
361 
362  /* While (u != 1. */
363  while (bn_cmp_dig(u, 1) != RLC_EQ) {
364  /* q = [v/u], r = v mod u. */
365  bn_div_rem(q, r, v, u);
366  /* v = u, u = r. */
367  bn_copy(v, u);
368  bn_copy(u, r);
369  /* r = g2 - q * g1. */
370  bn_mul(r, q, g1);
371  bn_sub(r, g2, r);
372  /* g2 = g1, g1 = r. */
373  bn_copy(g2, g1);
374  bn_copy(g1, r);
375  }
376 
377  if (bn_sign(g1) == RLC_NEG) {
378  bn_add(g1, g1, p);
379  }
380  fp_prime_conv(c, g1);
381  }
382  CATCH_ANY {
383  THROW(ERR_CAUGHT);
384  }
385  FINALLY {
386  bn_free(u);
387  bn_free(v);
388  bn_free(g1);
389  bn_free(g2);
390  bn_free(p);
391  bn_free(q);
392  bn_free(r);
393  }
394 }

402 void fp_inv_divst(fp_t c, const fp_t a) {
403  /* Compute number of iteratios based on modulus size. */
404 #if FP_PRIME < 46
405  int d = (49 * FP_PRIME + 80)/17;
406 #else
407  int d = (49 * FP_PRIME + 57)/17;
408 #endif
409  int delta = 1, g0, d0;
410  dig_t fs, gs;
411  bn_t _t;
412  dv_t f, g, t, u;
413  fp_t precomp, v, r;
414 
415  bn_null(_t);
416  dv_null(f);
417  dv_null(g);
418  dv_null(t);
419  dv_null(u);
420  fp_null(v);
421  fp_null(r);
422  fp_null(precomp);
423 
424  if (fp_is_zero(a)) {
425  THROW(ERR_NO_VALID);
426  }
427 
428  TRY {
429  bn_new(_t);
430  dv_new(f);
431  dv_new(g);
432  dv_new(t);
433  dv_new(u);
434  fp_new(v);
435  fp_new(r);
436  fp_new(precomp);
437 
438  bn_set_dig(_t, d);
439  dv_copy(precomp, fp_prime_get(), RLC_FP_DIGS);
440  fp_add_dig(precomp, precomp, 1);
441  fp_hlv(precomp, precomp);
442  fp_exp(precomp, precomp, _t);
443 
444  fp_zero(v);
445  fp_set_dig(r, 1);
446  fp_prime_back(_t, a);
447  dv_zero(g, RLC_FP_DIGS);
448  dv_copy(g, _t->dp, _t->used);
449  dv_copy(f, fp_prime_get(), RLC_FP_DIGS);
450  fs = gs = RLC_POS;
451 
452  for (int i = 0; i < d; i++) {
453  g0 = g[0] & 1;
454  d0 = g0 & (delta > 0);
455  delta = RLC_SEL(delta, -delta, d0);
456  /* Conditionally swap based on d0. */
457  dv_swap_cond(r, v, RLC_FP_DIGS, d0);
458  fp_negm_low(t, r);
459  dv_swap_cond(f, g, RLC_FP_DIGS, d0);
460  dv_copy_cond(r, t, RLC_FP_DIGS, d0);
461  for (int j = 0; j < RLC_FP_DIGS; j++) {
462  g[j] = RLC_SEL(g[j], ~g[j], d0);
463  }
464  fp_add1_low(g, g, d0);
465  t[0] = (fs ^ gs) & (-d0);
466  fs ^= t[0];
467  gs ^= t[0] ^ d0;
468 
469  delta++;
470  g0 = g[0] & 1;
471  for (int j = 0; j < RLC_FP_DIGS; j++) {
472  t[j] = v[j] & (-g0);
473  u[j] = f[j] & (-g0);
474  }
475  fp_addm_low(r, r, t);
476  fp_dbl(v, v);
477 
478  /* Compute g = (g + g0*f) div 2 by conditionally copying f to u and
479  * updating the sign of g. */
480  gs ^= g0 & (fs ^ fp_addn_low(g, g, u));
481  /* Shift and restore the sign. */
482  fp_rsh1_low(g, g);
483  g[RLC_FP_DIGS - 1] |= (dig_t)gs << (RLC_DIG - 1);
484  }
485  fp_neg(t, v);
486  dv_copy_cond(v, t, RLC_FP_DIGS, fs);
487  fp_mul(c, v, precomp);
488  } CATCH_ANY {
489  THROW(ERR_CAUGHT)
490  } FINALLY {
491  bn_free(_t);
492  dv_free(f);
493  dv_free(g);
494  dv_free(t);
495  dv_free(u);
496  fp_free(v);
497  fp_free(r);
498  fp_free(precomp);
499  }
500 }

514  fp_t u, *t = RLC_ALLOCA(fp_t, n);
515 
516  fp_null(u);
517 
518  TRY {

520  THROW(ERR_NO_MEMORY);
521  }

523  fp_null(t[i]);
524  fp_new(t[i]);
525  }

527

530

532  fp_copy(t[i], a[i]);

534  }
535 
536  fp_inv(u, c[n - 1]);
537 
538  for (i = n - 1; i > 0; i--) {
539  fp_mul(c[i], u, c[i - 1]);

543  }

546  }

553  }
