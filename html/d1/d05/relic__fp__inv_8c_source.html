44static void bn_mul2_low(dig_t *c, const dig_t *a, dis_t digit, size_t size) {
45 int sd = digit >> (RLC_DIG - 1);
46 digit = (digit ^ sd) - sd;
47 c[size] = bn_mul1_low(c, a, digit, size);
48}

52static dis_t jumpdivstep(dis_t m[4], dis_t delta, dig_t f, dig_t g, int s) {
53 dig_t u = 1, v = 0, q = 0, r = 1, c0, c1;
54
55 /* This is actually faster than my previous version, several tricks from
56 * https://github.com/bitcoin-core/secp256k1/blob/master/src/modinv64_impl.h
57 */
58 for (s--; s >= 0; s--) {
59 /* First handle the else part: if delta < 0, compute -(f,u,v). */
60 c0 = delta >> (RLC_DIG - 1);
61 c1 = -(g & 1);
62 c0 &= c1;
63 /* Conditionally add -(f,u,v) to (g,q,r) */
64 g += ((f ^ c0) - c0) & c1;
65 q += ((u ^ c0) - c0) & c1;
66 r += ((v ^ c0) - c0) & c1;
67 /* Now handle the 'if' part, so c0 will be (delta < 0) && (g & 1)) */
68 /* delta = RLC_SEL(delta, -delta, c0 & 1) - 2 (for half-divstep), thus
69 * delta = - delta - 2 or delta - 1 */
70 delta = (delta ^ c0) - 1;
71 f = f + (g & c0);
72 u = u + (q & c0);
73 v = v + (r & c0);
74 g >>= 1;
75 u += u;
76 v += v;
77 }
78 m[0] = u;
79 m[1] = v;
80 m[2] = q;
81 m[3] = r;
82 return delta;
83}

124void fp_inv_binar(fp_t c, const fp_t a) {
125 bn_t u, v, g1, g2, p;
126
127 bn_null(u);
128 bn_null(v);
129 bn_null(g1);
130 bn_null(g2);
131 bn_null(p);
132
133 if (fp_is_zero(a)) {
134 RLC_THROW(ERR_NO_VALID);
135 return;
136 }
137
138 RLC_TRY {
139 bn_new(u);
140 bn_new(v);
141 bn_new(g1);
142 bn_new(g2);
143 bn_new(p);
144
145 /* u = a, v = p, g1 = 1, g2 = 0. */
146 fp_prime_back(u, a);
147 p->used = RLC_FP_DIGS;
148 dv_copy(p->dp, fp_prime_get(), RLC_FP_DIGS);
149 bn_copy(v, p);
150 bn_set_dig(g1, 1);
151 bn_zero(g2);
152
153 /* While (u != 1 && v != 1). */
154 while (1) {
155 /* While u is even do. */
156 while (!(u->dp[0] & 1)) {
157 /* u = u/2. */
158 fp_rsh1_low(u->dp, u->dp);
159 /* If g1 is even then g1 = g1/2; else g1 = (g1 + p)/2. */
160 if (g1->dp[0] & 1) {
161 bn_add(g1, g1, p);
162 }
163 bn_hlv(g1, g1);
164 }
165
166 while (u->dp[u->used - 1] == 0) {
167 u->used--;
168 }
169 if (u->used == 1 && u->dp[0] == 1)
170 break;
171
172 /* While z divides v do. */
173 while (!(v->dp[0] & 1)) {
174 /* v = v/2. */
175 fp_rsh1_low(v->dp, v->dp);
176 /* If g2 is even then g2 = g2/2; else (g2 = g2 + p)/2. */
177 if (g2->dp[0] & 1) {
178 bn_add(g2, g2, p);
179 }
180 bn_hlv(g2, g2);
181 }
182
183 while (v->dp[v->used - 1] == 0) {
184 v->used--;
185 }
186 if (v->used == 1 && v->dp[0] == 1)
187 break;
188
189 /* If u > v then u = u - v, g1 = g1 - g2. */
190 if (bn_cmp(u, v) == RLC_GT) {
191 bn_sub(u, u, v);
192 bn_sub(g1, g1, g2);
193 } else {
194 bn_sub(v, v, u);
195 bn_sub(g2, g2, g1);
196 }
197 }
198 /* If u == 1 then return g1; else return g2. */
199 if (bn_cmp_dig(u, 1) == RLC_EQ) {
200 while (bn_sign(g1) == RLC_NEG) {
201 bn_add(g1, g1, p);
202 }
203 while (bn_cmp(g1, p) != RLC_LT) {
204 bn_sub(g1, g1, p);
205 }
206#if FP_RDC == MONTY
207 fp_prime_conv(c, g1);
208#else
209 dv_copy(c, g1->dp, RLC_FP_DIGS);
210#endif
211 } else {
212 while (bn_sign(g2) == RLC_NEG) {
213 bn_add(g2, g2, p);
214 }
215 while (bn_cmp(g2, p) != RLC_LT) {
216 bn_sub(g2, g2, p);
217 }
218#if FP_RDC == MONTY
219 fp_prime_conv(c, g2);
220#else
221 dv_copy(c, g2->dp, RLC_FP_DIGS);
222#endif
223 }
224 }
225 RLC_CATCH_ANY {
226 RLC_THROW(ERR_CAUGHT);
227 }
228 RLC_FINALLY {
229 bn_free(u);
230 bn_free(v);
231 bn_free(g1);
232 bn_free(g2);
233 bn_free(p);
234 }
235}

241void fp_inv_monty(fp_t c, const fp_t a) {
242 bn_t _a, _p, u, v, x1, x2;
243 const dig_t *p = NULL;
244 dig_t carry;
245 int i, k;
246
247 bn_null(_a);
248 bn_null(_p);
249 bn_null(u);
250 bn_null(v);
251 bn_null(x1);
252 bn_null(x2);
253
254 if (fp_is_zero(a)) {
255 RLC_THROW(ERR_NO_VALID);
256 return;
257 }
258
259 RLC_TRY {
260 bn_new(_a);
261 bn_new(_p);
262 bn_new(u);
263 bn_new(v);
264 bn_new(x1);
265 bn_new(x2);
266
267 p = fp_prime_get();
268
269 /* u = a, v = p, x1 = 1, x2 = 0, k = 0. */
270 k = 0;
271 bn_set_dig(x1, 1);
272 bn_zero(x2);
273
274#if FP_RDC != MONTY
275 bn_read_raw(_a, a, RLC_FP_DIGS);
276 bn_read_raw(_p, p, RLC_FP_DIGS);
277 bn_mod_monty_conv(u, _a, _p);
278#else
279 bn_read_raw(u, a, RLC_FP_DIGS);
280#endif
281 bn_read_raw(v, p, RLC_FP_DIGS);
282
283 while (!bn_is_zero(v)) {
284 /* If v is even then v = v/2, x1 = 2 * x1. */
285 if (!(v->dp[0] & 1)) {
286 fp_rsh1_low(v->dp, v->dp);
287 bn_dbl(x1, x1);
288 } else {
289 /* If u is even then u = u/2, x2 = 2 * x2. */
290 if (!(u->dp[0] & 1)) {
291 fp_rsh1_low(u->dp, u->dp);
292 bn_dbl(x2, x2);
293 /* If v >= u,then v = (v - u)/2, x2 += x1, x1 = 2 * x1. */
294 } else {
295 if (bn_cmp(v, u) != RLC_LT) {
296 fp_subn_low(v->dp, v->dp, u->dp);
297 fp_rsh1_low(v->dp, v->dp);
298 bn_add(x2, x2, x1);
299 bn_dbl(x1, x1);
300 } else {
301 /* u = (u - v)/2, x1 += x2, x2 = 2 * x2. */
302 fp_subn_low(u->dp, u->dp, v->dp);
303 fp_rsh1_low(u->dp, u->dp);
304 bn_add(x1, x1, x2);
305 bn_dbl(x2, x2);
306 }
307 }
308 }
309 bn_trim(u);
310 bn_trim(v);
311 k++;
312 }
313
314 /* If x1 > p then x1 = x1 - p. */
315 for (i = x1->used; i < RLC_FP_DIGS; i++) {
316 x1->dp[i] = 0;
317 }
318
319 while (x1->used > RLC_FP_DIGS) {
320 carry = bn_subn_low(x1->dp, x1->dp, fp_prime_get(), RLC_FP_DIGS);
321 bn_sub1_low(x1->dp + RLC_FP_DIGS, x1->dp + RLC_FP_DIGS, carry,
322 x1->used - RLC_FP_DIGS);
323 bn_trim(x1);
324 }
325 if (dv_cmp(x1->dp, fp_prime_get(), RLC_FP_DIGS) == RLC_GT) {
326 fp_subn_low(x1->dp, x1->dp, fp_prime_get());
327 }
328
329 /* If k < Wt then x1 = x1 * R^2 * R^{-1} mod p. */
330 if (k <= RLC_FP_DIGS * RLC_DIG) {
331 k = k + RLC_FP_DIGS * RLC_DIG;
332#if FP_RDC == MONTY
333 fp_mul(x1->dp, x1->dp, fp_prime_get_conv());
334#endif
335 }
336
337#if FP_RDC == MONTY
338 /* x1 = x1 * R^2 * R^{-1} mod p. */
339 fp_mul(x1->dp, x1->dp, fp_prime_get_conv());
340#endif
341 /* c = x1 * 2^(2Wt - k) * R^{-1} mod p. */
342 fp_copy(c, x1->dp);
343 dv_zero(x1->dp, RLC_FP_DIGS);
344 bn_set_2b(x1, 2 * RLC_FP_DIGS * RLC_DIG - k);
345 fp_mul(c, c, x1->dp);
346
347#if FP_RDC != MONTY
348 /*
349 * If we do not use Montgomery reduction, convert back.
350 */
351 bn_read_raw(_a, c, RLC_FP_DIGS);
352 bn_mod_monty_back(_a, _a, _p);
353
354 fp_zero(c);
355 dv_copy(c, _a->dp, _a->used);
356#endif
357 }
358 RLC_CATCH_ANY {
359 RLC_THROW(ERR_CAUGHT);
360 }
361 RLC_FINALLY {
362 bn_free(_a);
363 bn_free(_p);
364 bn_free(u);
365 bn_free(v);
366 bn_free(x1);
367 bn_free(x2);
368 }
369}

375void fp_inv_exgcd(fp_t c, const fp_t a) {
376 bn_t u, v, g1, g2, p, q, r;
377
378 bn_null(u);
379 bn_null(v);
380 bn_null(g1);
381 bn_null(g2);
382 bn_null(p);
383 bn_null(q);
384 bn_null(r);
385
386 if (fp_is_zero(a)) {
387 RLC_THROW(ERR_NO_VALID);
388 return;
389 }
390
391 RLC_TRY {
392 bn_new(u);
393 bn_new(v);
394 bn_new(g1);
395 bn_new(g2);
396 bn_new(p);
397 bn_new(q);
398 bn_new(r);
399
400 /* u = a, v = p, g1 = 1, g2 = 0. */
401 fp_prime_back(u, a);
402 p->used = RLC_FP_DIGS;
403 dv_copy(p->dp, fp_prime_get(), RLC_FP_DIGS);
404 bn_copy(v, p);
405 bn_set_dig(g1, 1);
406 bn_zero(g2);
407
408 /* While (u != 1. */
409 while (bn_cmp_dig(u, 1) != RLC_EQ) {
410 /* q = [v/u], r = v mod u. */
411 bn_div_rem(q, r, v, u);
412 /* v = u, u = r. */
413 bn_copy(v, u);
414 bn_copy(u, r);
415 /* r = g2 - q * g1. */
416 bn_mul(r, q, g1);
417 bn_sub(r, g2, r);
418 /* g2 = g1, g1 = r. */
419 bn_copy(g2, g1);
420 bn_copy(g1, r);
421 }
422
423 if (bn_sign(g1) == RLC_NEG) {
424 bn_add(g1, g1, p);
425 }
426 fp_prime_conv(c, g1);
427 }
428 RLC_CATCH_ANY {
429 RLC_THROW(ERR_CAUGHT);
430 }
431 RLC_FINALLY {
432 bn_free(u);
433 bn_free(v);
434 bn_free(g1);
435 bn_free(g2);
436 bn_free(p);
437 bn_free(q);
438 bn_free(r);
439 }
440}

446void fp_inv_divst(fp_t c, const fp_t a) {
447 /* Compute number of iterations based on modulus size. */
448#if FP_PRIME < 46
449 const int d = (49 * FP_PRIME + 80) / 17;
450#else
451 const int d = (49 * FP_PRIME + 57) / 17;
452#endif
453 int g0, d0;
454 dig_t fs, gs, delta = 1;
455 bn_t _t;
456 fp_t f, g, u, v, r;
457 dv_t t;
458
459 bn_null(_t);
460 dv_null(t);
461 fp_null(f);
462 fp_null(g);
463 fp_null(u);
464 fp_null(v);
465 fp_null(r);
466
467 if (fp_is_zero(a)) {
468 RLC_THROW(ERR_NO_VALID);
469 return;
470 }
471
472 RLC_TRY {
473 bn_new(_t);
474 dv_new(t);
475 fp_new(f);
476 fp_new(g);
477 fp_new(u);
478 fp_new(v);
479 fp_new(r);
480
481 fp_zero(v);
482 fp_set_dig(r, 1);
483 dv_copy(f, fp_prime_get(), RLC_FP_DIGS);
484#if FP_RDC == MONTY
485 /* Convert a from Montgomery form. */
486 dv_zero(t, 2 * RLC_FP_DIGS);
487 fp_copy(t, a);
488 fp_rdcn_low(g, t);
489#else
490 fp_copy(g, a);
491#endif
492 fs = gs = RLC_POS;
493
494 for (int i = 0; i < d; i++) {
495 g0 = g[0] & 1;
496 d0 = delta >> (RLC_DIG - 1);
497 d0 = g0 & ~d0;
498 /* Conditionally negate delta if d0 is set. */
499 delta = (delta ^ -d0) + d0;
500 /* Conditionally swap based on d0. */
501 dv_swap_sec(r, v, RLC_FP_DIGS, d0);
502 fp_negm_low(t, r);
503 dv_swap_sec(f, g, RLC_FP_DIGS, d0);
504 dv_copy_sec(r, t, RLC_FP_DIGS, d0);
505 for (int j = 0; j < RLC_FP_DIGS; j++) {
506 g[j] = RLC_SEL(g[j], ~g[j], d0);
507 }
508 fp_add1_low(g, g, d0);
509 t[0] = (fs ^ gs) & (-d0);
510 fs ^= t[0];
511 gs ^= t[0] ^ d0;
512
513 delta++;
514 g0 = g[0] & 1;
515 for (int j = 0; j < RLC_FP_DIGS; j++) {
516 t[j] = v[j] & (-g0);
517 u[j] = f[j] & (-g0);
518 }
519 fp_addm_low(r, r, t);
520 fp_dblm_low(v, v);
521
522 /* Compute g = (g + g0*f) div 2 by conditionally copying f to u and
523 * updating the sign of g. */
524 gs ^= g0 & (fs ^ bn_addn_low(g, g, u, RLC_FP_DIGS));
525 /* Shift and restore the sign. */
526 fp_rsh1_low(g, g);
527 g[RLC_FP_DIGS - 1] |= (dig_t)gs << (RLC_DIG - 1);
528 }
529 fp_neg(t, v);
530 fp_copy_sec(v, t, fs);
531
532 dv_copy(t, fp_prime_get(), RLC_FP_DIGS);
533 fp_add_dig(t, t, 1);
534 fp_hlv(t, t);
535#if WSIZE == 8
536 bn_set_dig(_t, d >> 8);
537 bn_lsh(_t, _t, 8);
538 bn_add_dig(_t, _t, d & 0xFF);
539#else
540 bn_set_dig(_t, d);
541#endif
542 fp_exp(t, t, _t);
543
544 fp_mul(c, v, t);
545 } RLC_CATCH_ANY {
546 RLC_THROW(ERR_CAUGHT)
547 } RLC_FINALLY {
548 bn_free(_t);
549 dv_free(t);
550 fp_free(f);
551 fp_free(g);
552 fp_free(u);
553 fp_free(v);
554 fp_free(r);
555 }
556}

562void fp_inv_jmpds(fp_t c, const fp_t a) {
563 dis_t m[4], d = -1;
564 /* Compute number of iterations based on modulus size. */
565 /* Iterations taken directly from https://github.com/sipa/safegcd-bounds */
566 const int iterations = (45907 * FP_PRIME + 26313) / 19929;
567 int loops, i, j = 0, s = RLC_DIG - 2;
568 dv_t f, g, t, p, t0, t1, u0, u1, v0, v1, p01, p11;
569 dig_t sf, sg;
570 fp_t pre;
571
572 if (fp_is_zero(a)) {
573 RLC_THROW(ERR_NO_VALID);
574 return;
575 }
576
577 dv_null(f);
578 dv_null(g);
579 dv_null(t);
580 dv_null(p);
581 dv_null(t0);
582 dv_null(t1);
583 dv_null(u0);
584 dv_null(u1);
585 dv_null(v0);
586 dv_null(v1);
587 dv_null(p01);
588 dv_null(p11);
589 fp_null(pre);
590
591 RLC_TRY {
592 dv_new(t0);
593 dv_new(f);
594 dv_new(t);
595 dv_new(p);
596 dv_new(g);
597 dv_new(t1);
598 dv_new(u0);
599 dv_new(u1);
600 dv_new(v0);
601 dv_new(v1);
602 dv_new(p01);
603 dv_new(p11);
604 fp_new(pre);
605
606 fp_copy(pre, core_get()->inv.dp);
607 dv_zero(t, 2 * RLC_FP_DIGS);
608 dv_zero(p, 2 * RLC_FP_DIGS);
609 dv_zero(u0, 2 * RLC_FP_DIGS);
610 dv_zero(u1, 2 * RLC_FP_DIGS);
611 dv_zero(v0, 2 * RLC_FP_DIGS);
612 dv_zero(v1, 2 * RLC_FP_DIGS);
613 dv_copy(f, fp_prime_get(), RLC_FP_DIGS);
614#if FP_RDC == MONTY
615 /* Convert a from Montgomery form. */
616 fp_copy(p, a);
617 fp_rdcn_low(g, p);
618#else
619 fp_copy(g, a);
620#endif
621 f[RLC_FP_DIGS] = g[RLC_FP_DIGS] = 0;
622 d = jumpdivstep(m, d, f[0] & RLC_MASK(s), g[0] & RLC_MASK(s), s);
623
624 t0[RLC_FP_DIGS] = bn_muls_low(t0, f, RLC_POS, m[0], RLC_FP_DIGS);
625 t1[RLC_FP_DIGS] = bn_muls_low(t1, g, RLC_POS, m[1], RLC_FP_DIGS);
626 bn_addn_low(t0, t0, t1, RLC_FP_DIGS + 1);
627
628 f[RLC_FP_DIGS] = bn_muls_low(f, f, RLC_POS, m[2], RLC_FP_DIGS);
629 t1[RLC_FP_DIGS] = bn_muls_low(t1, g, RLC_POS, m[3], RLC_FP_DIGS);
630 bn_addn_low(t1, t1, f, RLC_FP_DIGS + 1);
631
632 /* Update f and g. */
633 bn_rshs_low(f, t0, RLC_FP_DIGS + 1, s);
634 bn_rshs_low(g, t1, RLC_FP_DIGS + 1, s);
635
636 /* Update column vector below. */
637 v1[0] = RLC_SEL(m[1], -m[1], RLC_SIGN(m[1]));
638 fp_negm_low(t1, v1);
639 fp_copy_sec(v1, t1, RLC_SIGN(m[1]));
640 u1[0] = RLC_SEL(m[3], -m[3], RLC_SIGN(m[3]));
641 fp_negm_low(t1, u1);
642 fp_copy_sec(u1, t1, RLC_SIGN(m[3]));
643
644 dv_copy(p01, v1, 2 * RLC_FP_DIGS);
645 dv_copy(p11, u1, 2 * RLC_FP_DIGS);
646
647 loops = iterations / s;
648 loops = (iterations % s == 0 ? loops - 1 : loops);
649
650 for (i = 1; i < loops; i++) {
651 d = jumpdivstep(m, d, f[0] & RLC_MASK(s), g[0] & RLC_MASK(s), s);
652
653 sf = RLC_SIGN(f[RLC_FP_DIGS]);
654 sg = RLC_SIGN(g[RLC_FP_DIGS]);
655 bn_negs_low(u0, f, sf, RLC_FP_DIGS);
656 bn_negs_low(u1, g, sg, RLC_FP_DIGS);
657
658 t0[RLC_FP_DIGS] = bn_muls_low(t0, u0, sf, m[0], RLC_FP_DIGS);
659 t1[RLC_FP_DIGS] = bn_muls_low(t1, u1, sg, m[1], RLC_FP_DIGS);
660 bn_addn_low(t0, t0, t1, RLC_FP_DIGS + 1);
661 bn_rshs_low(f, t0, RLC_FP_DIGS + 1, s);
662
663 t0[RLC_FP_DIGS] = bn_muls_low(t0, u0, sf, m[2], RLC_FP_DIGS);
664 t1[RLC_FP_DIGS] = bn_muls_low(t1, u1, sg, m[3], RLC_FP_DIGS);
665 bn_addn_low(t1, t1, t0, RLC_FP_DIGS + 1);
666 bn_rshs_low(g, t1, RLC_FP_DIGS + 1, s);
667
668#ifdef RLC_FP_ROOM
669 p[j] = 0;
670 dv_copy(p + j + 1, fp_prime_get(), RLC_FP_DIGS);
671
672 /* Update column vector below. */
673 bn_mul2_low(v0, p01, m[0], RLC_FP_DIGS + j);
674 fp_subd_low(t, p, v0);
675 dv_copy_sec(v0, t, RLC_FP_DIGS + j + 1, RLC_SIGN(m[0]));
676
677 bn_mul2_low(v1, p11, m[1], RLC_FP_DIGS + j);
678 fp_subd_low(t, p, v1);
679 dv_copy_sec(v1, t, RLC_FP_DIGS + j + 1, RLC_SIGN(m[1]));
680
681 bn_mul2_low(u0, p01, m[2], RLC_FP_DIGS + j);
682 fp_subd_low(t, p, u0);
683 dv_copy_sec(u0, t, RLC_FP_DIGS + j + 1, RLC_SIGN(m[2]));
684
685 bn_mul2_low(u1, p11, m[3], RLC_FP_DIGS + j);
686 fp_subd_low(t, p, u1);
687 dv_copy_sec(u1, t, RLC_FP_DIGS + j + 1, RLC_SIGN(m[3]));
688
689 j = i % RLC_FP_DIGS;
690 if (j == 0) {
691 fp_addd_low(t, u0, u1);
692 fp_rdc(p11, t);
693 fp_addd_low(t, v0, v1);
694 fp_rdc(p01, t);
695 dv_zero(v0, 2 * RLC_FP_DIGS);
696 dv_zero(v1, 2 * RLC_FP_DIGS);
697 } else {
698 fp_addd_low(p11, u0, u1);
699 fp_addd_low(p01, v0, v1);
700 }
701#else
702 /* Update column vector below. */
703 t[0] = RLC_SEL(m[0], -m[0], RLC_SIGN(m[0]));
704 fp_mul(v0, p01, t);
705 fp_neg(t0, v0);
706 fp_copy_sec(v0, t0, RLC_SIGN(m[0]));
707
708 t[0] = RLC_SEL(m[1], -m[1], RLC_SIGN(m[1]));
709 fp_mul(v1, p11, t);
710 fp_neg(t1, v1);
711 fp_copy_sec(v1, t1, RLC_SIGN(m[1]));
712
713 t[0] = RLC_SEL(m[2], -m[2], RLC_SIGN(m[2]));
714 fp_mul(u0, p01, t);
715 fp_neg(t0, u0);
716 fp_copy_sec(u0, t0, RLC_SIGN(m[2]));
717
718 t[0] = RLC_SEL(m[3], -m[3], RLC_SIGN(m[3]));
719 fp_mul(u1, p11, t);
720 fp_neg(t1, u1);
721 fp_copy_sec(u1, t1, RLC_SIGN(m[3]));
722
723 fp_add(p11, u0, u1);
724 fp_add(p01, v0, v1);
725#endif
726 }
727
728 s = iterations - loops * s;
729 d = jumpdivstep(m, d, f[0] & RLC_MASK(s), g[0] & RLC_MASK(s), s);
730
731 sf = RLC_SIGN(f[RLC_FP_DIGS]);
732 sg = RLC_SIGN(g[RLC_FP_DIGS]);
733 bn_negs_low(u0, f, sf, RLC_FP_DIGS);
734 bn_negs_low(u1, g, sg, RLC_FP_DIGS);
735
736 t0[RLC_FP_DIGS] = bn_muls_low(t0, u0, sf, m[0], RLC_FP_DIGS);
737 t1[RLC_FP_DIGS] = bn_muls_low(t1, u1, sg, m[1], RLC_FP_DIGS);
738 bn_addn_low(t0, t0, t1, RLC_FP_DIGS + 1);
739 bn_rshs_low(f, t0, RLC_FP_DIGS + 1, s);
740
741 t0[RLC_FP_DIGS] = bn_muls_low(t0, u0, sf, m[2], RLC_FP_DIGS);
742 t1[RLC_FP_DIGS] = bn_muls_low(t1, u1, sg, m[3], RLC_FP_DIGS);
743 bn_addn_low(t1, t1, t0, RLC_FP_DIGS + 1);
744 bn_rshs_low(g, t1, RLC_FP_DIGS + 1, s);
745
746#ifdef RLC_FP_ROOM
747 p[j] = 0;
748 dv_copy(p + j + 1, fp_prime_get(), RLC_FP_DIGS);
749
750 /* Update column vector below. */
751 bn_mul2_low(v0, p01, m[0], RLC_FP_DIGS + j);
752 fp_subd_low(t, p, v0);
753 dv_copy_sec(v0, t, RLC_FP_DIGS + j + 1, RLC_SIGN(m[0]));
754
755 bn_mul2_low(v1, p11, m[1], RLC_FP_DIGS + j);
756 fp_subd_low(t, p, v1);
757 dv_copy_sec(v1, t, RLC_FP_DIGS + j + 1, RLC_SIGN(m[1]));
758
759 fp_addd_low(t, v0, v1);
760 fp_rdc(p01, t);
761#else
762 (void)j;
763
764 t[0] = RLC_SEL(m[0], -m[0], RLC_SIGN(m[0]));
765 fp_mul(v0, p01, t);
766 fp_neg(t0, v0);
767 fp_copy_sec(v0, t0, RLC_SIGN(m[0]));
768
769 t[0] = RLC_SEL(m[1], -m[1], RLC_SIGN(m[1]));
770 fp_mul(v1, p11, t);
771 fp_neg(t1, v1);
772 fp_copy_sec(v1, t1, RLC_SIGN(m[1]));
773
774 fp_add(p01, v0, v1);
775#endif
776
777 /* Negate based on sign of f at the end. */
778 fp_negm_low(t, p01);
779 dv_copy_sec(p01, t, RLC_FP_DIGS, f[RLC_FP_DIGS] >> (RLC_DIG - 1));
780
781 /* Multiply by (precomp * R^j) % p, one for each iteration of the loop,
782 * one for the constant, one more to be removed by reduction. */
783 fp_mul(c, p01, pre);
784 }
785 RLC_CATCH_ANY {
786 RLC_THROW(ERR_CAUGHT);
787 }
788 RLC_FINALLY {
789 dv_free(t0);
790 dv_free(f);
791 dv_free(t);
792 dv_free(p);
793 dv_free(g);
794 dv_free(t1);
795 dv_free(u0);
796 dv_free(u1);
797 dv_free(v0);
798 dv_free(v1);
799 dv_free(p01);
800 dv_free(p11);
801 fp_free(pre);
802 }
803}

845
