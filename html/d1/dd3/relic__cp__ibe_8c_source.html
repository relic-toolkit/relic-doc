71  char *id, int len, g1_t pub) {

96  bn_new(n);

99  g2_new(q);

101 
102  /* Allocate size for storing the output. */
103  l = gt_size_bin(e, 0);
104  buf = RLC_ALLOCA(uint8_t, l);
105  if (buf == NULL) {
106  THROW(ERR_NO_MEMORY);
107  }
108 
109  g1_get_ord(n);
110 
111  /* q = H_1(ID). */
112  g2_map(q, (uint8_t *)id, len);
113 
114  /* e = e(K_pub, q). */
115  pc_map(e, pub, q);

118  bn_rand_mod(r, n);

121  md_map(h, buf, l);
122

124  g1_mul_gen(p, r);
125  g1_write_bin(out, *out_len, p, 0);
126

128  out[l + (2 * RLC_FP_BYTES + 1)] = in[l] ^ h[l];
