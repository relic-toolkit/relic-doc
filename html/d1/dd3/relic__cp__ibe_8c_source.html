38int cp_ibe_gen(bn_t master, g1_t pub) {
39 bn_t n;
40 int result = RLC_OK;
41
42 bn_null(n);
43
44 RLC_TRY {
45 bn_new(n);
46
47 pc_get_ord(n);
48 bn_rand_mod(master, n);
49
50 /* K_pub = sG. */
51 g1_mul_gen(pub, master);
52 }
53 RLC_CATCH_ANY {
54 result = RLC_ERR;
55 }
56 RLC_FINALLY {
57 bn_free(n);
58 }
59 return result;
60}

68int cp_ibe_enc(uint8_t *out, size_t *out_len, const uint8_t *in, size_t in_len,
69 const char *id, const g1_t pub) {
70 int l, result = RLC_OK;
71 uint8_t *buf = NULL, h[RLC_MD_LEN];
72 bn_t n;
73 bn_t r;
74 g1_t p;
75 g2_t q;
76 gt_t e;
77
78 bn_null(n);
79 bn_null(r);
80 g1_null(p);
81 g2_null(q);
82 gt_null(e);
83
84 if (pub == NULL || in_len <= 0 || in_len > RLC_MD_LEN) {
85 return RLC_ERR;
86 }
87
88 l = 1 + 2 * RLC_FP_BYTES;
89 if (*out_len < (in_len + l)) {
90 return RLC_ERR;
91 }
92
93 RLC_TRY {
94 bn_new(n);
95 bn_new(r);
96 g1_new(p);
97 g2_new(q);
98 gt_new(e);
99
100 /* Allocate size for storing the output. */
101 l = gt_size_bin(e, 0);
102 buf = RLC_ALLOCA(uint8_t, l);
103 if (buf == NULL) {
104 RLC_THROW(ERR_NO_MEMORY);
105 }
106
107 pc_get_ord(n);
108
109 /* q = H_1(ID). */
110 g2_map(q, (uint8_t *)id, strlen(id));
111
112 /* e = e(K_pub, q). */
113 pc_map(e, pub, q);
114
115 /* h = H_2(e^r). */
116 bn_rand_mod(r, n);
117 gt_exp(e, e, r);
118 gt_write_bin(buf, l, e, 0);
119 md_map(h, buf, l);
120
121 /* P = [k]G. */
122 g1_mul_gen(p, r);
123 g1_write_bin(out, *out_len, p, 0);
124
125 for (l = 0; l < RLC_MIN(in_len, RLC_MD_LEN); l++) {
126 out[l + (2 * RLC_FP_BYTES + 1)] = in[l] ^ h[l];
127 }
128
129 *out_len = in_len + (2 * RLC_FP_BYTES + 1);
130 }
131 RLC_CATCH_ANY {
132 result = RLC_ERR;
133 }
134 RLC_FINALLY {
135 bn_free(n);
136 bn_free(r);
137 g1_free(p);
138 g2_free(q);
139 gt_free(e);
140 RLC_FREE(buf);
141 }
142 return result;
143}

145int cp_ibe_dec(uint8_t *out, size_t *out_len, const uint8_t *in, size_t in_len,
146 const g2_t prv) {
147 int l, result = RLC_OK;
148 uint8_t *buf = NULL, h[RLC_MD_LEN];
149 g1_t p;
150 gt_t e;
151
152 g1_null(p);
153 gt_null(e);
154
155 l = 2 * RLC_FP_BYTES + 1;
156 if (prv == NULL || in_len <= l || in_len > (RLC_MD_LEN + l)) {
157 return RLC_ERR;
158 }
159
160 if (*out_len < (in_len - l)) {
161 return RLC_ERR;
162 }
163
164 RLC_TRY {
165 g1_new(p);
166 gt_new(e);
167
168 g1_read_bin(p, in, l);
169
170 pc_map(e, p, prv);
171
172 /* Allocate size for storing the output. */
173 l = gt_size_bin(e, 0);
174 buf = RLC_ALLOCA(uint8_t, l);
175 if (buf == NULL) {
176 RLC_THROW(ERR_NO_MEMORY);
177 }
178 gt_write_bin(buf, l, e, 0);
179 md_map(h, buf, l);
180
181 in_len -= (2 * RLC_FP_BYTES + 1);
182 for (l = 0; l < RLC_MIN(in_len, RLC_MD_LEN); l++) {
183 out[l] = in[l + (2 * RLC_FP_BYTES + 1)] ^ h[l];
184 }
185
186 *out_len = in_len;
187 }
188 RLC_CATCH_ANY {
189 result = RLC_ERR;
190 }
191 RLC_FINALLY {
192 g1_free(p);
193 gt_free(e);
194 RLC_FREE(buf);
195 }
196
197 return result;
198}
