41void fp9_mul_basic(fp9_t c, const fp9_t a, const fp9_t b) {
42 fp3_t v0, v1, v2, t0, t1, t2;
43
44 fp3_null(v0);
45 fp3_null(v1);
46 fp3_null(v2);
47 fp3_null(t0);
48 fp3_null(t1);
49 fp3_null(t2);
50
51 RLC_TRY {
52 fp3_new(v0);
53 fp3_new(v1);
54 fp3_new(v2);
55 fp3_new(t0);
56 fp3_new(t1);
57 fp3_new(t2);
58
59 /* v0 = a_0b_0 */
60 fp3_mul(v0, a[0], b[0]);
61
62 /* v1 = a_1b_1 */
63 fp3_mul(v1, a[1], b[1]);
64
65 /* v2 = a_2b_2 */
66 fp3_mul(v2, a[2], b[2]);
67
68 /* t2 (c_0) = v0 + E((a_1 + a_2)(b_1 + b_2) - v1 - v2) */
69 fp3_add(t0, a[1], a[2]);
70 fp3_add(t1, b[1], b[2]);
71 fp3_mul(t2, t0, t1);
72 fp3_sub(t2, t2, v1);
73 fp3_sub(t2, t2, v2);
74 fp3_mul_nor(t0, t2);
75 fp3_add(t2, t0, v0);
76
77 /* c_1 = (a_0 + a_1)(b_0 + b_1) - v0 - v1 + Ev2 */
78 fp3_add(t0, a[0], a[1]);
79 fp3_add(t1, b[0], b[1]);
80 fp3_mul(c[1], t0, t1);
81 fp3_sub(c[1], c[1], v0);
82 fp3_sub(c[1], c[1], v1);
83 fp3_mul_nor(t0, v2);
84 fp3_add(c[1], c[1], t0);
85
86 /* c_2 = (a_0 + a_2)(b_0 + b_2) - v0 + v1 - v2 */
87 fp3_add(t0, a[0], a[2]);
88 fp3_add(t1, b[0], b[2]);
89 fp3_mul(c[2], t0, t1);
90 fp3_sub(c[2], c[2], v0);
91 fp3_add(c[2], c[2], v1);
92 fp3_sub(c[2], c[2], v2);
93
94 /* c_0 = t2 */
95 fp3_copy(c[0], t2);
96 } RLC_CATCH_ANY {
97 RLC_THROW(ERR_CAUGHT);
98 } RLC_FINALLY {
99 fp3_free(t2);
100 fp3_free(t1);
101 fp3_free(t0);
102 fp3_free(v2);
103 fp3_free(v1);
104 fp3_free(v0);
105 }
106}

112void fp9_mul_unr(dv9_t c, const fp9_t a, const fp9_t b) {
113 dv3_t u0, u1, u2, u3;
114 fp3_t t0, t1;
115
116 dv3_null(u0);
117 dv3_null(u1);
118 dv3_null(u2);
119 dv3_null(u3);
120 fp3_null(t0);
121 fp3_null(t1);
122
123 RLC_TRY {
124 dv3_new(u0);
125 dv3_new(u1);
126 dv3_new(u2);
127 dv3_new(u3);
128 fp3_new(t0);
129 fp3_new(t1);
130
131 /* v0 = a_0b_0, v1 = a_1b_1, v2 = a_2b_2,
132 * t0 = a_1 + a_2, t1 = b_1 + b_2,
133 * u4 = u1 + u2, u5 = u0 + u1, u6 = u0 + u2 */
134#ifdef RLC_FP_ROOM
135 fp3_muln_low(u0, a[0], b[0]);
136 fp3_muln_low(u1, a[1], b[1]);
137 fp3_muln_low(u2, a[2], b[2]);
138 fp3_addm_low(t0, a[1], a[2]);
139 fp3_addm_low(t1, b[1], b[2]);
140 fp3_addc_low(c[0], u1, u2);
141#else
142 fp3_muln_low(u0, a[0], b[0]);
143 fp3_muln_low(u1, a[1], b[1]);
144 fp3_muln_low(u2, a[2], b[2]);
145 fp3_addm_low(t0, a[1], a[2]);
146 fp3_addm_low(t1, b[1], b[2]);
147 fp3_addc_low(c[0], u1, u2);
148#endif
149 /* t2 (c_0) = v0 + E((a_1 + a_2)(b_1 + b_2) - v1 - v2) */
150 fp3_muln_low(u3, t0, t1);
151 fp3_subc_low(u3, u3, c[0]);
152 fp3_nord_low(c[0], u3);
153 fp3_addc_low(c[0], c[0], u0);
154
155 /* c_1 = (a_0 + a_1)(b_0 + b_1) - v0 - v1 + Ev2 */
156 fp3_addm_low(t0, a[0], a[1]);
157 fp3_addm_low(t1, b[0], b[1]);
158 fp3_addc_low(c[1], u0, u1);
159
160 fp3_muln_low(u3, t0, t1);
161 fp3_subc_low(u3, u3, c[1]);
162 fp3_nord_low(c[2], u2);
163 fp3_addc_low(c[1], u3, c[2]);
164
165 /* c_2 = (a_0 + a_2)(b_0 + b_2) - v0 + v1 - v2 */
166 fp3_addm_low(t0, a[0], a[2]);
167 fp3_addm_low(t1, b[0], b[2]);
168 fp3_addc_low(c[2], u0, u2);
169
170 fp3_muln_low(u3, t0, t1);
171 fp3_subc_low(u3, u3, c[2]);
172 fp3_addc_low(c[2], u3, u1);
173 } RLC_CATCH_ANY {
174 RLC_THROW(ERR_CAUGHT);
175 } RLC_FINALLY {
176 dv3_free(u0);
177 dv3_free(u1);
178 dv3_free(u2);
179 dv3_free(u3);
180 fp3_free(t0);
181 fp3_free(t1);
182 }
183}

205void fp9_mul_dxs(fp9_t c, const fp9_t a, const fp9_t b) {
206 fp3_t v0, v1, t0, t1, t2;
207
208 fp3_null(v0);
209 fp3_null(v1);
210 fp3_null(t0);
211 fp3_null(t1);
212 fp3_null(t2);
213
214 RLC_TRY {
215 fp3_new(v0);
216 fp3_new(v1);
217 fp3_new(t0);
218 fp3_new(t1);
219 fp3_new(t2);
220
221 /* v0 = a_0b_0 */
222 fp3_mul(v0, a[0], b[0]);
223
224 /* v1 = a_1b_1 */
225 fp3_mul(v1, a[1], b[1]);
226
227 /* v2 = a_2b_2 = 0 */
228
229 /* t2 (c0) = v0 + E((a_1 + a_2)(b_1 + b_2) - v1 - v2) */
230 fp3_add(t0, a[1], a[2]);
231 fp3_mul(t0, t0, b[1]);
232 fp3_sub(t0, t0, v1);
233 fp3_mul_nor(t2, t0);
234 fp3_add(t2, t2, v0);
235
236 /* c1 = (a_0 + a_1)(b_0 + b_1) - v0 - v1 + Ev2 */
237 fp3_add(t0, a[0], a[1]);
238 fp3_add(t1, b[0], b[1]);
239 fp3_mul(c[1], t0, t1);
240 fp3_sub(c[1], c[1], v0);
241 fp3_sub(c[1], c[1], v1);
242
243 /* c2 = (a_0 + a_2)(b_0 + b_2) - v0 + v1 - v2 */
244 fp3_add(t0, a[0], a[2]);
245 fp3_mul(c[2], t0, b[0]);
246 fp3_sub(c[2], c[2], v0);
247 fp3_add(c[2], c[2], v1);
248
249 /* c0 = t2 */
250 fp3_copy(c[0], t2);
251 } RLC_CATCH_ANY {
252 RLC_THROW(ERR_CAUGHT);
253 } RLC_FINALLY {
254 fp3_free(v0);
255 fp3_free(v1);
256 fp3_free(t0);
257 fp3_free(t1);
258 fp3_free(t2);
259 }
260}

262void fp9_mul_art(fp9_t c, const fp9_t a) {
263 fp3_t t0;
264
265 fp3_null(t0);
266
267 RLC_TRY {
268 fp3_new(t0);
269
270 /* (a_0 + a_1 * v + a_2 * v^2) * v = a_2 * v^3 + a_0 * v + a_1 * v^2 */
271 fp3_copy(t0, a[0]);
272 fp3_mul_nor(c[0], a[2]);
273 fp3_copy(c[2], a[1]);
274 fp3_copy(c[1], t0);
275 } RLC_CATCH_ANY {
276 RLC_THROW(ERR_CAUGHT);
277 } RLC_FINALLY {
278 fp3_free(t0);
279 }
