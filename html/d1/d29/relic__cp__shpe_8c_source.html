39int cp_shpe_gen(shpe_t pub, shpe_t prv, size_t sbits, size_t nbits) {
40 int result = RLC_OK;
41
42 if (sbits > (nbits / 2)) {
43 return RLC_ERR;
44 }
45
46 /* Generate primes p and q of equivalent length
47 * and (p-1) has a prime factor (the subgroup order) of length sbits
48 */
49 do {
50 bn_gen_prime_factor(prv->a, prv->crt->p, sbits, nbits / 2);
51 bn_gen_prime(prv->crt->q, nbits / 2);
52 } while (bn_cmp(prv->crt->p, prv->crt->q) == RLC_EQ);
53
54 /* Compute n = pq. */
55 bn_mul(prv->crt->n, prv->crt->p, prv->crt->q);
56
57 /* compute the subgroup size */
58 bn_sub_dig(prv->crt->p, prv->crt->p, 1);
59 bn_sub_dig(prv->crt->q, prv->crt->q, 1);
60 bn_mul(pub->g, prv->crt->p, prv->crt->q); // lambda = (p-1)(q-1)
61 bn_div(prv->b, pub->g, prv->a); // lambda = a*b
62
63 /* Restore p and q. */
64 bn_add_dig(prv->crt->p, prv->crt->p, 1);
65 bn_add_dig(prv->crt->q, prv->crt->q, 1);
66
67 /* Fix the generator to be g=(1+n)^b */
68
69 /* Compute dp and dq. */
70 /* dp is 1/((q-1)*lambda) mod p */
71 bn_mod(prv->crt->dp, pub->g, prv->crt->p);
72 bn_mul(prv->crt->dp, prv->crt->dp, prv->crt->q);
73 bn_mod(prv->crt->dp, prv->crt->dp, prv->crt->p);
74
75 /* dq is 1/((p-1)*lambda) mod q */
76 bn_mod(prv->crt->dq, pub->g, prv->crt->q);
77 bn_mul(prv->crt->dq, prv->crt->dq, prv->crt->p);
78 bn_mod(prv->crt->dq, prv->crt->dq, prv->crt->q);
79
80 /* invertions */
81 bn_mod_inv(prv->crt->dp, prv->crt->dp, prv->crt->p);
82 bn_mod_inv(prv->crt->dq, prv->crt->dq, prv->crt->q);
83
84 /* Precompute (1+n)^b)^n mod n^2 */
85 bn_sqr(prv->crt->qi, prv->crt->n); // n^2
86 bn_add_dig(pub->g, prv->crt->n, 1); // 1+n
87 bn_mxp(prv->g, pub->g, prv->b, prv->crt->qi); // (1+n)^b mod n^2
88 bn_mxp(prv->gn, prv->g, prv->crt->n, prv->crt->qi); // ((1+n)^b)^n mod n^2
89
90 /* qInv = q^(-1) mod p. */
91 bn_mod_inv(prv->crt->qi, prv->crt->q, prv->crt->p);
92
93 /* n=pq */
94 bn_copy(pub->crt->n, prv->crt->n);
95 bn_copy(pub->g, prv->g);
96
97 return result;
98}

101int cp_shpe_enc_prv(bn_t c, const bn_t m, const shpe_t prv) {
102 bn_t r, s;
103 int result = RLC_OK;
104
105 bn_null(r);
106 bn_null(s);
107
108 if (prv == NULL) {
109 return RLC_ERR;
110 }
111#if ALLOC != AUTO
112 if (prv->crt == NULL || prv->crt->n == NULL ||
113 bn_bits(m) > bn_bits(prv->crt->n)) {
114 return RLC_ERR;
115 }
116#endif
117
118 RLC_TRY {
119 bn_new(r);
120 bn_new(s);
121
122 /* Generate r in Z_alpha^*. */
123 bn_rand_mod(r, prv->a);
124 /* For G=(1+n)^b, compute c = (G^m)(G^n)^r mod n^2
125 * which is also c = (1+n*b*m)(G^n)^r mod n^2.
126 */
127 bn_sqr(s, prv->crt->n); // n^2
128 bn_mxp(r, prv->gn, r, s); // (G^n)^r
129 bn_mul(c, prv->crt->n, m); // n*m
130 bn_mod(c, c, s);
131 bn_mul(c, c, prv->b); // b*n*m
132 bn_add_dig(c, c, 1); // 1+b*n*m
133 bn_mod(c, c, s);
134 bn_mul(c, c, r); // (1+n*b*m)(G^n)^r
135 bn_mod(c, c, s);
136
137 }
138 RLC_CATCH_ANY {
139 result = RLC_ERR;
140 }
141 RLC_FINALLY {
142 bn_free(r);
143 bn_free(s);
144 }
145
146 return result;
147}

150 bn_t r, s;
151 int result = RLC_OK;
152
153 bn_null(r);
154 bn_null(s);
155
156 if (pub == NULL) {
157 return RLC_ERR;
158 }
159#if ALLOC != AUTO
160 if (pub->crt == NULL || pub->crt->n == NULL ||
161 bn_bits(m) > bn_bits(pub->crt->n)) {
162 return RLC_ERR;
163 }
164#endif
165
166 RLC_TRY {
167 bn_new(r);
168 bn_new(s);
169
170 /* Generate r in Z_alpha^*. */
171 bn_rand_mod(r, pub->crt->n);
172 /* For G=(1+n)^b, compute c = G^(m+nr) mod n^2
173 */
174 bn_sqr(s, pub->crt->n);
175 bn_mul(r, r, pub->crt->n); // n*r
176 bn_add(r, r, m); // m+n*r
177 bn_mxp(c, pub->g, r, s); // G^(m+n*r) mod n^2
178
179 }
180 RLC_CATCH_ANY {
181 result = RLC_ERR;
182 }
183 RLC_FINALLY {
184 bn_free(r);
185 bn_free(s);
186 }

191int cp_shpe_dec(bn_t m, const bn_t c, const shpe_t prv) {
192 bn_t t, u;
193 int result = RLC_OK;
194
195 if (prv == NULL || bn_bits(c) > 2 * bn_bits(prv->crt->n)) {
196 return RLC_ERR;
197 }
198
199 bn_null(t);
200 bn_null(u);
201
202 RLC_TRY {
203 bn_new(t);
204 bn_new(u);
205
206#if !defined(CP_CRT)
207 bn_sub_dig(t, prv->crt->p, 1);
208 bn_sub_dig(u, prv->crt->q, 1);
209 bn_mul(t, t, u);
210
211 /* Compute (c^l mod n^2) * u mod n. */
212 bn_sqr(u, prv->crt->n);
213 bn_mxp(m, c, prv->a, u);
214
215 bn_sub_dig(m, m, 1);
216 bn_div(m, m, prv->crt->n);
217 bn_mod_inv(u, t, prv->crt->n);
218 bn_mul(m, m, u);
219 bn_mod(m, m, prv->crt->n);
220#else
221 bn_mxp_crt(m, c, prv->a, prv->a, prv->crt, 1);
222#endif /* CP_CRT */
223 }
224 RLC_CATCH_ANY {
225 result = RLC_ERR;
226 }
227 RLC_FINALLY {
228 bn_free(t);
229 bn_free(u);
230 }
231
232 return result;
