39 void fp2_frb(fp2_t c, fp2_t a, int i) {
40  switch (i % 2) {
41  case 0:
42  fp2_copy(c, a);
43  break;
44  case 1:
45  /* (a_0 + a_1 * u)^p = a_0 - a_1 * u. */
46  fp_copy(c[0], a[0]);
47  fp_neg(c[1], a[1]);
48  break;
49  }
50 }

64 void fp4_frb(fp4_t c, fp4_t a, int i) {
65  /* Cost of a single multiplication in Fp^2 per Frobenius. */
