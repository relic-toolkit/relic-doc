39void fp2_frb(fp2_t c, const fp2_t a, int i) {
40 switch (i % 2) {
41 case 0:
42 fp2_copy(c, a);
43 break;
44 case 1:
45 /* (a_0 + a_1 * u)^p = a_0 - a_1 * u. */
46 fp_copy(c[0], a[0]);
47 fp_neg(c[1], a[1]);
48 break;
49 }
50}

52void fp3_frb(fp3_t c, const fp3_t a, int i) {
53 fp3_copy(c, a);
54 switch (i % 3) {
55 case 1:
56 fp3_mul_frb(c, c, 0, 1);
57 break;
58 case 2:
59 fp3_mul_frb(c, c, 0, 2);
60 break;
61 }
62}

64void fp4_frb(fp4_t c, const fp4_t a, int i) {
65 /* Cost of a single multiplication in Fp^2 per Frobenius. */
66 fp4_copy(c, a);
67 for (; i % 4 > 0; i--) {
68 fp2_frb(c[0], c[0], 1);
69 fp2_frb(c[1], c[1], 1);
70 if (fp_prime_get_mod18() % 3 == 1) {
71 fp2_mul_frb(c[1], c[1], 1, 3);
72 } else {
73 fp2_mul_frb(c[1], c[1], 2, 1);
74 fp2_mul_frb(c[1], c[1], 2, 1);
75 }
76 }
77}
