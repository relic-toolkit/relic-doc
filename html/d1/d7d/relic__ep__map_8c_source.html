43 static void ep_sw_b12(ep_t p, const fp_t t, int u, int negate) {
44  fp_t t0, t1, t2, t3;
45 
46  fp_null(t0);
47  fp_null(t1);
48  fp_null(t2);
49  fp_null(t3);
50 
51  TRY {
52  fp_new(t0);
53  fp_new(t1);
54  fp_new(t2);
55  fp_new(t3);
56 
57  /* t0 = t^2. */
58  fp_sqr(t0, t);
59  /* Compute f(u) such that u^3 + b is a square. */
60  fp_set_dig(p->x, -u);
61  fp_neg(p->x, p->x);
62  ep_rhs(t1, p);
63  /* Compute t1 = (-f(u) + t^2), t2 = t1 * t^2 and invert if non-zero. */
64  fp_add(t1, t1, t0);
65  fp_mul(t2, t1, t0);
66  if (!fp_is_zero(t2)) {
67  /* Compute inverse of u^3 * t2 and fix later. */
68  fp_mul(t2, t2, p->x);
69  fp_mul(t2, t2, p->x);
70  fp_mul(t2, t2, p->x);
71  fp_inv(t2, t2);
72  }
73  /* Compute t0 = t^4 * u * sqrt(-3)/t2. */
74  fp_sqr(t0, t0);
75  fp_mul(t0, t0, t2);
76  fp_mul(t0, t0, p->x);
77  fp_mul(t0, t0, p->x);
78  fp_mul(t0, t0, p->x);
79  /* Compute constant u * sqrt(-3). */
80  fp_copy(t3, core_get()->srm3);
81  for (int i = 1; i < -u; i++) {
82  fp_add(t3, t3, core_get()->srm3);
83  }
84  fp_mul(t0, t0, t3);
85  /* Compute (u * sqrt(-3) + u)/2 - t0. */
86  fp_add_dig(p->x, t3, -u);
87  fp_hlv(p->y, p->x);
88  fp_sub(p->x, p->y, t0);
89  ep_rhs(p->y, p);
90  if (!fp_srt(p->y, p->y)) {
91  /* Now try t0 - (u * sqrt(-3) - u)/2. */
92  fp_sub_dig(p->x, t3, -u);
93  fp_hlv(p->y, p->x);
94  fp_sub(p->x, t0, p->y);
95  ep_rhs(p->y, p);
96  if (!fp_srt(p->y, p->y)) {
97  /* Finally, try (u - t1^2 / t2). */
98  fp_sqr(p->x, t1);
99  fp_mul(p->x, p->x, t1);
100  fp_mul(p->x, p->x, t2);
101  fp_sub_dig(p->x, p->x, -u);
102  ep_rhs(p->y, p);
103  fp_srt(p->y, p->y);
104  }
105  }
106  if (negate) {
107  fp_neg(p->y, p->y);
108  }
109  fp_set_dig(p->z, 1);
110  p->norm = 1;
111  }
112  CATCH_ANY {
113  THROW(ERR_CAUGHT);
114  }
115  FINALLY {
116  fp_free(t0);
117  fp_free(t1);
118  fp_free(t2);
119  fp_free(t3);
120  }
121 }

129 static void ep_sw_bn(ep_t p, const fp_t t, int u, int negate) {
130  fp_t t0;
131  fp_t t1;
132  if (fp_is_zero(t)) {
133  ep_set_infty(p);
134  return;
135  }
136 
137  fp_null(t0);
138  fp_null(t1);
139 
140  TRY {
141  fp_new(t0);
142  fp_new(t1);
143 
144  /* w = t^2 + b + 1 */
145  fp_sqr(t0, t);
146  fp_add(t0, t0, ep_curve_get_b());
147  fp_add_dig(t0, t0, 1);
148 
149  if (fp_is_zero(t0)) {
150  ep_curve_get_gen(p);
151  return;
152  }
153 
154  /* (sqrt(-3) - u) / 2 */
155  fp_copy(t1, core_get()->srm3);
156  fp_sub_dig(t1, t1, -u);
157  fp_hlv(t1, t1);
158 
159  fp_inv(t0, t0);
160  fp_mul(t0, t0, core_get()->srm3);
161  fp_mul(t0, t0, t);
162 
163  /* x1 = -wt + sqrt(-3) */
164  fp_neg(p->x, t0);
165  fp_mul(p->x, p->x, t);
166  fp_add(p->x, p->x, t1);
167  ep_rhs(p->y, p);
168  if (!fp_srt(p->y, p->y)) {
169  /* x2 = - x1 - u */
170  fp_neg(p->x, p->x);
171  fp_sub_dig(p->x, p->x, -u);
172  ep_rhs(p->y, p);
173  if (!fp_srt(p->y, p->y)) {
174  /* x3 = (w^2 + u)/w^2 */
175  fp_sqr(p->x, t0);
176  fp_inv(p->x, p->x);
177  fp_add_dig(p->x, p->x, -u);
178  ep_rhs(p->y, p);
179  fp_srt(p->y, p->y);
180  p->norm = 0;
181  }
182  }
183 
184  if (negate) {
185  fp_neg(p->y, p->y);
186  }
187  fp_set_dig(p->z, 1);
188  p->norm = 1;
189  }
190  CATCH_ANY {
191  THROW(ERR_CAUGHT);
192  }
193  FINALLY {
194  fp_free(t0);
195  fp_free(t1);
196  }
197 }

203 void ep_map(ep_t p, const uint8_t *msg, int len) {
204  bn_t k, pm1o2;
205  fp_t t;
206  ep_t q;
207  uint8_t digest[RLC_MD_LEN];
208  int neg;
209 
210  bn_null(k);
211  bn_null(pm1o2);
212  fp_null(t);
213  ep_null(q);
214 
215  TRY {
216  bn_new(k);
217  bn_new(pm1o2);
218  fp_new(t);
219  ep_new(q);
220 
221  pm1o2->sign = RLC_POS;
222  pm1o2->used = RLC_FP_DIGS;
223  dv_copy(pm1o2->dp, fp_prime_get(), RLC_FP_DIGS);
224  bn_hlv(pm1o2, pm1o2);
225  md_map(digest, msg, len);
226  bn_read_bin(k, digest, RLC_MIN(RLC_FP_BYTES, RLC_MD_LEN));
227  fp_prime_conv(t, k);
228  fp_prime_back(k, t);
229  neg = (bn_cmp(k, pm1o2) == RLC_LT ? 0 : 1);
230 
231  switch (ep_curve_is_pairf()) {
232  case EP_BN:
233  ep_sw_bn(p, t, -1, neg);
234  md_map(digest, digest, RLC_MD_LEN);
235  bn_read_bin(k, digest, RLC_MIN(RLC_FP_BYTES, RLC_MD_LEN));
236  fp_prime_conv(t, k);
237  fp_prime_back(k, t);
238  neg = (bn_cmp(k, pm1o2) == RLC_LT ? 0 : 1);
239  ep_sw_bn(q, t, -1, neg);
240  ep_add(p, p, q);
241  ep_norm(p, p);
242  break;
243  case EP_B12:
244  ep_sw_b12(p, t, -3, neg);
245  md_map(digest, digest, RLC_MD_LEN);
246  bn_read_bin(k, digest, RLC_MIN(RLC_FP_BYTES, RLC_MD_LEN));
247  fp_prime_conv(t, k);
248  neg = (bn_cmp(k, pm1o2) == RLC_LT ? 0 : 1);
249  ep_sw_b12(q, t, -3, neg);
250  ep_add(p, p, q);
251  ep_norm(p, p);
252  /* Now, multiply by cofactor to get the correct group. */
253  fp_prime_get_par(k);
254  bn_neg(k, k);
255  bn_add_dig(k, k, 1);
256  if (bn_bits(k) < RLC_DIG) {
257  ep_mul_dig(p, p, k->dp[0]);
258  } else {
259  ep_mul(p, p, k);
260  }
261  break;
262  default:{
263  fp_prime_conv(p->x, k);
264  fp_zero(p->y);
265  fp_set_dig(p->z, 1);
266 
267  while (1) {
268  ep_rhs(t, p);
269 
270  if (fp_srt(p->y, t)) {
271  p->norm = 1;
272  break;
273  }
274  fp_add_dig(p->x, p->x, 1);
275  }
276 
277  /* Now, multiply by cofactor to get the correct group. */
278  ep_curve_get_cof(k);
279  if (bn_bits(k) < RLC_DIG) {
280  ep_mul_dig(p, p, k->dp[0]);
281  } else {
282  ep_mul_basic(p, p, k);
283  }
284  }
285  }
286  }
287  CATCH_ANY {
288  THROW(ERR_CAUGHT);
289  }
290  FINALLY {
291  bn_free(k);
292  bn_free(pm1o2);
293  fp_free(t);
294  ep_free(q);
295  }
296 }
