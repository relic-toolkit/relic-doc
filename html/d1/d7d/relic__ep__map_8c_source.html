77static void ep_map_basic_impl(ep_t p, const uint8_t *bytes, size_t len) {
78 bn_t x;
79 fp_t t0;
80
81 bn_null(x);
82 fp_null(t0);
83
84 RLC_TRY {
85 bn_new(x);
86 fp_new(t0);
87
88 bn_read_bin(x, bytes, len);
89 fp_prime_conv(p->x, x);
90 fp_set_dig(p->z, 1);
91
92 while (1) {
93 ep_rhs(t0, p->x);
94
95 if (fp_smb(t0) == 1) {
96 fp_srt(p->y, t0);
97 p->coord = BASIC;
98 break;
99 }
100
101 fp_add_dig(p->x, p->x, 1);
102 }
103
104 ep_mul_cof(p, p);
105 }
106 RLC_CATCH_ANY {
107 RLC_THROW(ERR_CAUGHT);
108 }
109 RLC_FINALLY {
110 bn_free(x);
111 fp_free(t0);
112 }
113}

126static void ep_map_sswum_impl(ep_t p, const uint8_t *bytes, size_t len,
127 void (*const map_fn)(ep_t, const fp_t)) {
128 bn_t k;
129 fp_t t;
130 ep_t q;
131 int neg;
132 /* enough space for two field elements plus extra bytes for uniformity */
133 const size_t elm = (FP_PRIME + ep_param_level() + 7) / 8;
134
135 bn_null(k);
136 fp_null(t);
137 ep_null(q);
138
139 RLC_TRY {
140 bn_new(k);
141 fp_new(t);
142 ep_new(q);
143
144#define EP_MAP_CONVERT_BYTES(IDX) \
145 do { \
146 bn_read_bin(k, bytes + IDX * elm, elm); \
147 fp_prime_conv(t, k); \
148 } while (0)
149
150#define EP_MAP_APPLY_MAP(PT) \
151 do { \
152 /* check sign of t */ \
153 neg = fp_is_even(t); \
154 /* convert */ \
155 map_fn(PT, t); \
156 /* compare sign of y and sign of t; fix if necessary */ \
157 neg = neg != fp_is_even(PT->y); \
158 fp_neg(t, PT->y); \
159 dv_copy_sec(PT->y, t, RLC_FP_DIGS, neg); \
160 } while (0)
161
162 /* first map invocation */
163 EP_MAP_CONVERT_BYTES(0);
164 EP_MAP_APPLY_MAP(p);
165 TMPL_MAP_CALL_ISOMAP(ep, p);
166
167 /* second map invocation */
168 EP_MAP_CONVERT_BYTES(1);
169 EP_MAP_APPLY_MAP(q);
170 TMPL_MAP_CALL_ISOMAP(ep, q);
171
172 /* XXX(rsw) could add p and q and then apply isomap,
173 * but need ep_add to support addition on isogeny curves */
174
175#undef EP_MAP_CONVERT_BYTES
176#undef EP_MAP_APPLY_MAP
177
178 /* sum the result */
179 ep_add(p, p, q);
180 ep_norm(p, p);
181 ep_mul_cof(p, p);
182 }
183 RLC_CATCH_ANY {
184 RLC_THROW(ERR_CAUGHT);
185 }
186 RLC_FINALLY {
187 bn_free(k);
188 fp_free(t);
189 ep_free(q);
190 }
191}

193static void ep_map_swift_impl(ep_t p, const uint8_t *random, size_t len) {
194 fp_t h[8], t1, t2, v, w, y, x1, x2, x3, d[3];
195 ctx_t *ctx = core_get();
196 bn_t k;
197 uint8_t s;
198
199 bn_null(k);
200 fp_null(v);
201 fp_null(w);
202 fp_null(y);
203 fp_null(t1);
204 fp_null(t2);
205 fp_null(x1);
206 fp_null(x2);
207 fp_null(x3);
208 fp_null(d[0]);
209 fp_null(d[1]);
210 fp_null(d[2]);
211
212 RLC_TRY {
213 bn_new(k);
214 fp_new(v);
215 fp_new(w);
216 fp_new(y);
217 fp_new(t1);
218 fp_new(t2);
219 fp_new(x1);
220 fp_new(x2);
221 fp_new(x3);
222 fp_new(d[0]);
223 fp_new(d[1]);
224 fp_new(d[2]);
225 for (size_t i = 0; i < 8; i++) {
226 fp_null(h[i]);
227 fp_new(h[i]);
228 }
229
230 bn_read_bin(k, random, len / 2);
231 fp_prime_conv(t1, k);
232 bn_read_bin(k, random + len / 2, len / 2);
233 fp_prime_conv(t2, k);
234 s = random[len - 1] & 1;
235
236 if (ep_curve_opt_b() == RLC_ZERO) {
237 /* h0 = t1^2, h1 = h0^2, h2 = 4a, h3 = h2^3, h4 = h0 * h1 + h3. */
238 fp_sqr(h[0], t1);
239 fp_sqr(h[1], h[0]);
240 fp_mul(h[4], h[0], h[1]);
241 if (ep_curve_opt_a() == RLC_ONE) {
242 fp_add_dig(h[4], h[4], 64);
243 } else {
244 fp_dbl(h[2], ep_curve_get_a());
245 fp_dbl(h[2], h[2]);
246 fp_sqr(h[3], h[2]);
247 fp_mul(h[3], h[3], h[2]);
248 fp_add(h[4], h[4], h[3]);
249 }
250 /* h5 = t2^2, h6 = \tau * t1, h7 = 24*h0*h5*h6. */
251 fp_sqr(h[5], t2);
252 fp_mul(h[6], ctx->ep_map_c[4], t1);
253 fp_mul(h[7], h[0], h[5]);
254 fp_mul(h[7], h[7], h[6]);
255 fp_mul_dig(h[7], h[7], 24);
256 /* \tau = (\omega - 1)/2. */
257 fp_sub_dig(p->x, ctx->ep_map_c[4], 1);
258 fp_hlv(p->x, p->x);
259 /* w = h9 = h1^2, v = h10 = \omega(h2h4 + h0h7). */
260 fp_sqr(w, h[1]);
261 fp_mul(v, h[0], h[7]);
262 if (ep_curve_opt_a() == RLC_ONE) {
263 fp_dbl(t1, h[4]);
264 fp_dbl(t1, t1);
265 } else {
266 fp_mul(t1, h[2], h[4]);
267 }
268 fp_add(v, v, t1);
269 fp_mul(v, v, p->x);
270
271 /* d0 = -2h6\omega(h4 + h7), d1 = d0\omega. */
272 fp_add(d[0], h[4], h[7]);
273 fp_mul(d[0], d[0], h[6]);
274 fp_mul(d[0], d[0], p->x);
275 fp_dbl(d[0], d[0]);
276 fp_neg(d[0], d[0]);
277 fp_mul(d[1], d[0], p->x);
278 if (ep_curve_opt_a() == RLC_ONE) {
279 fp_sub_dig(d[2], h[0], 4);
280 } else {
281 fp_sub(d[2], h[0], h[2]);
282 }
283 /* d2 = -432*h1h5(h0 - h2)^2. */
284 fp_sqr(d[2], d[2]);
285 fp_mul_dig(d[2], d[2], 216);
286 fp_dbl(d[2], d[2]);
287 fp_neg(d[2], d[2]);
288 fp_mul(d[2], d[2], h[1]);
289 fp_mul(d[2], d[2], h[5]);
290
291 if (fp_is_zero(d[0]) || fp_is_zero(d[1]) || fp_is_zero(d[2])) {
292 ep_set_infty(p);
293 } else {
294 if (ep_curve_opt_a() == RLC_ONE) {
295 /* n2 = 4(16h0 + h7). */
296 fp_dbl(h[0], h[0]);
297 fp_dbl(h[0], h[0]);
298 fp_dbl(h[0], h[0]);
299 fp_dbl(h[0], h[0]);
300 fp_add(x2, h[0], h[7]);
301 fp_dbl(x2, x2);
302 fp_dbl(x2, x2);
303 } else {
304 /* n2 = h8 + h9 + h2h7 + h10. */
305 fp_mul(t1, h[0], h[3]);
306 fp_mul(x2, h[2], h[7]);
307 fp_add(x2, x2, t1);
308 }
309 /* n1 = n2 + h9 + h10\omega, n2 = n1 + h10. */
310 fp_add(x2, x2, w);
311 fp_mul(x3, v, p->x);
312 fp_add(x1, x2, x3);
313 fp_add(x2, x2, v);
314 /* n3 = h1(h9 + 8*16h0) + 4096 - h7(h4 - 3(4h1 + 16h0) - h7). */
315 if (ep_curve_opt_a() == RLC_ONE) {
316 fp_dbl(h[2], h[1]);
317 fp_dbl(h[2], h[2]);
318 fp_add(x3, h[2], h[0]);
319 } else {
320 fp_mul(x3, h[2], h[0]);
321 fp_add(x3, x3, h[1]);
322 fp_mul(x3, x3, h[2]);
323 }
324 fp_dbl(t1, x3);
325 fp_add(x3, x3, t1);
326 fp_sub(x3, h[4], x3);
327 fp_sub(x3, x3, h[7]);
328 fp_mul(x3, x3, h[7]);
329 if (ep_curve_opt_a() == RLC_ONE) {
330 fp_dbl(h[0], h[0]);
331 fp_dbl(h[0], h[0]);
332 fp_dbl(h[0], h[0]);
333 fp_set_dig(t2, 64);
334 fp_sqr(t2, t2);
335 } else {
336 fp_dbl(h[0], t1);
337 fp_sqr(t2, h[3]);
338 }
339 fp_add(h[0], h[0], w);
340 fp_mul(t1, h[0], h[1]);
341 fp_sub(x3, t1, x3);
342 fp_add(x3, x3, t2);
343
344 /* Invert d0, d1 and d2 simultaneously. */
345 fp_inv_sim(d, d, 3);
346 fp_mul(p->x, x1, d[0]);
347 fp_mul(x2, x2, d[1]);
348 fp_mul(x3, x3, d[2]);
349 }
350 } else {
351 /* This is the SwiftEC case per se. */
352 if (ep_curve_opt_a() != RLC_ZERO) {
353 RLC_THROW(ERR_NO_VALID);
354 } else {
355 /* h_0 = t1^3, h1 = t2^2, h2 = h0 + b - h1, h3 = 2h1 + h2. */
356 fp_sqr(h[0], t1);
357 fp_mul(h[0], h[0], t1);
358 fp_sqr(h[1], t2);
359 fp_add(h[2], h[0], ctx->ep_b);
360 fp_sub(h[2], h[2], h[1]);
361 fp_dbl(h[3], h[1]);
362 fp_add(h[3], h[3], h[2]);
363 /* h6 = t1\tau, v = h7 = h2h6, h8 = 2h6t2.*/
364 fp_mul(x3, t1, ctx->ep_map_c[4]);
365 fp_mul(v, h[2], x3);
366 fp_mul(x3, x3, t2);
367 fp_dbl(x3, x3);
368
369 /* n1 = h8(h7 - t1h3), n2 = 2h3^2, d1 = 2h3h8*/
370 fp_mul(x1, t1, h[3]);
371 fp_sub(x1, v, x1);
372 fp_mul(x1, x1, x3);
373 fp_dbl(y, h[3]);
374 fp_sqr(y, y);
375 fp_mul(w, h[3], x3);
376 fp_dbl(w, w);
377
378 if (fp_is_zero(w)) {
379 ep_set_infty(p);
380 } else {
381 fp_inv(w, w);
382 fp_mul(p->x, x1, w);
383 fp_add(x2, t1, p->x);
384 fp_neg(x2, x2);
385 fp_mul(x3, y, w);
386 fp_sqr(x3, x3);
387 fp_add(x3, x3, t1);
388 }
389 }
390 }
391
392 ep_rhs(p->y, p->x);
393 ep_rhs(v, x2);
394 ep_rhs(w, x3);
395
396 int c2 = fp_is_sqr(v);
397 int c3 = fp_is_sqr(w);
398
399 fp_copy_sec(p->y, v, c2);
400 fp_copy_sec(p->x, x2, c2);
401 fp_copy_sec(p->y, w, c3);
402 fp_copy_sec(p->x, x3, c3);
403
404 if (!fp_srt(p->y, p->y)) {
405 RLC_THROW(ERR_NO_VALID);
406 }
407 fp_neg(w, p->y);
408 fp_copy_sec(p->y, w, fp_is_even(p->y) ^ s);
409 fp_set_dig(p->z, 1);
410 p->coord = BASIC;
411 /* Multiply by cofactor. */
412 ep_mul_cof(p, p);
413 }
414 RLC_CATCH_ANY {
415 RLC_THROW(ERR_CAUGHT);
416 }
417 RLC_FINALLY {
418 bn_free(k);
419 fp_free(v);
420 fp_free(w);
421 fp_free(y);
422 fp_free(t1);
423 fp_free(t2);
424 fp_free(x1);
425 fp_free(x2);
426 fp_free(x3);
427 fp_free(d[0]);
428 fp_free(d[1]);
429 fp_free(d[2]);
430 for (size_t i = 0; i < 8; i++) {
431 fp_free(h[i]);
432 }
433 }
434}
