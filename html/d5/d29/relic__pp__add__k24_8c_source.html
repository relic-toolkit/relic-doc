42void pp_add_k24_basic(fp24_t l, ep4_t r, const ep4_t q, const ep_t p) {
43 fp4_t s;
44 ep4_t t;
45 int two = 1, one = 1;
46
47 fp4_null(s);
48 ep4_null(t);
49
50 RLC_TRY {
51 fp4_new(s);
52 ep4_new(t);
53
54 ep4_copy(t, r);
55 ep4_add_slp_basic(r, s, r, q);
56
57 if (ep4_curve_is_twist() == RLC_EP_MTYPE) {
58 two += 1;
59 one ^= 1;
60 }
61
62 fp_mul(l[two][one][0][0], s[0][0], p->x);
63 fp_mul(l[two][one][0][1], s[0][1], p->x);
64 fp_mul(l[two][one][1][0], s[1][0], p->x);
65 fp_mul(l[two][one][1][1], s[1][1], p->x);
66
67 fp4_mul(l[0][0], s, t->x);
68 fp4_sub(l[0][0], t->y, l[0][0]);
69
70 if (ep4_curve_is_twist() != RLC_EP_MTYPE) {
71 fp4_mul_art(l[0][0], l[0][0]);
72 }
73
74 fp_neg(l[0][1][0][0], p->y);
75 }
76 RLC_CATCH_ANY {
77 RLC_THROW(ERR_CAUGHT);
78 }
79 RLC_FINALLY {
80 fp4_free(s);
81 ep4_free(t);
82 }
83}

89void pp_add_k24_projc(fp24_t l, ep4_t r, const ep4_t q, const ep_t p) {
90 fp4_t t0, t1, t2, t3, t4;
91 int two = 1, one = 1;
92
93 fp4_null(t0);
94 fp4_null(t1);
95 fp4_null(t2);
96 fp4_null(t3);
97 fp4_null(t4);
98
99 if (ep4_curve_is_twist() == RLC_EP_MTYPE) {
100 two += 1;
101 one ^= 1;
102 }
103
104 RLC_TRY {
105 fp4_new(t0);
106 fp4_new(t1);
107 fp4_new(t2);
108 fp4_new(t3);
109 fp4_new(t4);
110
111 /* B = t0 = x1 - x2 * z1. */
112 fp4_mul(t0, r->z, q->x);
113 fp4_sub(t0, r->x, t0);
114 /* A = t1 = y1 - y2 * z1. */
115 fp4_mul(t1, r->z, q->y);
116 fp4_sub(t1, r->y, t1);
117
118 /* D = B^2. */
119 fp4_sqr(t2, t0);
120 /* G = x1 * D. */
121 fp4_mul(r->x, r->x, t2);
122 /* E = B^3. */
123 fp4_mul(t2, t2, t0);
124 /* C = A^2. */
125 fp4_sqr(t3, t1);
126 /* F = E + z1 * C. */
127 fp4_mul(t3, t3, r->z);
128 fp4_add(t3, t2, t3);
129
130 /* l10 = - (A * xp). */
131 fp_neg(t4[0][0], p->x);
132 fp_mul(l[two][one][0][0], t1[0][0], t4[0][0]);
133 fp_mul(l[two][one][0][1], t1[0][1], t4[0][0]);
134 fp_mul(l[two][one][1][0], t1[1][0], t4[0][0]);
135 fp_mul(l[two][one][1][1], t1[1][1], t4[0][0]);
136
137 /* t4 = B * x2. */
138 fp4_mul(t4, q->x, t1);
139
140 /* H = E + F - 2 * G. */
141 fp4_sub(t3, t3, r->x);
142 fp4_sub(t3, t3, r->x);
143 /* y3 = A * (G - H) - y1 * E. */
144 fp4_sub(r->x, r->x, t3);
145 fp4_mul(t1, t1, r->x);
146 fp4_mul(r->y, t2, r->y);
147 fp4_sub(r->y, t1, r->y);
148 /* x3 = B * H. */
149 fp4_mul(r->x, t0, t3);
150 /* z3 = z1 * E. */
151 fp4_mul(r->z, r->z, t2);
152
153 /* l11 = J = B * x2 - A * y2. */
154 fp4_mul(t2, q->y, t0);
155 fp4_sub(l[0][0], t4, t2);
156
157 if (ep4_curve_is_twist() != RLC_EP_MTYPE) {
158 fp4_mul_art(l[0][0], l[0][0]);
159 }
160
161 /* l00 = B * yp. */
162 fp_mul(l[0][1][0][0], p->y, t0[0][0]);
163 fp_mul(l[0][1][0][1], p->y, t0[0][1]);
164 fp_mul(l[0][1][1][0], p->y, t0[1][0]);
165 fp_mul(l[0][1][1][1], p->y, t0[1][1]);
166 }
167 RLC_CATCH_ANY {
168 RLC_THROW(ERR_CAUGHT);
169 } RLC_FINALLY {
170 fp4_free(t0);
171 fp4_free(t1);
172 fp4_free(t2);
173 fp4_free(t3);
174 fp4_free(t4);
175 }
176}
