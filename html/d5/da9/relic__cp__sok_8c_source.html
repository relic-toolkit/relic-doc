43int cp_sokdl_sig(bn_t c, bn_t s, const uint8_t *msg, size_t len, const ec_t y,
44 const bn_t x) {
45 bn_t n, r;
46 ec_t t;
47 uint8_t h[RLC_MD_LEN];
48 uint8_t *buf, *m = RLC_ALLOCA(uint8_t, len + 3 * (RLC_FC_BYTES + 1));
49 int l, result = RLC_OK;
50
51 bn_null(n);
52 bn_null(r);
53 ec_null(t);
54
55 RLC_TRY {
56 bn_new(n);
57 bn_new(r);
58 ec_new(t);
59 if (m == NULL) {
60 RLC_THROW(ERR_NO_MEMORY);
61 }
62
63 buf = m;
64 ec_curve_get_ord(n);
65 ec_curve_get_gen(t);
66 memcpy(buf, msg, len);
67 buf += len;
68 l = ec_size_bin(t, 1);
69 ec_write_bin(buf, l, t, 1);
70 buf += l;
71 l = ec_size_bin(y, 1);
72 ec_write_bin(buf, l, y, 1);
73 buf += l;
74
75 bn_rand_mod(r, n);
76 ec_mul_gen(t, r);
77 l = ec_size_bin(t, 1);
78 ec_write_bin(buf, l, t, 1);
79 buf += l;
80
81 md_map(h, m, len + 3 * (RLC_FC_BYTES + 1));
82 bn_read_bin(c, h, RLC_MD_LEN);
83 bn_mod(c, c, n);
84
85 bn_mul(s, c, x);
86 bn_sub(s, r, s);
87 bn_mod(s, s, n);
88 }
89 RLC_CATCH_ANY {
90 result = RLC_ERR;
91 }
92 RLC_FINALLY {
93 bn_free(n);
94 bn_free(r);
95 ec_free(t);
96 RLC_FREE(m);
97 }
98 return result;
99}

101int cp_sokdl_ver(const bn_t c, const bn_t s, const uint8_t *msg, size_t len,
102 const ec_t y) {
103 bn_t n, v;
104 ec_t t;
105 uint8_t h[RLC_MD_LEN];
106 uint8_t *buf, *m = RLC_ALLOCA(uint8_t, len + 3 * (RLC_FC_BYTES + 1));
107 int l, result = 0;
108
109 bn_null(n);
110 bn_null(v);
111 ec_null(t);
112
113 RLC_TRY {
114 bn_new(n);
115 bn_new(v);
116 ec_new(t);
117 if (m == NULL) {
118 RLC_THROW(ERR_NO_MEMORY);
119 }
120
121 buf = m;
122 ec_curve_get_ord(n);
123 ec_curve_get_gen(t);
124 memcpy(buf, msg, len);
125 buf += len;
126 l = ec_size_bin(t, 1);
127 ec_write_bin(buf, l, t, 1);
128 buf += l;
129 l = ec_size_bin(y, 1);
130 ec_write_bin(buf, l, y, 1);
131 buf += l;
132
133 /* Compute t' = [r]G + [c]Y. */
134 ec_mul_sim_gen(t, s, y, c);
135 l = ec_size_bin(t, 1);
136 ec_write_bin(buf, l, t, 1);
137

139 bn_read_bin(v, h, RLC_MD_LEN);
140 bn_mod(v, v, n);
141
142 if (bn_cmp(v, c) == RLC_EQ) {
143 result = 1;
144 }
145 }
146 RLC_CATCH_ANY {
147 result = RLC_ERR;
148 }
149 RLC_FINALLY {
150 bn_free(n);
151 bn_free(v);
152 ec_free(t);
153 RLC_FREE(m);
154 }

156}

158int cp_sokor_sig(bn_t c[2], bn_t s[2], const uint8_t *msg, size_t len,
159 const ec_t y[2], const ec_t g[2], const bn_t x, int first) {
160 bn_t n, v[2], z;
161 ec_t u, t[2];
162 uint8_t h[RLC_MD_LEN];
163 uint8_t *buf, *m = RLC_ALLOCA(uint8_t, len + 6 * (RLC_FC_BYTES + 1));
164 int l, result = RLC_OK;
165 int one = 1 ^ first;
166 int zero = 0 ^ first;
167
168 bn_null(n);
169 bn_null(v[0]);
170 bn_null(v[1]);
171 bn_null(z);
172 ec_null(u);
173 ec_null(t[0]);
174 ec_null(t[1]);
175
176 RLC_TRY {
177 bn_new(n);
178 bn_new(v[0]);
179 bn_new(v[1]);
180 bn_new(z);
181 ec_new(u);
182 ec_new(t[0]);
183 ec_new(t[1]);
184 if (m == NULL) {
185 RLC_THROW(ERR_NO_MEMORY);
186 }
187
188 buf = m;
189 ec_curve_get_ord(n);
190 bn_rand_mod(c[zero], n);
191 memcpy(buf, msg, len);
192 buf += len;
193
194 bn_rand_mod(v[0], n);
195 bn_rand_mod(v[1], n);
196 if (first) {
197 if (g != NULL) {
198 ec_mul(t[0], g[0], v[0]);
199 ec_mul_sim(t[1], g[1], v[1], y[1], c[zero]);
200 } else {
201 ec_mul_gen(t[0], v[0]);
202 ec_mul_sim_gen(t[1], v[1], y[1], c[zero]);
203 }
204 } else {
205 if (g != NULL) {
206 /* T1 = [w]Y1 + [v1]G. */
207 ec_mul_sim(t[0], g[0], v[0], y[0], c[zero]);
208 /* T2 = [v2]G. */
209 ec_mul(t[1], g[1], v[1]);
210 } else {
211 /* T1 = [w]Y1 + [v1]G. */
212 ec_mul_sim_gen(t[0], v[0], y[0], c[zero]);
213 /* T2 = [v2]G. */
214 ec_mul_gen(t[1], v[1]);
215 }
216 }
217
218 ec_curve_get_gen(u);
219 for (int i = 0; i < 2; i++) {
220 if (g != NULL) {
221 ec_copy(u, g[i]);
222 }
223 l = ec_size_bin(u, 1);
224 ec_write_bin(buf, l, u, 1);
225 buf += l;
226 l = ec_size_bin(y[i], 1);
227 ec_write_bin(buf, l, y[i], 1);
228 buf += l;
229 l = ec_size_bin(t[i], 1);
230 ec_write_bin(buf, l, t[i], 1);
231 buf += l;
232 }
233 md_map(h, m, len + 6 * (RLC_FC_BYTES + 1));
234 bn_read_bin(z, h, RLC_MD_LEN);
235 bn_mod(z, z, n);
236
237 bn_sub(c[one], z, c[zero]);
238 bn_mod(c[one], c[one], n);
239
240 bn_copy(s[zero], v[zero]);
241 bn_mul(s[one], c[one], x);
242 bn_sub(s[one], v[one], s[one]);
243 bn_mod(s[one], s[one], n);
244 }
245 RLC_CATCH_ANY {
246 result = RLC_ERR;
247 }
248 RLC_FINALLY {
249 bn_free(n);
250 bn_free(v[0]);
251 bn_free(v[1]);
252 bn_free(z);
253 ec_free(u);
254 ec_free(t[0]);
255 ec_free(t[1]);
256 RLC_FREE(m);
257 }
258 return result;
259}

262 size_t len, const ec_t y[2], const ec_t g[2]) {
263 bn_t n, v[2], z;
264 ec_t t;

298 buf += l;
299 l = ec_size_bin(y[i], 1);
300 ec_write_bin(buf, l, y[i], 1);
301 buf += l;
302 if (g != NULL) {
303 ec_mul_sim(t, g[i], s[i], y[i], c[i]);
304 } else {
305 ec_mul_sim_gen(t, s[i], y[i], c[i]);
306 }
307 l = ec_size_bin(t, 1);
308 ec_write_bin(buf, l, t, 1);
309 buf += l;
310 }
311 md_map(h, m, len + 6 * (RLC_FC_BYTES + 1));
312 bn_read_bin(z, h, RLC_MD_LEN);
313 bn_mod(z, z, n);
314
315 bn_sub(z, z, c[0]);
316 bn_sub(z, z, c[1]);
317 bn_mod(z, z, n);
318
319 if (bn_is_zero(z)) {
320 result = 1;
321 }
322 }
323 RLC_CATCH_ANY {
324 result = RLC_ERR;
325 }
326 RLC_FINALLY {
327 bn_free(n);
328 bn_free(v[0]);
329 bn_free(v[1]);
330 bn_free(z);
331 ec_free(t);
332 RLC_FREE(m);
333 }
