47void bn_mxp_sim(bn_t c, const bn_t a, const bn_t b, const bn_t d, const bn_t e,
48 const bn_t m) {
49 bn_t t0[2], t1[2];
50
51 for (size_t i = 0; i < 2; i++) {
52 bn_null(t0[i]);
53 bn_null(t1[i]);
54 }
55
56 RLC_TRY {
57 for (size_t i = 0; i < 2; i++) {
58 bn_new(t0[i]);
59 bn_new(t1[i]);
60 }
61
62 bn_copy(t0[0], a);
63 bn_copy(t0[1], d);
64 bn_copy(t1[0], b);
65 bn_copy(t1[1], e);
66
67 bn_mxp_sim_few(c, t0, t1, m, 2);
68 }
69 RLC_CATCH_ANY {
70 RLC_THROW(ERR_CAUGHT);
71 }
72 RLC_FINALLY {
73 for (size_t i = 0; i < 2; i++) {
74 bn_free(t0[i]);
75 bn_free(t1[i]);
76 }
77 }
78}

80void bn_mxp_sim_few(bn_t c, const bn_t *a, const bn_t *b, const bn_t m,
81 size_t n) {
82 bn_t *t = NULL, u;
83 size_t l;
84 dig_t parities;
85
86 if (bn_cmp_dig(m, 1) == RLC_EQ) {
87 bn_zero(c);
88 return;
89 }
90
91 if (n == 0) {
92 return;
93 }
94
95 if (n > 8) {
96 RLC_THROW(ERR_NO_VALID);
97 return;
98 }
99
100 bn_null(u);
101 t = RLC_ALLOCA(bn_t, 1 << n);
102 if (t == NULL) {
103 RLC_THROW(ERR_NO_MEMORY);
104 return;
105 }
106
107 RLC_TRY {
108 for (size_t i = 0; i < (1 << n); i++) {
109 bn_null(t[i]);
110 bn_new(t[i]);
111 }
112 bn_new(u);
113 bn_mod_pre(u, m);
114
115 // Precompute all 2^n combinations
116 bn_set_dig(t[0], 1);
117#if BN_MOD == MONTY
118 bn_mod_monty_conv(t[0], t[0], m);
119#endif
120 for (size_t i = 0; i < n; i++) {
121 if (!bn_is_zero(b[i])) { // Otherwise will never need P[i]
122 const uint_t star = 1 << i;
123#if BN_MOD == MONTY
124 bn_mod_monty_conv(t[star], a[i], m);
125#else
126 bn_copy(t[star], a[i]);
127#endif
128 for(size_t j = star + 1; j < (star << 1); j++) {
129 bn_mul(t[j], t[star], t[j - star]);
130 bn_mod(t[j], t[j], m, u);
131 }
132 }
133 }
134
135 l = bn_bits(b[0]);
136 for(size_t j = 1; j < n; j++) {
137 l = RLC_MAX(l, bn_bits(b[j]));
138 }
139
140 bn_copy(c, t[0]);
141 for (int i = l - 1; i >= 0; i--) {
142 // One Squaring
143 bn_sqr(c, c);
144 bn_mod(c, c, m, u);
145 // Select odd exponents
146 parities = bn_get_bit(b[0], i);
147 for(size_t j = 1; j < n; j++) {
148 parities |= (uint_t)(bn_get_bit(b[j], i)) << j;
149 }
150 // One multiplication by the odd exponents
151 if (parities) {
152 bn_mul(c, c, t[parities]);
153 bn_mod(c, c, m, u);
154 }
155 }
156#if BN_MOD == MONTY
157 bn_mod_monty_back(c, c, m);
158#endif
159
160 } RLC_CATCH_ANY {
161 RLC_THROW(ERR_CAUGHT);
162 } RLC_FINALLY {
163 for (size_t i = 0; i < (1 << n); i++) {
164 bn_free(t[i]);
165 }
166 RLC_FREE(t);
167 bn_free(u);
168 }
169}

171void bn_mxp_sim_lot(bn_t c, const bn_t *a, const bn_t *b, const bn_t m, size_t n) {
172 uint_t i, j;
173 bn_t _a[XP_WIDTH], _b[XP_WIDTH], t;
174
175 if (bn_cmp_dig(m, 1) == RLC_EQ) {
176 bn_zero(c);
177 return;
178 }
179
180 RLC_TRY {
181 // Will use blocks of size XP_WIDTH
182 bn_null(t);
183 bn_new(t);
184 for(int j = 0; j < XP_WIDTH; j++) {
185 bn_null(_a[j]);
186 bn_null(_b[j]);
187 bn_new(_a[j]);
188 bn_new(_b[j]);
189 }
190
191 // Largest multiple of XP_WIDTH lower than n
192 const int endblockingloop = ((n / XP_WIDTH) * XP_WIDTH);
193 bn_set_dig(c, 1);
194 // Exponentiate by blocks of size RLC_WIDTH
195 for(i = 0; i < endblockingloop;) {
196 for(j = 0; j < 8; j++, i++) {
197 bn_copy(_a[j], a[i]);
198 bn_copy(_b[j], b[i]);
199 }
200 bn_mxp_sim_few(t, _a, _b, m, XP_WIDTH);
201 bn_mul(c, c, t);
202 bn_mod(c, c, m);
203 }
204
205 // Remaining (n - endblockingloop) exponentiations
206 if (n > i) {
207 if (n == i + 1) {
208 // A single exponent
209 bn_mxp(t, a[i], b[i], m);
210 } else {
211 j = 0;
212 for(; i < n; j++, i++) {
213 bn_copy(_a[j], a[i]);
214 bn_copy(_b[j], b[i]);
215 }
216 bn_mxp_sim_few(t, _a, _b, m, j);
217 }
218 bn_mul(c, c, t);
219 bn_mod(c, c, m);
220 }
221 } RLC_CATCH_ANY {
222 RLC_THROW(ERR_CAUGHT);
223 } RLC_FINALLY {
224 for(j = 0; j < XP_WIDTH; j++) {
225 bn_free(_a[j]);
226 bn_free(_b[j]);
227 }
228 bn_free(t);
229 }
