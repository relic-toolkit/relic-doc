40#define TMPL_MAP_HORNER(PFX, IN) \
41 static void PFX##_eval(PFX##_t c, PFX##_t a, IN *coeffs, int deg) { \
42 PFX##_copy(c, coeffs[deg]); \
43 for (int i = deg; i > 0; --i) { \
44 PFX##_mul(c, c, a); \
45 PFX##_add(c, c, coeffs[i - 1]); \
46 } \
47 }

49/* conditionally normalize result of isogeny map when not using projective coords */
50#if EP_ADD == JACOB
51
52#define TMPL_MAP_ISOMAP_NORM(PFX) \
53 do { \
54 /* Y = Ny * Dx * Z^2. */ \
55 PFX##_mul(q->y, p->y, t1); \
56 PFX##_mul(q->y, q->y, t3); \
57 /* Z = Dx * Dy, t1 = Z^2. */ \
58 PFX##_mul(q->z, t2, t3); \
59 PFX##_sqr(t1, q->z); \
60 PFX##_mul(q->y, q->y, t1); \
61 /* X = Nx * Dy * Z. */ \
62 PFX##_mul(q->x, t0, t2); \
63 PFX##_mul(q->x, q->x, q->z); \
64 q->coord = JACOB; \
65 } while (0) \
66
67#elif EP_ADD == PROJC
68
69#define TMPL_MAP_ISOMAP_NORM(PFX) \
70 /* Z = Dx * Dy. */ \
71 PFX##_mul(q->z, t2, t3); \
72 /* X = Nx * Dy. */ \
73 PFX##_mul(q->x, t0, t2); \
74 /* Y = Ny * Dx. */ \
75 PFX##_mul(q->y, p->y, t1); \
76 PFX##_mul(q->y, q->y, t3); \
77 q->coord = PROJC; \
78
79#else

81#define TMPL_MAP_ISOMAP_NORM(PFX) \
82 do { \
83 /* when working with affine coordinates, clear denominator */ \
84 PFX##_mul(q->z, t2, t3); \
85 PFX##_inv(q->z, q->z); \
86 /* y coord */ \
87 PFX##_mul(q->y, p->y, q->z); \
88 PFX##_mul(q->y, q->y, t3); \
89 PFX##_mul(q->y, q->y, t1); \
90 /* x coord */ \
91 PFX##_mul(q->x, t2, q->z); \
92 PFX##_mul(q->x, q->x, t0); \
93 /* z coord == 1 */ \
94 PFX##_set_dig(q->z, 1); \
95 q->coord = BASIC; \
96 } while (0) \

103#define TMPL_MAP_ISOGENY_MAP(CUR, PFX, ISO) \
104 /* declaring this function inline suppresses unused warnings */ \
105 static inline void CUR##_iso(CUR##_t q, CUR##_t p) { \
106 PFX##_t t0, t1, t2, t3; \
107 \
108 if (!CUR##_curve_is_ctmap()) { \
109 CUR##_copy(q, p); \
110 return; \
111 } \
112 /* XXX need to add real support for input projective points */ \
113 if (p->coord != BASIC) { \
114 CUR##_norm(p, p); \
115 } \
116 \
117 PFX##_null(t0); \
118 PFX##_null(t1); \
119 PFX##_null(t2); \
120 PFX##_null(t3); \
121 \
122 RLC_TRY { \
123 PFX##_new(t0); \
124 PFX##_new(t1); \
125 PFX##_new(t2); \
126 PFX##_new(t3); \
127 \
128 ISO##_t coeffs = CUR##_curve_get_iso(); \
129 \
130 /* numerators */ \
131 PFX##_eval(t0, p->x, coeffs->xn, coeffs->deg_xn); \
132 PFX##_eval(t1, p->x, coeffs->yn, coeffs->deg_yn); \
133 /* denominators */ \
134 PFX##_eval(t2, p->x, coeffs->yd, coeffs->deg_yd); \
135 PFX##_eval(t3, p->x, coeffs->xd, coeffs->deg_xd); \
136 \
137 /* normalize if necessary */ \
138 TMPL_MAP_ISOMAP_NORM(PFX); \
139 } \
140 RLC_CATCH_ANY { RLC_THROW(ERR_CAUGHT); } \
141 RLC_FINALLY { \
142 PFX##_free(t0); \
143 PFX##_free(t1); \
144 PFX##_free(t2); \
145 PFX##_free(t3); \
146 } \
147 } \

168#define TMPL_MAP_SSWU(CUR, PFX, PTR_TY) \
169 static void CUR##_map_sswu(CUR##_t p, const PFX##_t t) { \
170 PFX##_t t0, t1, t2, t3; \
171 ctx_t *ctx = core_get(); \
172 PTR_TY *mBoverA = ctx->CUR##_map_c[0]; \
173 PTR_TY *a = ctx->CUR##_map_c[2]; \
174 PTR_TY *b = ctx->CUR##_map_c[3]; \
175 PTR_TY *u = ctx->CUR##_map_u; \
176 \
177 PFX##_null(t0); \
178 PFX##_null(t1); \
179 PFX##_null(t2); \
180 PFX##_null(t3); \
181 \
182 RLC_TRY { \
183 PFX##_new(t0); \
184 PFX##_new(t1); \
185 PFX##_new(t2); \
186 PFX##_new(t3); \
187 \
188 /* start computing the map */ \
189 PFX##_sqr(t0, t); \
190 PFX##_mul(t0, t0, u); /* t0 = u * t^2 */ \
191 PFX##_sqr(t1, t0); /* t1 = u^2 * t^4 */ \
192 PFX##_add(t2, t1, t0); /* t2 = u^2 * t^4 + u * t^2 */ \
193 \
194 /* handle the exceptional cases */ \
195 /* XXX(rsw) should be done projectively */ \
196 { \
197 const int e1 = PFX##_is_zero(t2); \
198 PFX##_neg(t3, u); /* t3 = -u */ \
199 /* exception: -u instead of u^2t^4 + ut^2 */ \
200 PFX##_copy_sec(t2, t3, e1); \
201 /* t2 = -1/u or 1/(u^2 * t^4 + u*t^2) */ \
202 PFX##_inv(t2, t2); \
203 PFX##_add_dig(t3, t2, 1); /* t3 = 1 + t2 */ \
204 PFX##_copy_sec(t2, t3, e1 == 0); /* add 1 if t2 != -1/u */ \
205 } \
206 /* e1 goes out of scope */ \
207 /* compute x1, g(x1) */ \
208 /* -B / A * (1 + 1 / (u^2 * t^4 + u * t^2)) */ \
209 PFX##_mul(p->x, t2, mBoverA); \
210 PFX##_sqr(p->y, p->x); /* x^2 */ \
211 PFX##_add(p->y, p->y, a); /* x^2 + a */ \
212 PFX##_mul(p->y, p->y, p->x); /* x^3 + a x */ \
213 PFX##_add(p->y, p->y, b); /* x^3 + a x + b */ \
214 \
215 /* compute x2, g(x2) */ \
216 PFX##_mul(t2, t0, p->x); /* t2 = u * t^2 * x1 */ \
217 PFX##_mul(t1, t0, t1); /* t1 = u^3 * t^6 */ \
218 PFX##_mul(t3, t1, p->y); /* t5 = g(t2) = u^3 * t^6 * g(p->x) */ \
219 { \
220 /* try x2, g(x2) */ \
221 const int e1 = PFX##_is_sqr(p->y); \
222 PFX##_copy_sec(p->x, t2, e1 == 0); \
223 PFX##_copy_sec(p->y, t3, e1 == 0); \
224 } \
225 if (!PFX##_srt(p->y, p->y)) { \
226 RLC_THROW(ERR_NO_VALID); \
227 } \
228 PFX##_set_dig(p->z, 1); \
229 p->coord = BASIC; \
230 } \
231 RLC_CATCH_ANY { RLC_THROW(ERR_CAUGHT); } \
232 RLC_FINALLY { \
233 PFX##_free(t0); \
234 PFX##_free(t1); \
235 PFX##_free(t2); \
236 PFX##_free(t3); \
237 } \
238 } \

244#define TMPL_MAP_SVDW(CUR, PFX, PTR_TY) \
245 static void CUR##_map_svdw(CUR##_t p, const PFX##_t t) { \
246 PFX##_t t1, t2, t3, t4; \
247 ctx_t *ctx = core_get(); \
248 PTR_TY *gU = ctx->CUR##_map_c[0]; \
249 PTR_TY *mUover2 = ctx->CUR##_map_c[1]; \
250 PTR_TY *c3 = ctx->CUR##_map_c[2]; \
251 PTR_TY *c4 = ctx->CUR##_map_c[3]; \
252 PTR_TY *u = ctx->CUR##_map_u; \
253 \
254 PFX##_null(t1); \
255 PFX##_null(t2); \
256 PFX##_null(t3); \
257 PFX##_null(t4); \
258 \
259 RLC_TRY { \
260 PFX##_new(t1); \
261 PFX##_new(t2); \
262 PFX##_new(t3); \
263 PFX##_new(t4); \
264 \
265 /* start computing the map */ \
266 PFX##_sqr(t1, t); \
267 PFX##_mul(t1, t1, gU); \
268 PFX##_add_dig(t2, t1, 1); /* 1 + t^2 * g(u) */ \
269 PFX##_sub_dig(t1, t1, 1); \
270 PFX##_neg(t1, t1); /* 1 - t^2 * g(u) */ \
271 PFX##_mul(t3, t1, t2); /* (1+t^2*g(u)) * (1-t^2*g(u)) */ \
272 \
273 /* handle exceptional case */ \
274 { \
275 /* compute inv0(t3), i.e., 0 if t3 == 0, 1/t3 otherwise */ \
276 const int e0 = PFX##_is_zero(t3); \
277 PFX##_copy_sec(t3, gU, e0); /* g(u) is nonzero */ \
278 PFX##_inv(t3, t3); \
279 PFX##_zero(t4); \
280 PFX##_copy_sec(t3, t4, e0); \
281 } \
282 /* e0 goes out of scope */ \
283 PFX##_mul(t4, t, t1); \
284 PFX##_mul(t4, t4, t3); \
285 PFX##_mul(t4, t4, c3); \
286 \
287 /* compute x1 and g(x1) */ \
288 PFX##_sub(p->x, mUover2, t4); \
289 CUR##_rhs(p->y, p->x); \
290 { \
291 const int e0 = PFX##_is_sqr(p->y); \
292 /* compute x2 and g(x2) */ \
293 PFX##_add(t4, mUover2, t4); \
294 PFX##_copy_sec(p->x, t4, e0 == 0); \
295 CUR##_rhs(t1, p->x); \
296 PFX##_copy_sec(p->y, t1, e0 == 0); \
297 } \
298 { \
299 const int e1 = PFX##_is_sqr(p->y); \
300 /* compute x3 and g(x3) */ \
301 PFX##_sqr(t1, t2); \
302 PFX##_mul(t1, t1, t3); \
303 PFX##_sqr(t1, t1); \
304 PFX##_mul(t1, t1, c4); \
305 PFX##_add(t1, t1, u); \
306 PFX##_copy_sec(p->x, t1, e1 == 0); \
307 CUR##_rhs(t2, p->x); \
308 PFX##_copy_sec(p->y, t2, e1 == 0); \
309 } \
310 if (!PFX##_srt(p->y, p->y)) { \
311 RLC_THROW(ERR_NO_VALID); \
312 } \
313 PFX##_set_dig(p->z, 1); \
314 p->coord = BASIC; \
315 } \
316 RLC_CATCH_ANY { RLC_THROW(ERR_CAUGHT); } \
317 RLC_FINALLY { \
318 PFX##_free(t1); \
319 PFX##_free(t2); \
320 PFX##_free(t3); \
321 PFX##_free(t4); \
322 } \
323 } \
