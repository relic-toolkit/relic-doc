54 static void find_trace(void) {
55  fb_t t0, t1;
56  int counter;
57  ctx_t *ctx = core_get();
58 
59  fb_null(t0);
60  fb_null(t1);
61 
62  ctx->fb_ta = ctx->fb_tb = ctx->fb_tc = -1;
63 
64  TRY {
65  fb_new(t0);
66  fb_new(t1);
67 
68  counter = 0;
69  for (int i = 0; i < RLC_FB_BITS; i++) {
70  fb_zero(t0);
71  fb_set_bit(t0, i, 1);
72  fb_copy(t1, t0);
73  for (int j = 1; j < RLC_FB_BITS; j++) {
74  fb_sqr(t1, t1);
75  fb_add(t0, t0, t1);
76  }
77  if (!fb_is_zero(t0)) {
78  switch (counter) {
79  case 0:
80  ctx->fb_ta = i;
81  ctx->fb_tb = ctx->fb_tc = -1;
82  break;
83  case 1:
84  ctx->fb_tb = i;
85  ctx->fb_tc = -1;
86  break;
87  case 2:
88  ctx->fb_tc = i;
89  break;
90  default:
91  THROW(ERR_NO_VALID);
92  break;
93  }
94  counter++;
95  }
96  }
97  }
98  CATCH_ANY {
99  THROW(ERR_CAUGHT);
100  }
101  FINALLY {
102  fb_free(t0);
103  fb_free(t1);
104  }
105 }

116 static void find_solve(void) {
117  int i, j, k, l;
118  fb_t t0;
119  ctx_t *ctx = core_get();
120 
121  fb_null(t0);
122 
123  TRY {
124  fb_new(t0);
125 
126  l = 0;
127  for (i = 0; i < RLC_FB_BITS; i += 8, l++) {
128  for (j = 0; j < 16; j++) {
129  fb_zero(t0);
130  for (k = 0; k < 4; k++) {
131  if (j & (1 << k)) {
132  fb_set_bit(t0, i + 2 * k + 1, 1);
133  }
134  }
135  fb_copy(ctx->fb_half[l][j], t0);
136  for (k = 0; k < (RLC_FB_BITS - 1) / 2; k++) {
137  fb_sqr(ctx->fb_half[l][j], ctx->fb_half[l][j]);
138  fb_sqr(ctx->fb_half[l][j], ctx->fb_half[l][j]);
139  fb_add(ctx->fb_half[l][j], ctx->fb_half[l][j], t0);
140  }
141  }
142  fb_rsh(ctx->fb_half[l][j], ctx->fb_half[l][j], 1);
143  }
144  }
145  CATCH_ANY {
146  THROW(ERR_CAUGHT);
147  }
148  FINALLY {
149  fb_free(t0);
150  }
151 }

183 static void find_chain(void) {
184  int i, j, k, l, x, y, u[RLC_TERMS + 1];
185  ctx_t *ctx = core_get();
186 
187  ctx->chain_len = -1;
188  for (int i = 0; i < RLC_TERMS; i++) {
189  ctx->chain[i] = (i << 8) + i;
190  }
191  switch (RLC_FB_BITS) {
192  case 127:
193  ctx->chain[1] = (1 << 8) + 0;
194  ctx->chain[4] = (4 << 8) + 2;
195  ctx->chain[7] = (7 << 8) + 2;
196  ctx->chain_len = 9;
197  break;
198  case 193:
199  ctx->chain[1] = (1 << 8) + 0;
200  ctx->chain_len = 8;
201  break;
202  case 233:
203  ctx->chain[1] = (1 << 8) + 0;
204  ctx->chain[3] = (3 << 8) + 0;
205  ctx->chain[6] = (6 << 8) + 0;
206  ctx->chain_len = 10;
207  break;
208  case 251:
209  ctx->chain[1] = (1 << 8) + 0;
210  ctx->chain[2] = (2 << 8) + 1;
211  ctx->chain[4] = (4 << 8) + 3;
212  ctx->chain[5] = (5 << 8) + 4;
213  ctx->chain[7] = (7 << 8) + 6;
214  ctx->chain[8] = (8 << 8) + 7;
215  ctx->chain_len = 10;
216  break;
217  case 283:
218  ctx->chain[4] = (4 << 8) + 0;
219  ctx->chain[6] = (6 << 8) + 0;
220  ctx->chain[9] = (9 << 8) + 0;
221  ctx->chain_len = 11;
222  break;
223  case 353:
224  ctx->chain[2] = (2 << 8) + 0;
225  ctx->chain[4] = (4 << 8) + 0;
226  ctx->chain_len = 10;
227  break;
228  case 367:
229  ctx->chain[1] = (1 << 8) + 0;
230  ctx->chain[2] = (2 << 8) + 1;
231  ctx->chain[6] = (6 << 8) + 3;
232  ctx->chain[9] = (9 << 8) + 2;
233  ctx->chain_len = 11;
234  break;
235  case 1223:
236  ctx->chain[1] = (1 << 8) + 0;
237  ctx->chain[2] = (2 << 8) + 0;
238  ctx->chain[4] = (4 << 8) + 2;
239  ctx->chain[5] = (5 << 8) + 4;
240  ctx->chain[10] = (10 << 8) + 2;
241  ctx->chain[11] = (11 << 8) + 10;
242  ctx->chain_len = 13;
243  break;
244  default:
245  l = 0;
246  j = (RLC_FB_BITS - 1);
247  for (k = 16; k >= 0; k--) {
248  if (j & (1 << k)) {
249  break;
250  }
251  }
252  for (i = 1; i < k; i++) {
253  if (j & (1 << i)) {
254  l++;
255  }
256  }
257  i = 0;
258  ctx->chain_len = k + l;
259  while (j != 1) {
260  if ((j & 0x01) != 0) {
261  i++;
262  ctx->chain[ctx->chain_len - i] = ((ctx->chain_len - i) << 8) + 0;
263  }
264  i++;
265  j = j >> 1;
266  }
267  break;
268  }
269 
270  for (i = 0; i < RLC_TERMS; i++) {
271  for (j = 0; j < RLC_FB_TABLE; j++) {
272  ctx->fb_tab_ptr[i][j] = &(ctx->fb_tab_sqr[i][j]);
273  }
274  }
275 
276  u[0] = 1;
277  u[1] = 2;
278  for (i = 2; i <= ctx->chain_len; i++) {
279  x = ctx->chain[i - 1] >> 8;
280  y = ctx->chain[i - 1] - (x << 8);
281  if (x == y) {
282  u[i] = 2 * u[i - 1];
283  } else {
284  u[i] = u[x] + u[y];
285  }
286  }
287 
288  for (i = 0; i <= ctx->chain_len; i++) {
289  fb_itr_pre((fb_t *)fb_poly_tab_sqr(i), u[i]);
290  }
291 }

335 void fb_poly_add(fb_t c, const fb_t a) {
336  ctx_t *ctx = core_get();
337 
338  if (c != a) {
339  fb_copy(c, a);
340  }
341 
342  if (ctx->fb_pa != 0) {
343  c[RLC_FB_DIGS - 1] ^= ctx->fb_poly[RLC_FB_DIGS - 1];
344  if (ctx->fb_na != RLC_FB_DIGS - 1) {
345  c[ctx->fb_na] ^= ctx->fb_poly[ctx->fb_na];
346  }
347  if (ctx->fb_pb != 0 && ctx->fb_pc != 0) {
348  if (ctx->fb_nb != ctx->fb_na) {
349  c[ctx->fb_nb] ^= ctx->fb_poly[ctx->fb_nb];
350  }
351  if (ctx->fb_nc != ctx->fb_na && ctx->fb_nc != ctx->fb_nb) {
352  c[ctx->fb_nc] ^= ctx->fb_poly[ctx->fb_nc];
353  }
354  }
355  if (ctx->fb_na != 0 && ctx->fb_nb != 0 && ctx->fb_nc != 0) {
356  c[0] ^= 1;
357  }
358  } else {
359  fb_add(c, a, ctx->fb_poly);
360  }
361 }
