38void fp2_exp(fp2_t c, const fp2_t a, const bn_t b) {
39 fp2_t t;
40
41 if (bn_is_zero(b)) {
42 fp2_set_dig(c, 1);
43 return;
44 }
45
46 fp2_null(t);
47
48 RLC_TRY {
49 fp2_new(t);
50
51 fp2_copy(t, a);
52 for (int i = bn_bits(b) - 2; i >= 0; i--) {
53 fp2_sqr(t, t);
54 if (bn_get_bit(b, i)) {
55 fp2_mul(t, t, a);
56 }
57 }
58
59 if (bn_sign(b) == RLC_NEG) {
60 fp2_inv(c, t);
61 } else {
62 fp2_copy(c, t);
63 }
64 }
65 RLC_CATCH_ANY {
66 RLC_THROW(ERR_CAUGHT);
67 }
68 RLC_FINALLY {
69 fp2_free(t);
70 }
71}

252void fp8_exp_dig(fp8_t c, const fp8_t a, dig_t b) {
253 bn_t _b;
254 fp8_t t, v;
255 int8_t u, naf[RLC_DIG + 1];
256 size_t l;
257
258 if (b == 0) {
259 fp8_set_dig(c, 1);
260 return;
261 }
262
263 bn_null(_b);
264 fp8_null(t);
265 fp8_null(v);
266
267 RLC_TRY {
268 bn_new(_b);
269 fp8_new(t);
270 fp8_new(v);
271
272 fp8_copy(t, a);
273
274 if (fp8_test_cyc(a)) {
275 fp8_inv_cyc(v, a);
276 bn_set_dig(_b, b);
277
278 l = RLC_DIG + 1;
279 bn_rec_naf(naf, &l, _b, 2);
280
281 for (int i = bn_bits(_b) - 2; i >= 0; i--) {
282 fp8_sqr_cyc(t, t);
283
284 u = naf[i];
285 if (u > 0) {
286 fp8_mul(t, t, a);
287 } else if (u < 0) {
288 fp8_mul(t, t, v);
289 }
290 }
291 } else {
292 for (int i = util_bits_dig(b) - 2; i >= 0; i--) {
293 fp8_sqr(t, t);
294 if (b & ((dig_t)1 << i)) {
295 fp8_mul(t, t, a);
296 }
297 }
298 }
299
300 fp8_copy(c, t);
301 }
302 RLC_CATCH_ANY {
