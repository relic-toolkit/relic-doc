38 void fp2_exp(fp2_t c, fp2_t a, bn_t b) {
39  fp2_t t;
40 
41  if (bn_is_zero(b)) {
42  fp2_set_dig(c, 1);
43  return;
44  }
45 
46  fp2_null(t);
47 
48  TRY {
49  fp2_new(t);
50 
51  fp2_copy(t, a);
52  for (int i = bn_bits(b) - 2; i >= 0; i--) {
53  fp2_sqr(t, t);
54  if (bn_get_bit(b, i)) {
55  fp2_mul(t, t, a);
56  }
57  }
58 
59  if (bn_sign(b) == RLC_NEG) {
60  fp2_inv(c, t);
61  } else {
62  fp2_copy(c, t);
63  }
64  }
65  CATCH_ANY {
66  THROW(ERR_CAUGHT);
67  }
68  FINALLY {
69  fp2_free(t);
70  }
71 }
