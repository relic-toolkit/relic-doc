208static void ep4_mul_cof_fm16(ep4_t r, const ep4_t p) {
209 bn_t x;
210 ep4_t t0, t1, t2, t3;
211
212 ep4_null(t0);
213 ep4_null(t1);
214 ep4_null(t2);
215 ep4_null(t3);
216 bn_null(x);
217
218 RLC_TRY {
219 bn_new(x);
220 ep4_new(t0);
221 ep4_new(t1);
222 ep4_new(t2);
223 ep4_new(t3);
224
225 fp_prime_get_par(x);
226 if (bn_is_even(x)) {
227 bn_hlv(x, x);
228 } else {
229 RLC_THROW(ERR_NO_VALID);
230 }
231
232 /* [1 + u^7/2, u^6/2, u^5/2, u^4/2, u^3(2-u)/2, u^2(2-u)/2, u(2-u)/2, (2-u)/2] */
233
234 /* Compute [1 - u/2]\psi^7(P). */
235 ep4_mul_basic(t1, p, x);
236 ep4_sub(t0, p, t1);
237 ep4_frb(t0, t0, 7);
238 bn_dbl(x, x);
239 /* Compute [u - u^2/2]\psi^6(P). */
240 ep4_mul_basic(t2, t1, x);
241 ep4_dbl(t1, t1);
242 ep4_sub(t3, t1, t2);
243 ep4_frb(t3, t3, 6);
244 ep4_add(t0, t0, t3);
245 /* Compute [u^2 - u^3/2]\psi^5(P). */
246 ep4_mul_basic(t1, t2, x);
247 ep4_dbl(t2, t2);
248 ep4_sub(t3, t2, t1);
249 ep4_frb(t3, t3, 5);
250 ep4_add(t0, t0, t3);
251 /* Compute [u^3 - u^4/2]\psi^4(P). */
252 ep4_mul_basic(t2, t1, x);
253 ep4_dbl(t1, t1);
254 ep4_sub(t3, t1, t2);
255 ep4_frb(t3, t3, 4);
256 ep4_add(t0, t0, t3);
257 /* Compute [u^4/2]\psi^3(P). */
258 ep4_frb(t3, t2, 3);
259 ep4_add(t0, t0, t3);
260 /* Compute [u^5/2]\psi^2(P). */
261 ep4_mul_basic(t2, t2, x);
262 ep4_frb(t3, t2, 2);
263 ep4_add(t0, t0, t3);
264 /* Compute [u^6/2]\psi(P). */
265 ep4_mul_basic(t2, t2, x);
266 ep4_frb(t3, t2, 1);
267 ep4_add(t0, t0, t3);
268 /* Compute [1 + u^7/2]P. */
269 ep4_mul_basic(t2, t2, x);
270 ep4_add(t0, t0, t2);
271 ep4_add(t0, t0, p);
272
273 ep4_norm(r, t0);
274 } RLC_CATCH_ANY {
275 RLC_THROW(ERR_CAUGHT);
276 } RLC_FINALLY {
277 ep4_free(t0);
278 ep4_free(t1);
279 ep4_free(t2);
280 ep4_free(t3);
281 bn_free(x);
282 }

291void ep4_mul_cof_b24(ep4_t r, const ep4_t p) {
292 bn_t z;
293 ep4_t t0, t1, t2, t3;
294
295 ep4_null(t0);
296 ep4_null(t1);
297 ep4_null(t2);
298 ep4_null(t3);
299 bn_null(z);
300
301 RLC_TRY {
302 bn_new(z);
303 ep4_new(t0);
304 ep4_new(t1);
305 ep4_new(t2);
306 ep4_new(t3);
307
308 fp_prime_get_par(z);
309
310 bn_sub_dig(z, z, 1);
311 ep4_mul_basic(t0, p, z);
312 bn_add_dig(z, z, 1);
313 ep4_mul_basic(t1, t0, z);
314 ep4_mul_basic(t2, t1, z);
315 ep4_mul_basic(t3, t2, z);
316
317 /* Compute t0 = [u - 1]*\psi^3(P). */
318 ep4_frb(t0, t0, 3);
319 /* Compute t2 = [u^2*(u-1)]\psi(P). */
320 ep4_frb(t2, t2, 1);
321 /* Compute t1 = [u*(u-1)]\psi^2(P). */
322 ep4_frb(t1, t1, 2);
323 /* Compute t3 = [u^3(u-1) - 1]P. */
324 ep4_sub(t3, t3, p);
325
326 ep4_dbl(r, p);
327 ep4_frb(r, r, 4);
328 ep4_add(r, r, t0);
329 ep4_add(r, r, t1);
330 ep4_add(r, r, t2);
331 ep4_add(r, r, t3);
332
333 ep4_norm(r, r);
334 } RLC_CATCH_ANY {
335 RLC_THROW(ERR_CAUGHT);
336 } RLC_FINALLY {
337 ep4_free(t0);
338 ep4_free(t1);
339 ep4_free(t2);
340 ep4_free(t3);
341 bn_free(z);
342
343 }

350void ep4_mul_cof(ep4_t r, const ep4_t p) {
351 bn_t k;
352
353 bn_null(k);
354
355 RLC_TRY {
356 switch (ep_curve_is_pairf()) {
357 case EP_K16:
358 ep4_mul_cof_k16(r, p);
359 break;
360 case EP_AFG16:
361 ep4_mul_cof_afg16(r, p);
362 break;
363 case EP_FM16:
364 ep4_mul_cof_fm16(r, p);
365 break;
366 case EP_B24:
367 ep4_mul_cof_b24(r, p);
368 break;
369 default:
370 /* Now, multiply by cofactor to get the correct group. */
371 ep4_curve_get_cof(k);
372 ep4_mul_big(r, p, k);
373 break;
374 }
375 } RLC_CATCH_ANY {
376 RLC_THROW(ERR_CAUGHT);
377 } RLC_FINALLY {
378 bn_free(k);
379 }
