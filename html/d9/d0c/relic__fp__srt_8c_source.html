38int fp_is_sqr(const fp_t a) {

41 }
42

44}

46int fp_srt(fp_t c, const fp_t a) {
47 bn_t e;
48 fp_t t0, t1, t2, t3;
49 int f = 0, r = 0;
50
51 bn_null(e);
52 fp_null(t0);
53 fp_null(t1);
54 fp_null(t2);
55 fp_null(t3);
56
57 if (fp_is_zero(a)) {
58 fp_zero(c);
59 return 1;
60 }
61
62 RLC_TRY {
63 bn_new(e);
64 fp_new(t0);
65 fp_new(t1);
66 fp_new(t2);
67 fp_new(t3);
68
69 /* Make e = p. */
70 e->used = RLC_FP_DIGS;
71 dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
72
73 switch(fp_prime_get_mod8() % 4) {
74 case 3:
75 /* Easy case, compute a^((p + 1)/4). */
76 bn_add_dig(e, e, 1);
77 bn_rsh(e, e, 2);
78
79 fp_exp(t0, a, e);
80 fp_sqr(t1, t0);
81 r = (fp_cmp(t1, a) == RLC_EQ);
82 fp_copy(c, t0);
83 break;
84 default:
85 /* Implement constant-time version of Tonelli-Shanks algorithm
86 * as per https://eprint.iacr.org/2020/1497.pdf */
87
88 /* First check that a is a square. */
89 r = fp_is_sqr(a);
90
91 /* Compute progenitor as x^(p-1-2^f)/2^(f+1) where 2^f|(p-1). */
92
93 /* Write p - 1 as (e * 2^f), odd e. */
94 f = fp_prime_get_2ad();
95 bn_rsh(e, e, f);
96
97 /* Make it e = (p - 1 - 2^f)/2^(f + 1), compute t0 = a^e. */
98 bn_rsh(e, e, 1);
99 fp_exp(t0, a, e);
100
101 /* Recover 2^f-root of unity, and continue algorithm. */
102 fp_copy(t3, fp_prime_get_srt());
103
104 fp_sqr(t1, t0);
105 fp_mul(t1, t1, a);
106 fp_mul(c, t0, a);
107 for (int j = f; j > 1; j--) {
108 fp_copy(t2, t1);
109 for (int i = 1; i < j - 1; i++) {
110 fp_sqr(t2, t2);
111 }
112 fp_mul(t0, c, t3);
113 fp_copy_sec(c, t0, fp_cmp_dig(t2, 1) != RLC_EQ);
114 fp_sqr(t3, t3);
115 fp_mul(t0, t1, t3);
116 fp_copy_sec(t1, t0, fp_cmp_dig(t2, 1) != RLC_EQ);
117 }
118
119 fp_neg(t0, c);
120 fp_copy_sec(c, t0, fp_is_even(c) == 0);
121 break;
122 }
123 }
124 RLC_CATCH_ANY {
125 RLC_THROW(ERR_CAUGHT);
126 }
127 RLC_FINALLY {
128 bn_free(e);
129 fp_free(t0);
130 fp_free(t1);
131 fp_free(t2);
132 fp_free(t3);
133 }
134 return r;
135}
