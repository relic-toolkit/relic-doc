38 int fp_srt(fp_t c, const fp_t a) {
39  bn_t e;
40  fp_t t0;
41  fp_t t1;
42  int r = 0;
43 
44  bn_null(e);
45  fp_null(t0);
46  fp_null(t1);
47 
48  TRY {
49  bn_new(e);
50  fp_new(t0);
51  fp_new(t1);
52 
53  /* Make e = p. */
54  e->used = RLC_FP_DIGS;
55  dv_copy(e->dp, fp_prime_get(), RLC_FP_DIGS);
56 
57  if (fp_prime_get_mod8() == 3 || fp_prime_get_mod8() == 7) {
58  /* Easy case, compute a^((p + 1)/4). */
59  bn_add_dig(e, e, 1);
60  bn_rsh(e, e, 2);
61 
62  fp_exp(t0, a, e);
63  fp_sqr(t1, t0);
64  r = (fp_cmp(t1, a) == RLC_EQ);
65  fp_copy(c, t0);
66  } else {
67  int f = 0, m = 0;
68 
69  /* First, check if there is a root. Compute t1 = a^((p - 1)/2). */
70  bn_rsh(e, e, 1);
71  fp_exp(t0, a, e);
72 
73  if (fp_cmp_dig(t0, 1) != RLC_EQ) {
74  /* Nope, there is no square root. */
75  r = 0;
76  } else {
77  r = 1;
78  /* Find a quadratic non-residue modulo p, that is a number t2
79  * such that (t2 | p) = t2^((p - 1)/2)!= 1. */
80  do {
81  fp_rand(t1);
82  fp_exp(t0, t1, e);
83  } while (fp_cmp_dig(t0, 1) == RLC_EQ);
84 
85  /* Write p - 1 as (e * 2^f), odd e. */
86  bn_lsh(e, e, 1);
87  while (bn_is_even(e)) {
88  bn_rsh(e, e, 1);
89  f++;
90  }
91 
92  /* Compute t2 = t2^e. */
93  fp_exp(t1, t1, e);
94 
95  /* Compute t1 = a^e, c = a^((e + 1)/2) = a^(e/2 + 1), odd e. */
96  bn_rsh(e, e, 1);
97  fp_exp(t0, a, e);
98  fp_mul(e->dp, t0, a);
99  fp_sqr(t0, t0);
100  fp_mul(t0, t0, a);
101  fp_copy(c, e->dp);
102 
103  while (1) {
104  if (fp_cmp_dig(t0, 1) == RLC_EQ) {
105  break;
106  }
107  fp_copy(e->dp, t0);
108  for (m = 0; (m < f) && (fp_cmp_dig(t0, 1) != RLC_EQ); m++) {
109  fp_sqr(t0, t0);
110  }
111  fp_copy(t0, e->dp);
112  for (int i = 0; i < f - m - 1; i++) {
113  fp_sqr(t1, t1);
114  }
115  fp_mul(c, c, t1);
116  fp_sqr(t1, t1);
117  fp_mul(t0, t0, t1);
118  f = m;
119  }
120  }
121  }
122  }
123  CATCH_ANY {
124  THROW(ERR_CAUGHT);
125  }
126  FINALLY {
127  bn_free(e);
128  fp_free(t0);
129  fp_free(t1);
130  }
131  return r;
132 }
