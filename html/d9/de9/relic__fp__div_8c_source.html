41void fp_hlv_basic(fp_t c, const fp_t a) {
42 dig_t carry = 0;
43
44 if (a[0] & 1) {
45 carry = fp_addn_low(c, a, fp_prime_get());
46 } else {
47 fp_copy(c, a);
48 }
49 fp_rsh1_low(c, c);
50 if (carry) {
51 c[RLC_FP_DIGS - 1] ^= ((dig_t)1 << (RLC_DIG - 1));
52 }
53}

65void fp_trs(fp_t c, const fp_t a) {
66 const dig_t mask = (2 * RLC_3MASK + 1);
67 dig_t c0, c1, f0, f1;
68 fp_t t;
69
70 /* From "Efficient Multiplication in Finite Field Extensions of Degree 5"
71 * by El Mrabet, Guillevic and Ionica at ASIACRYPT 2011. */
72
73 fp_null(t);
74
75 RLC_TRY {
76 fp_new(t);
77
78 RLC_MUL_DIG(t[RLC_FP_DIGS - 1], f0, a[RLC_FP_DIGS - 1], mask);
79 t[RLC_FP_DIGS - 1] >>= 1;
80 c1 = a[RLC_FP_DIGS - 1] - 3 * t[RLC_FP_DIGS - 1];
81
82 for (size_t i = RLC_FP_DIGS - 1; i > 0; i--) {
83 c0 = c1;
84 RLC_MUL_DIG(t[i - 1], f0, a[i - 1], mask);
85 t[i - 1] >>= 1;
86 c1 = c0 + a[i - 1] - 3 * t[i - 1];
87 t[i - 1] += c0 * RLC_3MASK;
88 f0 = ((c1 >> 1) & c1); /* c1 == 3 */
89 f1 = ((c1 >> 2) & !(c1 & 0x11)); /* c1 == 4 */
90 f0 |= f1;
91 t[i - 1] += f0;
92 c1 = c1 - 3 * f0;
93 }
94
95 fp_copy(c, t);
96 fp_sub(t, c, core_get()->over3.dp);
97 fp_copy_sec(c, t, (c1 & 1) | (c1 >> 1)); // c1 >= 1
98 fp_sub(t, c, core_get()->over3.dp);
99 fp_copy_sec(c, t, c1 == 2);
100 } RLC_CATCH_ANY {
101 RLC_THROW(ERR_CAUGHT);
102 } RLC_FINALLY {
103 fp_free(t);
104 }
105}
