54 void arch_init(void) {

56 
57 #if defined(__ARM_ARCH_6__) || defined(__ARM_ARCH_6J__) || defined(__ARM_ARCH_6K__) || defined(__ARM_ARCH_6Z__) || defined(__ARM_ARCH_6ZK__) || defined(__ARM_ARCH_6T2__)
58  /*
59  * This relies on a Kernel module described in:
60  * http://blog.regehr.org/archives/794
61  */
62  asm("mcr p15, 0, %0, c15, c12, 0" : : "r" (1));
63 #elif __ARM_ARCH_7A__
64  /*
65  * This relies on a Kernel module described in:
66  * http://stackoverflow.com/questions/3247373/
67  * how-to-measure-program-execution-time-in-arm-cortex-a8-processor
68  */
69  asm("mcr p15, 0, %0, c9, c12, 0" :: "r"(17));
70  asm("mcr p15, 0, %0, c9, c12, 1" :: "r"(0x8000000f));
71  asm("mcr p15, 0, %0, c9, c12, 3" :: "r"(0x8000000f));
72 #elif __ARM_ARCH_7M__
73  *SCB_DEMCR = *SCB_DEMCR | 0x01000000;
74  *DWT_CYCCNT = 0; // reset the counter
75  *DWT_CONTROL = *DWT_CONTROL | 1 ; // enable the counter
76 #endif
77 
78 #endif /* TIMER = CYCLE */
