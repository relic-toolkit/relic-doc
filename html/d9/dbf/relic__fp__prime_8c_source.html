47 static void fp_prime_set(const bn_t p) {
48  dv_t s, q;
49  bn_t t;
50  fp_t r;
51  ctx_t *ctx = core_get();
52 
53  if (p->used != RLC_FP_DIGS) {
54  THROW(ERR_NO_VALID);
55  }
56 
57  dv_null(s);
58  bn_null(t);
59  dv_null(q);
60  fp_null(r);
61 
62  TRY {
63  dv_new(s);
64  bn_new(t);
65  dv_new(q);
66  fp_new(r);
67 
68  bn_copy(&(ctx->prime), p);
69 
70  #if FP_RDC == MONTY || !defined(STRIP)
71  bn_mod_pre_monty(t, &(ctx->prime));
72  ctx->u = t->dp[0];
73  dv_zero(s, 2 * RLC_FP_DIGS);
74  s[2 * RLC_FP_DIGS] = 1;
75  dv_zero(q, 2 * RLC_FP_DIGS + 1);
76  dv_copy(q, ctx->prime.dp, RLC_FP_DIGS);
77  bn_divn_low(t->dp, ctx->conv.dp, s, 2 * RLC_FP_DIGS + 1, q, RLC_FP_DIGS);
78  ctx->conv.used = RLC_FP_DIGS;
79  bn_trim(&(ctx->conv));
80  bn_set_dig(&(ctx->one), 1);
81  bn_lsh(&(ctx->one), &(ctx->one), ctx->prime.used * RLC_DIG);
82  bn_mod(&(ctx->one), &(ctx->one), &(ctx->prime));
83  #endif
84 
85  /* Now look for proper quadratic/cubic non-residues. */
86  ctx->qnr = ctx->cnr = 0;
87  bn_mod_dig(&(ctx->mod8), &(ctx->prime), 8);
88 
89  switch (ctx->mod8) {
90  case 3:
91  ctx->qnr = -1;
92  ctx->cnr = 2;
93  break;
94  case 7:
95  ctx->qnr = -1;
96  ctx->cnr = -2;
97  /* TODO: implement cube root to handle this better. */
98 #if FP_PRIME == 638
99  ctx->cnr = -3;
100 #endif
101  break;
102  case 1:
103  case 5:
104  ctx->qnr = -2;
105  ctx->cnr = 2;
106  /* Check if it is a quadratic non-residue or find another. */
107  fp_set_dig(r, -ctx->qnr);
108  fp_neg(r, r);
109  while (fp_srt(r, r) == 1) {
110  ctx->qnr--;
111  fp_set_dig(r, -ctx->qnr);
112  fp_neg(r, r);
113  /* We cannot guarantee a cubic extension anymore. */
114  ctx->cnr = 0;
115  };
116  break;
117  }
118 #ifdef FP_QNRES
119  if (ctx->mod8 != 3) {
120  THROW(ERR_NO_VALID);
121  }
122 #endif
123 
124  fp_prime_calc();
125  }
126  CATCH_ANY {
127  THROW(ERR_CAUGHT);
128  }
129  FINALLY {
130  bn_free(t);
131  dv_free(s);
132  dv_free(q);
133  fp_free(r);
134  }
135 }

241 void fp_prime_set_pairf(const bn_t x, int pairf) {
242  bn_t p, t0, t1;
243  ctx_t *ctx = core_get();
244  int len = bn_bits(x) + 1;
245  int8_t s[RLC_FP_BITS + 1];
246 
247  bn_null(p);
248  bn_null(t0);
249  bn_null(t1);
250 
251  TRY {
252  bn_new(p);
253  bn_new(t0);
254  bn_new(t1);
255 
256  bn_copy(&(ctx->par), x);
257  bn_copy(t0, x);
258 
259  switch (pairf) {
260  case EP_BN:
261  /* p = 36 * x^4 + 36 * x^3 + 24 * x^2 + 6 * x + 1. */
262  bn_set_dig(p, 1);
263  bn_mul_dig(t1, t0, 6);
264  bn_add(p, p, t1);
265  bn_mul(t1, t0, t0);
266  bn_mul_dig(t1, t1, 24);
267  bn_add(p, p, t1);
268  bn_mul(t1, t0, t0);
269  bn_mul(t1, t1, t0);
270  bn_mul_dig(t1, t1, 36);
271  bn_add(p, p, t1);
272  bn_mul(t0, t0, t0);
273  bn_mul(t1, t0, t0);
274  bn_mul_dig(t1, t1, 36);
275  bn_add(p, p, t1);
276  fp_prime_set_dense(p);
277  break;
278  case EP_B12:
279  /* p = (x^2 - 2x + 1) * (x^4 - x^2 + 1)/3 + x. */
280  bn_sqr(t1, t0);
281  bn_sqr(p, t1);
282  bn_sub(p, p, t1);
283  bn_add_dig(p, p, 1);
284  bn_sub(t1, t1, t0);
285  bn_sub(t1, t1, t0);
286  bn_add_dig(t1, t1, 1);
287  bn_mul(p, p, t1);
288  bn_div_dig(p, p, 3);
289  bn_add(p, p, t0);
290  fp_prime_set_dense(p);
291  break;
292  case EP_OT8:
293  /* p = (x^8 + x^6 + 5*x^4 + x^2 + 4*x + 4) / 4. */
294  bn_set_dig(p, 4);
295  bn_mul_dig(t1, t0, 4);
296  bn_add(p, p, t1);
297  bn_sqr(t0, t0);
298  bn_add(p, p, t0);
299  bn_sqr(t1, t0);
300  bn_add(p, p, t1);
301  bn_add(p, p, t1);
302  bn_add(p, p, t1);
303  bn_add(p, p, t1);
304  bn_add(p, p, t1);
305  bn_mul(t1, t1, t0);
306  bn_add(p, p, t1);
307  bn_mul(t1, t1, t0);
308  bn_add(p, p, t1);
309  bn_div_dig(p, p, 4);
310  fp_prime_set_dense(p);
311  break;
312  case EP_B48:
313  /* p = (x - 1)^2*(x^16 - x^8 + 1) / 3 + x. */
314  bn_sqr(t1, t0);
315  bn_sqr(t1, t1);
316  bn_sqr(p, t1);
317  bn_sqr(t1, p);
318  bn_sub(t1, t1, p);
319  bn_add_dig(t1, t1, 1);
320  bn_sub_dig(p, t0, 1);
321  bn_sqr(p, p);
322  bn_mul(p, p, t1);
323  bn_div_dig(p, p, 3);
324  bn_add(p, p, t0);
325  fp_prime_set_dense(p);
326  break;
327  case EP_K54:
328  /* p = (1+3*x+3*x^2+(3^5)*x^9+(3^5)*x^10+(3^6)*x^10+(3^6)*x^11+(3^9)*x^18+(3^10)*x^19+(3^10)*x^20) */
329  bn_set_dig(p, 1);
330  bn_mul_dig(t1, t0, 3);
331  bn_add(p, p, t1);
332  bn_sqr(t1, t0);
333  bn_add(p, p, t1);
334  bn_add(p, p, t1);
335  bn_add(p, p, t1);
336  bn_sqr(t1, t1);
337  bn_sqr(t1, t1);
338  bn_mul(t1, t1, t0);
339  bn_mul_dig(t1, t1, 243);
340  bn_add(p, p, t1);
341  bn_mul(t1, t1, t0);
342  bn_add(p, p, t1);
343  bn_mul_dig(t1, t1, 3);
344  bn_add(p, p, t1);
345  bn_mul(t1, t1, t0);
346  bn_add(p, p, t1);
347  bn_mul_dig(t1, t1, 27);
348  bn_mul(t1, t1, t0);
349  bn_mul(t1, t1, t0);
350  bn_mul(t1, t1, t0);
351  bn_mul(t1, t1, t0);
352  bn_mul(t1, t1, t0);
353  bn_mul(t1, t1, t0);
354  bn_mul(t1, t1, t0);
355  bn_add(p, p, t1);
356  bn_mul_dig(t1, t1, 3);
357  bn_mul(t1, t1, t0);
358  bn_add(p, p, t1);
359  bn_mul(t1, t1, t0);
360  bn_add(p, p, t1);
361  fp_prime_set_dense(p);
362  break;
363  }
364 
365  /* Store parameter in NAF form. */
366  ctx->par_len = 0;
367  bn_rec_naf(s, &len, &(ctx->par), 2);
368  /* Fix corner case to avoid problems with sparse representation. */
369  if (s[0] == -1) {
370  s[0] = 1;
371  s[1] = -1;
372  }
373  for (int i = 0; i < len && ctx->par_len < RLC_TERMS; i++) {
374  if (s[i] > 0) {
375  ctx->par_sps[ctx->par_len++] = i;
376  }
377  if (s[i] < 0) {
378  ctx->par_sps[ctx->par_len++] = -i;
379  }
380  }
381  if (ctx->par_len == RLC_TERMS) {
382  THROW(ERR_NO_VALID);
383  }
384  } CATCH_ANY {
385  THROW(ERR_CAUGHT);
386  } FINALLY {
387  bn_free(p);
388  bn_free(t0);
389  bn_free(t1);
390  }
391 }

393 void fp_prime_set_pmers(const int *f, int len) {
394  bn_t p, t;
395 
396  bn_null(p);
397  bn_null(t);
398 
399  TRY {
400  bn_new(p);
401  bn_new(t);
402 
403  if (len >= RLC_TERMS) {
404  THROW(ERR_NO_VALID);
405  }
406 
407  bn_set_2b(p, f[len - 1]);
408  for (int i = len - 2; i > 0; i--) {
409  if (f[i] > 0) {
410  bn_set_2b(t, f[i]);
411  bn_add(p, p, t);
412  } else {
413  bn_set_2b(t, -f[i]);
414  bn_sub(p, p, t);
415  }
416  }
417  if (f[0] > 0) {
418  bn_add_dig(p, p, f[0]);
419  } else {
420  bn_sub_dig(p, p, -f[0]);
421  }
422 
423 #if FP_RDC == QUICK || !defined(STRIP)
424  ctx_t *ctx = core_get();
425  for (int i = 0; i < len; i++) {
426  ctx->sps[i] = f[i];
427  }
428  ctx->sps[len] = 0;
429  ctx->sps_len = len;
430 #endif /* FP_RDC == QUICK */
431 
432  fp_prime_set(p);
433  }
434  CATCH_ANY {
435  THROW(ERR_CAUGHT);
436  }
437  FINALLY {
438  bn_free(p);
439  bn_free(t);
440  }

483 
484  TRY {

486
