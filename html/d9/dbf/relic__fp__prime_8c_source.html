47static void fp_prime_set(const bn_t p) {
48 bn_t t;
49 fp_t r;
50 ctx_t *ctx = core_get();
51 dig_t rem;
52
53 if (p->used != RLC_FP_DIGS) {
54 RLC_THROW(ERR_NO_VALID);
55 return;
56 }
57
58 bn_null(t);
59 fp_null(r);
60
61 RLC_TRY {
62 bn_new(t);
63 fp_new(r);
64
65 bn_copy(&(ctx->prime), p);
66 bn_sub_dig(&(ctx->over3), p, 1);
67 bn_div_dig(&(ctx->over3), &(ctx->over3), 3);
68
69#if FP_RDC == MONTY || !defined(STRIP)
70
71 bn_mod_pre_monty(t, &(ctx->prime));
72 ctx->u = t->dp[0];
73
74 /* compute R mod p */
75 bn_set_2b(&(ctx->one), ctx->prime.used * RLC_DIG);
76 bn_mod(&(ctx->one), &(ctx->one), &(ctx->prime));
77
78 /* compute the R^2 mod p */
79 fp_add(r, ctx->one.dp, ctx->one.dp);
80 bn_set_dig(t, RLC_FP_DIGS);
81 bn_lsh(t, t, RLC_DIG_LOG);
82 fp_exp(ctx->conv.dp, r, t);
83 ctx->conv.used = RLC_FP_DIGS;
84 bn_trim(&(ctx->conv));
85
86#endif /* FP_RDC == MONTY */
87
88#if FP_INV == JMPDS || !defined(STRIP)
89
90 int d = (45907 * FP_PRIME + 26313) / 19929;
91
92#if WSIZE == 8
93 bn_set_dig(t, d >> 8);
94 bn_lsh(t, t, 8);
95 bn_add_dig(t, t, d & 0xFF);
96#else
97 bn_set_dig(t, d);
98#endif
99 ctx->inv.used = RLC_FP_DIGS;
100 dv_copy(ctx->inv.dp, fp_prime_get(), RLC_FP_DIGS);
101 fp_add_dig(ctx->inv.dp, ctx->inv.dp, 1);
102 fp_hlv(ctx->inv.dp, ctx->inv.dp);
103 fp_exp(ctx->inv.dp, ctx->inv.dp, t);
104
105#if FP_RDC == MONTY
106 /* Convert to Montgomery form by multiplying by R^2. */
107 fp_mul(ctx->inv.dp, ctx->inv.dp, ctx->conv.dp);
108 fp_mul(ctx->inv.dp, ctx->inv.dp, ctx->conv.dp);
109#ifdef RLC_FP_ROOM
110 for (int i = 1, j = 0; i < d / (RLC_DIG - 2); i++) {
111 j = i % RLC_FP_DIGS;
112 if (j == 0) {
113 fp_mulm_low(ctx->inv.dp, ctx->inv.dp, ctx->conv.dp);
114 }
115 }
116#else
117 for (int i = 0; i < d / (RLC_DIG - 2) - 1; i++) {
118 fp_mulm_low(ctx->inv.dp, ctx->inv.dp, ctx->conv.dp);
119 }
120#endif
121
122#endif /* FP_RDC == MONTY */
123
124#endif /* FP_INV */
125
126 /* Now look for proper quadratic/cubic non-residues. */
127 ctx->qnr = ctx->cnr = 0;
128 bn_mod_dig(&(ctx->mod8), &(ctx->prime), 8);
129 bn_mod_dig(&(ctx->mod18), &(ctx->prime), 18);
130
131 switch (ctx->mod8) {
132 case 3:
133 ctx->qnr = -1;
134 ctx->cnr = 2;
135 break;
136 case 7:
137 ctx->qnr = -1;
138 /* Try this one, pick another later if not a CNR. */
139 ctx->cnr = -2;
140 break;
141 case 1:
142 case 5:
143 ctx->qnr = -2;
144 ctx->cnr = 2;
145#if FP_PRIME == 638
146 if (fp_param_get() == K18_638) {
147 ctx->qnr = -6;
148 } else {
149 ctx->qnr = -7;
150 }
151#endif
152 break;
153 }
154
155 /* Check if qnr is a quadratic non-residue or find another. */
156 fp_set_dig(r, -ctx->qnr);
157 fp_neg(r, r);
158 while (fp_is_sqr(r)) {
159 ctx->qnr--;
160 fp_set_dig(r, -ctx->qnr);
161 fp_neg(r, r);
162 };
163
164 /* Check if cnr is a cubic non-residue or find another. */
165 if (ctx->mod18 % 3 == 1) {
166 if (ctx->cnr > 0) {
167 fp_set_dig(r, ctx->cnr);
168 while (fp_is_cub(r)) {
169 ctx->cnr++;
170 fp_set_dig(r, ctx->cnr);
171 };
172 } else {
173 fp_set_dig(r, -ctx->cnr);
174 fp_neg(r, r);
175 while (fp_is_cub(r)) {
176 ctx->cnr--;
177 fp_set_dig(r, -ctx->cnr);
178 fp_neg(r, r);
179 };
180 }
181 } else {
182 ctx->cnr = 0;
183 }
184
185#ifdef FP_QNRES
186 if (ctx->mod8 != 3) {
187 RLC_THROW(ERR_NO_VALID);
188 }
189#endif
190
191 /* Compute root of unity by computing QNR to (p - 1)/2^f. */
192 ctx->ad2 = 0;
193 bn_sub_dig(t, p, 1);
194 while (bn_is_even(t)) {
195 ctx->ad2++;
196 bn_hlv(t, t);
197 }
198
199 ctx->srt.used = RLC_FP_DIGS;
200 if (ctx->qnr < 0) {
201 fp_set_dig(ctx->srt.dp, -ctx->qnr);
202 } else {
203 fp_set_dig(ctx->srt.dp, ctx->qnr);
204 }
205 fp_exp(ctx->srt.dp, ctx->srt.dp, t);
206
207 /* Write p - 1 as (e * 3^f), with e = 3l \pm 1. */
208 bn_sub_dig(t, p, 1);
209 bn_mod_dig(&rem, t, 3);
210 while (rem == 0) {
211 bn_div_dig(t, t, 3);
212 bn_mod_dig(&rem, t, 3);
213 }
214
215 /* Compute root of unity by computing CNR to (p - 1)/3^f. */
216 if (ctx->cnr < 0) {
217 fp_set_dig(ctx->crt.dp, -fp_prime_get_cnr());
218 } else {
219 fp_set_dig(ctx->crt.dp, fp_prime_get_cnr());
220 }
221 fp_exp(ctx->crt.dp, ctx->crt.dp, t);
222
223 fp_prime_calc();
224 }
225 RLC_CATCH_ANY {
226 RLC_THROW(ERR_CAUGHT);
227 }
228 RLC_FINALLY {
229 bn_free(t);
230 fp_free(r);
231 }
232}

238void fp_prime_init(void) {
239 ctx_t *ctx = core_get();
240 ctx->fp_id = 0;
241 bn_make(&(ctx->prime), RLC_FP_DIGS);
242 bn_make(&(ctx->over3), RLC_FP_DIGS);
243 bn_make(&(ctx->par), RLC_FP_DIGS);
244#if FP_RDC == QUICK || !defined(STRIP)
245 ctx->sps_len = 0;
246 memset(ctx->sps, 0, sizeof(ctx->sps));
247#endif
248#if FP_RDC == MONTY || !defined(STRIP)
249 bn_make(&(ctx->conv), RLC_FP_DIGS);
250 bn_make(&(ctx->one), RLC_FP_DIGS);
251#endif
252#if FP_INV == JMPDS || !defined(STRIP)
253 bn_make(&(ctx->inv), RLC_FP_DIGS);
254#endif /* FP_INV */
255 bn_make(&(ctx->srt), RLC_FP_DIGS);
256 bn_make(&(ctx->crt), RLC_FP_DIGS);
257}

259void fp_prime_clean(void) {
260 ctx_t *ctx = core_get();
261 if (ctx != NULL) {
262 ctx->fp_id = 0;
263#if FP_RDC == QUICK || !defined(STRIP)
264 ctx->sps_len = 0;
265 memset(ctx->sps, 0, sizeof(ctx->sps));
266#endif
267#if FP_RDC == MONTY || !defined(STRIP)
268 bn_clean(&(ctx->one));
269 bn_clean(&(ctx->conv));
270#endif
271#if FP_INV == JMPDS || !defined(STRIP)
272 bn_clean(&(ctx->inv));
273#endif /* FP_INV */
274 bn_clean(&(ctx->srt));
275 bn_clean(&(ctx->crt));
276 bn_clean(&(ctx->prime));
277 bn_clean(&(ctx->over3));
278 bn_clean(&(ctx->par));
279 }

370void fp_prime_set_pairf(const bn_t x, int pairf) {
371 bn_t p, t0, t1;
372 ctx_t *ctx = core_get();
373 size_t len = bn_bits(x) + 1;
374 int8_t s[RLC_FP_BITS + 1];
375
376 bn_null(p);
377 bn_null(t0);
378 bn_null(t1);
379
380 RLC_TRY {
381 bn_new(p);
382 bn_new(t0);
383 bn_new(t1);
384
385 bn_copy(&(ctx->par), x);
386 bn_copy(t0, x);
387
388 switch (pairf) {
389 case EP_BN:
390 /* p = 36 * x^4 + 36 * x^3 + 24 * x^2 + 6 * x + 1. */
391 bn_set_dig(p, 1);
392 bn_mul_dig(t1, t0, 6);
393 bn_add(p, p, t1);
394 bn_mul(t1, t0, t0);
395 bn_mul_dig(t1, t1, 24);
396 bn_add(p, p, t1);
397 bn_mul(t1, t0, t0);
398 bn_mul(t1, t1, t0);
399 bn_mul_dig(t1, t1, 36);
400 bn_add(p, p, t1);
401 bn_mul(t0, t0, t0);
402 bn_mul(t1, t0, t0);
403 bn_mul_dig(t1, t1, 36);
404 bn_add(p, p, t1);
405 fp_prime_set_dense(p);
406 break;
407 case EP_B12:
408 /* p = (x^2 - 2x + 1) * (x^4 - x^2 + 1)/3 + x. */
409 bn_sqr(t1, t0);
410 bn_sqr(p, t1);
411 bn_sub(p, p, t1);
412 bn_add_dig(p, p, 1);
413 bn_sub(t1, t1, t0);
414 bn_sub(t1, t1, t0);
415 bn_add_dig(t1, t1, 1);
416 bn_mul(p, p, t1);
417 bn_div_dig(p, p, 3);
418 bn_add(p, p, t0);
419 fp_prime_set_dense(p);
420 break;
421 case EP_AFG16:
422 /* p = (x^16 + 2*x^13 + x^10 + 5*x^8 + 6*x^5 + x^2 + 4)/4 */
423 bn_sqr(p, t0);
424 bn_mul(p, p, t0);
425 bn_add_dig(p, p, 2);
426 bn_mul(p, p, t0);
427 bn_mul(p, p, t0);
428 bn_mul(p, p, t0);
429 bn_add_dig(p, p, 1);
430 bn_mul(p, p, t0);
431 bn_mul(p, p, t0);
432 bn_add_dig(p, p, 5);
433 bn_mul(p, p, t0);
434 bn_mul(p, p, t0);
435 bn_mul(p, p, t0);
436 bn_add_dig(p, p, 6);
437 bn_mul(p, p, t0);
438 bn_mul(p, p, t0);
439 bn_mul(p, p, t0);
440 bn_add_dig(p, p, 1);
441 bn_mul(p, p, t0);
442 bn_mul(p, p, t0);
443 bn_add_dig(p, p, 4);
444 bn_div_dig(p, p, 4);
445 fp_prime_set_dense(p);
446 break;
447 case EP_FM16:
448 /* p = (x^16 + x^10 + 5*x^8 + x^2 + 4*x + 4)/4 */
449 bn_sqr(t1, t0);
450 bn_mul(p, t1, t0);
451 bn_sqr(p, p);
452 bn_add_dig(p, p, 1);
453 bn_mul(p, p, t1);
454 bn_add_dig(p, p, 5);
455 bn_mul(p, p, t1);
456 bn_mul(p, p, t1);
457 bn_mul(p, p, t1);
458 bn_add_dig(p, p, 1);
459 bn_mul(p, p, t0);
460 bn_add_dig(p, p, 4);
461 bn_mul(p, p, t0);
462 bn_add_dig(p, p, 4);
463 bn_div_dig(p, p, 4);
464 fp_prime_set_dense(p);
465 break;
466 case EP_K16:
467 /* p = (u^10 + 2*u^9 + 5*u^8 + 48*u^6 + 152*u^5 + 240*u^4 +
468 625*u^2 + 2398*u + 3125) div 980 */
469 bn_add_dig(p, t0, 2);
470 bn_mul(p, p, t0);
471 bn_add_dig(p, p, 5);
472 bn_mul(p, p, t0);
473 bn_mul(p, p, t0);
474 bn_add_dig(p, p, 48);
475 bn_mul(p, p, t0);
476 bn_add_dig(p, p, 152);
477 bn_mul(p, p, t0);
478 bn_add_dig(p, p, 240);
479 bn_mul(p, p, t0);
480 bn_mul(p, p, t0);
481 bn_add_dig(p, p, 256);
482 bn_add_dig(p, p, 256);
483 bn_add_dig(p, p, 113);
484 bn_mul(p, p, t0);
485 bn_set_dig(t1, 9);
486 bn_lsh(t1, t1, 8);
487 bn_add_dig(t1, t1, 94);
488 bn_add(p, p, t1);
489 bn_mul(p, p, t0);
490 bn_set_dig(t0, 12);
491 bn_lsh(t0, t0, 8);
492 bn_add_dig(t0, t0, 53);
493 bn_add(p, p, t0);
494 bn_set_dig(t1, 3);
495 bn_lsh(t1, t1, 8);
496 bn_add_dig(t1, t1, 212);
497 bn_div(p, p, t1);
498 fp_prime_set_dense(p);
499 break;
500 case EP_K18:
501 /* p = (x^8 + 5x^7 + 7x^6 + 37x^5 + 188x^4 + 259x^3 + 343x^2 +
502 1763x + 2401)/21 */

504 bn_mul(p, p, t0);
505 bn_add_dig(p, p, 7);
506 bn_mul(p, p, t0);
507 bn_add_dig(p, p, 37);
508 bn_mul(p, p, t0);
509 bn_add_dig(p, p, 188);
510 bn_mul(p, p, t0);
511 bn_add_dig(p, p, 256);

515 bn_add_dig(p, p, 87);
516 bn_mul(p, p, t0);

518 bn_lsh(t1, t1, 8);

520 bn_add(p, p, t1);
521 bn_mul(p, p, t0);

540 bn_mul(p, p, t0);
541 bn_add_dig(p, p, 7);
542 bn_mul(p, p, t0);
543 bn_sub_dig(p, p, 1);
544 bn_mul(p, p, t0);
545 bn_sub_dig(p, p, 1);
546 bn_mul(p, p, t0);

549 bn_add_dig(p, p, 1);

556 break;
557 case EP_SG18:
558 /* p = 243x^10 - 162x^8 + 81*x^7 + 27x^6 - 54x^5 + 9x^4 + 9x^3 -
559 3x^2 + 1 */
560 bn_sqr(p, t0);
561 bn_mul_dig(p, p, 243);
562 bn_sub_dig(p, p, 162);
563 bn_mul(p, p, t0);
564 bn_add_dig(p, p, 81);
565 bn_mul(p, p, t0);
566 bn_add_dig(p, p, 27);
567 bn_mul(p, p, t0);
568 bn_sub_dig(p, p, 54);
569 bn_mul(p, p, t0);
570 bn_add_dig(p, p, 9);
571 bn_mul(p, p, t0);
572 bn_add_dig(p, p, 9);
573 bn_mul(p, p, t0);
574 bn_sub_dig(p, p, 3);
575 bn_mul(p, p, t0);

577 bn_add_dig(p, p, 1);

579 break;

584 bn_sqr(p, t1);
585 bn_sub(p, p, t1);
586 bn_add_dig(p, p, 1);
587 bn_sub_dig(t1, t0, 1);
588 bn_sqr(t1, t1);
589 bn_mul(p, p, t1);
590 bn_div_dig(p, p, 3);
591 bn_add(p, p, t0);
592 fp_prime_set_dense(p);
593 break;
594 case EP_B48:
595 /* p = (x - 1)^2*(x^16 - x^8 + 1) / 3 + x. */
596 bn_sqr(t1, t0);
597 bn_sqr(t1, t1);
598 bn_sqr(p, t1);
599 bn_sqr(t1, p);
600 bn_sub(t1, t1, p);
601 bn_add_dig(t1, t1, 1);
602 bn_sub_dig(p, t0, 1);
603 bn_sqr(p, p);

605 bn_div_dig(p, p, 3);
606 bn_add(p, p, t0);
607 fp_prime_set_dense(p);
608 break;
609 case EP_SG54:
610 /* p = (1+3*x+3*x^2+(3^5)*x^9+(3^5)*x^10+(3^6)*x^10+(3^6)*x^11+
611 (3^9)*x^18+(3^10)*x^19+(3^10)*x^20) */
612 bn_set_dig(p, 1);
613 bn_mul_dig(t1, t0, 3);
614 bn_add(p, p, t1);
615 bn_sqr(t1, t0);
616 bn_add(p, p, t1);
617 bn_add(p, p, t1);
618 bn_add(p, p, t1);
619 bn_sqr(t1, t1);
620 bn_sqr(t1, t1);
621 bn_mul(t1, t1, t0);
622 bn_mul_dig(t1, t1, 243);
623 bn_add(p, p, t1);
624 bn_mul(t1, t1, t0);
625 bn_add(p, p, t1);
626 bn_mul_dig(t1, t1, 3);
627 bn_add(p, p, t1);
628 bn_mul(t1, t1, t0);
629 bn_add(p, p, t1);
630 bn_mul_dig(t1, t1, 27);
631 bn_mul(t1, t1, t0);
