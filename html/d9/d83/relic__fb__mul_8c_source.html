54static void fb_mul_basic_imp(dig_t *c, const dig_t *a, const dig_t *b,
55 size_t size) {
56 dv_t s;
57
58 dv_null(s);
59
60 RLC_TRY {
61 /* We need a temporary variable so that c can be a or b. */
62 dv_new(s);
63 dv_zero(s, 2 * RLC_FB_DIGS);
64
65 dv_copy(s, b, size);
66 dv_zero(c, 2 * size);
67
68 if (a[0] & 1) {
69 dv_copy(c, b, size);
70 }
71 for (uint_t i = 1; i <= (RLC_DIG * size) - 1; i++) {
72 fb_lsh1_low(s, s);
73 fb_rdc(s, s);
74 if (fb_get_bit(a, i)) {
75 fb_add(c, c, s);
76 }
77 }
78 }
79 RLC_CATCH_ANY {
80 RLC_THROW(ERR_CAUGHT);
81 }
82 RLC_FINALLY {
83 dv_free(s);
84 }
85}

88#if FB_KARAT > 0 || !defined(STRIP)
99static void fb_mul_karat_imp(dv_t c, const fb_t a, const fb_t b, size_t size,
100 uint_t level) {
101 int i, h, h1;
102 dv_t a1, b1, ab;
103 dig_t *a0b0, *a1b1;
104
105 dv_null(a1);
106 dv_null(b1);
107 dv_null(ab);
108
109 /* Compute half the digits of a or b. */
110 h = size >> 1;
111 h1 = size - h;
112
113 RLC_TRY {
114 /* Allocate the temp variables. */
115 dv_new(a1);
116 dv_new(b1);
117 dv_new(ab);
118 a0b0 = ab;
119 a1b1 = ab + 2 * h;
120
121 /* a0b0 = a0 * b0 and a1b1 = a1 * b1 */
122 if (level <= 1) {
123#if FB_MUL == BASIC
124 fb_mul_basic_imp(a0b0, a, b, h);
125 fb_mul_basic_imp(a1b1, a + h, b + h, h1);
126#elif FB_MUL == INTEG || FB_MUL == LODAH
127 fb_muld_low(a0b0, a, b, h);
128 fb_muld_low(a1b1, a + h, b + h, h1);
129#endif
130 } else {
131 fb_mul_karat_imp(a0b0, a, b, h, level - 1);
132 fb_mul_karat_imp(a1b1, a + h, b + h, h1, level - 1);
133 }
134
135 for (i = 0; i < 2 * size; i++) {
136 c[i] = ab[i];
137 }
138
139 /* c = c - (a0*b0 << h digits) */
140 fb_addd_low(c + h, c + h, a0b0, 2 * h);
141
142 /* c = c - (a1*b1 << h digits) */
143 fb_addd_low(c + h, c + h, a1b1, 2 * h1);
144
145 /* a1 = (a1 + a0) */
146 fb_addd_low(a1, a, a + h, h);
147
148 /* b1 = (b1 + b0) */
149 fb_addd_low(b1, b, b + h, h);
150
151 if (h1 > h) {
152 a1[h1 - 1] = a[h + h1 - 1];
153 b1[h1 - 1] = b[h + h1 - 1];
154 }
155
156 if (level <= 1) {
157 /* a1b1 = (a1 + a0)*(b1 + b0) */
158#if FB_MUL == BASIC
159 fb_mul_basic_imp(a1b1, a1, b1, h1);
160#elif FB_MUL == INTEG || FB_MUL == LODAH
161 fb_muld_low(a1b1, a1, b1, h1);
162#endif
163 } else {
164 fb_mul_karat_imp(a1b1, a1, b1, h1, level - 1);
165 }
166
167 /* c = c + [(a1 + a0)*(b1 + b0) << digits] */
168 fb_addd_low(c + h, c + h, a1b1, 2 * h1);
169 }
170 RLC_CATCH_ANY {
171 RLC_THROW(ERR_CAUGHT);
172 }
173 RLC_FINALLY {
174 dv_free(a1);
175 dv_free(b1);
176 dv_free(ab);
177 }
178}
179
180#endif /* FB_KARAT > 0 || !defined(STRIP) */

187
188void fb_mul_basic(fb_t c, const fb_t a, const fb_t b) {
189 int i;
190 dv_t s;
191 fb_t t;
192
193 dv_null(s);
194 fb_null(t);
195
196 RLC_TRY {
197 /* We need a temporary variable so that c can be a or b. */
198 fb_new(t);
199 dv_new(s);
200 fb_zero(t);
201 dv_zero(s + RLC_FB_DIGS, RLC_FB_DIGS);
202 fb_copy(s, b);
203
204 if (a[0] & 1) {
205 fb_copy(t, b);
206 }
207 for (i = 1; i < RLC_FB_BITS; i++) {
208 /* We are already shifting a temporary value, so this is more efficient
209 * than calling fb_lsh(). */
210 s[RLC_FB_DIGS] = fb_lsh1_low(s, s);
211 fb_rdc(s, s);
212 if (fb_get_bit(a, i)) {
213 fb_add(t, t, s);
214 }
215 }
216
217 if (fb_bits(t) > RLC_FB_BITS) {
218 fb_poly_add(c, t);
219 } else {
220 fb_copy(c, t);
221 }
222 }
223 RLC_CATCH_ANY {
224 RLC_THROW(ERR_CAUGHT);
225 }
226 RLC_FINALLY {
227 fb_free(t);
228 fb_free(s);
229 }
230}
