44void pp_dbl_k18_basic(fp18_t l, ep3_t r, const ep3_t q, const ep_t p) {
45 fp3_t s;
46 ep3_t t;
47 int one = 1, zero = 0;
48
49 fp3_null(s);
50 ep3_null(t);
51
52 RLC_TRY {
53 fp3_new(s);
54 ep3_new(t);
55 ep3_copy(t, q);
56 ep3_dbl_slp_basic(r, s, q);
57
58 if (ep3_curve_is_twist() == RLC_EP_MTYPE) {
59 one ^= 1;
60 zero ^= 1;
61 }
62
63 fp_mul(l[one][zero][0], s[0], p->x);
64 fp_mul(l[one][zero][1], s[1], p->x);
65 fp_mul(l[one][zero][2], s[2], p->x);
66 fp3_mul(l[one][one], s, t->x);
67 fp3_sub(l[one][one], t->y, l[one][one]);
68 fp_copy(l[zero][zero][0], p->y);
69 }
70 RLC_CATCH_ANY {
71 RLC_THROW(ERR_CAUGHT);
72 }
73 RLC_FINALLY {
74 fp3_free(s);
75 ep3_free(t);
76 }
77}

85void pp_dbl_k18_projc_basic(fp18_t l, ep3_t r, const ep3_t q, const ep_t p) {
86 fp3_t t0, t1, t2, t3, t4, t5, t6;
87 int one = 1, zero = 0;
88
89 fp3_null(t0);
90 fp3_null(t1);
91 fp3_null(t2);
92 fp3_null(t3);
93 fp3_null(t4);
94 fp3_null(t5);
95 fp3_null(t6);
96
97 RLC_TRY {
98 fp3_new(t0);
99 fp3_new(t1);
100 fp3_new(t2);
101 fp3_new(t3);
102 fp3_new(t4);
103 fp3_new(t5);
104 fp3_new(t6);
105
106 if (ep3_curve_is_twist() == RLC_EP_MTYPE) {
107 one ^= 1;
108 zero ^= 1;
109 }
110
111 /* A = x1^2. */
112 fp3_sqr(t0, q->x);
113 /* B = y1^2. */
114 fp3_sqr(t1, q->y);
115 /* C = z1^2. */
116 fp3_sqr(t2, q->z);
117 /* D = 3bC, general b. */
118 fp3_dbl(t3, t2);
119 fp3_add(t3, t3, t2);
120 fp3_mul(t3, t3, ep3_curve_get_b());
121 /* E = (x1 + y1)^2 - A - B. */
122 fp3_add(t4, q->x, q->y);
123 fp3_sqr(t4, t4);
124 fp3_sub(t4, t4, t0);
125 fp3_sub(t4, t4, t1);
126
127 /* F = (y1 + z1)^2 - B - C. */
128 fp3_add(t5, q->y, q->z);
129 fp3_sqr(t5, t5);
130 fp3_sub(t5, t5, t1);
131 fp3_sub(t5, t5, t2);
132
133 /* G = 3D. */
134 fp3_dbl(t6, t3);
135 fp3_add(t6, t6, t3);
136
137 /* x3 = E * (B - G). */
138 fp3_sub(r->x, t1, t6);
139 fp3_mul(r->x, r->x, t4);
140
141 /* y3 = (B + G)^2 -12D^2. */
142 fp3_add(t6, t6, t1);
143 fp3_sqr(t6, t6);
144 fp3_sqr(t2, t3);
145 fp3_dbl(r->y, t2);
146 fp3_dbl(t2, r->y);
147 fp3_dbl(r->y, t2);
148 fp3_add(r->y, r->y, t2);
149 fp3_sub(r->y, t6, r->y);
150
151 /* z3 = 4B * F. */
152 fp3_dbl(r->z, t1);
153 fp3_dbl(r->z, r->z);
154 fp3_mul(r->z, r->z, t5);
155
156 /* l11 = D - B. */
157 fp3_sub(l[one][one], t3, t1);
158
159 /* l10 = (3 * xp) * A. */
160 fp_mul(l[one][zero][0], p->x, t0[0]);
161 fp_mul(l[one][zero][1], p->x, t0[1]);
162 fp_mul(l[one][zero][2], p->x, t0[2]);
163
164 /* l00 = F * (-yp). */
165 fp_mul(l[zero][zero][0], t5[0], p->y);
166 fp_mul(l[zero][zero][1], t5[1], p->y);
167 fp_mul(l[zero][zero][2], t5[2], p->y);
168
169 r->coord = PROJC;
170 }
171 RLC_CATCH_ANY {
172 RLC_THROW(ERR_CAUGHT);
173 }
174 RLC_FINALLY {
175 fp3_free(t0);
176 fp3_free(t1);
177 fp3_free(t2);
178 fp3_free(t3);
179 fp3_free(t4);
180 fp3_free(t5);
181 fp3_free(t6);
182 }
183}
