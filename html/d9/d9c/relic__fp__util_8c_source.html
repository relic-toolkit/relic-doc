109 void fp_rand(fp_t a) {
110  int bits, digits;
111 
112  rand_bytes((uint8_t *)a, FP_DIGS * sizeof(dig_t));
113 
114  SPLIT(bits, digits, FP_BITS, FP_DIG_LOG);
115  if (bits > 0) {
116  dig_t mask = ((dig_t)1 << (dig_t)bits) - 1;
117  a[FP_DIGS - 1] &= mask;
118  }
119 
120  while (fp_cmpn_low(a, fp_prime_get()) != CMP_LT) {
121  fp_subn_low(a, a, fp_prime_get());
122  }
