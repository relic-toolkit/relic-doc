88 int fp_bits(const fp_t a) {
89  int i = RLC_FP_DIGS - 1;
90 
91  while (i >= 0 && a[i] == 0) {
92  i--;
93  }
94 
95  if (i > 0) {

106 void fp_rand(fp_t a) {
107  int bits, digits;
108 
109  rand_bytes((uint8_t *)a, RLC_FP_DIGS * sizeof(dig_t));
110 
111  RLC_RIP(bits, digits, RLC_FP_BITS);
112  if (bits > 0) {
113  dig_t mask = ((dig_t)1 << (dig_t)bits) - 1;
114  a[RLC_FP_DIGS - 1] &= mask;
115  }
116 
117  while (dv_cmp(a, fp_prime_get(), RLC_FP_DIGS) != RLC_LT) {
118  fp_subn_low(a, a, fp_prime_get());
119  }
120 }
