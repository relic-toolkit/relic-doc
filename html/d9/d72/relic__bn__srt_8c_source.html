39 void bn_srt(bn_t c, bn_t a) {
40  bn_t h, l, m, t;
41  int bits, cmp;
42 
43  if (bn_sign(a) == RLC_NEG) {
44  THROW(ERR_NO_VALID);
45  }
46 
47  bits = bn_bits(a);
48  bits += (bits % 2);
49 
50  bn_null(h);
51  bn_null(l);
52  bn_null(m);
53  bn_null(t);
54 
55  TRY{
56  bn_new(h);
57  bn_new(l);
58  bn_new(m);
59  bn_new(t);
60 
61  bn_set_2b(h, bits >> 1);
62  bn_set_2b(l, (bits >> 1) - 1);
63 
64  /* Trivial binary search approach. */
65  do {
66  bn_add(m, h, l);
67  bn_hlv(m, m);
68  bn_sqr(t, m);
69  cmp = bn_cmp(t, a);
70 
71  if (cmp == RLC_GT) {
72  bn_copy(h, m);
73  } else if (cmp == RLC_LT) {
74  bn_copy(l, m);
75  }
76  bn_sub(t, h, l);
77  } while (bn_cmp_dig(t, 1) == RLC_GT && cmp != RLC_EQ);
78 
79  bn_copy(c, m);
80  }
81  CATCH_ANY {
82  THROW(ERR_CAUGHT);
83  }
84  FINALLY {
85  bn_free(h);
86  bn_free(l);
87  bn_free(m);
88  bn_free(t);
89  }
90 }
