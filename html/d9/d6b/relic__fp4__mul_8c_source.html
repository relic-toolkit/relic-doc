42 void fp4_mul_basic(fp4_t c, fp4_t a, fp4_t b) {
43  fp2_t t0, t1, t2;
44 
45  fp2_null(t0);
46  fp2_null(t1);
47  fp2_null(t2);
48 
49  TRY {
50  fp2_new(t0);
51  fp2_new(t1);
52  fp2_new(t2);
53 
54  /* Karatsuba algorithm. */
55 
56  /* t0 = a_0 * b_0. */
57  fp2_mul(t0, a[0], b[0]);
58  /* t1 = a_1 * b_1. */
59  fp2_mul(t1, a[1], b[1]);
60  /* t2 = b_0 + b_1. */
61  fp2_add(t2, b[0], b[1]);
62 
63  /* c_1 = a_0 + a_1. */
64  fp2_add(c[1], a[0], a[1]);
65 
66  /* c_1 = (a_0 + a_1) * (b_0 + b_1) */
67  fp2_mul(c[1], c[1], t2);
68  fp2_sub(c[1], c[1], t0);
69  fp2_sub(c[1], c[1], t1);
70 
71  /* c_0 = a_0b_0 + v * a_1b_1. */
72  fp2_mul_nor(t2, t1);
73  fp2_add(c[0], t0, t2);
74  } CATCH_ANY {
75  THROW(ERR_CAUGHT);
76  } FINALLY {
77  fp2_free(t0);
78  fp2_free(t1);
79  fp2_free(t2);
80  }
81 }
