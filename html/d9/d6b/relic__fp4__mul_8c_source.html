42void fp4_mul_basic(fp4_t c, const fp4_t a, const fp4_t b) {
43 fp2_t t0, t1, t2;
44
45 fp2_null(t0);
46 fp2_null(t1);
47 fp2_null(t2);
48
49 RLC_TRY {
50 fp2_new(t0);
51 fp2_new(t1);
52 fp2_new(t2);
53
54 /* Karatsuba algorithm. */
55
56 /* t0 = a_0 * b_0. */
57 fp2_mul(t0, a[0], b[0]);
58 /* t1 = a_1 * b_1. */
59 fp2_mul(t1, a[1], b[1]);
60 /* t2 = b_0 + b_1. */
61 fp2_add(t2, b[0], b[1]);
62
63 /* c_1 = a_0 + a_1. */
64 fp2_add(c[1], a[0], a[1]);
65
66 /* c_1 = (a_0 + a_1) * (b_0 + b_1) */
67 fp2_mul(c[1], c[1], t2);
68 fp2_sub(c[1], c[1], t0);
69 fp2_sub(c[1], c[1], t1);
70
71 /* c_0 = a_0b_0 + v * a_1b_1. */
72 fp2_mul_nor(t2, t1);
73 fp2_add(c[0], t0, t2);
74 } RLC_CATCH_ANY {
75 RLC_THROW(ERR_CAUGHT);
76 } RLC_FINALLY {
77 fp2_free(t0);
78 fp2_free(t1);
79 fp2_free(t2);
80 }
81}

98 fp2_new(t1);
99 dv2_new(u0);
100 dv2_new(u1);
101
102 fp2_muln_low(u0, a[0], b[0]);
103 fp2_muln_low(u1, a[1], b[1]);
104 fp2_addm_low(t0, b[0], b[1]);
105 fp2_addm_low(t1, a[0], a[1]);
106
107 fp2_muln_low(c[1], t1, t0);
108 fp2_subc_low(c[1], c[1], u0);
109 fp2_subc_low(c[1], c[1], u1);
110
111 fp2_nord_low(c[0], u1);

113 } RLC_CATCH_ANY {
114 RLC_THROW(ERR_CAUGHT);
115 } RLC_FINALLY {
116 fp2_free(t0);

142void fp4_mul_art(fp4_t c, const fp4_t a) {
143 fp2_t t0;
144
145 fp2_null(t0);
146
147 RLC_TRY {
148 fp2_new(t0);
149
150 /* (a_0 + a_1 * v) * v = a_0 * v + a_1 * v^2 */
151 fp2_copy(t0, a[0]);
152 fp2_mul_nor(c[0], a[1]);
153 fp2_copy(c[1], t0);
154 } RLC_CATCH_ANY {
155 RLC_THROW(ERR_CAUGHT);
156 } RLC_FINALLY {
157 fp2_free(t0);
158 }

161void fp4_mul_frb(fp4_t c, const fp4_t a, int i, int j) {
162 fp2_t t;
163
164 fp2_null(t);
165
166 RLC_TRY {
167 fp2_new(t);
168
169 fp_copy(t[0], core_get()->fp4_p1[0]);
170 fp_copy(t[1], core_get()->fp4_p1[1]);
171
172 if (i == 1) {
173 fp4_copy(c, a);
174 for (int k = 0; k < j; k++) {
175 fp2_mul(c[0], c[0], t);
176 fp2_mul(c[1], c[1], t);
177 if (ep_curve_is_pairf() == EP_FM16) {
178 /* TODO: fix this ugly hack. */
179 fp4_mul_art(c, c);
180 }
181 /* If constant in base field, then second component is zero. */
182 if (core_get()->frb4 == 1) {
183 fp4_mul_art(c, c);
184 if (fp_prime_get_mod18() % 3 == 2) {
185 fp4_mul_art(c, c);
186 }
187 }
188 }
189 } else {
190 RLC_THROW(ERR_NO_VALID);
191 }
192 } RLC_CATCH_ANY {
193 RLC_THROW(ERR_CAUGHT);
194 } RLC_FINALLY {
195 fp2_free(t);
196 }
