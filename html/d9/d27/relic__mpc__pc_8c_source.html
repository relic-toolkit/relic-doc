40void g1_mul_lcl(bn_t d, g1_t q, const bn_t x, const g1_t p, const mt_t tri) {
41 bn_t n;
42
43 bn_null(n);
44
45 RLC_TRY {
46 bn_new(n);
47
48 /* Compute public values for transmission. */
49
50 /* [d] = [x] - [a]. */
51 g1_get_ord(n);
52 bn_sub(d, x, tri->a);
53 if (bn_sign(d) == RLC_NEG) {
54 bn_add(d, d, n);
55 }
56 bn_mod(d, d, n);
57
58 /* [Q] = [P] - [B] = [b]G. */
59 /* Copy first to avoid overwriting P. */
60 g1_copy(q, p);
61 g1_sub(q, q, *tri->b1);
62 g1_norm(q, q);
63 } RLC_CATCH_ANY {
64 RLC_THROW(ERR_CAUGHT);
65 } RLC_FINALLY {
66 bn_free(n);
67 }
68}

92void g1_mul_mpc(g1_t r, const bn_t d, const g1_t q, const mt_t tri, int party) {
93 g1_t t;
94
95 g1_null(t);
96
97 RLC_TRY {
98 g1_new(t);
99
100 if (party == 0) {
101 /* For one party, compute [B] + Q. */
102 g1_add(t, *tri->b1, q);
103 g1_norm(t, t);
104 } else {
105 g1_copy(t, *tri->b1);
106 }
107 /* Compute [a]Q + d([B] + Q) or d[B]. */
108 g1_mul_sim(r, q, tri->a, t, d);
109 /* R = [a]Q + d[B] + dQ + [C] = [xy] */
110 g1_add(r, r, *tri->c1);
111 g1_norm(r, r);
112 } RLC_CATCH_ANY {
113 RLC_THROW(ERR_CAUGHT);
114 } RLC_FINALLY {
115 g1_free(t);

323void pc_map_mpc(gt_t r, const g1_t d1, const g2_t d2, const pt_t triple,
324 int party) {
325 gt_t t;
326 g1_t _p[2];
327 g2_t _q[2];
328
329 gt_null(t);
330
331 RLC_TRY {
332 gt_new(t);
333 for (int i = 0; i < 2; i++) {
334 g1_null(_p[i]);
335 g2_null(_q[i]);
336 g1_new(_p[i]);
337 g2_new(_q[i]);
338 }
339
340 /* Compute the pairing in MPC. */
341 if (party == 0) {
342 g1_add(_p[0], triple->a, d1);
343 g1_norm(_p[0], _p[0]);
344 g2_copy(_q[0], d2);
345 } else {
346 g1_copy(_p[0], triple->a);
347 g2_copy(_q[0], d2);
348 }
349 g1_copy(_p[1], d1);
350 g2_copy(_q[1], triple->b);
351 pc_map_sim(t, _p, _q, 2);
352 gt_mul(r, triple->c, t);
353 } RLC_CATCH_ANY {
354 RLC_THROW(ERR_CAUGHT);
355 } RLC_FINALLY {
356 gt_free(t);
357 for (int i = 0; i < 2; i++) {
358 g1_free(_p[i]);
359 g2_free(_q[i]);
360 g1_free(_p[i]);
361 g2_free(_q[i]);
362 }
363 }
