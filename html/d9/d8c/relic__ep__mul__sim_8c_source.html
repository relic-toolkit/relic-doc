55static void ep_mul_sim_endom(ep_t r, const ep_t p, const bn_t k, const ep_t q,
56 const bn_t m, const ep_t *t) {
57 int i, sk0, sk1, sl0, sl1, w, g = 0;
58 int8_t naf0[RLC_FP_BITS + 1], naf1[RLC_FP_BITS + 1], *t0, *t1, u;
59 int8_t naf2[RLC_FP_BITS + 1], naf3[RLC_FP_BITS + 1], *t2, *t3;
60 bn_t n, k0, k1, m0, m1;
61 bn_t v1[3], v2[3];
62 ep_t v;
63 ep_t tab0[1 << (RLC_WIDTH - 2)];
64 ep_t tab1[1 << (RLC_WIDTH - 2)];
65 size_t l, l0, l1, l2, l3;
66
67 bn_null(n);
68 bn_null(k0);
69 bn_null(k1);
70 bn_null(m0);
71 bn_null(m1);
72 ep_null(v);
73
74 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
75 ep_null(tab0[i]);
76 ep_null(tab1[i]);
77 }
78
79 RLC_TRY {
80 bn_new(n);
81 bn_new(k0);
82 bn_new(k1);
83 bn_new(m0);
84 bn_new(m1);
85 ep_new(v);
86
87 for (i = 0; i < 3; i++) {
88 bn_null(v1[i]);
89 bn_null(v2[i]);
90 bn_new(v1[i]);
91 bn_new(v2[i]);
92 }
93
94 ep_curve_get_ord(n);
95 ep_curve_get_v1(v1);
96 ep_curve_get_v2(v2);
97
98 bn_rec_glv(k0, k1, k, n, (const bn_t *)v1, (const bn_t *)v2);
99 sk0 = bn_sign(k0);
100 sk1 = bn_sign(k1);
101 bn_abs(k0, k0);
102 bn_abs(k1, k1);
103
104 bn_rec_glv(m0, m1, m, n, (const bn_t *)v1, (const bn_t *)v2);
105 sl0 = bn_sign(m0);
106 sl1 = bn_sign(m1);
107 bn_abs(m0, m0);
108 bn_abs(m1, m1);
109
110 g = (t == NULL ? 0 : 1);
111 if (!g) {
112 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
113 ep_new(tab0[i]);
114 }
115 ep_tab(tab0, p, RLC_WIDTH);
116 t = (const ep_t *)tab0;
117 }
118
119 /* Prepare the precomputation table. */
120 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
121 ep_new(tab1[i]);
122 }
123 /* Compute the precomputation table. */
124 ep_tab(tab1, q, RLC_WIDTH);
125
126 /* Compute the w-TNAF representation of k and l */
127 if (g) {
128 w = RLC_DEPTH;
129 } else {
130 w = RLC_WIDTH;
131 }
132 l0 = l1 = l2 = l3 = RLC_FP_BITS + 1;
133 bn_rec_naf(naf0, &l0, k0, w);
134 bn_rec_naf(naf1, &l1, k1, w);
135 bn_rec_naf(naf2, &l2, m0, RLC_WIDTH);
136 bn_rec_naf(naf3, &l3, m1, RLC_WIDTH);
137
138 l = RLC_MAX(RLC_MAX(l0, l1), RLC_MAX(l2, l3));
139 t0 = naf0 + l - 1;
140 t1 = naf1 + l - 1;
141 t2 = naf2 + l - 1;
142 t3 = naf3 + l - 1;
143
144 if (bn_sign(k) == RLC_NEG) {
145 for (i = 0; i < l0; i++) {
146 naf0[i] = -naf0[i];
147 }
148 for (i = 0; i < l1; i++) {
149 naf1[i] = -naf1[i];
150 }
151 }
152 if (bn_sign(m) == RLC_NEG) {
153 for (i = 0; i < l2; i++) {
154 naf2[i] = -naf2[i];
155 }
156 for (i = 0; i < l3; i++) {
157 naf3[i] = -naf3[i];
158 }
159 }
160
161 ep_set_infty(r);
162 for (i = l - 1; i >= 0; i--, t0--, t1--, t2--, t3--) {
163 ep_dbl(r, r);
164
165 u = *t0;
166 if (u > 0) {
167 if (sk0 == RLC_POS) {
168 ep_add(r, r, t[u / 2]);
169 } else {
170 ep_sub(r, r, t[u / 2]);
171 }
172 }
173 if (u < 0) {
174 if (sk0 == RLC_POS) {
175 ep_sub(r, r, t[-u / 2]);
176 } else {
177 ep_add(r, r, t[-u / 2]);
178 }
179 }
180 u = *t1;
181 if (u > 0) {
182 ep_psi(v, t[u / 2]);
183 if (sk1 == RLC_NEG) {
184 ep_neg(v, v);
185 }
186 ep_add(r, r, v);
187 }
188 if (u < 0) {
189 ep_psi(v, t[-u / 2]);
190 if (sk1 == RLC_NEG) {
191 ep_neg(v, v);
192 }
193 ep_sub(r, r, v);
194 }
195
196 u = *t2;
197 if (u > 0) {
198 if (sl0 == RLC_POS) {
199 ep_add(r, r, tab1[u / 2]);
200 } else {
201 ep_sub(r, r, tab1[u / 2]);
202 }
203 }
204 if (u < 0) {
205 if (sl0 == RLC_POS) {
206 ep_sub(r, r, tab1[-u / 2]);
207 } else {
208 ep_add(r, r, tab1[-u / 2]);
209 }
210 }
211 u = *t3;
212 if (u > 0) {
213 ep_psi(v, tab1[u / 2]);
214 if (sl1 == RLC_NEG) {
215 ep_neg(v, v);
216 }
217 ep_add(r, r, v);
218 }
219 if (u < 0) {
220 ep_psi(v, tab1[-u / 2]);
221 if (sl1 == RLC_NEG) {
222 ep_neg(v, v);
223 }
224 ep_sub(r, r, v);
225 }
226 }
227 /* Convert r to affine coordinates. */
228 ep_norm(r, r);
229 }
230 RLC_CATCH_ANY {
231 RLC_THROW(ERR_CAUGHT);
232 }
233 RLC_FINALLY {
234 bn_free(n);
235 bn_free(k0);
236 bn_free(k1);
237 bn_free(m0);
238 bn_free(m1);
239 ep_free(v);
240
241 if (!g) {
242 for (i = 0; i < 1 << (RLC_WIDTH - 2); i++) {
243 ep_free(tab0[i]);
244 }
245 }
246 /* Free the precomputation tables. */
247 for (i = 0; i < 1 << (RLC_WIDTH - 2); i++) {
248 ep_free(tab1[i]);
249 }
250 for (i = 0; i < 3; i++) {
251 bn_free(v1[i]);
252 bn_free(v2[i]);
253 }
254 }
255}

266void ep_mul_sim_lot_endom(ep_t r, const ep_t p[], const bn_t k[], int n) {
267 const int len = RLC_FP_BITS + 1;
268 int i, j, m, sk;
269 bn_t _k[2], q, v1[3], v2[3];
270 int8_t ptr, *naf = RLC_ALLOCA(int8_t, 2 * n * len);
271 size_t l, _l[2];
272
273 bn_null(q);
274
275 if (n <= 10) {
276 ep_t *_p = RLC_ALLOCA(ep_t, 2 * n);
277
278 RLC_TRY {
279 if (naf == NULL || _p == NULL) {
280 RLC_THROW(ERR_NO_MEMORY);
281 }
282 bn_new(q);
283 for (j = 0; j < 2; j++) {
284 bn_null(_k[j]);
285 bn_new(_k[j]);
286 }
287 for (i = 0; i < 2 * n; i++) {
288 ep_null(_p[i]);
289 ep_new(_p[i]);
290 }
291
292 for (i = 0; i < 3; i++) {
293 bn_null(v1[i]);
294 bn_null(v2[i]);
295 bn_new(v1[i]);
296 bn_new(v2[i]);
297 }
298
299 l = 0;
300 ep_curve_get_ord(q);
301 ep_curve_get_v1(v1);
302 ep_curve_get_v2(v2);
303 for (i = 0; i < n; i++) {
304 ep_norm(_p[2*i], p[i]);
305 ep_psi(_p[2*i + 1], _p[2*i]);
306 bn_mod(_k[0], k[i], q);
307 sk = bn_sign(_k[0]);
308 bn_rec_glv(_k[0], _k[1], _k[0], q, (const bn_t *)v1, (const bn_t *)v2);
309 if (sk == RLC_NEG) {
310 bn_neg(_k[0], _k[0]);
311 bn_neg(_k[1], _k[1]);
312 }
313 for (j = 0; j < 2; j++) {
314 _l[j] = len;
315 bn_rec_naf(&naf[(2*i + j)*len], &_l[j], _k[j], 2);
316 if (bn_sign(_k[j]) == RLC_NEG) {
317 ep_neg(_p[2*i + j], _p[2*i + j]);
318 }
319 l = RLC_MAX(l, _l[j]);
320 }
321 }
322
323 ep_set_infty(r);
324 for (i = l - 1; i >= 0; i--) {
325 ep_dbl(r, r);
326 for (j = 0; j < n; j++) {
327 for (m = 0; m < 2; m++) {
328 if (naf[(2*j + m)*len + i] > 0) {
329 ep_add(r, r, _p[2*j + m]);
330 }
331 if (naf[(2*j + m)*len + i] < 0) {
332 ep_sub(r, r, _p[2*j + m]);
333 }
334 }
335 }
336 }
337 ep_norm(r, r);
338 } RLC_CATCH_ANY {
339 RLC_THROW(ERR_CAUGHT);
340 } RLC_FINALLY {
341 bn_free(q);
342 bn_free(_k[0]);
343 bn_free(_k[1]);
344 for (i = 0; i < 2 * n; i++) {
345 ep_free(_p[i]);
346 }
347 RLC_FREE(_p);
348 RLC_FREE(naf);
349 for (i = 0; i < 3; i++) {
350 bn_free(v1[i]);
351 bn_free(v2[i]);
352 }
353 }
354 } else {
355 const int w = RLC_MAX(2, util_bits_dig(n) - 2), c = (1 << (w - 2));
356 ep_t s, t, u, v, *_p = RLC_ALLOCA(ep_t, 2 * c);
357
358 ep_null(s);
359 ep_null(t);
360 ep_null(u);
361 ep_null(v);
362
363 RLC_TRY {
364 if (naf == NULL || _p == NULL) {
365 RLC_THROW(ERR_NO_MEMORY);
366 }
367 bn_new(q);
368 ep_new(s);
369 ep_new(t);
370 ep_new(u);
371 ep_new(v);
372 for (i = 0; i < 2; i++) {
373 bn_null(_k[i]);
374 bn_new(_k[i]);
375 for (j = 0; j < c; j++) {
376 ep_null(_p[i*c + j]);
377 ep_new(_p[i*c + j]);
378 ep_set_infty(_p[i*c + j]);
379 }
380 }
381 for (i = 0; i < 3; i++) {
382 bn_null(v1[i]);
383 bn_null(v2[i]);
384 bn_new(v1[i]);
385 bn_new(v2[i]);
386 }
387
388 l = 0;
389 ep_curve_get_ord(q);
390 ep_curve_get_v1(v1);
391 ep_curve_get_v2(v2);
392 for (i = 0; i < n; i++) {
393 bn_mod(_k[0], k[i], q);
394 sk = bn_sign(_k[0]);
395 bn_rec_glv(_k[0], _k[1], _k[0], q, (const bn_t *)v1, (const bn_t *)v2);
396 if (sk == RLC_NEG) {
397 bn_neg(_k[0], _k[0]);
398 bn_neg(_k[1], _k[1]);
399 }
400 for (j = 0; j < 2; j++) {
401 _l[j] = len;
402 bn_rec_naf(&naf[(2*i + j)*len], &_l[j], _k[j], w);
403 if (bn_sign(_k[j]) == RLC_NEG) {
404 for (m = 0; m < _l[j]; m++) {
405 naf[(2*i + j)*len + m] = -naf[(2*i + j)*len + m];
406 }
407 }
408 l = RLC_MAX(l, _l[j]);
409 }
410 }
411
412 ep_set_infty(s);
413 for (i = l - 1; i >= 0; i--) {
414 for (j = 0; j < n; j++) {
415 for (m = 0; m < 2; m++) {
416 ptr = naf[(2*j + m)*len + i];
417 if (ptr != 0) {
418 ep_copy(t, p[j]);
419 if (ptr < 0) {
420 ptr = -ptr;
421 ep_neg(t, t);
422 }
423 ep_add(_p[m*c + (ptr >> 1)], _p[m*c + (ptr >> 1)], t);
424 }
425 }
426 }
427
428 ep_set_infty(t);
429 for (m = 1; m >= 0; m--) {
430 ep_psi(t, t);
431 ep_set_infty(u);
432 ep_set_infty(v);
433 for (j = c - 1; j >= 0; j--) {
434 ep_add(u, u, _p[m*c + j]);
435 if (j == 0) {
436 ep_dbl(v, v);
437 }
438 ep_add(v, v, u);
439 ep_set_infty(_p[m*c + j]);
440 }
441 ep_add(t, t, v);
442 }
443 ep_dbl(s, s);
444 ep_add(s, s, t);
445 }
446
447 /* Convert r to affine coordinates. */
448 ep_norm(r, s);
449 } RLC_CATCH_ANY {
450 RLC_THROW(ERR_CAUGHT);
451 } RLC_FINALLY {
452 bn_free(q);
453 ep_free(s);
454 ep_free(t);
455 ep_free(u);
456 ep_free(v);
457 for (i = 0; i < 2; i++) {
458 bn_free(_k[i]);
459 for (j = 0; j < c; j++) {
460 ep_free(_p[i*c + j]);
461 }
462 }
463 RLC_FREE(_p);
464 RLC_FREE(naf);
465 for (i = 0; i < 3; i++) {
466 bn_free(v1[i]);
467 bn_free(v2[i]);
468 }
469 }
470 }
471}

489static void ep_mul_sim_plain(ep_t r, const ep_t p, const bn_t k, const ep_t q,
490 const bn_t m, const ep_t *t) {
491 int i, w, gen = (t == NULL ? 0 : 1);
492 int8_t naf0[RLC_FP_BITS + 1], naf1[RLC_FP_BITS + 1], n0, n1, *u, *v;
493 ep_t t0[1 << (RLC_WIDTH - 2)];
494 ep_t t1[1 << (RLC_WIDTH - 2)];
495 size_t l, l0, l1;
496
497 RLC_TRY {
498 if (!gen) {
499 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
500 ep_null(t0[i]);
501 ep_new(t0[i]);
502 }
503 ep_tab(t0, p, RLC_WIDTH);
504 t = (const ep_t *)t0;
505 }
506
507 /* Prepare the precomputation table. */
508 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
509 ep_null(t1[i]);
510 ep_new(t1[i]);
511 }
512
513 /* Compute the precomputation table. */
514 ep_tab(t1, q, RLC_WIDTH);
515
516 /* Compute the w-TNAF representation of k. */
517 if (gen) {
518 w = RLC_DEPTH;
519 } else {
520 w = RLC_WIDTH;
521 }
522 l0 = l1 = RLC_FP_BITS + 1;
523 bn_rec_naf(naf0, &l0, k, w);
524 bn_rec_naf(naf1, &l1, m, RLC_WIDTH);
525
526 l = RLC_MAX(l0, l1);
527 if (bn_sign(k) == RLC_NEG) {
528 for (i = 0; i < l0; i++) {
529 naf0[i] = -naf0[i];
530 }
531 }
532 if (bn_sign(m) == RLC_NEG) {
533 for (i = 0; i < l1; i++) {
534 naf1[i] = -naf1[i];
535 }
536 }
537
538 u = naf0 + l - 1;
539 v = naf1 + l - 1;
540 ep_set_infty(r);
541 for (i = l - 1; i >= 0; i--, u--, v--) {
542 ep_dbl(r, r);
543
544 n0 = *u;
545 n1 = *v;
546 if (n0 > 0) {
547 ep_add(r, r, t[n0 / 2]);
548 }
549 if (n0 < 0) {
550 ep_sub(r, r, t[-n0 / 2]);
551 }
552 if (n1 > 0) {
553 ep_add(r, r, t1[n1 / 2]);
554 }
555 if (n1 < 0) {
556 ep_sub(r, r, t1[-n1 / 2]);
557 }
558 }
559 /* Convert r to affine coordinates. */
560 ep_norm(r, r);
561 }
562 RLC_CATCH_ANY {
563 RLC_THROW(ERR_CAUGHT);
564 }
565 RLC_FINALLY {
566 /* Free the precomputation tables. */
567 if (!gen) {
568 for (i = 0; i < 1 << (RLC_WIDTH - 2); i++) {
569 ep_free(t0[i]);
570 }
571 }
572 for (i = 0; i < 1 << (RLC_WIDTH - 2); i++) {
573 ep_free(t1[i]);
574 }
575 }
576}

587void ep_mul_sim_lot_plain(ep_t r, const ep_t p[], const bn_t k[], int n) {
588 int i, j;
589 int8_t *naf = NULL;
590 ep_t *_p = RLC_ALLOCA(ep_t, n);
591 size_t l, *_l = RLC_ALLOCA(size_t, n);
592
593 RLC_TRY {
594 l = 0;
595 for (i = 0; i < n; i++) {
596 l = RLC_MAX(l, bn_bits(k[i]) + 1);
597 }
598 naf = RLC_ALLOCA(int8_t, n * l);
599 if (naf == NULL || _p == NULL || _l == NULL) {
600 RLC_THROW(ERR_NO_MEMORY);
601 }
602
603 for (i = 0; i < n; i++) {
604 ep_null(_p[i]);
605 ep_new(_p[i]);
606 }
607
608 for (i = 0; i < n; i++) {
609 _l[i] = l;
610 ep_norm(_p[i], p[i]);
611 bn_rec_naf(&naf[i*l], &_l[i], k[i], 2);
612 if (bn_sign(k[i]) == RLC_NEG) {
613 ep_neg(_p[i], _p[i]);
614 }
615 }
616
617 ep_set_infty(r);
618 for (i = l - 1; i >= 0; i--) {
619 ep_dbl(r, r);
620 for (j = 0; j < n; j++) {
621 if (naf[j*l + i] > 0) {
622 ep_add(r, r, _p[j]);
623 }
624 if (naf[j*l + i] < 0) {
625 ep_sub(r, r, _p[j]);
626 }
627 }
628 }
629
630 /* Convert r to affine coordinates. */
631 ep_norm(r, r);
632 } RLC_CATCH_ANY {
633 RLC_THROW(ERR_CAUGHT);
634 } RLC_FINALLY {
635 for (i = 0; i < n; i++) {
636 ep_free(_p[i]);
637 }
638 RLC_FREE(_l);
639 RLC_FREE(_p);
640 RLC_FREE(naf);
641 }
642}

679void ep_mul_sim_trick(ep_t r, const ep_t p, const bn_t k, const ep_t q,
680 const bn_t m) {
681 ep_t t0[1 << (RLC_WIDTH / 2)], t1[1 << (RLC_WIDTH / 2)], t[1 << RLC_WIDTH];
682 bn_t n, _k, _m;
683 size_t l0, l1, w = RLC_WIDTH / 2;
684 uint8_t w0[RLC_FP_BITS + 1], w1[RLC_FP_BITS + 1];
685
686 if (bn_is_zero(k) || ep_is_infty(p)) {
687 ep_mul(r, q, m);
688 return;
689 }
690 if (bn_is_zero(m) || ep_is_infty(q)) {
691 ep_mul(r, p, k);
692 return;
693 }
694
695 bn_null(n);
696 bn_null(_k);
697 bn_null(_m);
698
699 RLC_TRY {
700 bn_new(n);
701 bn_new(_k);
702 bn_new(_m);
703
704 for (int i = 0; i < (1 << w); i++) {
705 ep_null(t0[i]);
706 ep_null(t1[i]);
707 ep_new(t0[i]);
708 ep_new(t1[i]);
709 }
710 for (int i = 0; i < (1 << RLC_WIDTH); i++) {
711 ep_null(t[i]);
712 ep_new(t[i]);
713 }
714
715 ep_curve_get_ord(n);
716 bn_mod(_k, k, n);
717 bn_mod(_m, m, n);
718
719 ep_set_infty(t0[0]);
720 ep_copy(t0[1], p);
721 for (int i = 2; i < (1 << w); i++) {
722 ep_add(t0[i], t0[i - 1], t0[1]);
723 }
724
725 ep_set_infty(t1[0]);
726 ep_copy(t1[1], q);
727 for (int i = 2; i < (1 << w); i++) {
728 ep_add(t1[i], t1[i - 1], t1[1]);
729 }
730
731 for (int i = 0; i < (1 << w); i++) {
732 for (int j = 0; j < (1 << w); j++) {

828void ep_mul_sim_joint(ep_t r, const ep_t p, const bn_t k, const ep_t q,
829 const bn_t m) {
830 bn_t n, _k, _m;
831 ep_t t[5];
832 int i, u_i, offset;
833 int8_t jsf[2 * (RLC_FP_BITS + 1)];
834 size_t l;
835
836 if (bn_is_zero(k) || ep_is_infty(p)) {
837 ep_mul(r, q, m);
838 return;
839 }
840 if (bn_is_zero(m) || ep_is_infty(q)) {
841 ep_mul(r, p, k);
842 return;
843 }
844
845 bn_null(n);
846 bn_null(_k);
847 bn_null(_m);
848
849 RLC_TRY {
850 bn_new(n);
851 bn_new(_k);
852 bn_new(_m);
853 for (i = 0; i < 5; i++) {
854 ep_null(t[i]);
855 ep_new(t[i]);
856 }
857
858 ep_curve_get_ord(n);
859 bn_mod(_k, k, n);
860 bn_mod(_m, m, n);
861
862 ep_set_infty(t[0]);
863 ep_copy(t[1], q);
864 if (bn_sign(_m) == RLC_NEG) {
865 ep_neg(t[1], t[1]);
866 }
867 ep_copy(t[2], p);
868 if (bn_sign(_k) == RLC_NEG) {
869 ep_neg(t[2], t[2]);
870 }
871 ep_add(t[3], t[2], t[1]);
872 ep_sub(t[4], t[2], t[1]);
873#if defined(EP_MIXED)
874 ep_norm_sim(t + 3, (const ep_t *)t + 3, 2);
875#endif
876
877 l = 2 * (RLC_FP_BITS + 1);
878 bn_rec_jsf(jsf, &l, _k, _m);
879
880 ep_set_infty(r);
881
882 offset = RLC_MAX(bn_bits(_k), bn_bits(_m)) + 1;
883 for (i = l - 1; i >= 0; i--) {
884 ep_dbl(r, r);
885 if (jsf[i] != 0 && jsf[i] == -jsf[i + offset]) {
886 u_i = jsf[i] * 2 + jsf[i + offset];
887 if (u_i < 0) {
888 ep_sub(r, r, t[4]);
889 } else {
890 ep_add(r, r, t[4]);
891 }
892 } else {
893 u_i = jsf[i] * 2 + jsf[i + offset];
894 if (u_i < 0) {
895 ep_sub(r, r, t[-u_i]);
896 } else {
897 ep_add(r, r, t[u_i]);
898 }
899 }
900 }
901 ep_norm(r, r);
902 }
903 RLC_CATCH_ANY {
904 RLC_THROW(ERR_CAUGHT);
905 }
906 RLC_FINALLY {
907 bn_free(n);
908 bn_free(_k);
909 bn_free(_m);
910 for (i = 0; i < 5; i++) {
911 ep_free(t[i]);
912 }
913 }
914}
