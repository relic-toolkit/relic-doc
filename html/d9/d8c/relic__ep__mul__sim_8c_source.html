79  bn_new(k0);

105  sl1 = bn_sign(m1);
106  bn_abs(m0, m0);
107  bn_abs(m1, m1);
108

110  if (!g) {
111  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
112  ep_new(tab0[i]);
113  }
114  ep_tab(tab0, p, EP_WIDTH);
115  t = (const ep_t *)tab0;
116  }
117 
118  /* Prepare the precomputation table. */
119  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
120  ep_new(tab1[i]);
121  }
122  /* Compute the precomputation table. */
123  ep_tab(tab1, q, EP_WIDTH);
124 
125  /* Compute the w-TNAF representation of k and l */
126  if (g) {
127  w = EP_DEPTH;
128  } else {
129  w = EP_WIDTH;
130  }
131  l0 = l1 = l2 = l3 = RLC_FP_BITS + 1;
132  bn_rec_naf(naf0, &l0, k0, w);
133  bn_rec_naf(naf1, &l1, k1, w);
134  bn_rec_naf(naf2, &l2, m0, EP_WIDTH);

138  t0 = naf0 + l - 1;
139  t1 = naf1 + l - 1;
140  t2 = naf2 + l - 1;
141  t3 = naf3 + l - 1;
142  for (i = l0; i < l; i++) {
143  naf0[i] = 0;
144  }
145  for (i = l1; i < l; i++) {
146  naf1[i] = 0;
147  }
148  for (i = l2; i < l; i++) {
149  naf2[i] = 0;
150  }
151  for (i = l3; i < l; i++) {
152  naf3[i] = 0;
153  }
154 
155  if (bn_sign(k) == RLC_NEG) {
156  for (i = 0; i < l0; i++) {
157  naf0[i] = -naf0[i];
158  }
159  for (i = 0; i < l1; i++) {
160  naf1[i] = -naf1[i];
161  }
162  }
163  if (bn_sign(m) == RLC_NEG) {
164  for (i = 0; i < l2; i++) {
165  naf2[i] = -naf2[i];
166  }
167  for (i = 0; i < l3; i++) {
168  naf3[i] = -naf3[i];

170  }

174  ep_dbl(r, r);
175 
176  n = *t0;
177  if (n > 0) {
178  if (sk0 == RLC_POS) {
179  ep_add(r, r, t[n / 2]);
180  } else {
181  ep_sub(r, r, t[n / 2]);
182  }
183  }
184  if (n < 0) {
185  if (sk0 == RLC_POS) {
186  ep_sub(r, r, t[-n / 2]);
187  } else {
188  ep_add(r, r, t[-n / 2]);
189  }
190  }
191  n = *t1;
192  if (n > 0) {
193  ep_copy(u, t[n / 2]);
194  if (ep_curve_opt_a() == RLC_ZERO) {
195  fp_mul(u->x, u->x, ep_curve_get_beta());
196  } else {
197  fp_neg(u->x, u->x);
198  fp_mul(u->y, u->y, ep_curve_get_beta());
199  }
200  if (sk1 == RLC_NEG) {
201  ep_neg(u, u);
202  }
203  ep_add(r, r, u);
204  }
205  if (n < 0) {
206  ep_copy(u, t[-n / 2]);
207  if (ep_curve_opt_a() == RLC_ZERO) {
208  fp_mul(u->x, u->x, ep_curve_get_beta());
209  } else {
210  fp_neg(u->x, u->x);
211  fp_mul(u->y, u->y, ep_curve_get_beta());
212  }
213  if (sk1 == RLC_NEG) {
214  ep_neg(u, u);
215  }
216  ep_sub(r, r, u);
217  }
218 
219  n = *t2;
220  if (n > 0) {
221  if (sl0 == RLC_POS) {
222  ep_add(r, r, tab1[n / 2]);
223  } else {
224  ep_sub(r, r, tab1[n / 2]);
225  }
226  }
227  if (n < 0) {
228  if (sl0 == RLC_POS) {
229  ep_sub(r, r, tab1[-n / 2]);
230  } else {
231  ep_add(r, r, tab1[-n / 2]);
232  }
233  }
234  n = *t3;
235  if (n > 0) {
236  ep_copy(u, tab1[n / 2]);
237  if (ep_curve_opt_a() == RLC_ZERO) {
238  fp_mul(u->x, u->x, ep_curve_get_beta());
239  } else {
240  fp_neg(u->x, u->x);
241  fp_mul(u->y, u->y, ep_curve_get_beta());
242  }
243  if (sl1 == RLC_NEG) {
244  ep_neg(u, u);
245  }
246  ep_add(r, r, u);
247  }
248  if (n < 0) {
249  ep_copy(u, tab1[-n / 2]);
250  if (ep_curve_opt_a() == RLC_ZERO) {
251  fp_mul(u->x, u->x, ep_curve_get_beta());
252  } else {
253  fp_neg(u->x, u->x);
254  fp_mul(u->y, u->y, ep_curve_get_beta());
255  }
256  if (sl1 == RLC_NEG) {
257  ep_neg(u, u);
258  }
259  ep_sub(r, r, u);
260  }
261  }
262  /* Convert r to affine coordinates. */
263  ep_norm(r, r);
264  }
265  CATCH_ANY {
266  THROW(ERR_CAUGHT);
267  }
268  FINALLY {
269  bn_free(ord);
270  bn_free(k0);
271  bn_free(k1);
272  bn_free(m0);
273  bn_free(m1);
274  ep_free(u);
275 
276  if (!g) {
277  for (i = 0; i < 1 << (EP_WIDTH - 2); i++) {
278  ep_free(tab0[i]);
279  }
280  }
281  /* Free the precomputation tables. */
282  for (i = 0; i < 1 << (EP_WIDTH - 2); i++) {
283  ep_free(tab1[i]);
284  }
285  for (i = 0; i < 3; i++) {
286  bn_free(v1[i]);
287  bn_free(v2[i]);
288  }

308 static void ep_mul_sim_plain(ep_t r, const ep_t p, const bn_t k, const ep_t q,
309  const bn_t m, const ep_t *t) {
310  int i, l, l0, l1, n0, n1, w, gen;
311  int8_t naf0[RLC_FP_BITS + 1], naf1[RLC_FP_BITS + 1], *_k, *_m;
312  ep_t t0[1 << (EP_WIDTH - 2)];
313  ep_t t1[1 << (EP_WIDTH - 2)];
314 
315  TRY {
316  gen = (t == NULL ? 0 : 1);
317  if (!gen) {
318  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
319  ep_null(t0[i]);
320  ep_new(t0[i]);
321  }
322  ep_tab(t0, p, EP_WIDTH);
323  t = (const ep_t *)t0;
324  }
325 
326  /* Prepare the precomputation table. */
327  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
328  ep_null(t1[i]);
329  ep_new(t1[i]);
330  }
331  /* Compute the precomputation table. */
332  ep_tab(t1, q, EP_WIDTH);
333 
334  /* Compute the w-TNAF representation of k. */
335  if (gen) {
336  w = EP_DEPTH;
337  } else {
338  w = EP_WIDTH;
339  }
340  l0 = l1 = RLC_FP_BITS + 1;
341  bn_rec_naf(naf0, &l0, k, w);
342  bn_rec_naf(naf1, &l1, m, EP_WIDTH);
343 
344  l = RLC_MAX(l0, l1);
345  for (i = l0; i < l; i++) {
346  naf0[i] = 0;
347  }
348  for (i = l1; i < l; i++) {
349  naf1[i] = 0;
350  }
351 
352  if (bn_sign(k) == RLC_NEG) {
353  for (i = 0; i < l0; i++) {
354  naf0[i] = -naf0[i];
355  }
356  }
357  if (bn_sign(m) == RLC_NEG) {
358  for (i = 0; i < l1; i++) {
359  naf1[i] = -naf1[i];
360  }
361  }
362 
363  _k = naf0 + l - 1;
364  _m = naf1 + l - 1;
365  ep_set_infty(r);
366  for (i = l - 1; i >= 0; i--, _k--, _m--) {
367  ep_dbl(r, r);
368 
369  n0 = *_k;
370  n1 = *_m;
371  if (n0 > 0) {
372  ep_add(r, r, t[n0 / 2]);
373  }
374  if (n0 < 0) {
375  ep_sub(r, r, t[-n0 / 2]);
376  }
377  if (n1 > 0) {
378  ep_add(r, r, t1[n1 / 2]);
379  }
380  if (n1 < 0) {
381  ep_sub(r, r, t1[-n1 / 2]);
382  }
383  }
384  /* Convert r to affine coordinates. */
385  ep_norm(r, r);
386  }
387  CATCH_ANY {
388  THROW(ERR_CAUGHT);
389  }
390  FINALLY {
391  /* Free the precomputation tables. */
392  if (!gen) {
393  for (i = 0; i < 1 << (EP_WIDTH - 2); i++) {
394  ep_free(t0[i]);
395  }
396  }
397  for (i = 0; i < 1 << (EP_WIDTH - 2); i++) {
398  ep_free(t1[i]);
399  }
400  }
401 }

438 void ep_mul_sim_trick(ep_t r, const ep_t p, const bn_t k, const ep_t q,
439  const bn_t m) {
440  ep_t t0[1 << (EP_WIDTH / 2)], t1[1 << (EP_WIDTH / 2)], t[1 << EP_WIDTH];
441  bn_t n;
442  int l0, l1, w = EP_WIDTH / 2;
443  uint8_t w0[RLC_FP_BITS + 1], w1[RLC_FP_BITS + 1];
444 
445  bn_null(n);
446 
447  if (bn_is_zero(k) || ep_is_infty(p)) {
448  ep_mul(r, q, m);
449  return;
450  }
451  if (bn_is_zero(m) || ep_is_infty(q)) {
452  ep_mul(r, p, k);
453  return;
454  }
455 
456  TRY {
457  bn_new(n);
458 
459  ep_curve_get_ord(n);
460 
461  for (int i = 0; i < (1 << w); i++) {
462  ep_null(t0[i]);
463  ep_null(t1[i]);
464  ep_new(t0[i]);
465  ep_new(t1[i]);
466  }
467  for (int i = 0; i < (1 << EP_WIDTH); i++) {
468  ep_null(t[i]);
469  ep_new(t[i]);
470  }
471 
472  ep_set_infty(t0[0]);
473  ep_copy(t0[1], p);
474  if (bn_sign(k) == RLC_NEG) {
475  ep_neg(t0[1], t0[1]);
476  }
477  for (int i = 2; i < (1 << w); i++) {
478  ep_add(t0[i], t0[i - 1], t0[1]);
479  }
480 
481  ep_set_infty(t1[0]);
482  ep_copy(t1[1], q);
483  if (bn_sign(m) == RLC_NEG) {
484  ep_neg(t1[1], t1[1]);
485  }
486  for (int i = 2; i < (1 << w); i++) {
487  ep_add(t1[i], t1[i - 1], t1[1]);
488  }
489 
490  for (int i = 0; i < (1 << w); i++) {
491  for (int j = 0; j < (1 << w); j++) {
492  ep_add(t[(i << w) + j], t0[i], t1[j]);
493  }
494  }
495 
496 #if EP_WIDTH > 2 && defined(EP_MIXED)
497  ep_norm_sim(t + 1, (const ep_t *)(t + 1), (1 << EP_WIDTH) - 1);
498 #endif
499 
500  l0 = l1 = RLC_CEIL(RLC_FP_BITS + 1, w);
501  bn_rec_win(w0, &l0, k, w);
502  bn_rec_win(w1, &l1, m, w);
503  for (int i = l0; i < l1; i++) {
504  w0[i] = 0;
505  }
506  for (int i = l1; i < l0; i++) {
507  w1[i] = 0;
508  }
509 
510  ep_set_infty(r);
511  for (int i = RLC_MAX(l0, l1) - 1; i >= 0; i--) {
512  for (int j = 0; j < w; j++) {
513  ep_dbl(r, r);
514  }
515  ep_add(r, r, t[(w0[i] << w) + w1[i]]);
516  }
517  ep_norm(r, r);
518  } CATCH_ANY {
519  THROW(ERR_CAUGHT);
520  }
521  FINALLY {
522  bn_free(n);
523  for (int i = 0; i < (1 << w); i++) {
524  ep_free(t0[i]);
525  ep_free(t1[i]);
526  }
527  for (int i = 0; i < (1 << EP_WIDTH); i++) {
528  ep_free(t[i]);
529  }
530  }
531 }

564 void ep_mul_sim_joint(ep_t r, const ep_t p, const bn_t k, const ep_t q,
565  const bn_t m) {
566  ep_t t[5];
567  int i, l, u_i, offset;
568  int8_t jsf[2 * (RLC_FP_BITS + 1)];
569 
570  if (bn_is_zero(k) || ep_is_infty(p)) {
571  ep_mul(r, q, m);
572  return;
573  }
574  if (bn_is_zero(m) || ep_is_infty(q)) {
575  ep_mul(r, p, k);
576  return;
577  }
578 
579  TRY {
580  for (i = 0; i < 5; i++) {
581  ep_null(t[i]);
582  ep_new(t[i]);
583  }
584 
585  ep_set_infty(t[0]);
586  ep_copy(t[1], q);
587  if (bn_sign(m) == RLC_NEG) {
588  ep_neg(t[1], t[1]);
589  }
590  ep_copy(t[2], p);
591  if (bn_sign(k) == RLC_NEG) {
592  ep_neg(t[2], t[2]);
593  }
594  ep_add(t[3], t[2], t[1]);
595  ep_sub(t[4], t[2], t[1]);
596 #if defined(EP_MIXED)
597  ep_norm_sim(t + 3, (const ep_t *)t + 3, 2);
598 #endif
599 
600  l = 2 * (RLC_FP_BITS + 1);
601  bn_rec_jsf(jsf, &l, k, m);
602 
603  ep_set_infty(r);
604 
605  offset = RLC_MAX(bn_bits(k), bn_bits(m)) + 1;
606  for (i = l - 1; i >= 0; i--) {
607  ep_dbl(r, r);
608  if (jsf[i] != 0 && jsf[i] == -jsf[i + offset]) {
609  u_i = jsf[i] * 2 + jsf[i + offset];
610  if (u_i < 0) {
611  ep_sub(r, r, t[4]);
612  } else {
613  ep_add(r, r, t[4]);
614  }
615  } else {
616  u_i = jsf[i] * 2 + jsf[i + offset];
617  if (u_i < 0) {
618  ep_sub(r, r, t[-u_i]);
619  } else {
620  ep_add(r, r, t[u_i]);
621  }
622  }
623  }
624  ep_norm(r, r);
625  }
626  CATCH_ANY {
627  THROW(ERR_CAUGHT);
628  }
629  FINALLY {
630  for (i = 0; i < 5; i++) {
631  ep_free(t[i]);
632  }
633  }
634 }
