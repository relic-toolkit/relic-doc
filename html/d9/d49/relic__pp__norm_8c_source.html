50void pp_norm_k8(ep2_t r, const ep2_t p) {
51 if (ep2_is_infty(p)) {
52 ep2_set_infty(r);
53 return;
54 }
55
56 if (p->coord) {
57 /* If the point is represented in affine coordinates, we just copy it. */
58 ep2_copy(r, p);
59 }
60#if EP_ADD != BASIC || !defined(STRIP)
61 fp2_inv(r->z, p->z);
62 fp2_mul(r->x, p->x, r->z);
63 fp2_mul(r->y, p->y, r->z);
64 fp2_mul(r->y, r->y, r->z);
65 fp2_set_dig(r->z, 1);
66 r->coord = BASIC;
67#endif
68}
