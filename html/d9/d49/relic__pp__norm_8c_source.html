46 void pp_norm_k8(ep2_t r, ep2_t p) {
47  if (ep2_is_infty(p)) {
48  ep2_set_infty(r);
49  return;
50  }
51 
52  if (p->norm) {
53  /* If the point is represented in affine coordinates, we just copy it. */
54  ep2_copy(r, p);
55  }
56 #if EP_ADD == PROJC || !defined(STRIP)
57  fp2_inv(r->z, p->z);
58  fp2_mul(r->x, p->x, r->z);
59  fp2_mul(r->y, p->y, r->z);
60  fp2_mul(r->y, r->y, r->z);
61  fp2_set_dig(r->z, 1);
62  r->norm = 1;
63 #endif
64 }
