42 static void ep_mul_glv_imp(ep_t r, const ep_t p, const bn_t k) {
43  int l, l0, l1, i, n0, n1, s0, s1;
44  int8_t naf0[RLC_FP_BITS + 1], naf1[RLC_FP_BITS + 1], *t0, *t1;
45  bn_t n, k0, k1, v1[3], v2[3];
46  ep_t q, t[1 << (EP_WIDTH - 2)];
47 
48  if (bn_is_zero(k)) {
49  ep_set_infty(r);
50  return;
51  }
52 
53  bn_null(n);
54  bn_null(k0);
55  bn_null(k1);
56  ep_null(q);
57 
58  TRY {
59  bn_new(n);
60  bn_new(k0);
61  bn_new(k1);
62  ep_new(q);
63  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
64  ep_null(t[i]);
65  ep_new(t[i]);
66  }
67  for (i = 0; i < 3; i++) {
68  bn_null(v1[i]);
69  bn_null(v2[i]);
70  bn_new(v1[i]);
71  bn_new(v2[i]);
72  }
73 
74  ep_curve_get_ord(n);
75  ep_curve_get_v1(v1);
76  ep_curve_get_v2(v2);
77  bn_rec_glv(k0, k1, k, n, (const bn_t *)v1, (const bn_t *)v2);
78  s0 = bn_sign(k0);
79  s1 = bn_sign(k1);
80  bn_abs(k0, k0);
81  bn_abs(k1, k1);
82 
83  if (s0 == RLC_POS) {
84  ep_tab(t, p, EP_WIDTH);
85  } else {
86  ep_neg(q, p);
87  ep_tab(t, q, EP_WIDTH);
88  }
89 
90  l0 = l1 = RLC_FP_BITS + 1;
91  bn_rec_naf(naf0, &l0, k0, EP_WIDTH);
92  bn_rec_naf(naf1, &l1, k1, EP_WIDTH);
93 
94  l = RLC_MAX(l0, l1);
95  t0 = naf0 + l - 1;
96  t1 = naf1 + l - 1;
97  for (i = l0; i < l; i++)
98  naf0[i] = 0;
99  for (i = l1; i < l; i++)
100  naf1[i] = 0;
101 
102  ep_set_infty(r);
103  for (i = l - 1; i >= 0; i--, t0--, t1--) {
104  ep_dbl(r, r);
105 
106  n0 = *t0;
107  n1 = *t1;
108  if (n0 > 0) {
109  ep_add(r, r, t[n0 / 2]);
110  }
111  if (n0 < 0) {
112  ep_sub(r, r, t[-n0 / 2]);
113  }
114  if (n1 > 0) {
115  ep_copy(q, t[n1 / 2]);
116  if (ep_curve_opt_a() == RLC_ZERO) {
117  fp_mul(q->x, q->x, ep_curve_get_beta());
118  } else {
119  fp_neg(q->x, q->x);
120  fp_mul(q->y, q->y, ep_curve_get_beta());
121  }
122  if (s0 != s1) {
123  ep_neg(q, q);
124  }
125  ep_add(r, r, q);
126  }
127  if (n1 < 0) {
128  ep_copy(q, t[-n1 / 2]);
129  if (ep_curve_opt_a() == RLC_ZERO) {
130  fp_mul(q->x, q->x, ep_curve_get_beta());
131  } else {
132  fp_neg(q->x, q->x);
133  fp_mul(q->y, q->y, ep_curve_get_beta());
134  }
135 
136  if (s0 != s1) {
137  ep_neg(q, q);
138  }
139  ep_sub(r, r, q);
140  }
141  }
142  /* Convert r to affine coordinates. */
143  ep_norm(r, r);
144  if (bn_sign(k) == RLC_NEG) {
145  ep_neg(r, r);
146  }
147  }
148  CATCH_ANY {
149  THROW(ERR_CAUGHT);
150  }
151  FINALLY {
152  bn_free(n);
153  bn_free(k0);
154  bn_free(k1);
155  bn_free(n)
156  ep_free(q);
157  for (i = 0; i < 1 << (EP_WIDTH - 2); i++) {
158  ep_free(t[i]);
159  }
160  for (i = 0; i < 3; i++) {
161  bn_free(v1[i]);
162  bn_free(v2[i]);
163  }
164 
165  }
166 }

172 static void ep_mul_naf_imp(ep_t r, const ep_t p, const bn_t k) {
173  int i, l, n;
174  /* Some of the supported prime curves have order > field. */
175  int8_t naf[RLC_FP_BITS + 2];
176  ep_t t[1 << (EP_WIDTH - 2)];
177 
178  if (bn_is_zero(k)) {
179  ep_set_infty(r);
180  return;
181  }
182 
183  TRY {
184  /* Prepare the precomputation table. */
185  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
186  ep_null(t[i]);
187  ep_new(t[i]);
188  }
189  /* Compute the precomputation table. */
190  ep_tab(t, p, EP_WIDTH);
191 
192  /* Compute the w-NAF representation of k. */
193  l = RLC_FP_BITS + 2;
194  bn_rec_naf(naf, &l, k, EP_WIDTH);
195 
196  ep_set_infty(r);
197  for (i = l - 1; i >= 0; i--) {
198  ep_dbl(r, r);
199 
200  n = naf[i];
201  if (n > 0) {
202  ep_add(r, r, t[n / 2]);
203  } else if (n < 0) {
204  ep_sub(r, r, t[-n / 2]);
205  }
206  }
207  /* Convert r to affine coordinates. */
208  ep_norm(r, r);
209  if (bn_sign(k) == RLC_NEG) {
210  ep_neg(r, r);
211  }
212  }
213  CATCH_ANY {
214  THROW(ERR_CAUGHT);
215  }
216  FINALLY {
217  /* Free the precomputation table. */
218  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
219  ep_free(t[i]);
220  }
221  }
222 }

231 static void ep_mul_reg_glv(ep_t r, const ep_t p, const bn_t k) {
232  int i, j, l, n0, n1, s0, s1, b0, b1;
233  int8_t _s0, _s1, reg0[RLC_FP_BITS + 1], reg1[RLC_FP_BITS + 1];
234  bn_t n, k0, k1, v1[3], v2[3];
235  ep_t q, t[1 << (EP_WIDTH - 2)], u, v, w;
236 
237  if (bn_is_zero(k)) {
238  ep_set_infty(r);
239  return;
240  }
241 
242  bn_null(n);
243  bn_null(k0);
244  bn_null(k1);
245  ep_null(q);
246  ep_null(u);
247  ep_null(v);
248  ep_null(w);
249 
250  TRY {
251  bn_new(n);
252  bn_new(k0);
253  bn_new(k1);
254  ep_new(q);
255  ep_new(u);
256  ep_new(v);
257  ep_new(w);
258 
259  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
260  ep_null(t[i]);
261  ep_new(t[i]);
262  }
263  for (i = 0; i < 3; i++) {
264  bn_null(v1[i]);
265  bn_null(v2[i]);
266  bn_new(v1[i]);
267  bn_new(v2[i]);
268  }
269 
270  ep_curve_get_ord(n);
271  ep_curve_get_v1(v1);
272  ep_curve_get_v2(v2);
273  bn_rec_glv(k0, k1, k, n, (const bn_t *)v1, (const bn_t *)v2);
274  s0 = bn_sign(k0);
275  s1 = bn_sign(k1);
276  bn_abs(k0, k0);
277  bn_abs(k1, k1);
278  b0 = bn_is_even(k0);
279  b1 = bn_is_even(k1);
280  k0->dp[0] |= b0;
281  k1->dp[0] |= b1;
282 
283  ep_copy(q, p);
284  ep_neg(t[0], p);
285  dv_copy_cond(q->y, t[0]->y, RLC_FP_DIGS, s0 != RLC_POS);
286  ep_tab(t, q, EP_WIDTH);
287 
288  l = RLC_FP_BITS + 1;
289  bn_rec_reg(reg0, &l, k0, bn_bits(n)/2, EP_WIDTH);
290  l = RLC_FP_BITS + 1;
291  bn_rec_reg(reg1, &l, k1, bn_bits(n)/2, EP_WIDTH);
292 
293 #if defined(EP_MIXED)
294  fp_set_dig(u->z, 1);
295  fp_set_dig(w->z, 1);
296  u->norm = w->norm = 1;
297 #else
298  u->norm = w->norm = 0;
299 #endif
300  ep_set_infty(r);
301  for (i = l - 1; i >= 0; i--) {
302  for (j = 0; j < EP_WIDTH - 1; j++) {
303  ep_dbl(r, r);
304  }
305 
306  n0 = reg0[i];
307  _s0 = (n0 >> 7);
308  n0 = ((n0 ^ _s0) - _s0) >> 1;
309  n1 = reg1[i];
310  _s1 = (n1 >> 7);
311  n1 = ((n1 ^ _s1) - _s1) >> 1;
312 
313  for (j = 0; j < (1 << (EP_WIDTH - 2)); j++) {
314  dv_copy_cond(u->x, t[j]->x, RLC_FP_DIGS, j == n0);
315  dv_copy_cond(w->x, t[j]->x, RLC_FP_DIGS, j == n1);
316  dv_copy_cond(u->y, t[j]->y, RLC_FP_DIGS, j == n0);
317  dv_copy_cond(w->y, t[j]->y, RLC_FP_DIGS, j == n1);
318 #if !defined(EP_MIXED)
319  dv_copy_cond(u->z, t[j]->z, RLC_FP_DIGS, j == n0);
320  dv_copy_cond(w->z, t[j]->z, RLC_FP_DIGS, j == n1);
321 #endif
322  }
323  ep_neg(v, u);
324  dv_copy_cond(u->y, v->y, RLC_FP_DIGS, _s0 != 0);
325  ep_add(r, r, u);
326 
327  if (ep_curve_opt_a() == RLC_ZERO) {
328  fp_mul(w->x, w->x, ep_curve_get_beta());
329  } else {
330  fp_neg(w->x, w->x);
331  fp_mul(w->y, w->y, ep_curve_get_beta());
332  }
333  ep_neg(q, w);
334  dv_copy_cond(w->y, q->y, RLC_FP_DIGS, s0 != s1);
335  ep_neg(q, w);
336  dv_copy_cond(w->y, q->y, RLC_FP_DIGS, _s1 != 0);
337  ep_add(r, r, w);
338  }
339 
340  /* t[0] has an unmodified copy of p. */
341  ep_sub(u, r, t[0]);
342  dv_copy_cond(r->x, u->x, RLC_FP_DIGS, b0);
343  dv_copy_cond(r->y, u->y, RLC_FP_DIGS, b0);
344  dv_copy_cond(r->z, u->z, RLC_FP_DIGS, b0);
345 
346  ep_copy(w, t[0]);
347  if (ep_curve_opt_a() == RLC_ZERO) {
348  fp_mul(w->x, w->x, ep_curve_get_beta());
349  } else {
350  fp_neg(w->x, w->x);
351  fp_mul(w->y, w->y, ep_curve_get_beta());
352  }
353  ep_neg(q, w);
354  dv_copy_cond(w->y, q->y, RLC_FP_DIGS, s0 != s1);
355  ep_sub(u, r, w);
356  dv_copy_cond(r->x, u->x, RLC_FP_DIGS, b1);
357  dv_copy_cond(r->y, u->y, RLC_FP_DIGS, b1);
358  dv_copy_cond(r->z, u->z, RLC_FP_DIGS, b1);
359 
360  /* Convert r to affine coordinates. */
361  ep_norm(r, r);
362  ep_neg(u, r);
363  dv_copy_cond(r->y, u->y, RLC_FP_DIGS, bn_sign(k) == RLC_NEG);
364  }
365  CATCH_ANY {
366  THROW(ERR_CAUGHT);
367  }
368  FINALLY {
369  bn_free(n);
370  bn_free(k0);
371  bn_free(k1);
372  bn_free(n);
373  ep_free(q);
374  ep_free(u);
375  ep_free(v);
376  ep_free(w);
377  for (i = 0; i < 1 << (EP_WIDTH - 2); i++) {
378  ep_free(t[i]);
379  }
380  for (i = 0; i < 3; i++) {
381  bn_free(v1[i]);
382  bn_free(v2[i]);
383  }
384  }
385 }

391 static void ep_mul_reg_imp(ep_t r, const ep_t p, const bn_t k) {
392  bn_t _k;
393  int i, j, l, n;
394  int8_t s, reg[RLC_CEIL(RLC_FP_BITS + 1, EP_WIDTH - 1)];
395  ep_t t[1 << (EP_WIDTH - 2)], u, v;
396 
397  bn_null(_k);
398  if (bn_is_zero(k)) {
399  ep_set_infty(r);
400  return;
401  }
402 
403  TRY {
404  bn_new(_k);
405  ep_new(u);
406  ep_new(v);
407  /* Prepare the precomputation table. */
408  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
409  ep_null(t[i]);
410  ep_new(t[i]);
411  }
412  /* Compute the precomputation table. */
413  ep_tab(t, p, EP_WIDTH);
414 
415  ep_curve_get_ord(_k);
416  n = bn_bits(_k);
417 
418  /* Make a copy of the scalar for processing. */
419  bn_abs(_k, k);
420  _k->dp[0] |= bn_is_even(_k);
421 
422  /* Compute the regular w-NAF representation of k. */
423  l = RLC_CEIL(RLC_FP_BITS + 1, EP_WIDTH - 1);
424  bn_rec_reg(reg, &l, _k, n, EP_WIDTH);
425 
426 #if defined(EP_MIXED)
427  fp_set_dig(u->z, 1);
428  u->norm = 1;
429 #else
430  u->norm = 0;
431 #endif
432  ep_set_infty(r);
433  for (i = l - 1; i >= 0; i--) {
434  for (j = 0; j < EP_WIDTH - 1; j++) {
435  ep_dbl(r, r);
436  }
437 
438  n = reg[i];
439  s = (n >> 7);
440  n = ((n ^ s) - s) >> 1;
441 
442  for (j = 0; j < (1 << (EP_WIDTH - 2)); j++) {
443  dv_copy_cond(u->x, t[j]->x, RLC_FP_DIGS, j == n);
444  dv_copy_cond(u->y, t[j]->y, RLC_FP_DIGS, j == n);
445 #if !defined(EP_MIXED)
446  dv_copy_cond(u->z, t[j]->z, RLC_FP_DIGS, j == n);
447 #endif
448  }
449  ep_neg(v, u);
450  dv_copy_cond(u->y, v->y, RLC_FP_DIGS, s != 0);
451  ep_add(r, r, u);
452  }
453  /* t[0] has an unmodified copy of p. */
454  ep_sub(u, r, t[0]);
455  dv_copy_cond(r->x, u->x, RLC_FP_DIGS, bn_is_even(k));
456  dv_copy_cond(r->y, u->y, RLC_FP_DIGS, bn_is_even(k));
457  dv_copy_cond(r->z, u->z, RLC_FP_DIGS, bn_is_even(k));
458  /* Convert r to affine coordinates. */
459  ep_norm(r, r);
460  ep_neg(u, r);
461  dv_copy_cond(r->y, u->y, RLC_FP_DIGS, bn_sign(k) == RLC_NEG);
462  }
463  CATCH_ANY {
464  THROW(ERR_CAUGHT);
465  }
466  FINALLY {
467  /* Free the precomputation table. */
468  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
469  ep_free(t[i]);
470  }
471  bn_free(_k);
472  ep_free(u);
473  ep_free(v);
474  }
475 }

503  ep_add(t, t, p);
504  }

524 void ep_mul_slide(ep_t r, const ep_t p, const bn_t k) {
525  ep_t t[1 << (EP_WIDTH - 1)], q;
526  int i, j, l;
527  uint8_t win[RLC_FP_BITS + 1];
528 
529  ep_null(q);
530 
531  if (bn_is_zero(k) || ep_is_infty(p)) {
532  ep_set_infty(r);
533  return;
534  }
535 
536  TRY {
537  for (i = 0; i < (1 << (EP_WIDTH - 1)); i ++) {
538  ep_null(t[i]);
539  ep_new(t[i]);
540  }
541 
542  ep_new(q);
543 
544  ep_copy(t[0], p);
545  ep_dbl(q, p);
546 
547 #if defined(EP_MIXED)
548  ep_norm(q, q);
549 #endif
550 
551  /* Create table. */
552  for (i = 1; i < (1 << (EP_WIDTH - 1)); i++) {
553  ep_add(t[i], t[i - 1], q);
554  }
555 
556 #if defined(EP_MIXED)
557  ep_norm_sim(t + 1, (const ep_t *)t + 1, (1 << (EP_WIDTH - 1)) - 1);
558 #endif
559 
560  ep_set_infty(q);
561  l = RLC_FP_BITS + 1;
562  bn_rec_slw(win, &l, k, EP_WIDTH);
563  for (i = 0; i < l; i++) {
564  if (win[i] == 0) {
565  ep_dbl(q, q);
566  } else {
567  for (j = 0; j < util_bits_dig(win[i]); j++) {
568  ep_dbl(q, q);
569  }
570  ep_add(q, q, t[win[i] >> 1]);
571  }
572  }
573 
574  ep_norm(r, q);
575  if (bn_sign(k) == RLC_NEG) {
576  ep_neg(r, r);
577  }
578  }
579  CATCH_ANY {
580  THROW(ERR_CAUGHT);
581  }
582  FINALLY {
583  for (i = 0; i < (1 << (EP_WIDTH - 1)); i++) {
584  ep_free(t[i]);
585  }
586  ep_free(q);
587  }
588 }

594 void ep_mul_monty(ep_t r, const ep_t p, const bn_t k) {
595  ep_t t[2];
596  bn_t n, l;
597 
598  bn_null(n);
599  bn_null(l);
600  ep_null(t[0]);
601  ep_null(t[1]);
602 
603  if (bn_is_zero(k) || ep_is_infty(p)) {
604  ep_set_infty(r);
605  return;
606  }
607 
608  TRY {
609  bn_new(n);
610  bn_new(l);
611  ep_new(t[0]);
612  ep_new(t[1]);
613 
614  ep_curve_get_ord(n);
615  ep_norm(t[0], p);
616  ep_dbl(t[1], t[0]);
617 
618  bn_abs(l, k);
619  while (bn_bits(l) <= bn_bits(n)) {
620  bn_add(l, l, n);
621  }
622 
623 #if EP_ADD == PROJC
624  fp_rand(t[0]->z);
625  fp_sqr(t[0]->x, t[0]->z);
626  fp_mul(t[0]->y, t[0]->x, t[0]->z);
627  fp_mul(t[0]->x, t[0]->x, p->x);
628  fp_mul(t[0]->y, t[0]->y, p->y);
629 #endif
630 
631  for (int i = bn_bits(n) - 1; i >= 0; i--) {
632  int j = bn_get_bit(l, i);
