42static void ep_mul_glv_imp(ep_t r, const ep_t p, const bn_t k) {
43 int i, n0, n1, s0, s1;
44 int8_t naf0[RLC_FP_BITS + 1], naf1[RLC_FP_BITS + 1], *t0, *t1;
45 bn_t n, _k, k0, k1, v1[3], v2[3];
46 ep_t q, t[1 << (RLC_WIDTH - 2)];
47 size_t l, l0, l1;
48
49 bn_null(n);
50 bn_null(_k);
51 bn_null(k0);
52 bn_null(k1);
53 ep_null(q);
54
55 RLC_TRY {
56 bn_new(n);
57 bn_new(_k);
58 bn_new(k0);
59 bn_new(k1);
60 ep_new(q);
61 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
62 ep_null(t[i]);
63 ep_new(t[i]);
64 }
65 for (i = 0; i < 3; i++) {
66 bn_null(v1[i]);
67 bn_null(v2[i]);
68 bn_new(v1[i]);
69 bn_new(v2[i]);
70 }
71
72 ep_curve_get_ord(n);
73 ep_curve_get_v1(v1);
74 ep_curve_get_v2(v2);
75
76 bn_mod(_k, k, n);
77 bn_rec_glv(k0, k1, _k, n, (const bn_t *)v1, (const bn_t *)v2);
78 s0 = bn_sign(k0);
79 s1 = bn_sign(k1);
80
81 if (s0 == RLC_POS) {
82 ep_tab(t, p, RLC_WIDTH);
83 } else {
84 ep_neg(q, p);
85 ep_tab(t, q, RLC_WIDTH);
86 }
87
88 l0 = l1 = RLC_FP_BITS + 1;
89 bn_rec_naf(naf0, &l0, k0, RLC_WIDTH);
90 bn_rec_naf(naf1, &l1, k1, RLC_WIDTH);
91
92 l = RLC_MAX(l0, l1);
93 t0 = naf0 + l - 1;
94 t1 = naf1 + l - 1;
95
96 ep_set_infty(r);
97 for (i = l - 1; i >= 0; i--, t0--, t1--) {
98 ep_dbl(r, r);
99
100 n0 = *t0;
101 n1 = *t1;
102 if (n0 > 0) {
103 ep_add(r, r, t[n0 / 2]);
104 }
105 if (n0 < 0) {
106 ep_sub(r, r, t[-n0 / 2]);
107 }
108 if (n1 > 0) {
109 ep_psi(q, t[n1 / 2]);
110 if (s0 != s1) {
111 ep_neg(q, q);
112 }
113 ep_add(r, r, q);
114 }
115 if (n1 < 0) {
116 ep_psi(q, t[-n1 / 2]);
117 if (s0 != s1) {
118 ep_neg(q, q);
119 }
120 ep_sub(r, r, q);
121 }
122 }
123 /* Convert r to affine coordinates. */
124 ep_norm(r, r);
125 }
126 RLC_CATCH_ANY {
127 RLC_THROW(ERR_CAUGHT);
128 }
129 RLC_FINALLY {
130 bn_free(n);
131 bn_free(_k);
132 bn_free(k0);
133 bn_free(k1);
134 bn_free(n)
135 ep_free(q);
136 for (i = 0; i < 1 << (RLC_WIDTH - 2); i++) {
137 ep_free(t[i]);
138 }
139 for (i = 0; i < 3; i++) {
140 bn_free(v1[i]);
141 bn_free(v2[i]);
142 }
143 }
144}

150static void ep_mul_naf_imp(ep_t r, const ep_t p, const bn_t k) {
151 /* Some of the supported prime curves have order > field. */
152 int8_t u, naf[RLC_FP_BITS + 2];
153 ep_t t[1 << (RLC_WIDTH - 2)];
154 bn_t _k, n;
155 size_t l;
156
157 bn_null(n);
158 bn_null(_k);
159
160 RLC_TRY {
161 bn_new(n);
162 bn_new(_k);
163 /* Prepare the precomputation table. */
164 for (int i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
165 ep_null(t[i]);
166 ep_new(t[i]);
167 }
168
169 ep_curve_get_ord(n);
170 bn_mod(_k, k, n);
171
172 /* Compute the precomputation table. */
173 ep_tab(t, p, RLC_WIDTH);
174
175 /* Compute the w-NAF representation of k. */
176 l = RLC_FP_BITS + 2;
177 bn_rec_naf(naf, &l, _k, RLC_WIDTH);
178
179 ep_set_infty(r);
180 for (int i = l - 1; i >= 0; i--) {
181 ep_dbl(r, r);
182
183 u = naf[i];
184 if (u > 0) {
185 ep_add(r, r, t[u / 2]);
186 } else if (u < 0) {
187 ep_sub(r, r, t[-u / 2]);
188 }
189 }
190 /* Convert r to affine coordinates. */
191 ep_norm(r, r);
192 }
193 RLC_CATCH_ANY {
194 RLC_THROW(ERR_CAUGHT);
195 }
196 RLC_FINALLY {
197 bn_free(n);
198 bn_free(_k);
199 /* Free the precomputation table. */
200 for (int i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
201 ep_free(t[i]);
202 }
203 }
204}

212static void ep_mul_reg_glv(ep_t r, const ep_t p, const bn_t k) {
213 int8_t reg[2][RLC_FP_BITS + 1], s[2], b[2], c0, c1, n0, n1;
214 bn_t n, _k[2], v1[3], v2[3];
215 ep_t q, t[1 << (RLC_WIDTH - 2)], u, w;
216 size_t l;
217
218 bn_null(n);
219 bn_null(_k[0]);
220 bn_null(_k[1]);
221 ep_null(q);
222 ep_null(u);
223 ep_null(w);
224
225 RLC_TRY {
226 bn_new(n);
227 bn_new(_k[0]);
228 bn_new(_k[1]);
229 ep_new(q);
230 ep_new(u);
231 ep_new(w);
232
233 for (size_t i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
234 ep_null(t[i]);
235 ep_new(t[i]);
236 }
237 for (size_t i = 0; i < 3; i++) {
238 bn_null(v1[i]);
239 bn_null(v2[i]);
240 bn_new(v1[i]);
241 bn_new(v2[i]);
242 }
243
244 ep_curve_get_ord(n);
245 ep_curve_get_v1(v1);
246 ep_curve_get_v2(v2);
247
248 bn_mod(_k[0], k, n);
249
250 bn_rec_glv(_k[0], _k[1], _k[0], n, (const bn_t *)v1, (const bn_t *)v2);
251 for (size_t i = 0; i < 2; i++) {
252 s[i] = bn_sign(_k[i]);
253 bn_abs(_k[i], _k[i]);
254 b[i] = bn_is_even(_k[i]);
255 _k[i]->dp[0] |= b[i];
256 }
257
258 ep_norm(t[0], p);
259 ep_neg(q, t[0]);
260 dv_copy_sec(q->y, t[0]->y, RLC_FP_DIGS, s[0] == RLC_POS);
261 ep_tab(t, q, RLC_WIDTH);
262
263 l = RLC_FP_BITS + 1;
264 bn_rec_reg(reg[0], &l, _k[0], bn_bits(n) >> 1, RLC_WIDTH);
265 l = RLC_FP_BITS + 1;
266 bn_rec_reg(reg[1], &l, _k[1], bn_bits(n) >> 1, RLC_WIDTH);
267
268#if defined(EP_MIXED)
269 fp_set_dig(u->z, 1);
270 fp_set_dig(w->z, 1);
271 u->coord = w->coord = BASIC;
272#else
273 u->coord = w->coord = EP_ADD;
274#endif
275 ep_set_infty(r);
276 for (int i = l - 1; i >= 0; i--) {
277 for (size_t j = 0; j < RLC_WIDTH - 1; j++) {
278 ep_dbl(r, r);
279 }
280
281 n0 = reg[0][i];
282 c0 = (n0 >> 7);
283 n0 = ((n0 ^ c0) - c0) >> 1;
284 n1 = reg[1][i];
285 c1 = (n1 >> 7);
286 n1 = ((n1 ^ c1) - c1) >> 1;
287
288 for (size_t j = 0; j < (1 << (RLC_WIDTH - 2)); j++) {
289 fp_copy_sec(u->x, t[j]->x, j == n0);
290 fp_copy_sec(w->x, t[j]->x, j == n1);
291 fp_copy_sec(u->y, t[j]->y, j == n0);
292 fp_copy_sec(w->y, t[j]->y, j == n1);
293#if !defined(EP_fpXED)
294 fp_copy_sec(u->z, t[j]->z, j == n0);
295 fp_copy_sec(w->z, t[j]->z, j == n1);
296#endif
297 }
298 ep_neg(q, u);
299 fp_copy_sec(q->y, u->y, c0 == 0);
300 ep_add(r, r, q);
301
302 ep_psi(w, w);
303 ep_neg(q, w);
304 fp_copy_sec(w->y, q->y, (c1 != 0) ^ (s[0] != s[1]));
305 ep_add(r, r, w);
306 }
307
308 /* t[0] has an unmodified copy of p. */
309 ep_sub(u, r, t[0]);
310 fp_copy_sec(r->x, u->x, b[0]);
311 fp_copy_sec(r->y, u->y, b[0]);
312 fp_copy_sec(r->z, u->z, b[0]);
313
314 ep_psi(w, t[0]);
315 ep_neg(q, w);
316 fp_copy_sec(q->y, w->y, s[0] == s[1]);
317 ep_sub(u, r, q);
318 fp_copy_sec(r->x, u->x, b[1]);
319 fp_copy_sec(r->y, u->y, b[1]);
320 fp_copy_sec(r->z, u->z, b[1]);
321
322 /* Convert r to affine coordinates. */
323 ep_norm(r, r);
324 }
325 RLC_CATCH_ANY {
326 RLC_THROW(ERR_CAUGHT);
327 }
328 RLC_FINALLY {
329 bn_free(n);
330 bn_free(_k[0]);
331 bn_free(_k[1]);
332 bn_free(n);
333 ep_free(q);
334 ep_free(u);
335 ep_free(w);
336 for (size_t i = 0; i < 1 << (RLC_WIDTH - 2); i++) {
337 ep_free(t[i]);
338 }
339 for (size_t i = 0; i < 3; i++) {
340 bn_free(v1[i]);
341 bn_free(v2[i]);
342 }
343 }
344}

350static void ep_mul_reg_imp(ep_t r, const ep_t p, const bn_t k) {
351 bn_t _k;
352 int i, j, n;
353 int8_t s, reg[1 + RLC_CEIL(RLC_FP_BITS + 1, RLC_WIDTH - 1)];
354 ep_t t[1 << (RLC_WIDTH - 2)], u, v;
355 size_t l;
356
357 bn_null(_k);
358
359 RLC_TRY {
360 bn_new(_k);
361 ep_new(u);
362 ep_new(v);
363 /* Prepare the precomputation table. */
364 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
365 ep_null(t[i]);
366 ep_new(t[i]);
367 }
368 /* Compute the precomputation table. */
369 ep_tab(t, p, RLC_WIDTH);
370
371 ep_curve_get_ord(_k);
372 n = bn_bits(_k);
373
374 /* Make a copy of the scalar for processing. */
375 bn_abs(_k, k);
376 _k->dp[0] |= 1;
377
378 /* Compute the regular w-NAF representation of k. */
379 l = RLC_CEIL(n, RLC_WIDTH - 1) + 1;
380 bn_rec_reg(reg, &l, _k, n, RLC_WIDTH);
381
382#if defined(EP_MIXED)
383 fp_set_dig(u->z, 1);
384 u->coord = BASIC;
385#else
386 u->coord = EP_ADD;
387#endif
388 ep_set_infty(r);
389 for (i = l - 1; i >= 0; i--) {
390 for (j = 0; j < RLC_WIDTH - 1; j++) {
391 ep_dbl(r, r);
392 }
393
394 n = reg[i];
395 s = (n >> 7);
396 n = ((n ^ s) - s) >> 1;
397
398 for (j = 0; j < (1 << (RLC_WIDTH - 2)); j++) {
399 fp_copy_sec(u->x, t[j]->x, j == n);
400 fp_copy_sec(u->y, t[j]->y, j == n);
401#if !defined(EP_MIXED)
402 fp_copy_sec(u->z, t[j]->z, j == n);
403#endif
404 }
405 ep_neg(v, u);
406 fp_copy_sec(u->y, v->y, s != 0);
407 ep_add(r, r, u);
408 }
409 /* t[0] has an unmodified copy of p. */
410 ep_sub(u, r, t[0]);
411 fp_copy_sec(r->x, u->x, bn_is_even(k));
412 fp_copy_sec(r->y, u->y, bn_is_even(k));
413 fp_copy_sec(r->z, u->z, bn_is_even(k));
414 /* Convert r to affine coordinates. */
415 ep_norm(r, r);
416 ep_neg(u, r);
417 fp_copy_sec(r->y, u->y, bn_sign(k) == RLC_NEG);
418 }
419 RLC_CATCH_ANY {
420 RLC_THROW(ERR_CAUGHT);
421 }
422 RLC_FINALLY {
423 /* Free the precomputation table. */
424 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
425 ep_free(t[i]);
426 }
427 bn_free(_k);
428 ep_free(u);
429 ep_free(v);
430 }
431}

440void ep_mul_basic(ep_t r, const ep_t p, const bn_t k) {
441 ep_t t;
442 int8_t u, *naf = RLC_ALLOCA(int8_t, bn_bits(k) + 1);
443 size_t l;
444
445 ep_null(t);
446
447 if (bn_is_zero(k) || ep_is_infty(p)) {
448 ep_set_infty(r);
449 RLC_FREE(naf);
450 return;
451 }
452
453 if (bn_bits(k) <= RLC_DIG) {
454 ep_mul_dig(r, p, k->dp[0]);
455 if (bn_sign(k) == RLC_NEG) {
456 ep_neg(r, r);
457 }
458 RLC_FREE(naf);
459 return;
460 }
461
462 RLC_TRY {
463 ep_new(t);
464 if (naf == NULL) {
465 RLC_THROW(ERR_NO_BUFFER);
466 }
467
468 l = bn_bits(k) + 1;
469 bn_rec_naf(naf, &l, k, 2);
470 ep_set_infty(t);
471 for (int i = l - 1; i >= 0; i--) {
472 ep_dbl(t, t);
473
474 u = naf[i];
475 if (u > 0) {
476 ep_add(t, t, p);
477 } else if (u < 0) {
478 ep_sub(t, t, p);
479 }
480 }
481
482 ep_norm(r, t);
483 if (bn_sign(k) == RLC_NEG) {
484 ep_neg(r, r);
485 }
486 }
487 RLC_CATCH_ANY {
488 RLC_THROW(ERR_CAUGHT);
489 }
490 RLC_FINALLY {
491 ep_free(t);
492 RLC_FREE(naf);
493 }
494}

498void ep_mul_slide(ep_t r, const ep_t p, const bn_t k) {
499 bn_t _k, n;
500 ep_t t[1 << (RLC_WIDTH - 1)], q;
501 uint8_t win[RLC_FP_BITS + 1];
502 size_t l;
503
504 if (bn_is_zero(k) || ep_is_infty(p)) {
505 ep_set_infty(r);
506 return;
507 }
508
509 ep_null(q);
510 bn_null(n);
511 bn_null(_k);
512
513 RLC_TRY {
514 bn_new(n);
515 bn_new(_k);
516 for (size_t i = 0; i < (1 << (RLC_WIDTH - 1)); i ++) {
517 ep_null(t[i]);
518 ep_new(t[i]);
519 }
520 ep_new(q);
521
522 ep_copy(t[0], p);
523 ep_dbl(q, p);
524
525#if defined(EP_MIXED)
526 ep_norm(q, q);
527#endif
528
529 ep_curve_get_ord(n);
530 bn_mod(_k, k, n);
531
532 /* Create table. */
533 for (size_t i = 1; i < (1 << (RLC_WIDTH - 1)); i++) {
534 ep_add(t[i], t[i - 1], q);
535 }
536
537#if defined(EP_MIXED)
538 ep_norm_sim(t + 1, (const ep_t *)t + 1, (1 << (RLC_WIDTH - 1)) - 1);
539#endif
540
541 ep_set_infty(q);
542 l = RLC_FP_BITS + 1;
543 bn_rec_slw(win, &l, _k, RLC_WIDTH);
544 for (size_t i = 0; i < l; i++) {
545 if (win[i] == 0) {
546 ep_dbl(q, q);
547 } else {
548 for (size_t j = 0; j < util_bits_dig(win[i]); j++) {
549 ep_dbl(q, q);
550 }
551 ep_add(q, q, t[win[i] >> 1]);
552 }
553 }
554
555 ep_norm(r, q);
556 }
557 RLC_CATCH_ANY {
558 RLC_THROW(ERR_CAUGHT);
559 }
560 RLC_FINALLY {
561 bn_free(n);
562 bn_free(_k);
563 for (size_t i = 0; i < (1 << (RLC_WIDTH - 1)); i++) {
564 ep_free(t[i]);
565 }
566 ep_free(q);
567 }
568}

574void ep_mul_monty(ep_t r, const ep_t p, const bn_t k) {
575 ep_t t[2];
576 bn_t n, l, _k;
577 size_t bits;
578
579 bn_null(n);
580 bn_null(l);
581 bn_null(_k);
582 ep_null(t[0]);
583 ep_null(t[1]);
584
585 if (bn_is_zero(k) || ep_is_infty(p)) {
586 ep_set_infty(r);
587 return;
588 }
589
590 RLC_TRY {
591 bn_new(n);
592 bn_new(l);
593 bn_new(_k);
594 ep_new(t[0]);
595 ep_new(t[1]);
596
597 ep_curve_get_ord(n);
598 bits = bn_bits(n);
599
600 bn_mod(_k, k, n);
601 bn_abs(l, _k);
602 bn_add(l, l, n);
603 bn_add(n, l, n);
604 dv_swap_sec(l->dp, n->dp, RLC_MAX(l->used, n->used),
605 bn_get_bit(l, bits) == 0);
606 l->used = RLC_SEL(l->used, n->used, bn_get_bit(l, bits) == 0);
607
608 ep_norm(t[0], p);
609 ep_dbl(t[1], t[0]);
610
611 /* Blind both points independently. */
612 ep_blind(t[0], t[0]);
613 ep_blind(t[1], t[1]);
614
615 for (int i = bits - 1; i >= 0; i--) {
616 int j = bn_get_bit(l, i);
617 dv_swap_sec(t[0]->x, t[1]->x, RLC_FP_DIGS, j ^ 1);
618 dv_swap_sec(t[0]->y, t[1]->y, RLC_FP_DIGS, j ^ 1);
619 dv_swap_sec(t[0]->z, t[1]->z, RLC_FP_DIGS, j ^ 1);
620 ep_add(t[0], t[0], t[1]);
621 ep_dbl(t[1], t[1]);
622 dv_swap_sec(t[0]->x, t[1]->x, RLC_FP_DIGS, j ^ 1);
623 dv_swap_sec(t[0]->y, t[1]->y, RLC_FP_DIGS, j ^ 1);
624 dv_swap_sec(t[0]->z, t[1]->z, RLC_FP_DIGS, j ^ 1);
625 }
626
627 ep_norm(r, t[0]);
628 } RLC_CATCH_ANY {
629 RLC_THROW(ERR_CAUGHT);
630 }
631 RLC_FINALLY {
632 bn_free(n);
633 bn_free(l);
634 bn_free(_k);
635 ep_free(t[1]);
636 ep_free(t[0]);
637 }
638}

684void ep_mul_gen(ep_t r, const bn_t k) {
685 if (bn_is_zero(k)) {
686 ep_set_infty(r);
687 return;
688 }
689
690#ifdef EP_PRECO
691 ep_mul_fix(r, ep_curve_get_tab(), k);
692#else
693 ep_t g;
694
695 ep_null(g);
