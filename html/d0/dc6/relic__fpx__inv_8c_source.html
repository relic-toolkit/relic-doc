39 void fp2_inv(fp2_t c, fp2_t a) {
40  fp_t t0, t1;
41 
42  fp_null(t0);
43  fp_null(t1);
44 
45  TRY {
46  fp_new(t0);
47  fp_new(t1);
48 
49  /* t0 = a_0^2, t1 = a_1^2. */
50  fp_sqr(t0, a[0]);
51  fp_sqr(t1, a[1]);
52 
53  /* t1 = 1/(a_0^2 + a_1^2). */
54 #ifndef FP_QNRES
55  if (fp_prime_get_qnr() != -1) {
56  if (fp_prime_get_qnr() == -2) {
57  fp_dbl(t1, t1);
58  fp_add(t0, t0, t1);
59  } else {
60  if (fp_prime_get_qnr() < 0) {
61  fp_mul_dig(t1, t1, -fp_prime_get_qnr());
62  fp_add(t0, t0, t1);
63  } else {
64  fp_mul_dig(t1, t1, fp_prime_get_qnr());
65  fp_sub(t0, t0, t1);
66  }
67  }
68  } else {
69  fp_add(t0, t0, t1);
70  }
71 #else
72  fp_add(t0, t0, t1);
73 #endif
74 
75  fp_inv(t1, t0);
76 
77  /* c_0 = a_0/(a_0^2 + a_1^2). */
78  fp_mul(c[0], a[0], t1);
79  /* c_1 = - a_1/(a_0^2 + a_1^2). */
80  fp_mul(c[1], a[1], t1);
81  fp_neg(c[1], c[1]);
82  }
83  CATCH_ANY {
84  THROW(ERR_CAUGHT);
85  }
86  FINALLY {
87  fp_free(t0);
88  fp_free(t1);
89  }
90 }

97 void fp2_inv_sim(fp2_t *c, fp2_t *a, int n) {
98  int i;
99  fp2_t u, *t = RLC_ALLOCA(fp2_t, n);
100 
101  for (i = 0; i < n; i++) {
102  fp2_null(t[i]);
103  }
104  fp2_null(u);
105 
106  TRY {
107  for (i = 0; i < n; i++) {
108  fp2_new(t[i]);
109  }
110  fp2_new(u);
111 
112  fp2_copy(c[0], a[0]);
113  fp2_copy(t[0], a[0]);
114 
115  for (i = 1; i < n; i++) {
116  fp2_copy(t[i], a[i]);
117  fp2_mul(c[i], c[i - 1], t[i]);
118  }
119 
120  fp2_inv(u, c[n - 1]);
121 
122  for (i = n - 1; i > 0; i--) {
123  fp2_mul(c[i], c[i - 1], u);
124  fp2_mul(u, u, t[i]);
125  }
126  fp2_copy(c[0], u);
127  }
128  CATCH_ANY {
129  THROW(ERR_CAUGHT);
130  }
131  FINALLY {
132  for (i = 0; i < n; i++) {
133  fp2_free(t[i]);

139 void fp3_inv(fp3_t c, fp3_t a) {
140  fp_t v0;
141  fp_t v1;
142  fp_t v2;
143  fp_t t0;
144 
145  fp_null(v0);
146  fp_null(v1);
147  fp_null(v2);
148  fp_null(t0);
149 
150  TRY {
151  fp_new(v0);
152  fp_new(v1);
153  fp_new(v2);
154  fp_new(t0);
155 
156  /* v0 = a_0^2 - B * a_1 * a_2. */
157  fp_sqr(t0, a[0]);
158  fp_mul(v0, a[1], a[2]);
159  fp_copy(v2, v0);
160  for (int i = 1; i < fp_prime_get_cnr(); i++) {
161  fp_add(v2, v2, v0);
162  }
163  for (int i = 0; i >= fp_prime_get_cnr(); i--) {
164  fp_sub(v2, v2, v0);
165  }
166  fp_sub(v0, t0, v2);
167 
168  /* v1 = B * a_2^2 - a_0 * a_1. */
169  fp_sqr(t0, a[2]);
170  fp_copy(v2, t0);
171  for (int i = 1; i < fp_prime_get_cnr(); i++) {
172  fp_add(v2, v2, t0);
173  }
174  for (int i = 0; i >= fp_prime_get_cnr(); i--) {
175  fp_sub(v2, v2, t0);
176  }
177  fp_mul(v1, a[0], a[1]);
178  fp_sub(v1, v2, v1);
179 
180  /* v2 = a_1^2 - a_0 * a_2. */
181  fp_sqr(t0, a[1]);
182  fp_mul(v2, a[0], a[2]);
183  fp_sub(v2, t0, v2);
184 
185  fp_mul(t0, a[1], v2);
186  fp_copy(c[1], t0);
187  for (int i = 1; i < fp_prime_get_cnr(); i++) {
188  fp_add(c[1], c[1], t0);
189  }
190  for (int i = 0; i >= fp_prime_get_cnr(); i--) {
191  fp_sub(c[1], c[1], t0);
192  }
193 
194  fp_mul(c[0], a[0], v0);
195 
196  fp_mul(t0, a[2], v1);
197  fp_copy(c[2], t0);
198  for (int i = 1; i < fp_prime_get_cnr(); i++) {
199  fp_add(c[2], c[2], t0);
200  }
201  for (int i = 0; i >= fp_prime_get_cnr(); i--) {
202  fp_sub(c[2], c[2], t0);
203  }
204 
205  fp_add(t0, c[0], c[1]);
206  fp_add(t0, t0, c[2]);
207  fp_inv(t0, t0);
208 
209  fp_mul(c[0], v0, t0);
210  fp_mul(c[1], v1, t0);
211  fp_mul(c[2], v2, t0);
212  } CATCH_ANY {
213  THROW(ERR_CAUGHT);
214  } FINALLY {
215  fp_free(v0);
216  fp_free(v1);
217  fp_free(v2);
218  fp_free(t0);
219  }
220 }
