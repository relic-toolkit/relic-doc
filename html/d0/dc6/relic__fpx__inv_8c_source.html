39void fp2_inv(fp2_t c, const fp2_t a) {
40 fp_t t0, t1;
41
42 fp_null(t0);
43 fp_null(t1);
44
45 RLC_TRY {
46 fp_new(t0);
47 fp_new(t1);
48
49 /* t0 = a_0^2, t1 = a_1^2. */
50 fp_sqr(t0, a[0]);
51 fp_sqr(t1, a[1]);
52
53 /* t1 = 1/(a_0^2 + a_1^2). */
54#ifndef FP_QNRES
55 if (fp_prime_get_qnr() != -1) {
56 if (fp_prime_get_qnr() == -2) {
57 fp_dbl(t1, t1);
58 fp_add(t0, t0, t1);
59 } else {
60 if (fp_prime_get_qnr() < 0) {
61 fp_mul_dig(t1, t1, -fp_prime_get_qnr());
62 fp_add(t0, t0, t1);
63 } else {
64 fp_mul_dig(t1, t1, fp_prime_get_qnr());
65 fp_sub(t0, t0, t1);
66 }
67 }
68 } else {
69 fp_add(t0, t0, t1);
70 }
71#else
72 fp_add(t0, t0, t1);
73#endif
74
75 fp_inv(t1, t0);
76
77 /* c_0 = a_0/(a_0^2 + a_1^2). */
78 fp_mul(c[0], a[0], t1);
79 /* c_1 = - a_1/(a_0^2 + a_1^2). */
80 fp_mul(c[1], a[1], t1);
81 fp_neg(c[1], c[1]);
82 }
83 RLC_CATCH_ANY {
84 RLC_THROW(ERR_CAUGHT);
85 }
86 RLC_FINALLY {
87 fp_free(t0);
88 fp_free(t1);
89 }
90}

97void fp2_inv_sim(fp2_t *c, const fp2_t *a, int n) {
98 int i;
99 fp2_t u, *t = RLC_ALLOCA(fp2_t, n);
100
101 for (i = 0; i < n; i++) {
102 fp2_null(t[i]);
103 }
104 fp2_null(u);
105
106 RLC_TRY {
107 for (i = 0; i < n; i++) {
108 fp2_new(t[i]);
109 }
110 fp2_new(u);
111
112 fp2_copy(c[0], a[0]);
113 fp2_copy(t[0], a[0]);
114
115 for (i = 1; i < n; i++) {
116 fp2_copy(t[i], a[i]);
117 fp2_mul(c[i], c[i - 1], t[i]);
118 }
119
120 fp2_inv(u, c[n - 1]);
121
122 for (i = n - 1; i > 0; i--) {
123 fp2_mul(c[i], c[i - 1], u);
124 fp2_mul(u, u, t[i]);
125 }
126 fp2_copy(c[0], u);
127 }
128 RLC_CATCH_ANY {
129 RLC_THROW(ERR_CAUGHT);
130 }
131 RLC_FINALLY {
132 for (i = 0; i < n; i++) {
133 fp2_free(t[i]);
134 }
135 fp2_free(u);
136 RLC_FREE(t);
137 }

140void fp3_inv(fp3_t c, const fp3_t a) {
141 fp_t v0;
142 fp_t v1;
143 fp_t v2;
144 fp_t t0;
145
146 fp_null(v0);
147 fp_null(v1);
148 fp_null(v2);
149 fp_null(t0);
150
151 RLC_TRY {
152 fp_new(v0);
153 fp_new(v1);
154 fp_new(v2);
155 fp_new(t0);
156
157 /* v0 = a_0^2 - B * a_1 * a_2. */
158 fp_sqr(t0, a[0]);
159 fp_mul(v0, a[1], a[2]);
160 fp_copy(v2, v0);
161 for (int i = 1; i < fp_prime_get_cnr(); i++) {
162 fp_add(v2, v2, v0);
163 }
164 for (int i = 0; i >= fp_prime_get_cnr(); i--) {
165 fp_sub(v2, v2, v0);
166 }
167 fp_sub(v0, t0, v2);
168
169 /* v1 = B * a_2^2 - a_0 * a_1. */
170 fp_sqr(t0, a[2]);
171 fp_copy(v2, t0);
172 for (int i = 1; i < fp_prime_get_cnr(); i++) {
173 fp_add(v2, v2, t0);
174 }
175 for (int i = 0; i >= fp_prime_get_cnr(); i--) {
176 fp_sub(v2, v2, t0);
177 }
178 fp_mul(v1, a[0], a[1]);
179 fp_sub(v1, v2, v1);
180
181 /* v2 = a_1^2 - a_0 * a_2. */
182 fp_sqr(t0, a[1]);
183 fp_mul(v2, a[0], a[2]);
184 fp_sub(v2, t0, v2);
185
186 fp_mul(t0, a[1], v2);
187 fp_copy(c[1], t0);
188 for (int i = 1; i < fp_prime_get_cnr(); i++) {
189 fp_add(c[1], c[1], t0);
190 }
191 for (int i = 0; i >= fp_prime_get_cnr(); i--) {
192 fp_sub(c[1], c[1], t0);
193 }
194
195 fp_mul(c[0], a[0], v0);
196
197 fp_mul(t0, a[2], v1);
198 fp_copy(c[2], t0);
199 for (int i = 1; i < fp_prime_get_cnr(); i++) {
200 fp_add(c[2], c[2], t0);
201 }
202 for (int i = 0; i >= fp_prime_get_cnr(); i--) {
203 fp_sub(c[2], c[2], t0);
204 }
205
206 fp_add(t0, c[0], c[1]);
207 fp_add(t0, t0, c[2]);
208 fp_inv(t0, t0);
209
210 fp_mul(c[0], v0, t0);
211 fp_mul(c[1], v1, t0);
212 fp_mul(c[2], v2, t0);
213 } RLC_CATCH_ANY {
214 RLC_THROW(ERR_CAUGHT);
215 } RLC_FINALLY {
216 fp_free(v0);
217 fp_free(v1);
218 fp_free(v2);
219 fp_free(t0);
220 }
221}

342void fp6_inv(fp6_t c, const fp6_t a) {
343 fp2_t v0;
344 fp2_t v1;
345 fp2_t v2;
346 fp2_t t0;
347
348 fp2_null(v0);
349 fp2_null(v1);
350 fp2_null(v2);
351 fp2_null(t0);
352
353 RLC_TRY {
354 fp2_new(v0);
355 fp2_new(v1);
356 fp2_new(v2);
357 fp2_new(t0);
358
359 /* v0 = a_0^2 - E * a_1 * a_2. */
360 fp2_sqr(t0, a[0]);
361 fp2_mul(v0, a[1], a[2]);
362 fp2_mul_nor(v2, v0);
363 fp2_sub(v0, t0, v2);
364
365 /* v1 = E * a_2^2 - a_0 * a_1. */
366 fp2_sqr(t0, a[2]);
367 fp2_mul_nor(v2, t0);
368 fp2_mul(v1, a[0], a[1]);
369 fp2_sub(v1, v2, v1);
370
371 /* v2 = a_1^2 - a_0 * a_2. */
372 fp2_sqr(t0, a[1]);
373 fp2_mul(v2, a[0], a[2]);
374 fp2_sub(v2, t0, v2);
375
376 fp2_mul(t0, a[1], v2);
377 fp2_mul_nor(c[1], t0);
378
379 fp2_mul(c[0], a[0], v0);
380
381 fp2_mul(t0, a[2], v1);
382 fp2_mul_nor(c[2], t0);
383
384 fp2_add(t0, c[0], c[1]);
385 fp2_add(t0, t0, c[2]);
386 fp2_inv(t0, t0);
387
388 fp2_mul(c[0], v0, t0);
389 fp2_mul(c[1], v1, t0);
390 fp2_mul(c[2], v2, t0);
391 } RLC_CATCH_ANY {
392 RLC_THROW(ERR_CAUGHT);
393 } RLC_FINALLY {
394 fp2_free(v0);
395 fp2_free(v1);
396 fp2_free(v2);
397 fp2_free(t0);
398 }
