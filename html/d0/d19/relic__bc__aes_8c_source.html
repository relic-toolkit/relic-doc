43int bc_aes_cbc_enc(uint8_t *out, size_t *out_len, const uint8_t *in,
44 size_t in_len, const uint8_t *key, size_t key_len, const uint8_t *iv) {
45 keyInstance key_inst;
46 cipherInstance cipher_inst;
47
48 int pad_len = 16 - (in_len - 16 * (in_len/16));
49 if (*out_len < in_len + pad_len) {
50 return RLC_ERR;
51 }
52 if (makeKey2(&key_inst, DIR_ENCRYPT, 8 * key_len, (char *)key) != TRUE) {
53 return RLC_ERR;
54 }
55 if (cipherInit(&cipher_inst, MODE_CBC, NULL) != TRUE) {
56 return RLC_ERR;
57 }
58 memcpy(cipher_inst.IV, iv, RLC_BC_LEN);
59 pad_len = padEncrypt(&cipher_inst, &key_inst, (unsigned char *)in,
60 in_len, out);
61
62 *out_len = 0;
63 if (pad_len <= 0) {
64 return RLC_ERR;
65 } else {
66 *out_len = pad_len;
67 }
68 return RLC_OK;
69}
