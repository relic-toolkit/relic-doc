44  int in_len, uint8_t *key, int key_len, uint8_t *iv) {
45  keyInstance key_inst;

47 
48  int pad_len = 16 - (in_len - 16 * (in_len/16));
49  if (*out_len < in_len + pad_len) {
50  return RLC_ERR;
51  }
52  if (makeKey2(&key_inst, DIR_ENCRYPT, 8 * key_len, (char *)key) != TRUE) {
53  return RLC_ERR;
54  }
55  if (cipherInit(&cipher_inst, MODE_CBC, NULL) != TRUE) {
56  return RLC_ERR;
57  }
58  memcpy(cipher_inst.IV, iv, RLC_BC_LEN);
59  *out_len = padEncrypt(&cipher_inst, &key_inst, in, in_len, out);
60  if (*out_len <= 0) {
61  return RLC_ERR;
62  }

64 }
