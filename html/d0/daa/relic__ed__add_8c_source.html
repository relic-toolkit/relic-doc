44void ed_add_basic(ed_t r, const ed_t p, const ed_t q) {
45 fp_t t0, t1, t2;
46
47 fp_null(t0);
48 fp_null(t1);
49 fp_null(t2);
50
51 RLC_TRY {
52 fp_new(t0);
53 fp_new(t1);
54 fp_new(t2);
55
56 /* x3 = (x1*y2+y1*x2)/(1+d*x1*x2*y1*y2)
57 * y3 = (y1*y2-a*x1*x2)/(1-d*x1*x2*y1*y2) */
58
59 fp_mul(t0, p->x, q->y);
60 fp_mul(t1, p->y, q->x);
61 fp_add(t0, t0, t1);
62
63 fp_mul(t1, p->x, q->x);
64 fp_mul(t2, p->y, q->y);
65 fp_mul(t1, t1, t2);
66 fp_mul(t1, t1, core_get()->ed_d);
67 fp_add_dig(t2, t1, 1);
68 fp_inv(t2, t2);
69 fp_sub_dig(t1, t1, 1);
70 fp_neg(t1, t1);
71 fp_inv(t1, t1);
72
73 fp_mul(t0, t0, t2);
74
75 fp_mul(r->y, p->y, q->y);
76 fp_mul(t2, p->x, q->x);
77 fp_mul(t2, t2, core_get()->ed_a);
78 fp_sub(r->y, r->y, t2);
79 fp_mul(r->y, r->y, t1);
80
81 fp_copy(r->x, t0);
82 fp_copy(r->z, p->z);
83
84 r->coord = BASIC;
85 }
86 RLC_CATCH_ANY {
87 RLC_THROW(ERR_CAUGHT);
88 }
89 RLC_FINALLY {
90 fp_free(t0);
91 fp_free(t1);
92 fp_free(t2);
93 }
94}

125void ed_add_projc(ed_t r, const ed_t p, const ed_t q) {
126 fp_t t0, t1, t2, t3, t4, t5, t6, t7;
127
128 fp_null(t0);
129 fp_null(t1);
130 fp_null(t2);
131 fp_null(t3);
132 fp_null(t4);
133 fp_null(t5);
134 fp_null(t6);
135 fp_null(t7);
136
137 RLC_TRY {
138 fp_new(t0);
139 fp_new(t1);
140 fp_new(t2);
141 fp_new(t3);
142 fp_new(t4);
143 fp_new(t5);
144 fp_new(t6);
145 fp_new(t7);
146
147 /* A = z1 * z2, B = A^2 */
148 fp_mul(t0, p->z, q->z);
149 fp_sqr(t1, t0);
150
151 /* C = x1 * x2, D = y1 * y2 */
152 fp_mul(t2, p->x, q->x);
153 fp_mul(t3, p->y, q->y);
154
155 /* E = d * C * D */
156 fp_mul(t4, core_get()->ed_d, t2);
157 fp_mul(t4, t4, t3);
158
159 /* F = B - E */
160 fp_sub(t5, t1, t4);
161
162 /* G = B + E */
163 fp_add(t6, t1, t4);
164
165 /* x3 = A * F * ((x1 + y1) * (x2 + y2) - C - D) */
166 fp_mul(t7, t0, t5);
167 fp_add(r->z, p->x, p->y);
168 fp_add(r->x, q->x, q->y);
169 fp_mul(r->x, r->z, r->x);
170 fp_sub(r->x, r->x, t2);
171 fp_sub(r->x, r->x, t3);
172 fp_mul(r->x, t7, r->x);
173
174 /* y3 = A * G * (D - a * C) */
175 fp_mul(r->z, t0, t6);
176 fp_mul(r->y, core_get()->ed_a, t2);
177 fp_sub(r->y, t3, r->y);
178 fp_mul(r->y, r->z, r->y);
179
180 /* z3 = F * G */
181 fp_mul(r->z, t5, t6);
182
183 r->coord = PROJC;
184 } RLC_CATCH_ANY {
185 RLC_THROW(ERR_CAUGHT)
186 } RLC_FINALLY {
187 fp_free(t0);
188 fp_free(t1);
189 fp_free(t2);
190 fp_free(t3);
191 fp_free(t4);
192 fp_free(t5);
193 fp_free(t6);
194 fp_free(t7);
195 }
196}

230void ed_add_extnd(ed_t r, const ed_t p, const ed_t q) {
231 fp_t t0;
232 fp_t t1;
233 fp_t t2;
234 fp_t t3;
235 fp_t t4;
236
237 fp_null(t0);
238 fp_null(t1);
239 fp_null(t2);
240 fp_null(t3);
241 fp_null(t4);
242
243 RLC_TRY {
244 fp_new(t0);
245 fp_new(t1);
246 fp_new(t2);
247 fp_new(t3);
248 fp_new(t4);
249
250 /* A = x1 * x2, B = y1 * y2 */
251 fp_mul(t0, p->x, q->x);
252 fp_mul(t1, p->y, q->y);
253
254 /* C = d * t1 * t2 */
255 fp_mul(t2, core_get()->ed_d, p->t);
256 fp_mul(r->t, t2, q->t);
257
258 /* D = z1 * z2 */
259 fp_mul(r->z, p->z, q->z);
260
261 /* E = (x1 + y1) * (x2 + y2) - A - B */
262 fp_add(t2, p->x, p->y);
263 fp_add(t3, q->x, q->y);
264 fp_mul(t2, t2, t3);
265 fp_sub(t2, t2, t0);
266 fp_sub(t2, t2, t1);
267
268 /* F = D - C */
269 fp_sub(t3, r->z, r->t);
270
271 /* G = D + C */
272 fp_add(t4, r->z, r->t);
273
274 /* H = B - aA */
275 fp_mul(r->x, core_get()->ed_a, t0);
276 fp_sub(r->z, t1, r->x);
277
278 /* x3 = E * F, y3 = G * H, t3 = E * H, z3 = F * G */
279 fp_mul(r->x, t2, t3);
280 fp_mul(r->y, t4, r->z);
281 fp_mul(r->t, t2, r->z);
282 fp_mul(r->z, t3, t4);
283
284 r->coord = EXTND;
285 } RLC_CATCH_ANY {
286 RLC_THROW(ERR_CAUGHT)
287 } RLC_FINALLY {
288 fp_free(t0);
289 fp_free(t1);
290 fp_free(t2);
291 fp_free(t3);
292 fp_free(t4);
293 }
294}
