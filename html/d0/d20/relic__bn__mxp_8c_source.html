49 void bn_mxp_basic(bn_t c, const bn_t a, const bn_t b, const bn_t m) {
50  int i, l;
51  bn_t t, u, r;
52 
53  if (bn_is_zero(b)) {
54  bn_set_dig(c, 1);
55  return;
56  }
57 
58  bn_null(t);
59  bn_null(u);
60  bn_null(r);
61 
62  TRY {
63  bn_new(t);
64  bn_new(u);
65  bn_new(r);
66 
67  bn_mod_pre(u, m);
68 
69  l = bn_bits(b);
70 
71 #if BN_MOD == MONTY
72  bn_mod_monty_conv(t, a, m);
73 #else
74  bn_copy(t, a);
75 #endif
76 
77  bn_copy(r, t);
78 
79  for (i = l - 2; i >= 0; i--) {
80  bn_sqr(r, r);
81  bn_mod(r, r, m, u);
82  if (bn_get_bit(b, i)) {
83  bn_mul(r, r, t);
84  bn_mod(r, r, m, u);
85  }
86  }
87 
88 #if BN_MOD == MONTY
89  bn_mod_monty_back(r, r, m);
90 #endif
91 
92  if (bn_sign(b) == RLC_NEG) {
93  bn_gcd_ext(t, r, NULL, r, m);
94  if (bn_sign(r) == RLC_NEG) {
95  bn_add(r, r, m);
96  }
97  if (bn_cmp_dig(t, 1) == RLC_EQ) {
98  bn_copy(c, r);
99  } else {
100  bn_zero(c);
101  THROW(ERR_NO_VALID);
102  }
103  } else {
104  bn_copy(c, r);
105  }
106  }
107  CATCH_ANY {
108  THROW(ERR_CAUGHT);
109  }
110  FINALLY {
111  bn_free(t);
112  bn_free(u);
113  bn_free(r);
114  }
115 }

121 void bn_mxp_slide(bn_t c, const bn_t a, const bn_t b, const bn_t m) {
122  bn_t tab[RLC_TABLE_SIZE], t, u, r;
123  int i, j, l, w = 1;
124  uint8_t win[RLC_BN_BITS];
125 
126  if (bn_is_zero(b)) {
127  bn_set_dig(c, 1);
128  return;
129  }
130 
131  bn_null(t);
132  bn_null(u);
133  bn_null(r);
134  /* Initialize table. */
135  for (i = 0; i < RLC_TABLE_SIZE; i++) {
136  bn_null(tab[i]);
137  }
138 
139  TRY {
140  /* Find window size. */
141  i = bn_bits(b);
142  if (i <= 21) {
143  w = 2;
144  } else if (i <= 32) {
145  w = 3;
146  } else if (i <= 128) {
147  w = 4;
148  } else if (i <= 256) {
149  w = 5;
150  } else {
151  w = 6;
152  }
153 
154  for (i = 1; i < (1 << w); i += 2) {
155  bn_new(tab[i]);
156  }
157 
158  bn_new(t);
159  bn_new(u);
160  bn_new(r);
161  bn_mod_pre(u, m);
162 
163 #if BN_MOD == MONTY
164  bn_set_dig(r, 1);
165  bn_mod_monty_conv(r, r, m);
166  bn_mod_monty_conv(t, a, m);
167 #else /* BN_MOD == BARRT || BN_MOD == RADIX */
168  bn_set_dig(r, 1);
169  bn_copy(t, a);
170 #endif
171 
172  bn_copy(tab[1], t);
173  bn_sqr(t, tab[1]);
174  bn_mod(t, t, m, u);
175  /* Create table. */
176  for (i = 1; i < 1 << (w - 1); i++) {
177  bn_mul(tab[2 * i + 1], tab[2 * i - 1], t);
178  bn_mod(tab[2 * i + 1], tab[2 * i + 1], m, u);
179  }
180 
181  l = RLC_BN_BITS + 1;
182  bn_rec_slw(win, &l, b, w);
183  for (i = 0; i < l; i++) {
184  if (win[i] == 0) {
185  bn_sqr(r, r);
186  bn_mod(r, r, m, u);
187  } else {
188  for (j = 0; j < util_bits_dig(win[i]); j++) {
189  bn_sqr(r, r);
190  bn_mod(r, r, m, u);
191  }
192  bn_mul(r, r, tab[win[i]]);
193  bn_mod(r, r, m, u);
194  }
195  }
196  bn_trim(r);
197 #if BN_MOD == MONTY
198  bn_mod_monty_back(r, r, m);
199 #endif
200 
201  if (bn_sign(b) == RLC_NEG) {
202  bn_gcd_ext(t, r, NULL, r, m);
203  if (bn_sign(r) == RLC_NEG) {
204  bn_add(r, r, m);
205  }
206  if (bn_cmp_dig(t, 1) == RLC_EQ) {
207  bn_copy(c, r);
208  } else {
209  bn_zero(c);
210  THROW(ERR_NO_VALID);
211  }
212  } else {
213  bn_copy(c, r);
214  }
215  }
216  CATCH_ANY {
217  THROW(ERR_CAUGHT);
218  }
219  FINALLY {
220  for (i = 1; i < (1 << w); i++) {
221  bn_free(tab[i]);
222  }
223  bn_free(u);
224  bn_free(t);
225  bn_free(r);
226  }
227 }

233 void bn_mxp_monty(bn_t c, const bn_t a, const bn_t b, const bn_t m) {
234  bn_t tab[2], u;
235  dig_t mask;
236  int i, j, t;
237 
238  if (bn_is_zero(b)) {
239  bn_set_dig(c, 1);
240  return;
241  }
242 
243  bn_null(tab[0]);
244  bn_null(tab[1]);
245  bn_null(u);
246 
247  TRY {
248  bn_new(u);
249  bn_mod_pre(u, m);
250 
251  bn_new(tab[0]);
252  bn_new(tab[1]);
253 
254 #if BN_MOD == MONTY
255  bn_set_dig(tab[0], 1);
256  bn_mod_monty_conv(tab[0], tab[0], m);
257  bn_mod_monty_conv(tab[1], a, m);
258 #else
259  bn_set_dig(tab[0], 1);
260  bn_copy(tab[1], a);
261 #endif
262 
263  for (i = bn_bits(b) - 1; i >= 0; i--) {
264  j = bn_get_bit(b, i);
265  dv_swap_cond(tab[0]->dp, tab[1]->dp, RLC_BN_DIGS, j ^ 1);
266  mask = -(j ^ 1);
267  t = (tab[0]->used ^ tab[1]->used) & mask;
268  tab[0]->used ^= t;
269  tab[1]->used ^= t;
270  bn_mul(tab[0], tab[0], tab[1]);
271  bn_mod(tab[0], tab[0], m, u);
272  bn_sqr(tab[1], tab[1]);
273  bn_mod(tab[1], tab[1], m, u);
274  dv_swap_cond(tab[0]->dp, tab[1]->dp, RLC_BN_DIGS, j ^ 1);
275  mask = -(j ^ 1);
276  t = (tab[0]->used ^ tab[1]->used) & mask;
277  tab[0]->used ^= t;
278  tab[1]->used ^= t;
279  }
280 
281 #if BN_MOD == MONTY
282  bn_mod_monty_back(u, tab[0], m);
283 #else
284  bn_copy(u, tab[0]);
285 #endif
286 
287  /* Silly branchless code, since called functions not constant-time. */
288  bn_gcd_ext(tab[1], tab[0], NULL, u, m);
289  dv_swap_cond(u->dp, tab[0]->dp, RLC_BN_DIGS, bn_sign(b) == RLC_NEG);
290  if (bn_sign(b) == RLC_NEG) {
291  u->sign = tab[0]->sign;
292  if (bn_cmp_dig(tab[1], 1) != RLC_EQ) {
293  bn_zero(c);
294  THROW(ERR_NO_VALID);
295  }
296  }
297  bn_add(tab[1], u, m);
298  dv_swap_cond(u->dp, tab[1]->dp, RLC_BN_DIGS, bn_sign(b) == RLC_NEG && bn_sign(u) == RLC_NEG);
299  u->sign = RLC_POS;
300  bn_copy(c, u);
301  }
302  CATCH_ANY {
303  THROW(ERR_CAUGHT);
304  }
305  FINALLY {
306  bn_free(tab[1]);
307  bn_free(tab[0]);
308  bn_free(u);
309  }
310 }
