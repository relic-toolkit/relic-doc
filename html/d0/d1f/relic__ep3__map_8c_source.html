40void ep3_map(ep3_t p, const uint8_t *msg, size_t len) {
41 /* enough space for two field elements plus extra bytes for uniformity */
42 const size_t elm = (FP_PRIME + ep_param_level() + 7) / 8;
43 uint8_t t0z, t0, t1, t1z, t2, sign, *r = RLC_ALLOCA(uint8_t, 6 * elm + 1);
44 fp3_t t, u, v, w, y, x1, y1, z1;
45 ctx_t *ctx = core_get();
46 dig_t c2, c3;
47 bn_t k;
48
49 bn_null(k);
50 fp3_null(t);
51 fp3_null(u);
52 fp3_null(v);
53 fp3_null(w);
54 fp3_null(y);
55 fp3_null(x1);
56 fp3_null(y1);
57 fp3_null(z1);
58
59 RLC_TRY {
60 bn_new(k);
61 fp3_new(t);
62 fp3_new(u);
63 fp3_new(v);
64 fp3_new(w);
65 fp3_new(y);
66 fp3_new(x1);
67 fp3_new(y1);
68 fp3_new(z1);
69
70 md_xmd(r, 6 * elm + 1, msg, len, (const uint8_t *)"RELIC", 5);
71
72 for (int i = 0; i < 3; i++) {
73 bn_read_bin(k, r, elm);
74 fp_prime_conv(u[i], k);
75 r += elm;
76 bn_read_bin(k, r, elm);
77 fp_prime_conv(t[i], k);
78 r += elm;
79 }
80 sign = r[0] & 1;
81 r -= 6*elm;
82
83 /* Assume that a = 0. */
84 fp3_sqr(x1, u);
85 fp3_mul(x1, x1, u);
86 fp3_sqr(y1, t);
87 fp3_add(x1, x1, ctx->ep3_b);
88 fp3_sub(x1, x1, y1);
89 fp3_dbl(y1, y1);
90 fp3_add(y1, y1, x1);
91 fp3_copy(z1, u);
92 fp_mul(z1[0], z1[0], ctx->ep_map_c[4]);
93 fp_mul(z1[1], z1[1], ctx->ep_map_c[4]);
94 fp_mul(z1[2], z1[2], ctx->ep_map_c[4]);
95 fp3_mul(x1, x1, z1);
96 fp3_mul(z1, z1, t);
97 fp3_dbl(z1, z1);
98
99 fp3_dbl(y, y1);
100 fp3_sqr(y, y);
101 fp3_mul(v, y1, u);
102 fp3_sub(v, x1, v);
103 fp3_mul(v, v, z1);
104 fp3_mul(w, y1, z1);
105 fp3_dbl(w, w);
106
107 if (fp3_is_zero(w)) {
108 ep3_set_infty(p);
109 } else {
110 fp3_inv(w, w);
111 fp3_mul(x1, v, w);
112 fp3_add(y1, u, x1);
113 fp3_neg(y1, y1);
114 fp3_mul(z1, y, w);
115 fp3_sqr(z1, z1);
116 fp3_add(z1, z1, u);
117
118 ep3_rhs(t, x1);
119 ep3_rhs(u, y1);
120 ep3_rhs(v, z1);
121
122 c2 = fp3_is_sqr(u);
123 c3 = fp3_is_sqr(v);
124
125 fp3_copy_sec(t, u, c2);
126 fp3_copy_sec(x1, y1, c2);
127 fp3_copy_sec(t, v, c3);
128 fp3_copy_sec(x1, z1, c3);
129
130 if (!fp3_srt(t, t)) {
131 RLC_THROW(ERR_NO_VALID);
132 }
133
134 t0z = fp_is_zero(t[0]);
135 fp_prime_back(k, t[0]);
136 t0 = bn_get_bit(k, 0);
137 t1z = fp_is_zero(t[1]);
138 fp_prime_back(k, t[1]);
139 t1 = bn_get_bit(k, 0);
140 fp_prime_back(k, t[2]);
141 t2 = bn_get_bit(k, 0);
142
143 /* t[0] == 0 ? sgn0(t[1]) : sgn0(t[0]) */
144 sign ^= (t0 | (t0z & (t1 | (t1z & t2))));
145
146 fp3_neg(u, t);
147 fp3_copy_sec(t, u, sign);
148
149 fp3_copy(p->x, x1);
150 fp3_copy(p->y, t);
151 fp3_set_dig(p->z, 1);
152 p->coord = BASIC;
153
154 ep3_mul_cof(p, p);
155 }
156 }
157 RLC_CATCH_ANY {
158 RLC_THROW(ERR_CAUGHT);
159 }
160 RLC_FINALLY {
161 bn_free(k);
162 fp3_free(t);
163 fp3_free(u);
164 fp3_free(v);
165 fp3_free(w);
166 fp3_free(y);
167 fp3_free(x1);
168 fp3_free(y1);
169 fp3_free(z1);
170 RLC_FREE(r);
171 }
172}
