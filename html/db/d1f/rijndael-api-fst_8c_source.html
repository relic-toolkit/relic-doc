46int makeKey(keyInstance * key, BYTE direction, int keyLen, char *keyMaterial) {
47 int i;
48 char *keyMat;
49 u8 cipherKey[MAXKB];
50
51 if (key == NULL) {
52 return BAD_KEY_INSTANCE;
53 }
54
55 if ((direction == DIR_ENCRYPT) || (direction == DIR_DECRYPT)) {
56 key->direction = direction;
57 } else {
58 return BAD_KEY_DIR;
59 }
60
61 if ((keyLen == 128) || (keyLen == 192) || (keyLen == 256)) {
62 key->keyLen = keyLen;
63 } else {
64 return BAD_KEY_MAT;
65 }
66
67 if (keyMaterial != NULL) {
68 strncpy(key->keyMaterial, keyMaterial, keyLen / 4);
69 }
70
71 /* initialize key schedule: */
72 keyMat = key->keyMaterial;
73 for (i = 0; i < key->keyLen / 8; i++) {
74 int t, v;
75
76 t = *keyMat++;
77 if ((t >= '0') && (t <= '9'))
78 v = (t - '0') << 4;
79 else if ((t >= 'a') && (t <= 'f'))
80 v = (t - 'a' + 10) << 4;
81 else if ((t >= 'A') && (t <= 'F'))
82 v = (t - 'A' + 10) << 4;
83 else
84 return BAD_KEY_MAT;
85
86 t = *keyMat++;
87 if ((t >= '0') && (t <= '9'))
88 v ^= (t - '0');
89 else if ((t >= 'a') && (t <= 'f'))
90 v ^= (t - 'a' + 10);
91 else if ((t >= 'A') && (t <= 'F'))
92 v ^= (t - 'A' + 10);
93 else
94 return BAD_KEY_MAT;
95
96 cipherKey[i] = (u8) v;
97 }
98 if (direction == DIR_ENCRYPT) {
99 key->Nr = rijndaelKeySetupEnc(key->rk, cipherKey, keyLen);
100 } else {
101 key->Nr = rijndaelKeySetupDec(key->rk, cipherKey, keyLen);
102 }
103 rijndaelKeySetupEnc(key->ek, cipherKey, keyLen);
