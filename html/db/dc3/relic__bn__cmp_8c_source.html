39int bn_cmp_abs(const bn_t a, const bn_t b) {

75int bn_cmp(const bn_t a, const bn_t b) {
76 if (bn_is_zero(a) && bn_is_zero(b)) {
77 return RLC_EQ;

79
80 if (a->sign == RLC_POS && b->sign == RLC_NEG) {

84 return RLC_LT;
85 }
