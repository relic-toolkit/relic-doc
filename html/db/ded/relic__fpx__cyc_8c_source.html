84void fp2_exp_cyc(fp2_t c, const fp2_t a, const bn_t b) {
85 fp2_t r, s, t[1 << (RLC_WIDTH - 2)];
86 int8_t naf[RLC_FP_BITS + 1], *k;
87 size_t l;
88
89 if (bn_is_zero(b)) {
90 return fp2_set_dig(c, 1);
91 }
92
93 if (bn_bits(b) <= RLC_DIG) {
94 fp2_exp_dig(c, a, b->dp[0]);
95 if (bn_sign(b) == RLC_NEG) {
96 fp2_inv_cyc(c, c);
97 }
98 return;
99 }
100
101 fp2_null(r);
102 fp2_null(s);
103
104 RLC_TRY {
105 fp2_new(r);
106 fp2_new(s);
107 for (int i = 0; i < (1 << (RLC_WIDTH - 2)); i ++) {
108 fp2_null(t[i]);
109 fp2_new(t[i]);
110 }
111
112#if RLC_WIDTH > 2
113 fp2_sqr(t[0], a);
114 fp2_mul(t[1], t[0], a);
115 for (int i = 2; i < (1 << (RLC_WIDTH - 2)); i++) {
116 fp2_mul(t[i], t[i - 1], t[0]);
117 }
118#endif
119 fp2_copy(t[0], a);
120
121 l = RLC_FP_BITS + 1;
122 fp2_set_dig(r, 1);
123 bn_rec_naf(naf, &l, b, RLC_WIDTH);
124
125 k = naf + l - 1;
126 for (int i = l - 1; i >= 0; i--, k--) {
127 fp2_sqr(r, r);
128
129 if (*k > 0) {
130 fp2_mul(r, r, t[*k / 2]);
131 }
132 if (*k < 0) {
133 fp2_inv_cyc(s, t[-*k / 2]);
134 fp2_mul(r, r, s);
135 }
136 }
137
138 if (bn_sign(b) == RLC_NEG) {
139 fp2_inv_cyc(c, r);
140 } else {
141 fp2_copy(c, r);
142 }
143 }
144 RLC_CATCH_ANY {
145 RLC_THROW(ERR_CAUGHT);
146 }
147 RLC_FINALLY {
148 fp2_free(r);
149 fp2_free(s);
150 for (int i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
151 fp2_free(t[i]);
152 }
153 }

156void fp2_exp_cyc_sim(fp2_t e, const fp2_t a, const bn_t b, const fp2_t c,
157 const bn_t d) {
158 int n0, n1;
159 int8_t naf0[RLC_FP_BITS + 1], naf1[RLC_FP_BITS + 1], *_k, *_m;
160 fp2_t r, t0[1 << (RLC_WIDTH - 2)];
161 fp2_t s, t1[1 << (RLC_WIDTH - 2)];
162 size_t l, l0, l1;
163
164 if (bn_is_zero(b)) {
165 return fp2_exp_cyc(e, c, d);
166 }
167
168 if (bn_is_zero(d)) {
169 return fp2_exp_cyc(e, a, b);
170 }
171
172 fp2_null(r);
173 fp2_null(s);
174
175 RLC_TRY {
176 fp2_new(r);
177 fp2_new(s);
178 for (int i = 0; i < (1 << (RLC_WIDTH - 2)); i ++) {
179 fp2_null(t0[i]);
180 fp2_null(t1[i]);
181 fp2_new(t0[i]);
182 fp2_new(t1[i]);
183 }
184
185#if RLC_WIDTH > 2
186 fp2_sqr(t0[0], a);
187 fp2_mul(t0[1], t0[0], a);
188 for (int i = 2; i < (1 << (RLC_WIDTH - 2)); i++) {
189 fp2_mul(t0[i], t0[i - 1], t0[0]);
190 }
191
192 fp2_sqr(t1[0], c);
193 fp2_mul(t1[1], t1[0], c);
194 for (int i = 2; i < (1 << (RLC_WIDTH - 2)); i++) {
195 fp2_mul(t1[i], t1[i - 1], t1[0]);
196 }
197#endif
198 fp2_copy(t0[0], a);
199 fp2_copy(t1[0], c);
200
201 l0 = l1 = RLC_FP_BITS + 1;
202 bn_rec_naf(naf0, &l0, b, RLC_WIDTH);
203 bn_rec_naf(naf1, &l1, d, RLC_WIDTH);
204
205 l = RLC_MAX(l0, l1);
206 if (bn_sign(b) == RLC_NEG) {
207 for (size_t i = 0; i < l0; i++) {
208 naf0[i] = -naf0[i];
209 }
210 }
211 if (bn_sign(d) == RLC_NEG) {
212 for (size_t i = 0; i < l1; i++) {
213 naf1[i] = -naf1[i];
214 }
215 }
216
217 _k = naf0 + l - 1;
218 _m = naf1 + l - 1;
219
220 fp2_set_dig(r, 1);
221 for (int i = l - 1; i >= 0; i--, _k--, _m--) {
222 fp2_sqr(r, r);
223
224 n0 = *_k;
225 n1 = *_m;
226
227 if (n0 > 0) {
228 fp2_mul(r, r, t0[n0 / 2]);
229 }
230 if (n0 < 0) {
231 fp2_inv_cyc(s, t0[-n0 / 2]);
232 fp2_mul(r, r, s);
233 }
234 if (n1 > 0) {
235 fp2_mul(r, r, t1[n1 / 2]);
236 }
237 if (n1 < 0) {
238 fp2_inv_cyc(s, t1[-n1 / 2]);
239 fp2_mul(r, r, s);
240 }
241 }
242
243 fp2_copy(e, r);
244 }
245 RLC_CATCH_ANY {
246 RLC_THROW(ERR_CAUGHT);
247 }
248 RLC_FINALLY {
249 fp2_free(r);
250 fp2_free(s);
251 for (int i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
252 fp2_free(t0[i]);
253 fp2_free(t1[i]);
254 }
255 }

473void fp12_conv_cyc(fp12_t c, const fp12_t a) {
474 fp12_t t;
475
476 fp12_null(t);
477
478 RLC_TRY {
479 fp12_new(t);
480
481 /* First, compute c = a^(p^6 - 1). */
482 /* t = a^{-1}. */
483 fp12_inv(t, a);
484 /* c = a^(p^6). */
485 fp12_inv_cyc(c, a);
486 /* c = a^(p^6 - 1). */
487 fp12_mul(c, c, t);
488
489 /* Second, compute c^(p^2 + 1). */
490 /* t = c^(p^2). */
491 fp12_frb(t, c, 2);
492
493 /* c = c^(p^2 + 1). */
494 fp12_mul(c, c, t);
495 }
496 RLC_CATCH_ANY {
497 RLC_THROW(ERR_CAUGHT);
498 }
499 RLC_FINALLY {
500 fp12_free(t);
501 }

533void fp12_back_cyc(fp12_t c, const fp12_t a) {
534 fp2_t t0, t1, t2;
535
536 fp2_null(t0);
537 fp2_null(t1);
538 fp2_null(t2);
539
540 RLC_TRY {
541 fp2_new(t0);
542 fp2_new(t1);
543 fp2_new(t2);
544
545 int f = fp2_is_zero(a[1][0]);
546 /* If f, t0 = 2 * g4 * g5, t1 = g3. */
547 fp2_copy(t2, a[0][1]);
548 fp2_copy_sec(t2, a[1][2], f);
549 /* t0 = g4^2. */
550 fp2_mul(t0, a[0][1], t2);
551 fp2_dbl(t2, t0);
552 fp2_copy_sec(t0, t2, f);
553 /* t1 = 3 * g4^2 - 2 * g3. */
554 fp2_sub(t1, t0, a[0][2]);
555 fp2_dbl(t1, t1);
556 fp2_add(t1, t1, t0);
557 /* t0 = E * g5^2 + t1. */
558 fp2_sqr(t2, a[1][2]);
559 fp2_mul_nor(t0, t2);
560 fp2_add(t0, t0, t1);
561 /* t1 = (4 * g2). */
562 fp2_dbl(t1, a[1][0]);
563 fp2_dbl(t1, t1);
564 fp2_copy_sec(t1, a[0][2], f);
565 /* If unity, decompress to unity as well. */
566 f = fp12_cmp_dig(a, 1) == RLC_EQ;
567 fp2_set_dig(t2, 1);
568 fp2_copy_sec(t1, t2, f);
569
570 /* t1 = 1/g3 or 1/(4*g2), depending on the above. */
571 fp2_inv(t1, t1);
572 /* c_1 = g1. */
573 fp2_mul(c[1][1], t0, t1);
574
575 /* t1 = g3 * g4. */
576 fp2_mul(t1, a[0][2], a[0][1]);
577 /* t2 = 2 * g1^2 - 3 * g3 * g4. */
578 fp2_sqr(t2, c[1][1]);
579 fp2_sub(t2, t2, t1);
580 fp2_dbl(t2, t2);
581 fp2_sub(t2, t2, t1);
582 /* t1 = g2 * g5. */
583 fp2_mul(t1, a[1][0], a[1][2]);
584 /* c_0 = E * (2 * g1^2 + g2 * g5 - 3 * g3 * g4) + 1. */
585 fp2_add(t2, t2, t1);
586 fp2_mul_nor(c[0][0], t2);
587 fp_add_dig(c[0][0][0], c[0][0][0], 1);
588
589 fp2_copy(c[0][1], a[0][1]);
590 fp2_copy(c[0][2], a[0][2]);
591 fp2_copy(c[1][0], a[1][0]);
592 fp2_copy(c[1][2], a[1][2]);
593 }
594 RLC_CATCH_ANY {
595 RLC_THROW(ERR_CAUGHT);
596 }
597 RLC_FINALLY {
598 fp2_free(t0);
599 fp2_free(t1);
600 fp2_free(t2);
601 }
602}

604void fp12_back_cyc_sim(fp12_t c[], const fp12_t a[], int n) {
605 fp2_t *t = RLC_ALLOCA(fp2_t, n * 3);
606 fp2_t *t0 = t + 0 * n, *t1 = t + 1 * n, *t2 = t + 2 * n;
607
608 if (n == 0) {
609 RLC_FREE(t);
610 return;
611 }
612
613 RLC_TRY {
614 if (t == NULL) {
615 RLC_THROW(ERR_NO_MEMORY);
616 }
617 for (int i = 0; i < n; i++) {
618 fp2_null(t0[i]);
619 fp2_null(t1[i]);
620 fp2_null(t2[i]);
621 fp2_new(t0[i]);
622 fp2_new(t1[i]);
623 fp2_new(t2[i]);
624 }
625
626 for (int i = 0; i < n; i++) {
627 int f = fp2_is_zero(a[i][1][0]);
628 /* If f, t0 = 2 * g4 * g5, t1 = g3. */
629 fp2_copy(t2[i], a[i][0][1]);
630 fp2_copy_sec(t2[i], a[i][1][2], f);
631 /* t0 = g4^2. */
632 fp2_mul(t0[i], a[i][0][1], t2[i]);
633 fp2_dbl(t2[i], t0[i]);
634 fp2_copy_sec(t0[i], t2[i], f);
635 /* t1 = 3 * g4^2 - 2 * g3. */
636 fp2_sub(t1[i], t0[i], a[i][0][2]);
637 fp2_dbl(t1[i], t1[i]);
638 fp2_add(t1[i], t1[i], t0[i]);
639 /* t0 = E * g5^2 + t1. */
640 fp2_sqr(t2[i], a[i][1][2]);
641 fp2_mul_nor(t0[i], t2[i]);
642 fp2_add(t0[i], t0[i], t1[i]);
643 /* t1 = (4 * g2). */
644 fp2_dbl(t1[i], a[i][1][0]);
645 fp2_dbl(t1[i], t1[i]);
646 fp2_copy_sec(t1[i], a[i][0][2], f);
647 /* If unity, decompress to unity as well. */
648 f = (fp12_cmp_dig(a[i], 1) == RLC_EQ);
649 fp2_set_dig(t2[i], 1);
650 fp2_copy_sec(t1[i], t2[i], f);
651 }
652
653 /* t1 = 1 / t1. */
654 fp2_inv_sim(t1, t1, n);
655
656 for (int i = 0; i < n; i++) {
657 /* t0 = g1. */
658 fp2_mul(c[i][1][1], t0[i], t1[i]);
659
660 /* t1 = g3 * g4. */
661 fp2_mul(t1[i], a[i][0][2], a[i][0][1]);
662 /* t2 = 2 * g1^2 - 3 * g3 * g4. */
663 fp2_sqr(t2[i], c[i][1][1]);
664 fp2_sub(t2[i], t2[i], t1[i]);
665 fp2_dbl(t2[i], t2[i]);
666 fp2_sub(t2[i], t2[i], t1[i]);
667 /* t1 = g2 * g5. */
668 fp2_mul(t1[i], a[i][1][0], a[i][1][2]);
669 /* t2 = E * (2 * g1^2 + g2 * g5 - 3 * g3 * g4) + 1. */
670 fp2_add(t2[i], t2[i], t1[i]);
671 fp2_mul_nor(c[i][0][0], t2[i]);
672 fp_add_dig(c[i][0][0][0], c[i][0][0][0], 1);
673

695 return fp12_set_dig(c, 1);
696 }
697
698 if ((bn_bits(b) > RLC_DIG) && ((bn_ham(b) << 3) > bn_bits(b))) {
699 fp12_t r, s, t[1 << (RLC_WIDTH - 2)];

732 fp12_sqr_cyc(r, r);
733
734 if (*k > 0) {
735 fp12_mul(r, r, t[*k / 2]);
736 }
737 if (*k < 0) {
738 fp12_inv_cyc(s, t[-*k / 2]);
739 fp12_mul(r, r, s);
740 }
741 }

743 if (bn_sign(b) == RLC_NEG) {

746 fp12_copy(c, r);

749 RLC_THROW(ERR_CAUGHT);

796 }

799 fp12_mul(c, c, u[i]);

803 fp12_inv_cyc(c, c);
804 }
805 }
806 RLC_CATCH_ANY {
807 RLC_THROW(ERR_CAUGHT);
808 }
809 RLC_FINALLY {

811 fp12_free(u[i]);
812 }
813 fp12_free(t);

909 fp12_sqr_cyc(e, e);

1286void fp18_back_cyc(fp18_t c, const fp18_t a) {
1287 fp3_t t0, t1, t2;
1288
1289 fp3_null(t0);
1290 fp3_null(t1);
1291 fp3_null(t2);
1292
1293 RLC_TRY {
1294 fp3_new(t0);
1295 fp3_new(t1);
1296 fp3_new(t2);
1297
1298 int f = fp3_is_zero(a[1][0]);
1299 /* If f, t0 = 2 * g4 * g5, t1 = g3. */
1300 fp3_copy(t2, a[0][1]);
1301 fp3_copy_sec(t2, a[1][2], f);
1302 /* t0 = g4^2. */
1303 fp3_mul(t0, a[0][1], t2);
1304 fp3_dbl(t2, t0);
1305 fp3_copy_sec(t0, t2, f);
1306 /* t1 = 3 * g4^2 - 2 * g3. */
1307 fp3_sub(t1, t0, a[0][2]);
1308 fp3_dbl(t1, t1);
1309 fp3_add(t1, t1, t0);
1310 /* t0 = E * g5^2 + t1. */
1311 fp3_sqr(t2, a[1][2]);
1312 fp3_mul_nor(t0, t2);
1313 fp3_add(t0, t0, t1);
1314 /* t1 = (4 * g2). */
1315 fp3_dbl(t1, a[1][0]);
1316 fp3_dbl(t1, t1);
1317 fp3_copy_sec(t1, a[0][2], f);
1318 /* If unity, decompress to unity as well. */
1319 f = fp18_cmp_dig(a, 1) == RLC_EQ;
1320 fp3_set_dig(t2, 1);
1321 fp3_copy_sec(t1, t2, f);
1322
1323 /* t1 = 1/g3 or 1/(4 * g2), depending on the above. */
1324 fp3_inv(t1, t1);
1325 /* c_1 = g1. */
1326 fp3_mul(c[1][1], t0, t1);
1327
1328 /* t1 = g3 * g4. */
1329 fp3_mul(t1, a[0][2], a[0][1]);
1330 /* t2 = 2 * g1^2 - 3 * g3 * g4. */
1331 fp3_sqr(t2, c[1][1]);
1332 fp3_sub(t2, t2, t1);
1333 fp3_dbl(t2, t2);
1334 fp3_sub(t2, t2, t1);
1335 /* t1 = g2 * g5. */
1336 fp3_mul(t1, a[1][0], a[1][2]);
1337 /* c_0 = E * (2 * g1^2 + g2 * g5 - 3 * g3 * g4) + 1. */
1338 fp3_add(t2, t2, t1);
1339 fp3_mul_nor(c[0][0], t2);
1340 fp_add_dig(c[0][0][0], c[0][0][0], 1);
1341
1342 fp3_copy(c[0][1], a[0][1]);
1343 fp3_copy(c[0][2], a[0][2]);
1344 fp3_copy(c[1][0], a[1][0]);
1345 fp3_copy(c[1][2], a[1][2]);
1346 }
1347 RLC_CATCH_ANY {
1348 RLC_THROW(ERR_CAUGHT);
1349 }
1350 RLC_FINALLY {
1351 fp3_free(t0);
1352 fp3_free(t1);
1353 fp3_free(t2);
1354 }

1357void fp18_back_cyc_sim(fp18_t c[], const fp18_t a[], int n) {
1358 fp3_t *t = RLC_ALLOCA(fp3_t, n * 3);
1359 fp3_t *t0 = t + 0 * n, *t1 = t + 1 * n, *t2 = t + 2 * n;
1360
1361 if (n == 0) {
1362 RLC_FREE(t);
1363 return;
1364 }
1365
1366 RLC_TRY {
1367 if (t == NULL) {
1368 RLC_THROW(ERR_NO_MEMORY);
1369 }
1370 for (int i = 0; i < n; i++) {
1371 fp3_null(t0[i]);
1372 fp3_null(t1[i]);
1373 fp3_null(t2[i]);
1374 fp3_new(t0[i]);
1375 fp3_new(t1[i]);
1376 fp3_new(t2[i]);
1377 }
1378
1379 for (int i = 0; i < n; i++) {
1380 int f = fp3_is_zero(a[i][1][0]);
1381 /* If f, t0 = 2 * g4 * g5, t1 = g3. */
1382 fp3_copy(t2[i], a[i][0][1]);
1383 fp3_copy_sec(t2[i], a[i][1][2], f);
1384 /* t0 = g4^2. */
1385 fp3_mul(t0[i], a[i][0][1], t2[i]);
1386 fp3_dbl(t2[i], t0[i]);
1387 fp3_copy_sec(t0[i], t2[i], f);
1388 /* t1 = 3 * g4^2 - 2 * g3. */
1389 fp3_sub(t1[i], t0[i], a[i][0][2]);
1390 fp3_dbl(t1[i], t1[i]);
1391 fp3_add(t1[i], t1[i], t0[i]);
1392 /* t0 = E * g5^2 + t1. */
1393 fp3_sqr(t2[i], a[i][1][2]);
1394 fp3_mul_nor(t0[i], t2[i]);
1395 fp3_add(t0[i], t0[i], t1[i]);
1396 /* t1 = (4 * g2). */
1397 fp3_dbl(t1[i], a[i][1][0]);
1398 fp3_dbl(t1[i], t1[i]);
1399 fp3_copy_sec(t1[i], a[i][0][2], f);
1400 /* If unity, decompress to unity as well. */
1401 f = (fp18_cmp_dig(a[i], 1) == RLC_EQ);
1402 fp3_set_dig(t2[i], 1);
1403 fp3_copy_sec(t1[i], t2[i], f);
1404 }
1405
1406 /* t1 = 1 / t1. */
1407 fp3_inv_sim(t1, t1, n);
1408
1409 for (int i = 0; i < n; i++) {
1410 /* t0 = g1. */
1411 fp3_mul(c[i][1][1], t0[i], t1[i]);
1412
1413 /* t1 = g3 * g4. */
1414 fp3_mul(t1[i], a[i][0][2], a[i][0][1]);
1415 /* t2 = 2 * g1^2 - 3 * g3 * g4. */
1416 fp3_sqr(t2[i], c[i][1][1]);
1417 fp3_sub(t2[i], t2[i], t1[i]);
1418 fp3_dbl(t2[i], t2[i]);
1419 fp3_sub(t2[i], t2[i], t1[i]);
1420 /* t1 = g2 * g5. */
1421 fp3_mul(t1[i], a[i][1][0], a[i][1][2]);
1422 /* t2 = E * (2 * g1^2 + g2 * g5 - 3 * g3 * g4) + 1. */
1423 fp3_add(t2[i], t2[i], t1[i]);
1424 fp3_mul_nor(c[i][0][0], t2[i]);
1425 fp_add_dig(c[i][0][0][0], c[i][0][0][0], 1);
1426
1427 fp3_copy(c[i][0][1], a[i][0][1]);
1428 fp3_copy(c[i][0][2], a[i][0][2]);
1429 fp3_copy(c[i][1][0], a[i][1][0]);
1430 fp3_copy(c[i][1][2], a[i][1][2]);
1431 }
1432 } RLC_CATCH_ANY {
1433 RLC_THROW(ERR_CAUGHT);
1434 } RLC_FINALLY {
1435 for (int i = 0; i < n; i++) {
1436 fp3_free(t0[i]);
1437 fp3_free(t1[i]);
1438 fp3_free(t2[i]);
1439 }
1440 RLC_FREE(t);
1441 }

1444void fp18_exp_cyc(fp18_t c, const fp18_t a, const bn_t b) {
1445 size_t l, w = bn_ham(b);
1446
1447 if (bn_is_zero(b)) {
1448 return fp18_set_dig(c, 1);
1449 }
1450
1451 if ((bn_bits(b) > RLC_DIG) && ((w << 3) > bn_bits(b))) {
1452 fp18_t r, s, t[1 << (RLC_WIDTH - 2)];
1453 int8_t naf[RLC_FP_BITS + 1], *k;
1454
1455 if (bn_bits(b) <= RLC_DIG) {
1456 w = 2;
1457 } else {
1458 w = RLC_WIDTH;
1459 }
1460
1461 fp18_null(r);
1462 fp18_null(s);
1463
1464 RLC_TRY {
1465 fp18_new(r);
1466 fp18_new(s);
1467 for (size_t i = 0; i < (1 << (RLC_WIDTH - 2)); i ++) {
1468 fp18_null(t[i]);
1469 fp18_new(t[i]);
1470 }
1471
1472#if RLC_WIDTH > 2
1473 fp18_sqr_cyc(t[0], a);
1474 fp18_mul(t[1], t[0], a);
1475 for (size_t i = 2; i < (1 << (w - 2)); i++) {
1476 fp18_mul(t[i], t[i - 1], t[0]);
1477 }
1478#endif
1479 fp18_copy(t[0], a);
1480
1481 l = RLC_FP_BITS + 1;
1482 fp18_set_dig(r, 1);
1483 bn_rec_naf(naf, &l, b, w);
1484
1485 k = naf + l - 1;
1486 for (int i = l - 1; i >= 0; i--, k--) {
1487 fp18_sqr_cyc(r, r);
1488
1489 if (*k > 0) {
1490 fp18_mul(r, r, t[*k / 2]);
1491 }
1492 if (*k < 0) {
1493 fp18_inv_cyc(s, t[-*k / 2]);
1494 fp18_mul(r, r, s);
1495 }
1496 }
1497
1498 if (bn_sign(b) == RLC_NEG) {
1499 fp18_inv_cyc(c, r);
1500 } else {
1501 fp18_copy(c, r);
1502 }
1503 } RLC_CATCH_ANY {
1504 RLC_THROW(ERR_CAUGHT);
1505 }
1506 RLC_FINALLY {
1507 fp18_free(r);
1508 fp18_free(s);
1509 for (size_t i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
1510 fp18_free(t[i]);
1511 }
1512 }
1513 } else {
1514 size_t j, k;
1515 fp18_t t, *u = RLC_ALLOCA(fp18_t, w);
1516
1517 fp18_null(t);
1518
1519 RLC_TRY {
1520 if (u == NULL) {
1521 RLC_THROW(ERR_NO_MEMORY);
1522 }
1523 for (int i = 0; i < w; i++) {
1524 fp18_null(u[i]);
1525 fp18_new(u[i]);
1526 }
1527 fp18_new(t);
1528
1529 j = 0;
1530 fp18_copy(t, a);
1531 for (size_t i = 1; i < bn_bits(b); i++) {
1532 fp18_sqr_pck(t, t);
1533 if (bn_get_bit(b, i)) {
1534 fp18_copy(u[j++], t);
1535 }
1536 }
1537
1538 if (!bn_is_even(b)) {
1539 j = 0;
1540 k = w - 1;
1541 } else {
1542 j = 1;
1543 k = w;
1544 }
1545
1546 fp18_back_cyc_sim(u, u, k);
1547
1548 if (!bn_is_even(b)) {
1549 fp18_copy(c, a);
1550 } else {
1551 fp18_copy(c, u[0]);
1552 }
1553
1554 for (size_t i = j; i < k; i++) {
1555 fp18_mul(c, c, u[i]);
1556 }
1557
1558 if (bn_sign(b) == RLC_NEG) {
1559 fp18_inv_cyc(c, c);
1560 }
1561 }
1562 RLC_CATCH_ANY {
1563 RLC_THROW(ERR_CAUGHT);
1564 }
1565 RLC_FINALLY {
1566 for (size_t i = 0; i < w; i++) {
1567 fp18_free(u[i]);
1568 }
1569 fp18_free(t);
1570 RLC_FREE(u);
1571 }
1572 }

1575void fp18_exp_cyc_sim(fp18_t e, const fp18_t a, const bn_t b, const fp18_t c,
1576 const bn_t d) {
1577 int i, n0, n1;
1578 size_t l, l0, l1;
1579 int8_t naf0[RLC_FP_BITS + 1], naf1[RLC_FP_BITS + 1], *_k, *_m;
1580 fp18_t r, t0[1 << (RLC_WIDTH - 2)];
1581 fp18_t s, t1[1 << (RLC_WIDTH - 2)];
1582
1583 if (bn_is_zero(b)) {
1584 return fp18_exp_cyc(e, c, d);
1585 }
1586
1587 if (bn_is_zero(d)) {
1588 return fp18_exp_cyc(e, a, b);
1589 }
1590
1591 fp18_null(r);
1592 fp18_null(s);
1593
1594 RLC_TRY {
1595 fp18_new(r);
1596 fp18_new(s);
1597 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i ++) {
1598 fp18_null(t0[i]);
1599 fp18_null(t1[i]);
1600 fp18_new(t0[i]);
1601 fp18_new(t1[i]);
1602 }
1603
1604#if RLC_WIDTH > 2
1605 fp18_sqr(t0[0], a);
1606 fp18_mul(t0[1], t0[0], a);
1607 for (int i = 2; i < (1 << (RLC_WIDTH - 2)); i++) {
1608 fp18_mul(t0[i], t0[i - 1], t0[0]);
1609 }
1610
1611 fp18_sqr(t1[0], c);
1612 fp18_mul(t1[1], t1[0], c);
1613 for (int i = 2; i < (1 << (RLC_WIDTH - 2)); i++) {
1614 fp18_mul(t1[i], t1[i - 1], t1[0]);
1615 }
1616#endif
1617 fp18_copy(t0[0], a);
1618 fp18_copy(t1[0], c);
1619
1620 l0 = l1 = RLC_FP_BITS + 1;
1621 bn_rec_naf(naf0, &l0, b, RLC_WIDTH);
1622 bn_rec_naf(naf1, &l1, d, RLC_WIDTH);
1623
1624 l = RLC_MAX(l0, l1);
1625 if (bn_sign(b) == RLC_NEG) {
1626 for (i = 0; i < l0; i++) {
1627 naf0[i] = -naf0[i];
1628 }
1629 }
1630 if (bn_sign(d) == RLC_NEG) {
1631 for (i = 0; i < l1; i++) {
1632 naf1[i] = -naf1[i];
1633 }
1634 }
1635
1636 _k = naf0 + l - 1;
1637 _m = naf1 + l - 1;
1638
1639 fp18_set_dig(r, 1);
1640 for (i = l - 1; i >= 0; i--, _k--, _m--) {
1641 fp18_sqr(r, r);
1642
1643 n0 = *_k;
1644 n1 = *_m;
1645
1646 if (n0 > 0) {
1647 fp18_mul(r, r, t0[n0 / 2]);
1648 }
1649 if (n0 < 0) {
1650 fp18_inv_cyc(s, t0[-n0 / 2]);
1651 fp18_mul(r, r, s);
1652 }
1653 if (n1 > 0) {
1654 fp18_mul(r, r, t1[n1 / 2]);
1655 }
1656 if (n1 < 0) {
1657 fp18_inv_cyc(s, t1[-n1 / 2]);
1658 fp18_mul(r, r, s);
1659 }
1660 }
1661
1662 fp18_copy(e, r);
1663 }
1664 RLC_CATCH_ANY {
1665 RLC_THROW(ERR_CAUGHT);
1666 }
1667 RLC_FINALLY {
1668 fp18_free(r);
1669 fp18_free(s);
1670 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
1671 fp18_free(t0[i]);
1672 fp18_free(t1[i]);
1673 }
1674 }
