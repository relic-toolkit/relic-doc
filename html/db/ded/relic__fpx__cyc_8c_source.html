96

101  fp2_null(t[i]);

103  }
104

121  fp2_sqr(r, r);
122

124  fp2_mul(r, r, t[*k / 2]);
125  }

133  fp2_inv_cyc(c, r);

136  }
137  }
138  CATCH_ANY {
139  THROW(ERR_CAUGHT);
140  }
141  FINALLY {
142  fp2_free(r);

315 void fp12_back_cyc(fp12_t c, fp12_t a) {
316  fp2_t t0, t1, t2;
317 
318  fp2_null(t0);
319  fp2_null(t1);
320  fp2_null(t2);
321 
322  TRY {
323  fp2_new(t0);
324  fp2_new(t1);
325  fp2_new(t2);
326 
327  /* t0 = g4^2. */
328  fp2_sqr(t0, a[0][1]);
329  /* t1 = 3 * g4^2 - 2 * g3. */
330  fp2_sub(t1, t0, a[0][2]);
331  fp2_dbl(t1, t1);
332  fp2_add(t1, t1, t0);
333  /* t0 = E * g5^2 + t1. */
334  fp2_sqr(t2, a[1][2]);
335  fp2_mul_nor(t0, t2);
336  fp2_add(t0, t0, t1);
337  /* t1 = 1/(4 * g2). */
338  fp2_dbl(t1, a[1][0]);
339  fp2_dbl(t1, t1);
340  fp2_inv(t1, t1);
341  /* c_1 = g1. */
342  fp2_mul(c[1][1], t0, t1);
343 
344  /* t1 = g3 * g4. */
345  fp2_mul(t1, a[0][2], a[0][1]);
346  /* t2 = 2 * g1^2 - 3 * g3 * g4. */
347  fp2_sqr(t2, c[1][1]);
348  fp2_sub(t2, t2, t1);
349  fp2_dbl(t2, t2);
350  fp2_sub(t2, t2, t1);
351  /* t1 = g2 * g5. */
352  fp2_mul(t1, a[1][0], a[1][2]);
353  /* c_0 = E * (2 * g1^2 + g2 * g5 - 3 * g3 * g4) + 1. */
354  fp2_add(t2, t2, t1);
355  fp2_mul_nor(c[0][0], t2);
356  fp_add_dig(c[0][0][0], c[0][0][0], 1);
357 
358  fp2_copy(c[0][1], a[0][1]);
359  fp2_copy(c[0][2], a[0][2]);
360  fp2_copy(c[1][0], a[1][0]);
361  fp2_copy(c[1][2], a[1][2]);
362  }
363  CATCH_ANY {
364  THROW(ERR_CAUGHT);
365  }
366  FINALLY {
367  fp2_free(t0);
368  fp2_free(t1);
369  fp2_free(t2);
370  }

373 void fp12_back_cyc_sim(fp12_t c[], fp12_t a[], int n) {
374  fp2_t *t = RLC_ALLOCA(fp2_t, n * 3);
375  fp2_t
376  *t0 = t + 0 * n,
377  *t1 = t + 1 * n,
378  *t2 = t + 2 * n;
379 
380  if (n == 0) {
381  return;
382  }
383 
384  TRY {
385  if (t == NULL) {
386  THROW(ERR_NO_MEMORY);
387  }
388  for (int i = 0; i < n; i++) {
389  fp2_null(t0[i]);
390  fp2_null(t1[i]);
391  fp2_null(t2[i]);
392  fp2_new(t0[i]);
393  fp2_new(t1[i]);
394  fp2_new(t2[i]);
395  }
396 
397  for (int i = 0; i < n; i++) {
398  /* t0 = g4^2. */
399  fp2_sqr(t0[i], a[i][0][1]);
400  /* t1 = 3 * g4^2 - 2 * g3. */
401  fp2_sub(t1[i], t0[i], a[i][0][2]);
402  fp2_dbl(t1[i], t1[i]);
403  fp2_add(t1[i], t1[i], t0[i]);
404  /* t0 = E * g5^2 + t1. */
405  fp2_sqr(t2[i], a[i][1][2]);
406  fp2_mul_nor(t0[i], t2[i]);
407  fp2_add(t0[i], t0[i], t1[i]);
408  /* t1 = (4 * g2). */
409  fp2_dbl(t1[i], a[i][1][0]);
410  fp2_dbl(t1[i], t1[i]);
411  }
412 
413  /* t1 = 1 / t1. */
414  fp2_inv_sim(t1, t1, n);
415 
416  for (int i = 0; i < n; i++) {
417  /* t0 = g1. */
418  fp2_mul(c[i][1][1], t0[i], t1[i]);
419 
420  /* t1 = g3 * g4. */
421  fp2_mul(t1[i], a[i][0][2], a[i][0][1]);
422  /* t2 = 2 * g1^2 - 3 * g3 * g4. */
423  fp2_sqr(t2[i], c[i][1][1]);
424  fp2_sub(t2[i], t2[i], t1[i]);
425  fp2_dbl(t2[i], t2[i]);
426  fp2_sub(t2[i], t2[i], t1[i]);
427  /* t1 = g2 * g5. */
428  fp2_mul(t1[i], a[i][1][0], a[i][1][2]);
429  /* t2 = E * (2 * g1^2 + g2 * g5 - 3 * g3 * g4) + 1. */
430  fp2_add(t2[i], t2[i], t1[i]);
431  fp2_mul_nor(c[i][0][0], t2[i]);
432  fp_add_dig(c[i][0][0][0], c[i][0][0][0], 1);
433 
434  fp2_copy(c[i][0][1], a[i][0][1]);
435  fp2_copy(c[i][0][2], a[i][0][2]);
436  fp2_copy(c[i][1][0], a[i][1][0]);
437  fp2_copy(c[i][1][2], a[i][1][2]);
438  }
439  } CATCH_ANY {
440  THROW(ERR_CAUGHT);
441  } FINALLY {
442  for (int i = 0; i < n; i++) {
443  fp2_free(t0[i]);
444  fp2_free(t1[i]);
445  fp2_free(t2[i]);
446  }
447  RLC_FREE(t);
448  }
449 }

610  fp12_sqr_cyc(t[0], t[0]);
611  if (bn_get_bit(b, i)) {
612  fp12_mul(t[0], t[0], a);
