44int ed_is_infty(const ed_t p) {
45 if (p->coord == BASIC) {
46 return (fp_is_zero(p->x) && (fp_cmp_dig(p->y, 1) == RLC_EQ));
47 }
48
49 if (p->coord != BASIC) {
50 return (fp_is_zero(p->x) && (fp_cmp(p->y, p->z) == RLC_EQ));
51 }
52
53 if (fp_is_zero(p->z)) {
54 RLC_THROW(ERR_NO_VALID);
55 return 0;
56 }
57
58 return 0;
59}

104void ed_blind(ed_t r, const ed_t p) {
105 fp_t rand;
106
107 fp_null(rand);
108
109 RLC_TRY {
110 fp_new(rand);
111 fp_rand(rand);
112#if ED_ADD == BASIC
113 (void)rand;
114 ed_copy(r, p);
115#elif ED_ADD == PROJC || ED_ADD == EXTND
116 fp_mul(r->x, p->x, rand);
117 fp_mul(r->y, p->y, rand);
118 fp_mul(r->z, p->z, rand);
119 r->coord = PROJC;
120#endif
121#if ED_ADD == EXTND
122 fp_mul(r->t, p->t, rand);
123#endif
124 } RLC_CATCH_ANY {
125 RLC_THROW(ERR_CAUGHT);
126 } RLC_FINALLY {
127 fp_free(rand);

131void ed_rhs(fp_t rhs, const fp_t x) {
132 /* y^2 * (d * x^2 - 1) = 1a * x^2 - 1. */
133 fp_sqr(rhs, x);
134 fp_mul(rhs, rhs, core_get()->ed_a);
135 fp_sub_dig(rhs, rhs, 1);
136}

138int ed_on_curve(const ed_t p) {
139 ed_t t;
140 int r = 1;
141
142 ed_null(t);
143
144 if (fp_is_zero(p->z)) {
145 return 0;
146 }
147
148 RLC_TRY {
149 ed_new(t);
150 ed_norm(t, p);
151
152 /* Compute y^2 * (d * x^2 - 1) */
153#if ED_ADD == EXTND
154 fp_mul(t->z, t->x, t->y);
155 r &= (fp_cmp(t->z, t->t) == RLC_EQ);
156#endif
157 fp_sqr(t->z, t->y);
158 fp_sqr(t->t, t->x);
159 fp_mul(t->t, t->t, core_get()->ed_d);
160 fp_sub_dig(t->t, t->t, 1);
161 fp_mul(t->t, t->t, t->z);
162 ed_rhs(t->z, t->x);
163 r &= (fp_cmp(t->t, t->z) == RLC_EQ);
164 r |= ed_is_infty(p);
165 }
166 RLC_CATCH_ANY {
167 RLC_THROW(ERR_CAUGHT);
168 }
169 RLC_FINALLY {
170 ed_free(t);
171 }
172 return r;
173}

175void ed_tab(ed_t * t, const ed_t p, int w) {
176 if (w > 2) {
177 ed_dbl(t[0], p);
178#if defined(ED_MIXED)
179 ed_norm(t[0], t[0]);
180#endif
181 ed_add(t[1], t[0], p);
182 for (int i = 2; i < (1 << (w - 2)); i++) {
183 ed_add(t[i], t[i - 1], t[0]);
184 }
185#if defined(ED_MIXED)
186 ed_norm_sim(t + 1, (const ed_t *)t + 1, (1 << (w - 2)) - 1);
187#endif
188 }
189 ed_copy(t[0], p);
190}

201size_t ed_size_bin(const ed_t a, int pack) {
202 size_t size = 0;
203
204 if (ed_is_infty(a)) {
205 return 1;
206 }
207
208 size = 1 + RLC_FP_BYTES;
209 if (!pack) {

216void ed_read_bin(ed_t a, const uint8_t *bin, size_t len) {
217 if (len == 1) {
218 if (bin[0] == 0) {
219 ed_set_infty(a);
220 return;
221 } else {
222 RLC_THROW(ERR_NO_BUFFER);
223 return;
224 }
225 }
226
227 if (len != (RLC_FP_BYTES + 1) && len != (2 * RLC_FP_BYTES + 1)) {
228 RLC_THROW(ERR_NO_BUFFER);
229 return;
230 }
231
232 a->coord = BASIC;
233 fp_set_dig(a->z, 1);
234 fp_read_bin(a->y, bin + 1, RLC_FP_BYTES);
235 if (len == RLC_FP_BYTES + 1) {
236 switch (bin[0]) {
237 case 2:
238 fp_zero(a->x);
239 break;
240 case 3:
241 fp_zero(a->x);
242 fp_set_bit(a->x, 0, 1);
243 break;
244 default:
245 RLC_THROW(ERR_NO_VALID);
246 break;
247 }
248 ed_upk(a, a);
249 }
250
251 if (len == 2 * RLC_FP_BYTES + 1) {
252 if (bin[0] == 4) {
253 fp_read_bin(a->x, bin + RLC_FP_BYTES + 1, RLC_FP_BYTES);
254 } else {
255 RLC_THROW(ERR_NO_VALID);
256 return;
257 }
258 }
259#if ED_ADD == EXTND
260 fp_mul(a->t, a->x, a->y);
261 fp_mul(a->x, a->x, a->z);
262 fp_mul(a->y, a->y, a->z);
263 fp_sqr(a->z, a->z);
264#endif
265
266 if (!ed_on_curve(a)) {
267 RLC_THROW(ERR_NO_VALID);
268 }
269}

271void ed_write_bin(uint8_t *bin, size_t len, const ed_t a, int pack) {
272 ed_t t;
273
274 ed_null(t);
275
276 memset(bin, 0, len);
277
278 if (ed_is_infty(a)) {
279 if (len < 1) {
280 RLC_THROW(ERR_NO_BUFFER);
281 return;
282 } else {
283 return;
284 }
285 }
286
287 RLC_TRY {
288 ed_new(t);
289
290 ed_norm(t, a);
291
292 if (pack) {
293 if (len < RLC_FP_BYTES + 1) {
294 RLC_THROW(ERR_NO_BUFFER);
295 } else {
296 ed_pck(t, t);
297 bin[0] = 2 | fp_get_bit(t->x, 0);
298 fp_write_bin(bin + 1, RLC_FP_BYTES, t->y);
299 }
300 } else {
301 if (len < 2 * RLC_FP_BYTES + 1) {
302 RLC_THROW(ERR_NO_BUFFER);
303 } else {
304 bin[0] = 4;
305 fp_write_bin(bin + 1, RLC_FP_BYTES, t->y);
306 fp_write_bin(bin + RLC_FP_BYTES + 1, RLC_FP_BYTES, t->x);
307 }
308 }
309 } RLC_CATCH_ANY {
310 RLC_THROW(ERR_CAUGHT);
311 }
312 RLC_FINALLY {
313 ed_free(t);
314 }
