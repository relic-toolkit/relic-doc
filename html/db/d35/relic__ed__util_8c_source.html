44 int ed_is_infty(const ed_t p) {
45  fp_t t;
46  int r = 0;
47 
48  fp_null(t);
49 
50  if (p->norm) {
51  return (fp_is_zero(p->x) && (fp_cmp_dig(p->y, 1) == RLC_EQ));
52  }
53 
54  if (fp_is_zero(p->z)) {
55  THROW(ERR_NO_VALID);
56  }
57 
58  TRY {
59  fp_new(t);
60  fp_inv(t, p->z);
61  fp_mul(t, p->y, t);
62  if (fp_is_zero(p->x) && (fp_cmp_dig(t, 1) == RLC_EQ)) {
63  r = 1;
64  }
65  } CATCH_ANY {
66  THROW(ERR_CAUGHT);
67  } FINALLY {
68  fp_free(t);
69  }
70 
71  return r;
72 }

95 int ed_cmp(const ed_t p, const ed_t q) {
96  ed_t r, s;
97  int result = RLC_EQ;
98 
99  ed_null(r);
100  ed_null(s);
101 
102  TRY {
103  ed_new(r);
104  ed_new(s);
105 
106  if ((!p->norm) && (!q->norm)) {
107  /* If the two points are not normalized, it is faster to compare
108  * x1 * z2 == x2 * z1 and y1 * z2 == y2 * z1. */
109  fp_mul(r->x, p->x, q->z);
110  fp_mul(s->x, q->x, p->z);
111  fp_mul(r->y, p->y, q->z);
112  fp_mul(s->y, q->y, p->z);
113 #if ED_ADD == EXTND
114  fp_mul(r->t, p->t, q->z);
115  fp_mul(s->t, q->t, p->z);
116  if (fp_cmp(r->t, s->t) != RLC_EQ) {
117  result = RLC_NE;
118  }
119 #endif
120  } else {
121  ed_copy(r, p);
122  ed_copy(s, q);
123  if (!p->norm) {
124  ed_norm(r, p);
125  }
126  if (!q->norm) {
127  ed_norm(s, q);
128  }
129  }
130 
131  if (fp_cmp(r->x, s->x) != RLC_EQ) {
132  result = RLC_NE;
133  }
134  if (fp_cmp(r->y, s->y) != RLC_EQ) {
135  result = RLC_NE;
136  }
137  } CATCH_ANY {
138  THROW(ERR_CAUGHT);
139  } FINALLY {
140  ep_free(r);
141  ep_free(s);
142  }
143 
144  return result;
145 }

169 void ed_rhs(fp_t rhs, const ed_t p) {
170  fp_t t0, t1;
171 
172  fp_null(t0);
173  fp_null(t1);
174 
175  TRY {
176  fp_new(t0);
177  fp_new(t1);
178 
179  // 1 = a * X^2 + Y^2 - d * X^2 * Y^2
180  fp_sqr(t0, p->x);
181  fp_mul(t0, t0, core_get()->ed_a);
182  fp_sqr(t1, p->y);
183  fp_add(t1, t1, t0);
184  fp_mul(t0, p->x, p->y);
185  fp_sqr(t0, t0);
186  fp_mul(t0, t0, core_get()->ed_d);
187  fp_sub(rhs, t1, t0);
188  } CATCH_ANY {
189  THROW(ERR_CAUGHT);
190  } FINALLY {
191  fp_free(t0);
192  fp_free(t1);
193  }
194 }

196 int ed_is_valid(const ed_t p) {
197  ed_t t;
198  int r = 0;
199 
200  ed_null(t);
201 
202  if (fp_is_zero(p->z)) {
203  r = 0;
204  } else {
205  TRY {
206  ed_new(t);
207  ed_norm(t, p);
208 
209  ed_rhs(t->z, t);
210 #if ED_ADD == EXTND
211  fp_mul(t->y, t->x, t->y);
212  r = ((fp_cmp_dig(t->z, 1) == RLC_EQ) &&
213  (fp_cmp(t->y, t->t) == RLC_EQ)) || ed_is_infty(p);
214 #else
215  r = (fp_cmp_dig(t->z, 1) == RLC_EQ) || ed_is_infty(p);
216 #endif
217  }
218  CATCH_ANY {
219  THROW(ERR_CAUGHT);
220  }
221  FINALLY {
222  ed_free(t);
223  }
224  }
225  return r;
226 }

228 void ed_tab(ed_t * t, const ed_t p, int w) {
229  if (w > 2) {
230  ed_dbl(t[0], p);
231 #if defined(ED_MIXED)
232  ed_norm(t[0], t[0]);
233 #endif
234  ed_add(t[1], t[0], p);
235  for (int i = 2; i < (1 << (w - 2)); i++) {
236  ed_add(t[i], t[i - 1], t[0]);
237  }
238 #if defined(ED_MIXED)
239  ed_norm_sim(t + 1, (const ed_t *)t + 1, (1 << (w - 2)) - 1);
240 #endif
241  }
242  ed_copy(t[0], p);
243 }

269 void ed_read_bin(ed_t a, const uint8_t *bin, int len) {
270  if (len == 1) {
271  if (bin[0] == 0) {
272  ed_set_infty(a);
273  return;
274  } else {
275  THROW(ERR_NO_BUFFER);
276  return;
277  }
278  }
279 
280  if (len != (RLC_FP_BYTES + 1) && len != (2 * RLC_FP_BYTES + 1)) {
281  THROW(ERR_NO_BUFFER);
282  return;
283  }
284 
285  a->norm = 1;
286  fp_set_dig(a->z, 1);
287  fp_read_bin(a->y, bin + 1, RLC_FP_BYTES);
288  if (len == RLC_FP_BYTES + 1) {
289  switch (bin[0]) {
290  case 2:
291  fp_zero(a->x);
292  break;
293  case 3:
294  fp_zero(a->x);
295  fp_set_bit(a->x, 0, 1);
296  break;
297  default:
298  THROW(ERR_NO_VALID);
299  break;
300  }
301  ed_upk(a, a);
302  }
303 
304  if (len == 2 * RLC_FP_BYTES + 1) {
305  if (bin[0] == 4) {
306  fp_read_bin(a->x, bin + RLC_FP_BYTES + 1, RLC_FP_BYTES);
307  } else {
308  THROW(ERR_NO_VALID);
309  }
310  }
311 #if ED_ADD == EXTND
312  fp_mul(a->t, a->x, a->y);
313  fp_mul(a->x, a->x, a->z);
314  fp_mul(a->y, a->y, a->z);
315  fp_sqr(a->z, a->z);
316 #endif
317 }

319 void ed_write_bin(uint8_t *bin, int len, const ed_t a, int pack) {
320  ed_t t;
321 
322  ed_null(t);
323 
324  if (ed_is_infty(a)) {
325  if (len < 1) {
326  THROW(ERR_NO_BUFFER);
327  } else {
328  bin[0] = 0;
329  return;
330  }
331  }
332 
333  TRY {
334  ed_new(t);
335 
336  ed_norm(t, a);
337 
338  if (pack) {
339  if (len != RLC_FP_BYTES + 1) {
340  THROW(ERR_NO_BUFFER);
341  } else {
342  ed_pck(t, t);
343  bin[0] = 2 | fp_get_bit(t->x, 0);
344  fp_write_bin(bin + 1, RLC_FP_BYTES, t->y);
345  }
346  } else {
347  if (len != 2 * RLC_FP_BYTES + 1) {
348  THROW(ERR_NO_BUFFER);
349  } else {
350  bin[0] = 4;
351  fp_write_bin(bin + 1, RLC_FP_BYTES, t->y);
352  fp_write_bin(bin + RLC_FP_BYTES + 1, RLC_FP_BYTES, t->x);
353  }
354  }
355  }
356  CATCH_ANY {
357  THROW(ERR_CAUGHT);
358  }
359  FINALLY {
360  ed_free(t);
361  }
