102int core_init(void) {
103 if (core_ctx == NULL) {
104 core_ctx = &(first_ctx);
105 }
106
107#ifdef CHECK
108 core_ctx->reason[ERR_NO_MEMORY] = MSG_NO_MEMORY;
109 core_ctx->reason[ERR_NO_PRECI] = MSG_NO_PRECI;
110 core_ctx->reason[ERR_NO_FILE] = MSG_NO_FILE;
111 core_ctx->reason[ERR_NO_READ] = MSG_NO_READ;
112 core_ctx->reason[ERR_NO_VALID] = MSG_NO_VALID;
113 core_ctx->reason[ERR_NO_BUFFER] = MSG_NO_BUFFER;
114 core_ctx->reason[ERR_NO_FIELD] = MSG_NO_FIELD;
115 core_ctx->reason[ERR_NO_CURVE] = MSG_NO_CURVE;
116 core_ctx->reason[ERR_NO_CONFIG] = MSG_NO_CONFIG;
117 core_ctx->reason[ERR_NO_RAND] = MSG_NO_RAND;
118 core_ctx->last = NULL;
119#endif /* CHECK */
120
121 core_ctx->code = RLC_OK;
122
123 RLC_TRY {
124 arch_init();
125 rand_init();
126
127#if BENCH > 0
128 bench_init();
129#endif
130
131#ifdef WITH_FP
132 fp_prime_init();
133#endif
134#ifdef WITH_FB
135 fb_poly_init();
136#endif
137#ifdef WITH_EP
138 ep_curve_init();
139#endif
140#ifdef WITH_EB
141 eb_curve_init();
142#endif
143#ifdef WITH_ED
144 ed_curve_init();
145#endif
146#ifdef WITH_PP
147 pp_map_init();
148#endif
149#ifdef WITH_PC
150 pc_core_init();
151#endif
152 } RLC_CATCH_ANY {
153 return RLC_ERR;
154 }
155
156 return RLC_OK;
157}

182#if BENCH > 0
183 bench_clean();
184#endif

186 arch_clean();
187 rand_clean();
188

191 core_ctx = NULL;
192 return result;
193 }
194 return RLC_OK;
195}
