38int cp_pbpsi_gen(bn_t sk, g1_t ss, g2_t *s, size_t m) {
39 int i, result = RLC_OK;
40 bn_t q;
41
42 bn_null(q);
43
44 RLC_TRY {
45 bn_new(q);
46
47 pc_get_ord(q);
48 bn_rand_mod(sk, q);
49 g1_mul_gen(ss, sk);
50
51 g2_get_gen(s[0]);
52 for (i = 1; i <= m; i++) {
53 g2_mul(s[i], s[i - 1], sk);
54 }
55 }
56 RLC_CATCH_ANY {
57 result = RLC_ERR;
58 }
59 RLC_FINALLY {
60 bn_free(q);
61 }
62 return result;
63}

65int cp_pbpsi_ask(g2_t *d, bn_t r, const bn_t *x, const g2_t *s, size_t m) {
66 int i, result = RLC_OK;
67 bn_t t, q, *p = RLC_ALLOCA(bn_t, m + 1), *_x = RLC_ALLOCA(bn_t, m + 1);
68
69 bn_null(q);
70 bn_null(t);
71
72 RLC_TRY {
73 bn_new(q);
74 bn_new(t);
75 if (p == NULL) {
76 RLC_THROW(ERR_NO_MEMORY);
77 }
78 for (i = 0; i <= m; i++) {
79 bn_null(p[i]);
80 bn_new(p[i]);
81 bn_null(_x[i]);
82 bn_new(_x[i]);
83 }
84
85 pc_get_ord(q);
86 bn_rand_mod(r, q);
87 if (m == 0) {
88 g2_mul_gen(d[0], r);
89 } else {
90 bn_lag(p, x, q, m);
91 g2_mul_sim_lot(d[0], s, p, m + 1);
92 g2_mul(d[0], d[0], r);
93 for (i = 0; i < m; i++) {
94 bn_copy(_x[i], x[i]);
95 }
96 for (i = 0; i < m; i++) {
97 bn_copy(t, _x[i]);
98 bn_copy(_x[i], _x[m - 1]);
99 bn_lag(p, _x, q, m - 1);
100 g2_mul_sim_lot(d[i + 1], s, p, m);
101 g2_mul(d[i + 1], d[i + 1], r);
102 bn_copy(_x[i], t);
103 }
104 }
105 }
106 RLC_CATCH_ANY {
107 result = RLC_ERR;
108 }
109 RLC_FINALLY {
110 bn_free(q);
111 bn_free(t);
112 for (i = 0; i <= m; i++) {
113 bn_free(p[i]);
114 bn_free(_x[i]);
115 }
116 RLC_FREE(p);
117 RLC_FREE(_x);
118 }
119 return result;
120}

122int cp_pbpsi_ans(gt_t *t, g1_t *u, const g1_t ss, const g2_t d,
123 const bn_t *y, size_t n) {
124 int j, result = RLC_OK;
125 bn_t q, tj;
126 g1_t g1;
127 g2_t g2;
128 uint_t *shuffle = RLC_ALLOCA(uint_t, n);
129
130 bn_null(q);
131 bn_null(tj);
132 g1_null(g1);
133 g2_null(g2);
134
135 RLC_TRY {
136 bn_new(q);
137 bn_new(tj);
138 g1_new(g1);
139 g2_new(g2);
140 if (shuffle == NULL) {
141 RLC_THROW(ERR_NO_MEMORY);
142 }
143
144 util_perm(shuffle, n);
145
146 pc_get_ord(q);
147 g2_get_gen(g2);
148 for (j = 0; j < n; j++) {
149 bn_rand_mod(tj, q);
150 g1_mul_gen(g1, tj);
151 pc_map(t[j], g1, d);
152 g1_mul_gen(u[j], y[shuffle[j]]);
153 g1_sub(u[j], ss, u[j]);
154 g1_mul(u[j], u[j], tj);
155 }
156 }
157 RLC_CATCH_ANY {
158 result = RLC_ERR;
159 }
160 RLC_FINALLY {
161 bn_free(q);
162 bn_free(tj);
163 g1_free(g1);
164 g2_free(g2);
165 RLC_FREE(shuffle);
166 }
167 return result;
168}

170int cp_pbpsi_int(bn_t *z, size_t *len, const g2_t *d, const bn_t *x,
171 size_t m, const gt_t *t, const g1_t *u, size_t n) {
172 int j, k, result = RLC_OK;
173 gt_t e;
174
175 gt_null(e);
176
177 RLC_TRY {
178 gt_new(e);
179
180 *len = 0;
181 if (m > 0) {
182 for (k = 0; k < m; k++) {
183 for (j = 0; j < n; j++) {
184 pc_map(e, u[j], d[k + 1]);
185 if (gt_cmp(e, t[j]) == RLC_EQ && !gt_is_unity(e)) {
186 bn_copy(z[*len], x[k]);
187 (*len)++;
188 }
189 }
190 }
191 }
192 }
193 RLC_CATCH_ANY {
194 result = RLC_ERR;
195 }
196 RLC_FINALLY {
197 gt_free(e);
198 }
199 return result;
