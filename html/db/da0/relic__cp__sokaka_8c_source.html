39int cp_sokaka_gen(bn_t master) {

73int cp_sokaka_key(uint8_t *key, size_t key_len, const char *id1,
74 const sokaka_t k, const char *id2) {
75 size_t size, len1 = strlen(id1), len2 = strlen(id2);
76 int first = 0, result = RLC_OK;
77 uint8_t *buf;
78 g1_t p;
79 g2_t q;
80 gt_t e;
81
82 g1_null(p);
83 g2_null(q);
84 gt_null(e);
85
86 RLC_TRY {
87 g1_new(p);
88 g2_new(q);
89 gt_new(e);
90 size = gt_size_bin(e, 0);
91 buf = RLC_ALLOCA(uint8_t, size);
92 if (buf == NULL) {
93 RLC_THROW(ERR_NO_MEMORY);
94 }
95
96 if (len1 == len2) {
97 if (strncmp(id1, id2, len1) == 0) {
98 RLC_THROW(ERR_NO_VALID);
99 }
100 first = (strncmp(id1, id2, len1) < 0 ? 1 : 2);
101 } else {
102 if (len1 < len2) {
103 if (strncmp(id1, id2, len1) == 0) {
104 first = 1;
105 } else {
106 first = (strncmp(id1, id2, len1) < 0 ? 1 : 2);
107 }
108 } else {
109 if (strncmp(id1, id2, len2) == 0) {
110 first = 2;
111 } else {
112 first = (strncmp(id1, id2, len2) < 0 ? 1 : 2);
113 }
114 }
115 }
116
117 if (pc_map_is_type1()) {
118 g2_map(q, (uint8_t *)id2, len2);
119 pc_map(e, k->s1, q);
120 } else {
121 if (first == 1) {
122 g2_map(q, (uint8_t *)id2, len2);
123 pc_map(e, k->s1, q);
124 } else {
125 g1_map(p, (uint8_t *)id2, len2);
126 pc_map(e, p, k->s2);
127 }
128 }
129
130 /* Allocate size for storing the output. */
131 gt_write_bin(buf, size, e, 0);
132 md_kdf(key, key_len, buf, size);
133 }
134 RLC_CATCH_ANY {
135 result = RLC_ERR;
136 }
137 RLC_FINALLY {
138 g1_free(p);
139 g2_free(q);
140 gt_free(e);
141 RLC_FREE(buf);
142 }
143 return result;
144}
