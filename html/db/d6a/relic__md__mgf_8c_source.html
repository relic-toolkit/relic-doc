43 void md_mgf(uint8_t *key, int key_len, const uint8_t *in,
44  int in_len) {
45  uint32_t i, j, d;
46  uint8_t *buffer = RLC_ALLOCA(uint8_t, in_len + sizeof(uint32_t));
47  uint8_t *t = RLC_ALLOCA(uint8_t, key_len + RLC_MD_LEN);
48 
49  if (buffer == NULL || t == NULL) {
50  RLC_FREE(buffer);
51  RLC_FREE(t);
52  THROW(ERR_NO_MEMORY);
53  }
54 
55  /* d = ceil(kLen/hLen). */
56  d = RLC_CEIL(key_len, RLC_MD_LEN);
57  memcpy(buffer, in, in_len);
58  for (i = 0; i < d; i++) {
59  j = util_conv_big(i);
60  /* c = integer_to_string(c, 4). */
61  memcpy(buffer + in_len, &j, sizeof(uint32_t));
62  /* t = t || hash(z || c). */
63  md_map(t + i * RLC_MD_LEN, buffer, in_len + sizeof(uint32_t));
64  }
65  memcpy(key, t, key_len);
66 
67  RLC_FREE(buffer);
68  RLC_FREE(t);
69 }
