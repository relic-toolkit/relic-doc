71void ed_param_set(int param) {
72 ctx_t *ctx = core_get();
73 char str[2 * RLC_FP_BYTES + 2];
74
75 ed_t g;
76 bn_t r;
77 bn_t h;
78
79 ed_null(g);
80 bn_null(r);
81 bn_null(h);
82
83 RLC_TRY {
84 ed_new(g);
85 bn_new(r);
86 bn_new(h);
87
88 ctx->ed_id = 0;
89
90 switch (param) {
91#if FP_PRIME == 255
92 case CURVE_ED25519:
93 ASSIGN_ED(CURVE_ED25519, PRIME_25519);
94 break;
95#endif
96 default:
97 (void)str;
98 RLC_THROW(ERR_NO_VALID);
99 break;
100 }
101 fp_set_dig(g->z, 1);
102 fp_neg(g->z, g->z);
103 fp_neg(g->z, g->z);
104#if ED_ADD == EXTND
105 fp_mul(g->t, g->x, g->y);
106#endif
107 g->coord = BASIC;
108
109 bn_copy(&ctx->ed_h, h);
110 bn_copy(&ctx->ed_r, r);
111 ed_copy(&ctx->ed_g, g);
112
113#ifdef ED_PRECO
114 for (int i = 0; i < RLC_ED_TABLE; i++) {
115 ctx->ed_ptr[i] = &(ctx->ed_pre[i]);
116 }
117 ed_mul_pre((ed_t *)ed_curve_get_tab(), &ctx->ed_g);
118#endif
119 ctx->ed_id = param;
120
121 /* compute constants for p == 5 mod 8 */
122 if (fp_prime_get_2ad() == 2) {
123 h->used = RLC_FP_DIGS;
124 h->sign = RLC_POS;
125 dv_copy(h->dp, fp_prime_get(), RLC_FP_DIGS); /* p */
126 bn_add_dig(h, h, 3); /* p + 3 */
127 bn_rsh(h, h, 3); /* (p + 3) / 8 */
128 /* compute 2^((p+3)/8) */
129 fp_set_dig(ctx->ed_map_c[0], 2);
130 fp_exp(ctx->ed_map_c[0], ctx->ed_map_c[0], h);
131
132 /* compute sqrt(-1) */
133 fp_set_dig(ctx->ed_map_c[1], 1);
134 fp_neg(ctx->ed_map_c[1], ctx->ed_map_c[1]);
135 if (!fp_srt(ctx->ed_map_c[1], ctx->ed_map_c[1])) {
136 /* should never happen because 2adicity > 1 */
137 RLC_THROW(ERR_NO_VALID);
138 }
139
140 if (param == CURVE_ED25519) {
141 /* compute sqrt(-486664) for mapping curve25519 -> edwards25519 */
142 fp_read_str(ctx->ed_map_c[3], "486662", 6, 10);
143 fp_add_dig(ctx->ed_map_c[2], ctx->ed_map_c[3], 2);
144 fp_neg(ctx->ed_map_c[2], ctx->ed_map_c[2]);
145 if (!fp_srt(ctx->ed_map_c[2], ctx->ed_map_c[2])) {
146 RLC_THROW(ERR_NO_VALID);
147 }
148 fp_prime_back(h, ctx->ed_map_c[2]);
149 /* make sure sgn0(ctx->ed_map_c[2]) == 0 */
150 if (bn_get_bit(h, 0) != 0) {
151 fp_neg(ctx->ed_map_c[2], ctx->ed_map_c[2]);
152 }
153 } else {
154 /* don't know how to compute remaining constants for other curves */
155 RLC_THROW(ERR_NO_VALID);
156 }
157 } else {
158 /* map impl only supports p = 5 mod 8! */
159 RLC_THROW(ERR_NO_VALID);
160 }
161 }
162 RLC_CATCH_ANY {
163 RLC_THROW(ERR_CAUGHT);
164 }
165 RLC_FINALLY {
166 bn_free(r);
167 bn_free(h);
168 ed_free(g);
169 }
170}
