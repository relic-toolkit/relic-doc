47static void pp_exp_bn(fp12_t c, fp12_t a) {
48 fp12_t t0, t1, t2, t3;
49 bn_t x;
50 const int *b;
51 int l;
52
53 fp12_null(t0);
54 fp12_null(t1);
55 fp12_null(t2);
56 fp12_null(t3);
57 bn_null(x);
58
59 RLC_TRY {
60 fp12_new(t0);
61 fp12_new(t1);
62 fp12_new(t2);
63 fp12_new(t3);
64 bn_new(x);
65
66 /*
67 * New final exponentiation following Fuentes-Castañeda, Knapp and
68 * Rodríguez-Henríquez: Fast Hashing to G_2.
69 */
70 fp_prime_get_par(x);
71 b = fp_prime_get_par_sps(&l);
72
73 /* First, compute m = f^(p^6 - 1)(p^2 + 1). */
74 fp12_conv_cyc(c, a);
75
76 /* Now compute m^((p^4 - p^2 + 1) / r). */
77 /* t0 = m^2x. */
78 fp12_exp_cyc_sps(t0, c, b, l, RLC_POS);
79 fp12_sqr_cyc(t0, t0);
80 /* t1 = m^6x. */
81 fp12_sqr_cyc(t1, t0);
82 fp12_mul(t1, t1, t0);
83 /* t2 = m^6x^2. */
84 fp12_exp_cyc_sps(t2, t1, b, l, RLC_POS);
85 /* t3 = m^12x^3. */
86 fp12_sqr_cyc(t3, t2);
87 fp12_exp_cyc_sps(t3, t3, b, l, RLC_POS);
88
89 if (bn_sign(x) == RLC_NEG) {
90 fp12_inv_cyc(t0, t0);
91 fp12_inv_cyc(t1, t1);
92 fp12_inv_cyc(t3, t3);
93 }
94
95 /* t3 = a = m^12x^3 * m^6x^2 * m^6x. */
96 fp12_mul(t3, t3, t2);
97 fp12_mul(t3, t3, t1);
98
99 /* t0 = b = 1/(m^2x) * t3. */
100 fp12_inv_cyc(t0, t0);
101 fp12_mul(t0, t0, t3);
102
103 /* Compute t2 * t3 * m * b^p * a^p^2 * [b * 1/m]^p^3. */
104 fp12_mul(t2, t2, t3);
105 fp12_mul(t2, t2, c);
106 fp12_inv_cyc(c, c);
107 fp12_mul(c, c, t0);
108 fp12_frb(c, c, 3);
109 fp12_mul(c, c, t2);
110 fp12_frb(t0, t0, 1);
111 fp12_mul(c, c, t0);
112 fp12_frb(t3, t3, 2);
113 fp12_mul(c, c, t3);
114 }
115 RLC_CATCH_ANY {
116 RLC_THROW(ERR_CAUGHT);
117 }
118 RLC_FINALLY {
119 fp12_free(t0);
120 fp12_free(t1);
121 fp12_free(t2);
122 fp12_free(t3);
123 bn_free(x);
124 }
125}

134static void pp_exp_sm9(fp12_t c, fp12_t a) {
135 fp12_t t0, t1, t2, t3, t4, t5, t6, r0, r1;
136 bn_t x;
137 const int *b;
138 int l;
139
140 fp12_null(t0);
141 fp12_null(t1);
142 fp12_null(t2);
143 fp12_null(t3);
144 fp12_null(t4);
145 fp12_null(t5);
146 fp12_null(t6);
147 fp12_null(r0);
148 fp12_null(r1);
149 bn_null(x);

151 RLC_TRY {
152 fp12_new(t0);
153 fp12_new(t1);
154 fp12_new(t2);
155 fp12_new(t3);
156 fp12_new(t4);
157 fp12_new(t5);
158 fp12_new(t6);
159 fp12_new(r0);
160 fp12_new(r1);
161 bn_new(x);
162
163 fp_prime_get_par(x);
164 b = fp_prime_get_par_sps(&l);
165
166 /* First, compute m = f^(p^6 - 1)(p^2 + 1). */
167 fp12_conv_cyc(c, a);
168
169 /* Now compute m^((p^4 - p^2 + 1) / r) using addition chain method. */
170 fp12_frb(r0, c, 1);
171 fp12_frb(r1, c, 2);
172 fp12_frb(t0, c, 3);
173 fp12_mul(t0, t0, r0);
174 fp12_mul(t0, t0, r1);
175
176 fp12_copy(t1, c);
177
178 fp12_exp_cyc_sps(t4, c, b, l, RLC_POS);
179 fp12_exp_cyc_sps(t5, t4, b, l, RLC_POS);
180 fp12_exp_cyc_sps(t6, t5, b, l, RLC_POS);
181
182 fp12_frb(t3, t4, 1);
183 fp12_frb(t2, t5, 2);
184 fp12_frb(r0, t6, 1);
185 fp12_mul(t6, t6, r0);
186 fp12_frb(r0, t5, 1);
187 fp12_mul(t4, t4, r0);
188
189 fp12_inv_cyc(t1, t1);
190 fp12_inv_cyc(t3, t3);
191 fp12_inv_cyc(t4, t4);
192 fp12_inv_cyc(t5, t5);
193 fp12_inv_cyc(t6, t6);
194
195 fp12_sqr_cyc(r0, t6);
196 fp12_mul(r0, r0, t4);
197 fp12_mul(r0, r0, t5);
198 fp12_mul(r1, t5, t3);
199 fp12_mul(r1, r0, r1);
200 fp12_mul(r0, r0, t2);
201 fp12_sqr_cyc(r1, r1);
202 fp12_mul(r1, r1, r0);
203 fp12_sqr_cyc(r1, r1);
204 fp12_mul(r0, r1, t0);
205 fp12_mul(r1, r1, t1);
206 fp12_sqr_cyc(r1, r1);
207 fp12_mul(c, r0, r1);
208 }

211 }
212 RLC_FINALLY {

233static void pp_exp_b12(fp12_t c, fp12_t a) {
234 fp12_t t0, t1, t2, t3;
235 bn_t x;
236 const int *b;
237 int l;
238
239 fp12_null(t0);
240 fp12_null(t1);
241 fp12_null(t2);
242 fp12_null(t3);
243 bn_null(x);
244
245 RLC_TRY {
246 fp12_new(t0);
247 fp12_new(t1);
248 fp12_new(t2);
249 fp12_new(t3);
250 bn_new(x);
251
252 /*
253 * Final exponentiation following Hayashida, Hayasaka and Teruya:
254 * Efficient Final Exponentiation via Cyclotomic Structure for Pairings
255 * over Families of Elliptic Curves
256 */
257 fp_prime_get_par(x);
258 b = fp_prime_get_par_sps(&l);
259
260 /* First, compute m^(p^6 - 1)(p^2 + 1). */
261 fp12_conv_cyc(c, a);
262
263 /* Now compute m^((p^4 - p^2 + 1) / r). */
264 /* t0 = f^2. */
265 fp12_sqr_cyc(t0, c);
266
267 /* t3 = 1/f. */
268 fp12_inv_cyc(t3, c);
269
270 if (b[0] == 0) {
271 /* t1 = f^x. */
272 fp12_exp_cyc_sps(t1, c, b, l, bn_sign(x));
273
274 /* t2 = f^(x^2). */
275 fp12_exp_cyc_sps(t2, t1, b, l, bn_sign(x));
276
277 /* t1 = t2/(t1^2 * f). */
278 fp12_sqr_cyc(t1, t1);
279 } else {
280 /*
281 * Variant from Ghammam and Fouotsa saves computing (x-1)^2 power:
282 * Improving the computation of the optimal ate pairing for a high
283 * security level
284 */
285 int _b[RLC_TERMS + 1];
286
287 for (int i = 0; i < l; i++) {
288 if (b[i] > 0) {
289 _b[i] = b[i] - 1;
290 } else {
291 _b[i] = b[i] + 1;
292 }
293 }
294
295 /* t1 = f^x. */
296 fp12_exp_cyc_sps(t1, t0, b, l, bn_sign(x));
297
298 /* t2 = f^(x^2). */
299 fp12_exp_cyc_sps(t2, t1, _b, l, bn_sign(x));
300 }
301
302 /* t1 = f^(u^2 - 2u + 1). */
303 fp12_mul(t1, t1, t3);
304 fp12_inv_cyc(t1, t1);
305 fp12_mul(t1, t1, t2);
306
307 /* t2 = t1^(x + p). */
308 fp12_exp_cyc_sps(t2, t1, b, l, bn_sign(x));
309 fp12_frb(t3, t1, 1);
310 fp12_mul(t2, t2, t3);
311
312 /* t3 = t2^(x^2 + p^2 - 1) */
313 fp12_exp_cyc_sps(t3, t2, b, l, bn_sign(x));
314 fp12_exp_cyc_sps(t3, t3, b, l, bn_sign(x));
315 fp12_frb(t1, t2, 2);
316 fp12_inv_cyc(t2, t2);
317 fp12_mul(t1, t1, t2);
318 fp12_mul(t3, t3, t1);
319
320 /* c = t3 * f^3. */
321 fp12_mul(t0, t0, c);
322 fp12_mul(c, t0, t3);
323 }
324 RLC_CATCH_ANY {
325 RLC_THROW(ERR_CAUGHT);
326 }
327 RLC_FINALLY {
328 fp12_free(t0);
329 fp12_free(t1);
330 fp12_free(t2);
331 fp12_free(t3);
332 bn_free(x);
333 }
334}

340void pp_exp_k12(fp12_t c, fp12_t a) {
341 switch (ep_curve_is_pairf()) {
342 case EP_BN:
343 if (ep_param_get() == SM9_P256) {
344 pp_exp_sm9(c, a);
345 } else {
346 pp_exp_bn(c, a);
347 }
348 break;
349 case EP_B12:
350 pp_exp_b12(c, a);
351 break;
352 }
