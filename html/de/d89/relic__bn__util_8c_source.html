182 void bn_rand(bn_t a, int sign, int bits) {
183  int digits;
184 
185  RLC_RIP(bits, digits, bits);
186  digits += (bits > 0 ? 1 : 0);
187 
188  bn_grow(a, digits);
189 
190  rand_bytes((uint8_t *)a->dp, digits * sizeof(dig_t));
191 
192  a->used = digits;
193  a->sign = sign;
194  if (bits > 0) {
195  dig_t mask = ((dig_t)1 << (dig_t)bits) - 1;
196  a->dp[a->used - 1] &= mask;
197  }
198  bn_trim(a);

232 int bn_size_str(const bn_t a, int radix) {
233  int digits = 0;
234  bn_t t;
235 
236  bn_null(t);
237 
238  /* Binary case requires the bits, a sign and the null terminator. */
239  if (radix == 2) {
240  return bn_bits(a) + (a->sign == RLC_NEG ? 1 : 0) + 1;
241  }
242 
243  /* Check the radix. */
244  if (radix < 2 || radix > 64) {
245  THROW(ERR_NO_VALID);
246  }
247 
248  if (bn_is_zero(a)) {
249  return 2;
250  }
251 
252  if (a->sign == RLC_NEG) {
253  digits++;
254  }
255 
256  TRY {
257  bn_new(t);
258  bn_copy(t, a);
259 
260  t->sign = RLC_POS;
261 
262  while (!bn_is_zero(t)) {
263  bn_div_dig(t, t, (dig_t)radix);
264  digits++;
265  }
266  } CATCH_ANY {
267  THROW(ERR_CAUGHT);
268  } FINALLY {
269  bn_free(t);
270  }
271 
272  return digits + 1;
273 }

275 void bn_read_str(bn_t a, const char *str, int len, int radix) {
276  int sign, i, j;
277  char c;
278 
279  bn_zero(a);
280 
281  if (radix < 2 || radix > 64) {
282  THROW(ERR_NO_VALID)
283  }
284 
285  j = 0;
286  if (str[0] == '-') {
287  j++;
288  sign = RLC_NEG;
289  } else {
290  sign = RLC_POS;
291  }
292 
293  while (j < len) {
294  if (str[j] == 0) {
295  break;
296  }
297  c = (char)((radix < 36) ? RLC_UPP(str[j]) : str[j]);
298  for (i = 0; i < 64; i++) {
299  if (c == util_conv_char(i)) {
300  break;
301  }
302  }
303 
304  if (i < radix) {
305  bn_mul_dig(a, a, (dig_t)radix);
306  bn_add_dig(a, a, (dig_t)i);
307  } else {
308  break;
309  }
310  j++;
311  }
312 
313  a->sign = sign;
314 }

316 void bn_write_str(char *str, int len, const bn_t a, int radix) {
317  bn_t t;
318  dig_t d;
319  int digits, l, i, j;
320  char c;
321 
322  bn_null(t);
323 
324  l = bn_size_str(a, radix);
325  if (len < l) {
326  THROW(ERR_NO_BUFFER);
327  }
328 
329  if (radix < 2 || radix > 64) {
330  THROW(ERR_NO_VALID)
331  }
332 
333  if (bn_is_zero(a) == 1) {
334  *str++ = '0';
335  *str = '\0';
336  return;
337  }
338 
339  TRY {
340  bn_new(t);
341  bn_copy(t, a);
342 
343  j = 0;
344  if (t->sign == RLC_NEG) {
345  str[j] = '-';
346  j++;
347  t->sign = RLC_POS;
348  }
349 
350  digits = 0;
351  while (!bn_is_zero(t)) {
352  bn_div_rem_dig(t, &d, t, (dig_t)radix);
353  str[j] = util_conv_char(d);
354  digits++;
355  j++;
356  }
357 
358  /* Reverse the digits of the string. */
359  i = 0;
360  if (str[0] == '-') {
361  i = 1;
362  }
363 
364  j = l - 2;
365  while (i < j) {
366  c = str[i];
367  str[i] = str[j];
368  str[j] = c;
369  ++i;
370  --j;
371  }
372 
373  str[l - 1] = '\0';
374  }
375  CATCH_ANY {
376  THROW(ERR_CAUGHT);
377  }
378  FINALLY {
379  bn_free(t);
380  }
381 }

397 void bn_read_bin(bn_t a, const uint8_t *bin, int len) {
398  int i, j;
399  dig_t d = (RLC_DIG / 8);
400  int digs = (len % d == 0 ? len / d : len / d + 1);
401 
402  bn_grow(a, digs);
403  bn_zero(a);
404  a->used = digs;
405 
406  for (i = 0; i < digs - 1; i++) {
407  d = 0;
408  for (j = (RLC_DIG / 8) - 1; j >= 0; j--) {
409  d = d << 8;
410  d |= bin[len - 1 - (i * (RLC_DIG / 8) + j)];
411  }
412  a->dp[i] = d;
413  }
414  d = 0;
415  for (j = (RLC_DIG / 8) - 1; j >= 0; j--) {
416  if ((int)(i * (RLC_DIG / 8) + j) < len) {
417  d = d << 8;
418  d |= bin[len - 1 - (i * (RLC_DIG / 8) + j)];
419  }
420  }
421  a->dp[i] = d;
422 
423  a->sign = RLC_POS;
424  bn_trim(a);
425 }
