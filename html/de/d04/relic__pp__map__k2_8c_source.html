51 static void pp_mil_k2(fp2_t r, ep_t *t, ep_t *p, ep_t *q, int m, bn_t a) {
52  fp2_t l;
53  ep_t *_q = RLC_ALLOCA(ep_t, m);
54  int i, j;
55 
56  fp2_null(l);
57 
58  TRY {
59  if (_q == NULL) {
60  THROW(ERR_NO_MEMORY);
61  }
62  fp2_new(l);
63  for (j = 0; j < m; j++) {
64  ep_null(_q[j]);
65  ep_new(_q[j]);
66  ep_copy(t[j], p[j]);
67  ep_neg(_q[j], q[j]);
68  }
69 
70  fp2_zero(l);
71  for (i = bn_bits(a) - 2; i >= 0; i--) {
72  fp2_sqr(r, r);
73  for (j = 0; j < m; j++) {
74  pp_dbl_k2(l, t[j], t[j], _q[j]);
75  fp2_mul(r, r, l);
76  if (bn_get_bit(a, i)) {
77  pp_add_k2(l, t[j], p[j], q[j]);
78  fp2_mul(r, r, l);
79  }
80  }
81  }
82  }
83  CATCH_ANY {
84  THROW(ERR_CAUGHT);
85  }
86  FINALLY {
87  fp2_free(l);
88  for (j = 0; j < m; j++) {
89  ep_free(_q[j]);
90  }
91  RLC_FREE(_q);
92  }
93 }

161 void pp_map_tatep_k2(fp2_t r, ep_t p, ep_t q) {
162  ep_t _p[1], _q[1], t[1];
163  bn_t n;
164 
165  ep_null(_p[0]);
166  ep_null(_q[0]);
167  ep_null(t[0]);
168  bn_null(n);
169 
170  TRY {
171  ep_new(t[0]);
172  bn_new(n);
173 
174  ep_norm(_p[0], p);
175  ep_norm(_q[0], q);
176  ep_curve_get_ord(n);
177  /* Since p has order n, we do not have to perform last iteration. */
178  bn_sub_dig(n, n, 1);
179  fp2_set_dig(r, 1);
180 
181  if (!ep_is_infty(p) && !ep_is_infty(q)) {
182  pp_mil_k2(r, t, _p, _q, 1, n);
183  pp_exp_k2(r, r);
184  }
185  }
186  CATCH_ANY {
187  THROW(ERR_CAUGHT);
188  }
189  FINALLY {
190  ep_free(_p[0]);
191  ep_free(_q[0]);
192  ep_free(t[0]);
193  bn_free(n);
194  }
195 }
