221static void pp_exp_afg16(fp16_t c, fp16_t a) {
222 fp16_t t0, t1, t2, t3, t4;
223 bn_t x, x_;
224
225 bn_null(x);
226 bn_null(x_);
227 fp16_null(t0);
228 fp16_null(t1);
229 fp16_null(t2);
230 fp16_null(t3);
231 fp16_null(t4);
232
233 RLC_TRY {
234 bn_new(x);
235 bn_new(x_);
236 fp16_new(t0);
237 fp16_new(t1);
238 fp16_new(t2);
239 fp16_new(t3);
240 fp16_new(t4);
241
242 fp_prime_get_par(x);
243
244 /* First, compute m = f^(p^8 - 1). */
245 fp16_conv_cyc(c, a);
246
247 /* Now compute m^((p^8 + 1) / r). */
248 bn_abs(x_, x);
249
250 /* Compute eq t0 = m^(u*p * (1+u*p^3)). */
251 fp16_exp_cyc(t0, c, x_);
252 fp16_frb(t0, t0, 3);
253 fp16_inv_cyc(t1, c);
254 if (bn_sign(x) == RLC_POS) {
255 fp16_mul(t0, t0, c);
256 } else {
257 fp16_mul(t0, t0, t1);
258 }
259 fp16_exp_cyc(t0, t0, x_);
260 fp16_frb(t0, t0, 1);
261
262 /* Compute t2 = m^(p^5 + u*(-1 + u^2 * u*p*(1+u*p^3))). */
263 fp16_exp_cyc(t2, t0, x_);
264 fp16_exp_cyc(t2, t2, x_);
265 fp16_mul(t2, t2, t1);
266 fp16_exp_cyc(t2, t2, x_);
267 if (bn_sign(x) == RLC_NEG) {
268 fp16_inv_cyc(t2, t2);
269 }
270 fp16_frb(t3, c, 5);
271 fp16_mul(t2, t2, t3);
272
273 /* Compute t2 = m2^(((u^2 div 4)*(u^3+1)^2 +1) */
274 if (bn_is_even(x_)) {
275 bn_hlv(x_, x_);
276 fp16_exp_cyc(t3, t2, x_);
277 fp16_exp_cyc(t3, t3, x_);
278 bn_dbl(x_, x_);
279 if (bn_sign(x) == RLC_NEG) {
280 fp16_inv_cyc(t4, t3);
281 } else {
282 fp16_copy(t4, t3);
283 }
284 fp16_exp_cyc(t3, t3, x_);
285 fp16_exp_cyc(t3, t3, x_);
286 fp16_exp_cyc(t3, t3, x_);
287 fp16_mul(t3, t3, t4);
288 if (bn_sign(x) == RLC_NEG) {
289 fp16_inv_cyc(t4, t3);
290 } else {
291 fp16_copy(t4, t3);
292 }
293 fp16_exp_cyc(t3, t3, x_);
294 fp16_exp_cyc(t3, t3, x_);
295 fp16_exp_cyc(t3, t3, x_);
296 fp16_mul(t3, t3, t4);
297 } else {
298 fp16_exp_cyc(t3, t2, x_);
299 fp16_exp_cyc(t3, t3, x_);
300 bn_sqr(x, x_);
301 bn_mul(x, x, x_);
302 bn_add_dig(x, x, 1);
303 bn_hlv(x, x);
304 bn_abs(x, x);
305 fp16_exp_cyc(t3, t3, x);
306 fp16_exp_cyc(t3, t3, x);
307 }
308 fp16_mul(t2, t2, t3);
309
310 /* Compute t2 = (t0 * m2)^((p^2-u^2). */
311 fp16_mul(t2, t2, t0);
312 fp16_frb(t4, t2, 2);
313 fp16_exp_cyc(t2, t2, x_);
314 fp16_exp_cyc(t2, t2, x_);
315 fp16_inv_cyc(t2, t2);
316 fp16_mul(t2, t4, t2);
317
318 fp16_mul(c, c, t2);
319 if (!bn_is_even(x_)) {
320 fp16_sqr_cyc(c, c);
321 }
322 }
323 RLC_CATCH_ANY {
324 RLC_THROW(ERR_CAUGHT);
325 }
326 RLC_FINALLY {
327 bn_free(x);
328 bn_free(x_);
329 fp16_free(t0);
330 fp16_free(t1);
331 fp16_free(t2);
332 fp16_free(t3);
333 fp16_free(t4);
334 }
335}
