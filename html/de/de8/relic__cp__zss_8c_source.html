41
42 bn_null(n);

64int cp_zss_sig(g2_t s, const uint8_t *msg, size_t len, int hash, const bn_t d) {
65 bn_t m, n, r, t;
66 uint8_t h[RLC_MD_LEN];
67 int result = RLC_OK;
68
69 bn_null(m);
70 bn_null(n);
71 bn_null(r);
72 bn_null(t);
73
74 RLC_TRY {
75 bn_new(m);
76 bn_new(n);
77 bn_new(r);
78 bn_new(t);
79
80 pc_get_ord(n);
81
82 /* m = H(msg). */
83 if (hash) {
84 bn_read_bin(m, msg, len);
85 } else {
86 md_map(h, msg, len);
87 bn_read_bin(m, h, RLC_MD_LEN);
88 }
89 bn_mod(m, m, n);
90
91 /* Compute (H(m) + d) and invert. */
92 bn_add(t, m, d);
93 bn_mod(t, t, n);
94 bn_mod_inv(t, t, n);
95
96 /* Compute the sinature. */
97 g2_mul_gen(s, t);
98 }
99 RLC_CATCH_ANY {
100 result = RLC_ERR;
101 }
102 RLC_FINALLY {
103 bn_free(m);
104 bn_free(n);
105 bn_free(r);
106 bn_free(t);
107 }
108 return result;
109}

111int cp_zss_ver(const g2_t s, const uint8_t *msg, size_t len, int hash,
112 const g1_t q, const gt_t z) {
113 bn_t m, n;

118

123
124 RLC_TRY {
125 bn_new(m);
126 bn_new(n);
127 g1_new(g);
128 gt_new(e);
129
130 pc_get_ord(n);
131
132 /* m = H(msg). */
133 if (hash) {
134 bn_read_bin(m, msg, len);
135 } else {
136 md_map(h, msg, len);
137 bn_read_bin(m, h, RLC_MD_LEN);
138 }
139 bn_mod(m, m, n);
140
141 g1_mul_gen(g, m);
142 g1_add(g, g, q);
143 g1_norm(g, g);
144
145 pc_map(e, g, s);
146
147 if (gt_cmp(e, z) == RLC_EQ) {
148 result = 1;
149 }
150 }
151 RLC_CATCH_ANY {
152 RLC_THROW(ERR_CAUGHT);
153 }
154 RLC_FINALLY {

156 bn_free(n);
157 g1_free(g);
