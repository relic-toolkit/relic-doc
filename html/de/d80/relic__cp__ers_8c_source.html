66 const ec_t pk, const ec_t pp) {

94 result = RLC_ERR;
95 }
96 RLC_FINALLY {
97 bn_free(n);
98 ec_free(t);
99 ec_free(y[0]);
100 ec_free(y[1]);
101 }

105int cp_ers_ver(const bn_t td, const ers_t *s, size_t size, const uint8_t *msg,
106 size_t len, const ec_t pp) {
107 bn_t n;
108 ec_t t, y[2];
109 int flag = 0, result = 0;
110
111 bn_null(n);
112 ec_null(t);
113 ec_null(y[0]);
114 ec_null(y[1]);
115
116 RLC_TRY {
117 bn_new(n);
118 ec_new(t);
119 ec_new(y[0]);
120 ec_new(y[1]);
121
122 ec_curve_get_ord(n);
123 ec_mul_gen(t, td);
124
125 for (int i = 0; i < size; i++) {
126 ec_add(t, t, s[i]->h);
127 }
128 if (ec_cmp(pp, t) == RLC_EQ) {
129 flag = 1;
130 for (int i = 0; i < size; i++) {
131 ec_copy(y[0], s[i]->h);
132 ec_copy(y[1], s[i]->pk);
133 flag &= cp_sokor_ver(s[i]->c, s[i]->r, msg, len, y, NULL);
134 }
135 }
136 result = flag;
137 }
138 RLC_CATCH_ANY {
139 RLC_THROW(ERR_CAUGHT);
140 }
141 RLC_FINALLY {
142 bn_free(n);
143 ec_free(t);
144 ec_free(y[0]);
145 ec_free(y[1]);
146 }
147 return result;
148}

150int cp_ers_ext(bn_t td, ers_t *p, size_t *size, const uint8_t *msg, size_t len,
151 const ec_t pk, const ec_t pp) {
152 bn_t n, r;
153 ec_t y[2];
154 int result = RLC_OK;
155
156 bn_null(n);
157 bn_null(r);
158 ec_null(y[0]);
159 ec_null(y[1]);
160
161 for (int i = 0; i < *size; i++) {
162 if (ec_cmp(pk, p[i]->pk) == RLC_EQ) {
163 return RLC_ERR;
164 }
165 }
166
167 RLC_TRY {
168 bn_new(n);
169 bn_new(r);
170 ec_new(y[0]);
171 ec_new(y[1]);
172
173 ec_curve_get_ord(n);
174 bn_rand_mod(r, n);
175 bn_sub(td, td, r);
176 bn_mod(td, td, n);
177 ec_mul_gen(p[*size]->h, r);
178
179 ec_copy(p[*size]->pk, pk);
180 ec_copy(y[0], p[*size]->h);
181 ec_copy(y[1], p[*size]->pk);
182 cp_sokor_sig(p[*size]->c, p[*size]->r, msg, len, y, NULL, r, 1);
183 (*size)++;
184 result = RLC_OK;
185 }
186 RLC_CATCH_ANY {
187 result = RLC_ERR;
188 }
189 RLC_FINALLY {
190 bn_free(n);
191 bn_free(r);
192 ec_free(y[0]);
193 ec_free(y[1]);
194 }
195 return result;
196}
