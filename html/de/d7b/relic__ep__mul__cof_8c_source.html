40void ep_mul_cof(ep_t r, const ep_t p) {
41 ep_t u, v;
42 bn_t k, l;
43
44 bn_null(k);
45 bn_null(l);
46 ep_null(u);
47 ep_null(v);
48
49 RLC_TRY {
50 bn_new(k);
51 bn_new(l);
52 ep_new(u);
53 ep_new(v);
54
55 fp_prime_get_par(k);
56
57 switch (ep_curve_is_pairf()) {
58#if defined(EP_ENDOM) && !defined(STRIP)
59 case EP_BN:
60 /* h = 1 */
61 break;
62 case EP_B12:
63 case EP_B24:
64 case EP_B48:
65 /* Multiply by (1-x) to get the correct group, as proven in
66 * Piellard. https://eprint.iacr.org/2022/352.pdf */
67 bn_neg(k, k);
68 bn_add_dig(k, k, 1);
69 ep_mul_basic(r, p, k);
70 break;
71 case EP_AFG16:
72 /* if (u % 2) == 0, compute = (u * (u**3+1)/2)*P
73 * else Compute (u * (u**3+1))*P */
74 bn_sqr(l, k);
75 bn_mul(l, l, k);
76 bn_add_dig(l, l, 1);
77 bn_mul(k, l, k);
78 if (bn_is_even(k)) {
79 bn_hlv(k, k);
80 }
81 ep_mul_basic(r, p, k);
82 break;
83 case EP_FM16:
84 /* Compute (u/2)*P + [u^3]*phi([u/2]P) */
85 bn_sqr(l, k);
86 bn_mul(l, l, k);
87 if (bn_is_even(k)) {
88 bn_hlv(k, k);
89 }
90 ep_mul_basic(r, p, k);
91 ep_psi(v, r);
92 ep_mul_basic(v, v, l);
93 ep_add(r, r, v);
94 ep_norm(r, r);
95 break;
96 case EP_K16:
97 /* Compute 1250*(P + [(u+1)/2]phi(P)) */
98 bn_add_dig(k, k, 1);
99 bn_hlv(k, k);
100 ep_dbl(r, p);
101 ep_norm(r, r);
102 ep_psi(v, r);
103 ep_neg(v, v);
104 ep_mul_dig(v, v, 182);
105 ep_add(r, r, v);
106 ep_norm(r, r);
107 ep_psi(v, r);
108 ep_neg(v, v);
109 ep_mul_basic(v, v, k);
110 ep_add(r, r, v);
111 ep_norm(r, r);
112 break;
113 case EP_K18:
114 /* Compute 343*(P + [u+3]psi(P)). */
115 bn_add_dig(k, k, 3);
116 ep_psi(v, p);
117 ep_mul_basic(v, v, k);
118 ep_add(v, v, p);
119 ep_norm(r, v);
120 ep_mul_dig(r, r, 49);
121 ep_mul_dig(r, r, 7);
122 break;
123 case EP_FM18:
124 /* Compute S = [(u-1)/3]P, P - 2\phi(S) + [(u^2+u+1)*3 - 1]*S */
125 bn_sqr(l, k);
126 bn_add(l, l, k);
127 bn_add_dig(l, l, 1);
128 bn_mul_dig(l, l, 3);
129 bn_sub_dig(l, l, 1);
130 bn_sub_dig(k, k, 1);
131 bn_div_dig(k, k, 3);
132 ep_mul_basic(v, p, k);
133 ep_psi(u, v);
134 ep_dbl(u, u);
135 ep_sub(r, p, u);
136 ep_mul_basic(v, v, l);
137 ep_add(r, r, v);
138 ep_norm(r, r);
139 break;
140 case EP_SG18:
141 /* Compute [3u^2-1]P. */
142 ep_mul_basic(v, p, k);
143 bn_mul_dig(k, k, 3);
144 ep_mul_basic(v, v, k);
145 ep_sub(v, v, p);
146 ep_norm(r, v);
147 break;
148#endif
149 default:
150 /* multiply by cofactor to get the correct group. */
151 ep_curve_get_cof(k);
152 ep_mul_big(r, p, k);
153 }
154 } RLC_CATCH_ANY {
155 RLC_THROW(ERR_CAUGHT);
156 } RLC_FINALLY {
157 bn_free(k);
158 bn_free(l);
159 ep_free(u);
160 ep_free(v);
161 }
162}
