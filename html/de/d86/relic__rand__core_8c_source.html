82 void rand_init(void) {
83  uint8_t buf[SEED_SIZE];
84 
85 #if RAND == UDEV
86  int *fd = (int *)&(core_get()->rand);
87 
88  *fd = open(RAND_PATH, O_RDONLY);
89  if (*fd == -1) {
90  THROW(ERR_NO_FILE);
91  }
92 #else
93 
94 #if !defined(SEED)
95 
96  memset(buf, 0, SEED_SIZE);
97 
98 #elif SEED == DEV || SEED == UDEV
99  int fd, c, l;
100 
101  fd = open(RAND_PATH, O_RDONLY);
102  if (fd == -1) {
103  THROW(ERR_NO_FILE);
104  }
105 
106  l = 0;
107  do {
108  c = read(fd, buf + l, SEED_SIZE - l);
109  l += c;
110  if (c == -1) {
111  THROW(ERR_NO_READ);
112  }
113  } while (l < SEED_SIZE);
114 
115  if (fd != -1) {
116  close(fd);
117  }
118 #elif SEED == LIBC
119 
120 #if OPSYS == FREEBSD
121  /* This is better than using a fixed value. */
122  srandomdev();
123  for (int i = 0; i < SEED_SIZE; i++) {
124  buf[i] = (uint8_t)random();
125  }
126 #else
127  /* This is horribly insecure, serves only for benchmarking. */
128  srand(1);
129  for (int i = 0; i < SEED_SIZE; i++) {
130  buf[i] = (uint8_t)rand();
131  }
132 #endif
133 
134 #elif SEED == WCGR
135 
136  HCRYPTPROV hCryptProv;
137 
138  if (!CryptAcquireContext(&hCryptProv, NULL, NULL, PROV_RSA_FULL,
139  CRYPT_VERIFYCONTEXT | CRYPT_SILENT)) {
140  THROW(ERR_NO_FILE);
141  }
142  if (hCryptProv && !CryptGenRandom(hCryptProv, SEED_SIZE, buf)) {
143  THROW(ERR_NO_READ);
144  }
145  if (hCryptProv && !CryptReleaseContext(hCryptProv, 0)) {
146  THROW(ERR_NO_READ);
147  }
148 
149 #elif SEED == RDRND
150 
151  int i, j;
152  ull_t r;
153 
154  while (i < SEED_SIZE) {
155 #ifdef __RDRND__
156  while (_rdrand64_step(&r) == 0);
157 #else
158 #error "RdRand not available, check your compiler settings."
159 #endif
160  for (j = 0; i < SEED_SIZE && j < sizeof(ull_t); i++, j++) {
161  buf[i] = r & 0xFF;
162  }
163  }
164 
165 #endif
166 
167 #endif /* RAND == UDEV */
168 
169 #if RAND != CALL
170  core_get()->seeded = 0;
171  rand_seed(buf, SEED_SIZE);
172 #else
173  rand_seed(NULL, NULL);
174 #endif
175 }
