85void rand_init(void) {
86 uint8_t buf[RLC_RAND_SEED];
87
88#if RAND == UDEV
89 int *fd = (int *)&(core_get()->rand);
90
91 *fd = open(RLC_RAND_PATH, O_RDONLY);
92 if (*fd == -1) {
93 RLC_THROW(ERR_NO_FILE);
94 return;
95 }
96#else
97
98#if !defined(SEED)
99
100 memset(buf, 0, RLC_RAND_SEED);
101
102#elif SEED == DEV || SEED == UDEV
103 int fd, c, l;
104
105 fd = open(RLC_RAND_PATH, O_RDONLY);
106 if (fd == -1) {
107 RLC_THROW(ERR_NO_FILE);
108 return;
109 }
110
111 l = 0;
112 do {
113 c = read(fd, buf + l, RLC_RAND_SEED - l);
114 l += c;
115 if (c == -1) {
116 RLC_THROW(ERR_NO_READ);
117 return;
118 }
119 } while (l < RLC_RAND_SEED);
120
121 if (fd != -1) {
122 close(fd);
123 }
124#elif SEED == LIBC
125
126#if OPSYS == FREEBSD || OPSYS == NETBSD
127 /* This is better than using a fixed value. */
128 srandomdev();
129 for (int i = 0; i < RLC_RAND_SEED; i++) {
130 buf[i] = (uint8_t)random();
131 }
132#else
133 /* This is horribly insecure, serves only for benchmarking. */
134 srand(1);
135 for (int i = 0; i < RLC_RAND_SEED; i++) {
136 buf[i] = (uint8_t)rand();
137 }
138#endif
139
140#elif SEED == WCGR
141
142 HCRYPTPROV hCryptProv;
143
144 if (!CryptAcquireContext(&hCryptProv, NULL, NULL, PROV_RSA_FULL,
145 CRYPT_VERIFYCONTEXT | CRYPT_SILENT)) {
146 RLC_THROW(ERR_NO_FILE);
147 return;
148 }
149 if (hCryptProv && !CryptGenRandom(hCryptProv, RLC_RAND_SEED, buf)) {
150 RLC_THROW(ERR_NO_READ);
151 return;
152 }
153 if (hCryptProv && !CryptReleaseContext(hCryptProv, 0)) {
154 RLC_THROW(ERR_NO_READ);
155 return;
156 }
157
158#elif SEED == RDRND
159
160 int i, j;
161 ull_t r;
162
163 while (i < RLC_RAND_SEED) {
164#ifdef __RDRND__
165 while (_rdrand64_step(&r) == 0);
166#else
167#error "RdRand not available, check your compiler settings."
168#endif
169 for (j = 0; i < RLC_RAND_SEED && j < sizeof(ull_t); i++, j++) {
170 buf[i] = r & 0xFF;
171 }
172 }
173
174#endif
175
176#endif /* RAND == UDEV */
177
178#if RAND != CALL
179 core_get()->seeded = 0;
180 rand_seed(buf, RLC_RAND_SEED);
181#else
182 rand_seed(NULL, NULL);
183#endif
184}

186int rand_check(uint8_t *buf, size_t size) {
187 int count = 0;
188
189 for (int i = 1; i < size; i++) {
190 if (buf[i] == buf[i - 1]) {
191 count++;
192 } else {
193 count = 0;
194 }
195 }
196
197 if (count > RAND_REP) {
198 return RLC_ERR;
199 }
200 return RLC_OK;
201}

203void rand_clean(void) {
204 ctx_t *ctx = core_get();
205 if (ctx != NULL) {
206#if RAND == UDEV
207 int *fd = (int *)&(ctx->rand);
208 close(*fd);
209#endif
210#if RAND != CALL
211 memset(ctx->rand, 0, sizeof(ctx->rand));
212#else
213 ctx->rand_call = NULL;
214 ctx->rand_args = NULL;
215#endif
216 ctx->seeded = 0;
217 }
