42 void fp3_sqr_basic(fp3_t c, fp3_t a) {
43  dv_t t0, t1, t2, t3, t4;
44 
45  dv_null(t0);
46  dv_null(t1);
47  dv_null(t2);
48  dv_null(t3);
49  dv_null(t4);
50 
51  TRY {
52  dv_new(t0);
53  dv_new(t1);
54  dv_new(t2);
55  dv_new(t3);
56  dv_new(t4);
57 
58  /* t0 = a_0^2. */
59  fp_sqrn_low(t0, a[0]);
60 
61  /* t1 = 2 * a_1 * a_2. */
62  fp_dbl(t2, a[1]);
63  fp_muln_low(t1, t2, a[2]);
64 
65  /* t3 = (a_0 + a_2 + a_1)^2, t4 = (a_0 + a_2 - a_1)^2. */
66  fp_add(t3, a[0], a[2]);
67  fp_add(t4, t3, a[1]);
68  fp_sub(t2, t3, a[1]);
69  fp_sqrn_low(t3, t4);
70  fp_sqrn_low(t4, t2);
71 
72  /* t2 = a_2^2. */
73  fp_sqrn_low(t2, a[2]);
74 
75  /* t4 = (t4 + t3)/2. */
76 #ifdef RLC_FP_ROOM
77  fp_addd_low(t4, t4, t3);
78 #else
79  fp_addc_low(t4, t4, t3);
80 #endif
81  fp_hlvd_low(t4, t4);
82 
83  /* t3 = t3 - t4 - t1. */
84  fp_subc_low(t3, t3, t4);
85  fp_subc_low(t3, t3, t1);
86 
87  /* c_2 = t4 - t0 - t2. */
88  fp_subc_low(t4, t4, t0);
89  fp_subc_low(t4, t4, t2);
90  fp_rdc(c[2], t4);
91 
92  /* c_0 = t0 + t1 * B. */
93  for (int i = 1; i <= fp_prime_get_cnr(); i++) {
94  fp_addc_low(t0, t0, t1);
95  }
96  for (int i = -1; i >= fp_prime_get_cnr(); i--) {
97  fp_subc_low(t0, t0, t1);
98  }
99  fp_rdc(c[0], t0);
100 
101  /* c_1 = t3 + t2 * B. */
102  for (int i = 1; i <= fp_prime_get_cnr(); i++) {
103  fp_addc_low(t3, t3, t2);
104  }
105  for (int i = -1; i >= fp_prime_get_cnr(); i--) {
106  fp_subc_low(t3, t3, t2);
107  }
108  fp_rdc(c[1], t3);
109  }
110  CATCH_ANY {
111  THROW(ERR_CAUGHT);
112  }
113  FINALLY {
114  dv_free(t0);
115  dv_free(t1);
116  dv_free(t2);
117  dv_free(t3);
118  dv_free(t4);
119  }
120 }
