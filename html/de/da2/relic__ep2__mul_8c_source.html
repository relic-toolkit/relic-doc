43 static void ep2_mul_glv_imp(ep2_t r, ep2_t p, const bn_t k) {
44  int i, j, l;
45  bn_t n, _k[4], u[4], v[4];
46  ep2_t q[4];
47 
48  bn_null(n);
49 
50  TRY {
51  bn_new(n);
52  for (i = 0; i < 4; i++) {
53  bn_null(u[i]);
54  bn_null(v[i]);
55  bn_null(_k[i]);
56  ep2_null(q[i]);
57  bn_new(u[i]);
58  bn_new(v[i]);
59  bn_new(_k[i]);
60  ep2_new(q[i]);
61  }
62 
63  ep2_curve_get_ord(n);
64 
65  switch (ep_curve_is_pairf()) {
66  case EP_BN:
67  ep2_curve_get_vs(v);
68 
69  for (i = 0; i < 4; i++) {
70  bn_mul(v[i], v[i], k);
71  bn_div(v[i], v[i], n);
72  if (bn_sign(v[i]) == RLC_NEG) {
73  bn_add_dig(v[i], v[i], 1);
74  }
75  bn_zero(_k[i]);
76  }
77 
78  /* u0 = x + 1, u1 = 2x + 1, u2 = 2x, u3 = x - 1. */
79  fp_prime_get_par(u[0]);
80  bn_dbl(u[2], u[0]);
81  bn_add_dig(u[1], u[2], 1);
82  bn_sub_dig(u[3], u[0], 1);
83  bn_add_dig(u[0], u[0], 1);
84  bn_copy(_k[0], k);
85  for (i = 0; i < 4; i++) {
86  bn_mul(u[i], u[i], v[i]);
87  bn_mod(u[i], u[i], n);
88  bn_add(_k[0], _k[0], n);
89  bn_sub(_k[0], _k[0], u[i]);
90  bn_mod(_k[0], _k[0], n);
91  }
92 
93  /* u0 = x, u1 = -x, u2 = 2x + 1, u3 = 4x + 2. */
94  fp_prime_get_par(u[0]);
95  bn_neg(u[1], u[0]);
96  bn_dbl(u[2], u[0]);
97  bn_add_dig(u[2], u[2], 1);
98  bn_dbl(u[3], u[2]);
99  for (i = 0; i < 4; i++) {
100  bn_mul(u[i], u[i], v[i]);
101  bn_mod(u[i], u[i], n);
102  bn_add(_k[1], _k[1], n);
103  bn_sub(_k[1], _k[1], u[i]);
104  bn_mod(_k[1], _k[1], n);
105  }
106 
107  /* u0 = x, u1 = -(x + 1), u2 = 2x + 1, u3 = -(2x - 1). */
108  fp_prime_get_par(u[0]);
109  bn_add_dig(u[1], u[0], 1);
110  bn_neg(u[1], u[1]);
111  bn_dbl(u[2], u[0]);
112  bn_add_dig(u[2], u[2], 1);
113  bn_sub_dig(u[3], u[2], 2);
114  bn_neg(u[3], u[3]);
115  for (i = 0; i < 4; i++) {
116  bn_mul(u[i], u[i], v[i]);
117  bn_mod(u[i], u[i], n);
118  bn_add(_k[2], _k[2], n);
119  bn_sub(_k[2], _k[2], u[i]);
120  bn_mod(_k[2], _k[2], n);
121  }
122 
123  /* u0 = -2x, u1 = -x, u2 = 2x + 1, u3 = x - 1. */
124  fp_prime_get_par(u[1]);
125  bn_dbl(u[0], u[1]);
126  bn_neg(u[0], u[0]);
127  bn_dbl(u[2], u[1]);
128  bn_add_dig(u[2], u[2], 1);
129  bn_sub_dig(u[3], u[1], 1);
130  bn_neg(u[1], u[1]);
131  for (i = 0; i < 4; i++) {
132  bn_mul(u[i], u[i], v[i]);
133  bn_mod(u[i], u[i], n);
134  bn_add(_k[3], _k[3], n);
135  bn_sub(_k[3], _k[3], u[i]);
136  bn_mod(_k[3], _k[3], n);
137  }
138 
139  for (i = 0; i < 4; i++) {
140  l = bn_bits(_k[i]);
141  bn_sub(_k[i], n, _k[i]);
142  if (bn_bits(_k[i]) > l) {
143  bn_sub(_k[i], _k[i], n);
144  _k[i]->sign = RLC_POS;
145  } else {
146  _k[i]->sign = RLC_NEG;
147  }
148  }
149  break;
150  default:
151  bn_abs(v[0], k);
152  fp_prime_get_par(u[0]);
153  bn_copy(u[1], u[0]);
154  if (bn_sign(u[0]) == RLC_NEG) {
155  bn_neg(u[0], u[0]);
156  }
157 
158  for (i = 0; i < 4; i++) {
159  bn_mod(_k[i], v[0], u[0]);
160  bn_div(v[0], v[0], u[0]);
161  if ((bn_sign(u[1]) == RLC_NEG) && (i % 2 != 0)) {
162  bn_neg(_k[i], _k[i]);
163  }
164  if (bn_sign(k) == RLC_NEG) {
165  bn_neg(_k[i], _k[i]);
166  }
167  }
168 
169  break;
170  }
171 
172  ep2_norm(q[0], p);
173  ep2_frb(q[1], q[0], 1);
174  ep2_frb(q[2], q[1], 1);
175  ep2_frb(q[3], q[2], 1);
176 
177  for (i = 0; i < 4; i++) {
178  if (bn_sign(_k[i]) == RLC_NEG) {
179  ep2_neg(q[i], q[i]);
180  }
181  }
182 
183  l = RLC_MAX(bn_bits(_k[0]), bn_bits(_k[1]));
184  l = RLC_MAX(l, RLC_MAX(bn_bits(_k[2]), bn_bits(_k[3])));
185  ep2_set_infty(r);
186  for (i = l - 1; i >= 0; i--) {
187  ep2_dbl(r, r);
188  for (j = 0; j < 4; j++) {
189  if (bn_get_bit(_k[j], i)) {
190  ep2_add(r, r, q[j]);
191  }
192  }
193  }
194 
195  /* Convert r to affine coordinates. */
196  ep2_norm(r, r);
197  }
198  CATCH_ANY {
199  THROW(ERR_CAUGHT);
200  }
201  FINALLY {
202  bn_free(n);
203  for (i = 0; i < 4; i++) {
204  bn_free(u[i]);
205  bn_free(v[i]);
206  bn_free(_k[i]);
207  ep2_free(q[i]);
208  }
209 
210  }

217 static void ep2_mul_naf_imp(ep2_t r, ep2_t p, const bn_t k) {
218  int l, i, n;
219  int8_t naf[RLC_FP_BITS + 1];
220  ep2_t t[1 << (EP_WIDTH - 2)];
221 
222  TRY {
223  /* Prepare the precomputation table. */
224  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
225  ep2_null(t[i]);
226  ep2_new(t[i]);
227  }
228  /* Compute the precomputation table. */
229  ep2_tab(t, p, EP_WIDTH);
230 
231  /* Compute the w-NAF representation of k. */
232  l = sizeof(naf);
233  bn_rec_naf(naf, &l, k, EP_WIDTH);
234 
235  ep2_set_infty(r);
236  for (i = l - 1; i >= 0; i--) {
237  ep2_dbl(r, r);
238 
239  n = naf[i];
240  if (n > 0) {
241  ep2_add(r, r, t[n / 2]);
242  }
243  if (n < 0) {
244  ep2_sub(r, r, t[-n / 2]);
245  }
246  }
247  /* Convert r to affine coordinates. */
248  ep2_norm(r, r);
249  if (bn_sign(k) == RLC_NEG) {
250  ep2_neg(r, r);
251  }
252  }
253  CATCH_ANY {
254  THROW(ERR_CAUGHT);
255  }
256  FINALLY {
257  /* Free the precomputation table. */
258  for (i = 0; i < (1 << (EP_WIDTH - 2)); i++) {
259  ep2_free(t[i]);
260  }
261  }
262 }

319 void ep2_mul_slide(ep2_t r, ep2_t p, const bn_t k) {
320  ep2_t t[1 << (EP_WIDTH - 1)], q;
321  int i, j, l;
322  uint8_t win[RLC_FP_BITS + 1];
323 
324  ep2_null(q);
325 
326  if (bn_is_zero(k) || ep2_is_infty(p)) {
327  ep2_set_infty(r);
328  return;
329  }
330 
331  TRY {
332  for (i = 0; i < (1 << (EP_WIDTH - 1)); i ++) {
333  ep2_null(t[i]);
334  ep2_new(t[i]);
335  }
336 
337  ep2_new(q);
338 
339  ep2_copy(t[0], p);
340  ep2_dbl(q, p);
341 
342 #if defined(EP_MIXED)
343  ep2_norm(q, q);
344 #endif
345 
346  /* Create table. */
347  for (i = 1; i < (1 << (EP_WIDTH - 1)); i++) {
348  ep2_add(t[i], t[i - 1], q);
349  }
350 
351 #if defined(EP_MIXED)
352  ep2_norm_sim(t + 1, t + 1, (1 << (EP_WIDTH - 1)) - 1);
353 #endif
354 
355  ep2_set_infty(q);
356  l = RLC_FP_BITS + 1;
357  bn_rec_slw(win, &l, k, EP_WIDTH);
358  for (i = 0; i < l; i++) {
359  if (win[i] == 0) {
360  ep2_dbl(q, q);
361  } else {
362  for (j = 0; j < util_bits_dig(win[i]); j++) {
363  ep2_dbl(q, q);
364  }
365  ep2_add(q, q, t[win[i] >> 1]);
366  }
367  }
368 
369  ep2_norm(r, q);
370  if (bn_sign(k) == RLC_NEG) {
371  ep2_neg(r, r);
372  }
373  }
374  CATCH_ANY {
375  THROW(ERR_CAUGHT);
376  }
377  FINALLY {
378  for (i = 0; i < (1 << (EP_WIDTH - 1)); i++) {
379  ep2_free(t[i]);
380  }
381  ep2_free(q);
382  }
383 }
