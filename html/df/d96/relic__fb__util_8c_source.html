48 static int valid_radix(int radix) {
49  while (radix > 0) {
50  if (radix != 1 && radix % 2 == 1)
51  return 0;
52  radix = radix / 2;
53  }
54  return 1;
55 }

190 void fb_read_str(fb_t a, const char *str, int len, int radix) {
191  int i, j, l;
192  char c;
193  dig_t carry;
194 
195  fb_zero(a);
196 
197  l = log_radix(radix);
198  if (!valid_radix(radix)) {
199  THROW(ERR_NO_VALID);
200  }
201 
202  j = 0;
203  while (j < len) {
204  if (str[j] == 0) {
205  break;
206  }
207  c = (char)((radix < 36) ? RLC_UPP(str[j]) : str[j]);
208  for (i = 0; i < 64; i++) {
209  if (c == util_conv_char(i)) {
210  break;
211  }
212  }
213 
214  if (i < radix) {
215  carry = fb_lshb_low(a, a, l);
216  if (carry != 0) {
217  THROW(ERR_NO_BUFFER);
218  }
219  fb_add_dig(a, a, (dig_t)i);
220  } else {
221  break;
222  }
223  j++;
224  }
225 }

227 void fb_write_str(char *str, int len, const fb_t a, int radix) {
228  fb_t t;
229  int d, l, i, j;
230  char c;
231 
232  fb_null(t);
233 
234  l = fb_size_str(a, radix);
235  if (len < l) {
236  THROW(ERR_NO_BUFFER);
237  }
238  len = l;
239 
240  l = log_radix(radix);
241  if (!valid_radix(radix)) {
242  THROW(ERR_NO_VALID)
243  }
244 
245  if (fb_is_zero(a) == 1) {
246  *str++ = '0';
247  *str = '\0';
248  return;
249  }
250 
251  TRY {
252  fb_new(t);
253  fb_copy(t, a);
254 
255  j = 0;
256  while (!fb_is_zero(t)) {
257  d = t[0] % radix;
258  fb_rshb_low(t, t, l);
259  str[j] = util_conv_char(d);
260  j++;
261  }
262 
263  /* Reverse the digits of the string. */
264  i = 0;
265  j = len - 2;
266  while (i < j) {
267  c = str[i];
268  str[i] = str[j];
269  str[j] = c;
270  ++i;
271  --j;
272  }
273 
274  str[len - 1] = '\0';
275  }
276  CATCH_ANY {
277  THROW(ERR_CAUGHT);
278  }
279  FINALLY {
280  fb_free(t);
281  }
282 }
