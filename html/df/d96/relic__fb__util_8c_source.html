48static int valid_radix(uint_t radix) {
49 while (radix > 0) {
50 if (radix != 1 && radix % 2 == 1)
51 return 0;
52 radix = radix / 2;
53 }
54 return 1;
55}
56
63static int log_radix(uint_t radix) {
64 int l = 0;
65
66 while (radix > 0) {
67 radix = radix / 2;
68 l++;
69 }
70 return --l;
71}

85int fb_is_zero(const fb_t a) {

104void fb_set_bit(fb_t a, uint_t bit, int value) {
105 int d;
106 dig_t mask;
107
108 RLC_RIP(bit, d, bit);
109
110 mask = (dig_t)1 << bit;
111
112 if (value == 1) {
113 a[d] |= mask;
114 } else {
115 a[d] &= ~mask;
116 }
117}

119size_t fb_bits(const fb_t a) {
120 int i = RLC_FB_DIGS - 1;
121
122 while (i > 0 && a[i] == 0) {
123 i--;
124 }
125
126 return (i << RLC_DIG_LOG) + util_bits_dig(a[i]);
127}

134void fb_rand(fb_t a) {
135 size_t bits, digits;
136
137 rand_bytes((uint8_t *)a, RLC_FB_DIGS * sizeof(dig_t));
138
139 RLC_RIP(bits, digits, RLC_FB_BITS);
140 if (bits > 0) {

142 a[RLC_FB_DIGS - 1] &= mask;

157size_t fb_size_str(const fb_t a, uint_t radix) {
158 bn_t t;
159 size_t digits = 0;
160
161 bn_null(t);
162
163 if (!valid_radix(radix)) {
164 RLC_THROW(ERR_NO_VALID);
165 return 0;
166 }
167
168 RLC_TRY {
169 bn_new(t);
170
171 bn_read_raw(t, a, RLC_FB_DIGS);
172
173 digits = bn_size_str(t, radix);
174 }
175 RLC_CATCH_ANY {
176 RLC_THROW(ERR_CAUGHT);
177 }
178 RLC_FINALLY {
179 bn_free(t);
180 }
181
182 return digits;
183}

214void fb_write_str(char *str, size_t len, const fb_t a, uint_t radix) {
215 fb_t t;
216 int d, l, i, j;
217 char c;
218
219 fb_null(t);
220
221 l = fb_size_str(a, radix);
222 if (len < l) {
223 RLC_THROW(ERR_NO_BUFFER);
224 return;
225 }
226 len = l;
227
228 l = log_radix(radix);
229 if (!valid_radix(radix)) {
230 RLC_THROW(ERR_NO_VALID);
231 return;
232 }
233
234 if (fb_is_zero(a) == 1) {
235 *str++ = '0';
236 *str = '\0';
237 return;
238 }
239
240 RLC_TRY {
241 fb_new(t);
242 fb_copy(t, a);
243
244 j = 0;
245 while (!fb_is_zero(t)) {
246 d = t[0] % radix;
247 fb_rshb_low(t, t, l);
248 str[j] = util_conv_char(d);
249 j++;
250 }
251
252 /* Reverse the digits of the string. */
253 i = 0;
254 j = len - 2;
255 while (i < j) {
256 c = str[i];
257 str[i] = str[j];
258 str[j] = c;
259 ++i;
260 --j;
261 }
262
263 str[len - 1] = '\0';
264 }
265 RLC_CATCH_ANY {
266 RLC_THROW(ERR_CAUGHT);
267 }
268 RLC_FINALLY {
269 fb_free(t);
270 }
271}
