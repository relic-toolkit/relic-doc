135 static void bn_exp(bn_t c, const bn_t a, const bn_t b, const bn_t m) {
136  int i, l;
137  bn_t t;
138 
139  bn_null(t);
140 
141  TRY {
142  bn_new(t);
143 
144  l = bn_bits(b);
145 
146  bn_copy(t, a);
147 
148  for (i = l - 2; i >= 0; i--) {
149  bn_sqr(t, t);
150  bn_mod(t, t, m);
151  if (bn_get_bit(b, i)) {
152  bn_mul(t, t, a);
153  bn_mod(t, t, m);
154  }
155  }
156 
157  bn_copy(c, t);
158  }
159  CATCH_ANY {
160  THROW(ERR_CAUGHT);
161  }
162  FINALLY {
163  bn_free(t);
164  }
165 }

197 int bn_is_prime_basic(const bn_t a) {
198  dig_t t;
199  int i, result;
200 
201  result = 1;
202 
203  if (bn_cmp_dig(a, 1) == RLC_EQ) {
204  return 0;
205  }
206 
207  /* Trial division. */
208  for (i = 0; i < BASIC_TESTS; i++) {
209  bn_mod_dig(&t, a, primes[i]);
210  if (t == 0 && bn_cmp_dig(a, primes[i]) != RLC_EQ) {
211  result = 0;
212  break;
213  }
214  }
215  return result;
216 }

218 int bn_is_prime_rabin(const bn_t a) {
219  bn_t t, n1, y, r;
220  int i, s, j, result, b, tests = 0, cmp2;
221 
222  tests = 0;
223  result = 1;
224 
225  bn_null(t);
226  bn_null(n1);
227  bn_null(y);
228  bn_null(r);
229 
230  cmp2 = bn_cmp_dig(a, 2);
231  if (cmp2 == RLC_LT) {
232  /* Numbers 1 or smaller are not prime */
233  return 0;
234  }
235  if (cmp2 == RLC_EQ) {
236  /* The number 2 is prime */
237  return 1;
238  }
239 
240  if (bn_is_even(a) == 1) {
241  /* Even numbers > 2 are not prime */
242  return 0;
243  }
244 
245  TRY {
246  /*
247  * These values are taken from Table 4.4 inside Handbook of Applied
248  * Cryptography.
249  */
250  b = bn_bits(a);
251  if (b >= 1300) {
252  tests = 2;
253  } else if (b >= 850) {
254  tests = 3;
255  } else if (b >= 650) {
256  tests = 4;
257  } else if (b >= 550) {
258  tests = 5;
259  } else if (b >= 450) {
260  tests = 6;
261  } else if (b >= 400) {
262  tests = 7;
263  } else if (b >= 350) {
264  tests = 8;
265  } else if (b >= 300) {
266  tests = 9;
267  } else if (b >= 250) {
268  tests = 12;
269  } else if (b >= 200) {
270  tests = 15;
271  } else if (b >= 150) {
272  tests = 18;
273  } else {
274  tests = 27;
275  }
276 
277  bn_new(t);
278  bn_new(n1);
279  bn_new(y);
280  bn_new(r);
281 
282  /* r = (n - 1)/2^s. */
283  bn_sub_dig(n1, a, 1);
284  bn_copy(r, n1);
285  s = 0;
286  while (bn_is_even(r)) {
287  s++;
288  bn_rsh(r, r, 1);
289  }
290 
291  for (i = 0; i < tests; i++) {
292  /* Fix the basis as the first few primes. */
293  bn_set_dig(t, primes[i]);
294 
295  /* Ensure t <= n - 2 as per HAC */
296  if( bn_cmp(t, n1) != RLC_LT ) {
297  result = 1;
298  break;
299  }
300 
301  /* y = b^r mod a. */
302 #if BN_MOD != PMERS
303  bn_mxp(y, t, r, a);
304 #else
305  bn_exp(y, t, r, a);
306 #endif
307 
308  if (bn_cmp_dig(y, 1) != RLC_EQ && bn_cmp(y, n1) != RLC_EQ) {
309  j = 1;
310  while ((j <= (s - 1)) && bn_cmp(y, n1) != RLC_EQ) {
311  bn_sqr(y, y);
312  bn_mod(y, y, a);
313 
314  /* If y == 1 then composite. */
315  if (bn_cmp_dig(y, 1) == RLC_EQ) {
316  result = 0;
317  break;
318  }
319  ++j;
320  }
321 
322  /* If y != n1 then composite. */
323  if (bn_cmp(y, n1) != RLC_EQ) {
324  result = 0;
325  break;
326  }
327  }
328  }
329  }
330  CATCH_ANY {
331  result = 0;
332  THROW(ERR_CAUGHT);
333  }
334  FINALLY {
335  bn_free(r);
336  bn_free(y);
337  bn_free(n1);
338  bn_free(t);
339  }
340  return result;
341 }

343 int bn_is_prime_solov(const bn_t a) {
344  bn_t t0, t1, t2;
345  int i, result;
346 
347  bn_null(t0);
348  bn_null(t1);
349  bn_null(t2);
350 
351  result = 1;
352 
353  TRY {
354  bn_new(t0);
355  bn_new(t1);
356  bn_new(t2);
357 
358  for (i = 0; i < 100; i++) {
359  /* Generate t0, 2 <= t0, <= a - 2. */
360  do {
361  bn_rand(t0, RLC_POS, bn_bits(a));
362  bn_mod(t0, t0, a);
363  } while (bn_cmp_dig(t0, 2) == RLC_LT);
364  /* t2 = a - 1. */
365  bn_copy(t2, a);
366  bn_sub_dig(t2, t2, 1);
367  /* t1 = (a - 1)/2. */
368  bn_rsh(t1, t2, 1);
369  /* t1 = t0^(a - 1)/2 mod a. */
370 #if BN_MOD != PMERS
371  bn_mxp(t1, t0, t1, a);
372 #else
373  bn_exp(t1, t0, t1, a);
374 #endif
375  /* If t1 != 1 and t1 != n - 1 return 0 */
376  if (bn_cmp_dig(t1, 1) != RLC_EQ && bn_cmp(t1, t2) != RLC_EQ) {
377  result = 0;
378  break;
379  }
380 
381  /* t2 = (t0|a). */
382  bn_smb_jac(t2, t0, a);
383  if (bn_sign(t2) == RLC_NEG) {
384  bn_add(t2, t2, a);
385  }
386  /* If t1 != t2 (mod a) return 0. */
387  bn_mod(t1, t1, a);
388  bn_mod(t2, t2, a);
389  if (bn_cmp(t1, t2) != RLC_EQ) {
390  result = 0;
391  break;
392  }
393  }
394  }
395  CATCH_ANY {
396  result = 0;
397  THROW(ERR_CAUGHT);
398  }
399  FINALLY {
400  bn_free(t0);
401  bn_free(t1);
402  bn_free(t2);
403  }
404  return result;
405 }

448 void bn_gen_prime_stron(bn_t a, int bits) {
449  dig_t i, j;
450  int found, k;
451  bn_t r, s, t;
452 
453  bn_null(r);
454  bn_null(s);
455  bn_null(t);
456 
457  TRY {
458  bn_new(r);
459  bn_new(s);
460  bn_new(t);
461 
462  do {
463  do {
464  /* Generate two large primes r and s. */
465  bn_rand(s, RLC_POS, bits / 2 - RLC_DIG / 2);
466  bn_rand(t, RLC_POS, bits / 2 - RLC_DIG / 2);
467  } while (!bn_is_prime(s) || !bn_is_prime(t));
468  found = 1;
469  bn_rand(a, RLC_POS, bits / 2 - bn_bits(t) - 1);
470  i = a->dp[0];
471  bn_dbl(t, t);
472  do {
473  /* Find first prime r = 2 * i * t + 1. */
474  bn_mul_dig(r, t, i);
475  bn_add_dig(r, r, 1);
476  i++;
477  if (bn_bits(r) > bits / 2 - 1) {
478  found = 0;
479  break;
480  }
481  } while (!bn_is_prime(r));
482  if (found == 0) {
483  continue;
484  }
485  /* Compute t = 2 * (s^(r-2) mod r) * s - 1. */
486  bn_sub_dig(t, r, 2);
487 #if BN_MOD != PMERS
488  bn_mxp(t, s, t, r);
489 #else
490  bn_exp(t, s, t, r);
491 #endif
492 
493  bn_mul(t, t, s);
494  bn_dbl(t, t);
495  bn_sub_dig(t, t, 1);
496 
497  k = bits - bn_bits(r);
498  k -= bn_bits(s);
499  bn_rand(a, RLC_POS, k);
500  j = a->dp[0];
501  do {
502  /* Find first prime a = t + 2 * j * r * s. */
503  bn_mul(a, r, s);
504  bn_mul_dig(a, a, j);
505  bn_dbl(a, a);
506  bn_add(a, a, t);
507  j++;
508  if (bn_bits(a) > bits) {
509  found = 0;
510  break;
511  }
512  } while (!bn_is_prime(a));
513  } while (found == 0 && bn_bits(a) != bits);
514  }
515  CATCH_ANY {
516  THROW(ERR_CAUGHT);
517  }
518  FINALLY {
519  bn_free(r);
520  bn_free(s);
521  bn_free(t);
522  }
523 }
