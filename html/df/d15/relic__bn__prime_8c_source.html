135static void bn_exp(bn_t c, const bn_t a, const bn_t b, const bn_t m) {
136 int i, l;
137 bn_t t;
138
139 bn_null(t);
140
141 RLC_TRY {
142 bn_new(t);
143
144 l = bn_bits(b);
145
146 bn_copy(t, a);
147
148 for (i = l - 2; i >= 0; i--) {
149 bn_sqr(t, t);
150 bn_mod(t, t, m);
151 if (bn_get_bit(b, i)) {
152 bn_mul(t, t, a);
153 bn_mod(t, t, m);
154 }
155 }
156
157 bn_copy(c, t);
158 }
159 RLC_CATCH_ANY {
160 RLC_THROW(ERR_CAUGHT);
161 }
162 RLC_FINALLY {
163 bn_free(t);
164 }
165}

173int bn_is_prime(const bn_t a) {
174 int result;
175
176 result = 0;
177 if (!bn_is_prime_basic(a)) {
178 goto end;
179 }
180
181 if (!bn_is_prime_rabin(a)) {
182 goto end;
183 }
184
185 result = 1;
186 end:
187 return result;
188}

190int bn_is_prime_basic(const bn_t a) {
191 dig_t t;
192 int i, result;
193
194 result = 1;
195
196 if (bn_cmp_dig(a, 1) == RLC_EQ) {
197 return 0;
198 }
199
200 /* Trial division. */
201 for (i = 0; i < BASIC_TESTS; i++) {
202 bn_mod_dig(&t, a, primes[i]);
203 if (t == 0 && bn_cmp_dig(a, primes[i]) != RLC_EQ) {
204 result = 0;
205 break;
206 }
207 }
208 return result;
209}

211int bn_is_prime_rabin(const bn_t a) {
212 bn_t t, n1, y, r;
213 int i, s, j, result, b, tests = 0, cmp2;
214
215 tests = 0;
216 result = 1;
217
218 bn_null(t);
219 bn_null(n1);
220 bn_null(y);
221 bn_null(r);
222
223 cmp2 = bn_cmp_dig(a, 2);
224 if (cmp2 == RLC_LT) {
225 /* Numbers 1 or smaller are not prime */
226 return 0;
227 }
228 if (cmp2 == RLC_EQ) {
229 /* The number 2 is prime */
230 return 1;
231 }
232
233 if (bn_is_even(a) == 1) {
234 /* Even numbers > 2 are not prime */
235 return 0;
236 }
237
238 RLC_TRY {
239 /*
240 * These values are taken from Table 4.4 inside Handbook of Applied
241 * Cryptography.
242 */
243 b = bn_bits(a);
244 if (b >= 1300) {
245 tests = 2;
246 } else if (b >= 850) {
247 tests = 3;
248 } else if (b >= 650) {
249 tests = 4;
250 } else if (b >= 550) {
251 tests = 5;
252 } else if (b >= 450) {
253 tests = 6;
254 } else if (b >= 400) {
255 tests = 7;
256 } else if (b >= 350) {
257 tests = 8;
258 } else if (b >= 300) {
259 tests = 9;
260 } else if (b >= 250) {
261 tests = 12;
262 } else if (b >= 200) {
263 tests = 15;
264 } else if (b >= 150) {
265 tests = 18;
266 } else {
267 tests = 27;
268 }
269
270 bn_new(t);
271 bn_new(n1);
272 bn_new(y);
273 bn_new(r);
274
275 /* r = (n - 1)/2^s. */
276 bn_sub_dig(n1, a, 1);
277 bn_copy(r, n1);
278 s = 0;
279 while (!bn_is_zero(r) && bn_is_even(r)) {
280 s++;
281 bn_rsh(r, r, 1);
282 }
283
284 for (i = 0; i < tests; i++) {
285 /* Fix the basis as the first few primes. */
286 bn_set_dig(t, primes[i]);
287
288 /* Ensure t <= n - 2 as per HAC */
289 if (bn_cmp(t, n1) != RLC_LT) {
290 result = 1;
291 break;
292 }
293
294 /* y = b^r mod a. */
295#if BN_MOD != PMERS
296 bn_mxp(y, t, r, a);
297#else
298 bn_exp(y, t, r, a);
299#endif
300
301 if (bn_cmp_dig(y, 1) != RLC_EQ && bn_cmp(y, n1) != RLC_EQ) {
302 j = 1;
303 while ((j <= (s - 1)) && bn_cmp(y, n1) != RLC_EQ) {
304 bn_sqr(y, y);
305 bn_mod(y, y, a);
306
307 /* If y == 1 then composite. */
308 if (bn_cmp_dig(y, 1) == RLC_EQ) {
309 result = 0;
310 break;
311 }
312 ++j;
313 }
314
315 /* If y != n1 then composite. */
316 if (bn_cmp(y, n1) != RLC_EQ) {
317 result = 0;
318 break;
319 }
320 }
321 }
322 }
323 RLC_CATCH_ANY {
324 result = 0;
325 RLC_THROW(ERR_CAUGHT);
326 }
327 RLC_FINALLY {
328 bn_free(r);
329 bn_free(y);
330 bn_free(n1);
331 bn_free(t);
332 }
333 return result;
334}

336int bn_is_prime_solov(const bn_t a) {
337 bn_t t0, t1, t2;
338 int i, result;
339
340 bn_null(t0);
341 bn_null(t1);
342 bn_null(t2);
343
344 result = 1;
345
346 RLC_TRY {
347 bn_new(t0);
348 bn_new(t1);
349 bn_new(t2);
350
351 for (i = 0; i < 100; i++) {
352 /* Generate t0, 2 <= t0, <= a - 2. */
353 do {
354 bn_rand(t0, RLC_POS, bn_bits(a));
355 bn_mod(t0, t0, a);
356 } while (bn_cmp_dig(t0, 2) == RLC_LT);
357 /* t2 = a - 1. */
358 bn_copy(t2, a);
359 bn_sub_dig(t2, t2, 1);
360 /* t1 = (a - 1)/2. */
361 bn_rsh(t1, t2, 1);
362 /* t1 = t0^(a - 1)/2 mod a. */
363#if BN_MOD != PMERS
364 bn_mxp(t1, t0, t1, a);
365#else
366 bn_exp(t1, t0, t1, a);
367#endif
368 /* If t1 != 1 and t1 != n - 1 return 0 */
369 if (bn_cmp_dig(t1, 1) != RLC_EQ && bn_cmp(t1, t2) != RLC_EQ) {
370 result = 0;
371 break;
372 }
373
374 /* Lend result here, but restore afterwards, for t2 = (t0|a). */
375 result = bn_smb_jac(t0, a);
376 bn_set_dig(t2, (result < 0 ? -result : result));
377 if (result < 0) {
378 bn_neg(t2, t2);
379 }
380 result = 1;
381 /* If t1 != t2 (mod a) return 0. */
382 bn_mod(t1, t1, a);
383 bn_mod(t2, t2, a);
384 if (bn_cmp(t1, t2) != RLC_EQ) {
385 result = 0;
386 break;
387 }
388 }
389 }
390 RLC_CATCH_ANY {
391 result = 0;
392 RLC_THROW(ERR_CAUGHT);
393 }
394 RLC_FINALLY {
395 bn_free(t0);
396 bn_free(t1);
397 bn_free(t2);
398 }
399 return result;
400}

419void bn_gen_prime_safep(bn_t a, size_t bits) {
420 while (1) {
421 do {
422 bn_rand(a, RLC_POS, bits);
423 } while (bn_bits(a) != bits);
424 /* Check if (a - 1)/2 is prime. */
425 bn_sub_dig(a, a, 1);

428 /* Restore a. */

443void bn_gen_prime_stron(bn_t a, size_t bits) {
444 dig_t i, j;
445 int found, k;
446 bn_t r, s, t;
447
448 bn_null(r);
449 bn_null(s);
450 bn_null(t);
451
452 RLC_TRY {
453 bn_new(r);
454 bn_new(s);
455 bn_new(t);
456
457 do {
458 do {
459 /* Generate two large primes r and s. */
460 bn_rand(s, RLC_POS, bits / 2 - RLC_DIG / 2);
461 bn_rand(t, RLC_POS, bits / 2 - RLC_DIG / 2);
462 } while (!bn_is_prime(s) || !bn_is_prime(t));
463 found = 1;
464 bn_rand(a, RLC_POS, bits / 2 - bn_bits(t) - 1);
465 i = a->dp[0];
466 bn_dbl(t, t);
467 do {
468 /* Find first prime r = 2 * i * t + 1. */
469 bn_mul_dig(r, t, i);
470 bn_add_dig(r, r, 1);
471 i++;
472 if (bn_bits(r) > bits / 2 - 1) {
473 found = 0;
474 break;
475 }
476 } while (!bn_is_prime(r));
477 if (found == 0) {
478 continue;
479 }
480 /* Compute t = 2 * (s^(r-2) mod r) * s - 1. */
481 bn_sub_dig(t, r, 2);
482#if BN_MOD != PMERS
483 bn_mxp(t, s, t, r);
484#else
485 bn_exp(t, s, t, r);
486#endif
487
488 bn_mul(t, t, s);
489 bn_dbl(t, t);
490 bn_sub_dig(t, t, 1);
491
492 k = bits - bn_bits(r);
493 k -= bn_bits(s);
494 bn_rand(a, RLC_POS, k);
495 j = a->dp[0];
496 do {
497 /* Find first prime a = t + 2 * j * r * s. */
498 bn_mul(a, r, s);
499 bn_mul_dig(a, a, j);
500 bn_dbl(a, a);
501 bn_add(a, a, t);
502 j++;
503 if (bn_bits(a) > bits) {
504 found = 0;
505 break;
506 }
507 } while (!bn_is_prime(a));
508 } while (found == 0 && bn_bits(a) != bits);
509 }
510 RLC_CATCH_ANY {
511 RLC_THROW(ERR_CAUGHT);
512 }
513 RLC_FINALLY {
514 bn_free(r);
515 bn_free(s);
516 bn_free(t);
517 }
518}

522int bn_gen_prime_factor(bn_t a, bn_t b, size_t abits, size_t bbits) {
523 bn_t t, u;
524 int result = RLC_OK;
525
526 if (!(bbits > abits)) {
527 return RLC_ERR;
528 }
529
530 bn_null(t);
531 bn_null(u);
532
533 RLC_TRY {
534 bn_new(t);
535 bn_new(u);
536
537 bn_gen_prime(a, abits);
538 bn_set_dig(t, 1);
539 bn_lsh(t, t, bbits - bn_bits(a) - 1);
540 do {
541 bn_rand(u, RLC_POS, bbits - bn_bits(a) - 1);
542 bn_add(u, u, t);
543 bn_mul(b, a, u);
544 bn_add_dig(b, b, 1);
545 } while ((bn_bits(b) != bbits) || (!bn_is_prime(b)));
546 }
547 RLC_CATCH_ANY {
548 result = RLC_ERR;
549 }
550 RLC_FINALLY {
551 bn_free(t);
552 bn_free(u);
553 }
554
555 return result;
