41static void ed_mul_naf_imp(ed_t r, const ed_t p, const bn_t k) {
42 int i, n;
43 int8_t naf[RLC_FP_BITS + 1];
44 ed_t t[1 << (RLC_WIDTH - 2)];
45 size_t l;
46
47 if (bn_is_zero(k)) {
48 ed_set_infty(r);
49 return;
50 }
51
52 RLC_TRY {
53 /* Prepare the precomputation table. */
54 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
55 ed_null(t[i]);
56 ed_new(t[i]);
57 }
58 /* Compute the precomputation table. */
59 ed_tab(t, p, RLC_WIDTH);
60
61 /* Compute the w-NAF representation of k. */
62 l = sizeof(naf);
63 bn_rec_naf(naf, &l, k, RLC_WIDTH);
64
65 ed_set_infty(r);
66 for (i = l - 1; i > 0; i--) {
67 n = naf[i];
68 if (n == 0) {
69 /* This point will be doubled in the next iteration. */
70#if ED_ADD == EXTND
71 r->coord = EXTND;
72#endif
73 }
74 ed_dbl(r, r);
75 if (n > 0) {
76 ed_add(r, r, t[n / 2]);
77 } else if (n < 0) {
78 ed_sub(r, r, t[-n / 2]);
79 }
80 }
81
82 /* Last iteration. */
83 n = naf[0];
84 ed_dbl(r, r);
85 if (n > 0) {
86 ed_add(r, r, t[n / 2]);
87 } else if (n < 0) {
88 ed_sub(r, r, t[-n / 2]);
89 }
90
91 /* Convert r to affine coordinates. */
92 ed_norm(r, r);
93 if (bn_sign(k) == RLC_NEG) {
94 ed_neg(r, r);
95 }
96 }
97 RLC_CATCH_ANY {
98 RLC_THROW(ERR_CAUGHT);
99 }
100 RLC_FINALLY {
101 /* Free the precomputation table. */
102 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
103 ed_free(t[i]);
104 }
105 }
106}
107
108#endif /* ED_MUL == LWNAF */

112static void ed_mul_reg_imp(ed_t r, const ed_t p, const bn_t k) {
113 bn_t _k;
114 int i, j, n;
115 int8_t s, reg[RLC_CEIL(RLC_FP_BITS + 1, RLC_WIDTH - 1)];
116 ed_t t[1 << (RLC_WIDTH - 2)], u, v;
117 size_t l;
118
119 bn_null(_k);
120
121 RLC_TRY {
122 bn_new(_k);
123 ed_new(u);
124 ed_new(v);
125 /* Prepare the precomputation table. */
126 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
127 ed_null(t[i]);
128 ed_new(t[i]);
129 }
130 /* Compute the precomputation table. */
131 ed_tab(t, p, RLC_WIDTH);
132
133 /* Make a copy of the scalar for processing. */
134 bn_abs(_k, k);
135 _k->dp[0] |= bn_is_even(_k);
136
137 /* Compute the w-NAF representation of k. */
138 l = RLC_CEIL(RLC_FP_BITS + 1, RLC_WIDTH - 1) + 1;
139 bn_rec_reg(reg, &l, _k, RLC_FP_BITS, RLC_WIDTH);
140
141 ed_set_infty(r);
142 for (i = l - 1; i >= 0; i--) {
143 for (j = 0; j < RLC_WIDTH - 1; j++) {
144 ed_dbl(r, r);
145 }
146
147 n = reg[i];
148 s = (n >> 7);
149 n = ((n ^ s) - s) >> 1;
150
151 for (j = 0; j < (1 << (RLC_WIDTH - 2)); j++) {
152 fp_copy_sec(u->x, t[j]->x, j == n);
153 fp_copy_sec(u->y, t[j]->y, j == n);
154 fp_copy_sec(u->z, t[j]->z, j == n);
155#if ED_ADD == EXTND
156 fp_copy_sec(u->t, t[j]->t, j == n);
157#endif
158 }
159 ed_neg(v, u);
160 fp_copy_sec(u->x, v->x, s != 0);
161#if ED_ADD == EXTND
162 fp_copy_sec(u->t, v->t, s != 0);
163#endif
164 ed_add(r, r, u);
165 }
166
167 /* t[0] has an unmodified copy of p. */
168 ed_sub(u, r, t[0]);
169 fp_copy_sec(r->x, u->x, bn_is_even(k));
170 fp_copy_sec(r->y, u->y, bn_is_even(k));
171 fp_copy_sec(r->z, u->z, bn_is_even(k));
172#if ED_Afp == EXTND
173 fp_copy_sec(r->t, u->t, bn_is_even(k));
174#endif
175 /* Convert r to affine coordinates. */
176 ed_norm(r, r);
177 ed_neg(u, r);
178 fp_copy_sec(r->x, u->x, bn_sign(k) == RLC_NEG);
179#if ED_ADD == EXTND
180 fp_copy_sec(r->t, u->t, bn_sign(k) == RLC_NEG);
181#endif
182 }
183 RLC_CATCH_ANY {
184 RLC_THROW(ERR_CAUGHT);
185 }
186 RLC_FINALLY {
187 /* Free the precomputation table. */
188 for (i = 0; i < (1 << (RLC_WIDTH - 2)); i++) {
189 ed_free(t[i]);
190 }
191 bn_free(_k);
192 }
193}
194
195#endif /* ED_MUL == LWNAF */

203void ed_mul_basic(ed_t r, const ed_t p, const bn_t k) {
204 ed_t t;
205 int8_t u, *naf = RLC_ALLOCA(int8_t, bn_bits(k) + 1);
206 size_t l;
207
208 ed_null(t);
209
210 if (bn_is_zero(k) || ed_is_infty(p)) {
211 RLC_FREE(naf);
212 ed_set_infty(r);
213 return;
214 }
215
216 RLC_TRY {
217 ed_new(t);
218 if (naf == NULL) {
219 RLC_THROW(ERR_NO_BUFFER);
220 }
221
222 l = bn_bits(k) + 1;
223 bn_rec_naf(naf, &l, k, 2);
224 ed_set_infty(t);
225 for (int i = l - 1; i >= 0; i--) {
226 ed_dbl(t, t);
227
228 u = naf[i];
229 if (u > 0) {
230 ed_add(t, t, p);
231 } else if (u < 0) {
232 ed_sub(t, t, p);
233 }
234 }
235
236 ed_norm(r, t);
237 if (bn_sign(k) == RLC_NEG) {
238 ed_neg(r, r);
239 }
240 }
241 RLC_CATCH_ANY {
242 RLC_THROW(ERR_CAUGHT);
243 }
244 RLC_FINALLY {
245 ed_free(t);
246 RLC_FREE(naf);
247 }
248}

254void ed_mul_slide(ed_t r, const ed_t p, const bn_t k) {
255 ed_t t[1 << (RLC_WIDTH - 1)], q;
256 uint8_t win[RLC_FP_BITS + 1];
257 size_t l;
258
259 ed_null(q);
260
261 if (bn_is_zero(k) || ed_is_infty(p)) {
262 ed_set_infty(r);
263 return;
264 }
265
266 RLC_TRY {
267 for (size_t i = 0; i < (1 << (RLC_WIDTH - 1)); i ++) {
268 ed_null(t[i]);
269 ed_new(t[i]);
270 }
271
272 ed_new(q);
273
274 ed_copy(t[0], p);
275 ed_dbl(q, p);
276
277#if defined(EP_MIXED)
278 ed_norm(q, q);
279#endif
280
281 /* Create table. */
282 for (size_t i = 1; i < (1 << (RLC_WIDTH - 1)); i++) {
283 ed_add(t[i], t[i - 1], q);
284 }
285
286#if defined(EP_MIXED)
287 ed_norm_sim(t + 1, (const ed_t *)t + 1, (1 << (RLC_WIDTH - 1)) - 1);
288#endif
289
290 ed_set_infty(q);
291 l = RLC_FP_BITS + 1;
292 bn_rec_slw(win, &l, k, RLC_WIDTH);
293 for (size_t i = 0; i < l; i++) {
294 if (win[i] == 0) {
295 ed_dbl(q, q);
296 } else {
297 for (size_t j = 0; j < util_bits_dig(win[i]); j++) {
298 ed_dbl(q, q);
299 }
300 ed_add(q, q, t[win[i] >> 1]);
301 }
302 }
303
304 ed_norm(r, q);
305 if (bn_sign(k) == RLC_NEG) {
306 ed_neg(r, r);
307 }
308 }
309 RLC_CATCH_ANY {
310 RLC_THROW(ERR_CAUGHT);
311 }
312 RLC_FINALLY {
313 for (size_t i = 0; i < (1 << (RLC_WIDTH - 1)); i++) {
314 ed_free(t[i]);
315 }
316 ed_free(q);
317 }
318}
