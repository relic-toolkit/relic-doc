42void fp3_mul_basic(fp3_t c, const fp3_t a, const fp3_t b) {
43 dv_t t, t0, t1, t2, t3, t4, t5, t6;
44
45 dv_null(t);
46 dv_null(t0);
47 dv_null(t1);
48 dv_null(t2);
49 dv_null(t3);
50 dv_null(t4);
51 dv_null(t5);
52 dv_null(t6);
53
54 RLC_TRY {
55 dv_new(t);
56 dv_new(t0);
57 dv_new(t1);
58 dv_new(t2);
59 dv_new(t3);
60 dv_new(t4);
61 dv_new(t5);
62 dv_new(t6);
63
64 /* Karatsuba algorithm. */
65
66 /* t0 = a_0 * b_0, t1 = a_1 * b_1, t2 = a_2 * b_2. */
67 fp_muln_low(t0, a[0], b[0]);
68 fp_muln_low(t1, a[1], b[1]);
69 fp_muln_low(t2, a[2], b[2]);
70
71 /* t3 = (a_1 + a_2) * (b_1 + b_2). */
72 fp_add(t3, a[1], a[2]);
73 fp_add(t4, b[1], b[2]);
74 fp_muln_low(t, t3, t4);
75#ifdef RLC_FP_ROOM
76 fp_addd_low(t6, t1, t2);
77#else
78 fp_addc_low(t6, t1, t2);
79#endif
80 fp_subc_low(t4, t, t6);
81 fp_addc_low(t3, t0, t4);
82 for (int i = 1; i < fp_prime_get_cnr(); i++) {
83 fp_addc_low(t3, t3, t4);
84 }
85 for (int i = 0; i >= fp_prime_get_cnr(); i--) {
86 fp_subc_low(t3, t3, t4);
87 }
88
89 fp_add(t4, a[0], a[1]);
90 fp_add(t5, b[0], b[1]);
91 fp_muln_low(t, t4, t5);
92#ifdef RLC_FP_ROOM
93 fp_addd_low(t4, t0, t1);
94#else
95 fp_addc_low(t4, t0, t1);
96#endif
97 fp_subc_low(t4, t, t4);
98 fp_addc_low(t4, t4, t2);
99 for (int i = 1; i < fp_prime_get_cnr(); i++) {
100 fp_addc_low(t4, t4, t2);
101 }
102 for (int i = 0; i >= fp_prime_get_cnr(); i--) {
103 fp_subc_low(t4, t4, t2);
104 }
105
106 fp_add(t5, a[0], a[2]);
107 fp_add(t6, b[0], b[2]);
108 fp_muln_low(t, t5, t6);
109#ifdef RLC_FP_ROOM
110 fp_addd_low(t6, t0, t2);
111#else
112 fp_addc_low(t6, t0, t2);
113#endif
114 fp_addc_low(t6, t0, t2);
115 fp_subc_low(t5, t, t6);
116 fp_addc_low(t5, t5, t1);
117
118 /* c_0 = t3 mod p. */
119 fp_rdc(c[0], t3);
120
121 /* c_1 = t4 mod p. */
122 fp_rdc(c[1], t4);
123
124 /* c_2 = t5 mod p. */
125 fp_rdc(c[2], t5);
126 }
127 RLC_CATCH_ANY {
128 RLC_THROW(ERR_CAUGHT);
129 }
130 RLC_FINALLY {
131 dv_free(t);
132 dv_free(t0);
133 dv_free(t1);
134 dv_free(t2);
135 dv_free(t3);
136 dv_free(t4);
137 dv_free(t5);
138 dv_free(t6);
139 }
140}

152void fp3_mul_art(fp3_t c, const fp3_t a) {
153 fp_t t;
154
155 fp_null(t);
156
157 RLC_TRY {
158 fp_new(t);
159
160 /* (a_0 + a_1 * u + a_2 * u^2) * u = a_0 * u + a_1 * u^2 + a_2 * u^3. */
161 fp_copy(t, a[0]);
162 fp_copy(c[0], a[2]);
163 for (int i = 1; i < fp_prime_get_cnr(); i++) {
164 fp_add(c[0], c[0], a[2]);
165 }
166 for (int i = 0; i >= fp_prime_get_cnr(); i--) {
167 fp_sub(c[0], c[0], a[2]);
168 }
169 fp_copy(c[2], a[1]);
170 fp_copy(c[1], t);
171 }
172 RLC_CATCH_ANY {
173 RLC_THROW(ERR_CAUGHT);
174 }
175 RLC_FINALLY {
176 fp_free(t);
177 }
178}

180void fp3_mul_nor(fp3_t c, const fp3_t a) {
181 fp3_t t, u;
182
183 fp3_null(t);
184 fp3_null(u);
185
186 RLC_TRY {
187 fp3_new(t);
188 fp3_new(u);
189
190 fp3_mul_art(t, a);
191
192 int cnr = fp3_field_get_cnr();
193 cnr = (cnr < 0 ? -cnr : cnr);
194 switch (fp_prime_get_mod18()) {
195 case 1:
196 case 7:
197 if (cnr != 0) {
198 fp3_copy(u, a);
199 while (cnr > 1) {
200 fp3_dbl(u, u);
201 if (cnr & 1) {
202 fp3_add(u, u, a);
203 }
204 cnr = cnr >> 1;
205 }
206 if (fp3_field_get_cnr() > 0) {
207 fp3_add(t, t, u);
208 } else {
209 fp3_sub(t, t, u);
210 }
211 }
212 break;
213 }
214
215 fp3_copy(c, t);
216 }
217 RLC_CATCH_ANY {
218 RLC_THROW(ERR_CAUGHT);
219 }
220 RLC_FINALLY {
221 fp3_free(t);
222 fp3_free(u);
223 }

226void fp3_mul_frb(fp3_t c, const fp3_t a, int i, int j) {
227 ctx_t *ctx = core_get();
228
229 fp3_copy(c, a);
230 if (i % 3 == 0) {
231 if (j % 3 == 1) {
232 fp_mul(c[1], c[1], ctx->fp3_p0[0]);
233 fp_mul(c[2], c[2], ctx->fp3_p0[1]);
234 }
235 if (j % 3 == 2) {
236 fp_mul(c[1], c[1], ctx->fp3_p0[1]);
237 fp_mul(c[2], c[2], ctx->fp3_p0[0]);
238 }
239 }
240
241 if (fp3_field_get_cnr() == 0) {
242 switch (i % 3) {
243 case 1:
244 fp_mul(c[0], c[0], ctx->fp3_p1[j - 1][0]);
245 fp_mul(c[1], c[1], ctx->fp3_p1[j - 1][0]);
246 fp_mul(c[2], c[2], ctx->fp3_p1[j - 1][0]);
247 for (int k = 0; k < (j * ctx->frb3[0]) % 3; k++) {
248 fp3_mul_nor(c, c);
249 }
250 break;
251 case 2:
252 fp_mul(c[0], c[0], ctx->fp3_p2[j - 1][0]);
253 fp_mul(c[1], c[1], ctx->fp3_p2[j - 1][0]);
254 fp_mul(c[2], c[2], ctx->fp3_p2[j - 1][0]);
255 for (int k = 0; k < ctx->frb3[j]; k++) {
256 fp3_mul_nor(c, c);
257 }
258 break;
259 }
260 } else {
261#if ALLOC == AUTO
262 switch (i) {
263 case 1:
264 fp3_mul(c, c, ctx->fp3_p1[j - 1]);
265 break;
266 case 2:
267 fp3_mul(c, c, ctx->fp3_p2[j - 1]);
268 break;
269 }
270#else
271 fp3_t t;
272
273 fp3_null(t);
274
275 RLC_TRY {
276 fp3_new(t);
277
278 switch (i) {
279 case 1:
280 fp_copy(t[0], ctx->fp3_p1[j - 1][0]);
281 fp_copy(t[1], ctx->fp3_p1[j - 1][1]);
282 fp_copy(t[2], ctx->fp3_p1[j - 1][2]);
283 fp3_mul(c, c, t);
284 break;
285 case 2:
286 fp_copy(t[0], ctx->fp3_p2[j - 1][0]);
287 fp_copy(t[1], ctx->fp3_p2[j - 1][1]);
288 fp_copy(t[2], ctx->fp3_p2[j - 1][2]);
289 fp3_mul(c, c, t);
290 break;
291 }
292 }
293 RLC_CATCH_ANY {
294 RLC_THROW(ERR_CAUGHT);
295 }
296 RLC_FINALLY {
297 fp3_free(t);
298 }
299#endif
300 }
