42 void fp3_mul_basic(fp3_t c, fp3_t a, fp3_t b) {
43  dv_t t, t0, t1, t2, t3, t4, t5, t6;
44 
45  dv_null(t);
46  dv_null(t0);
47  dv_null(t1);

50  dv_null(t4);

52  dv_null(t6);
53 
54  TRY {
55  dv_new(t);
56  dv_new(t0);
57  dv_new(t1);
58  dv_new(t2);
59  dv_new(t3);
60  dv_new(t4);
61  dv_new(t5);
62  dv_new(t6);
63 
64  /* Karatsuba algorithm. */
65 
66  /* t0 = a_0 * b_0, t1 = a_1 * b_1, t2 = a_2 * b_2. */
67  fp_muln_low(t0, a[0], b[0]);
68  fp_muln_low(t1, a[1], b[1]);
69  fp_muln_low(t2, a[2], b[2]);
70 
71  /* t3 = (a_1 + a_2) * (b_1 + b_2). */
72  fp_add(t3, a[1], a[2]);
73  fp_add(t4, b[1], b[2]);
74  fp_muln_low(t, t3, t4);
75 #ifdef RLC_FP_ROOM
76  fp_addd_low(t6, t1, t2);
77 #else
78  fp_addc_low(t6, t1, t2);
79 #endif
80  fp_subc_low(t4, t, t6);
81  fp_addc_low(t3, t0, t4);
82  for (int i = 1; i < fp_prime_get_cnr(); i++) {
83  fp_addc_low(t3, t3, t4);
84  }
85  for (int i = 0; i >= fp_prime_get_cnr(); i--) {
86  fp_subc_low(t3, t3, t4);
87  }
88 
89  fp_add(t4, a[0], a[1]);
90  fp_add(t5, b[0], b[1]);
91  fp_muln_low(t, t4, t5);
92 #ifdef RLC_FP_ROOM
93  fp_addd_low(t4, t0, t1);
94 #else
95  fp_addc_low(t4, t0, t1);
96 #endif
97  fp_subc_low(t4, t, t4);
98  fp_addc_low(t4, t4, t2);
99  for (int i = 1; i < fp_prime_get_cnr(); i++) {
100  fp_addc_low(t4, t4, t2);
101  }
102  for (int i = 0; i >= fp_prime_get_cnr(); i--) {
103  fp_subc_low(t4, t4, t2);
104  }
105 
106  fp_add(t5, a[0], a[2]);
107  fp_add(t6, b[0], b[2]);
108  fp_muln_low(t, t5, t6);
109 #ifdef RLC_FP_ROOM
110  fp_addd_low(t6, t0, t2);
111 #else
112  fp_addc_low(t6, t0, t2);
113 #endif
114  fp_addc_low(t6, t0, t2);
115  fp_subc_low(t5, t, t6);
116  fp_addc_low(t5, t5, t1);
117 
118  /* c_0 = t3 mod p. */
119  fp_rdc(c[0], t3);
120 
121  /* c_1 = t4 mod p. */
122  fp_rdc(c[1], t4);
123 
124  /* c_2 = t5 mod p. */
125  fp_rdc(c[2], t5);
126  }
127  CATCH_ANY {
128  THROW(ERR_CAUGHT);
129  }
130  FINALLY {
131  dv_free(t);
132  dv_free(t0);
133  dv_free(t1);
134  dv_free(t2);
135  dv_free(t3);
136  dv_free(t4);
137  dv_free(t5);
138  dv_free(t6);
139  }
140 }

152 void fp3_mul_nor(fp3_t c, fp3_t a) {
153  fp_t t;
154 
155  fp_null(t);
156 
157  TRY {
158  fp_new(t);
159 
160  /* (a_0 + a_1 * u + a_2 * u^2) * u = a_0 * u + a_1 * u^2 + a_2 * u^3. */
161  fp_copy(t, a[0]);
162  fp_copy(c[0], a[2]);
163  for (int i = 1; i < fp_prime_get_cnr(); i++) {
164  fp_add(c[0], c[0], a[2]);
165  }
166  for (int i = 0; i >= fp_prime_get_cnr(); i--) {
167  fp_sub(c[0], c[0], a[2]);
168  }
169  fp_copy(c[2], a[1]);
170  fp_copy(c[1], t);
171  }
172  CATCH_ANY {
173  THROW(ERR_CAUGHT);
174  }
175  FINALLY {
176  fp_free(t);
177  }
178 }

180 void fp3_mul_frb(fp3_t c, fp3_t a, int i, int j) {
181  ctx_t *ctx = core_get();
182  fp3_copy(c, a);
183  switch (i % 3) {
184  case 0:
185  if (j % 3 == 1) {
186  fp_mul(c[1], c[1], ctx->fp3_p0[0]);
187  fp_mul(c[2], c[2], ctx->fp3_p0[1]);
188  }
189  if (j % 3 == 2) {
190  fp_mul(c[1], c[1], ctx->fp3_p0[1]);
191  fp_mul(c[2], c[2], ctx->fp3_p0[0]);
192  }
193  break;
194  case 1:
195  fp_mul(c[0], c[0], ctx->fp3_p1[j - 1]);
196  fp_mul(c[1], c[1], ctx->fp3_p1[j - 1]);
197  fp_mul(c[2], c[2], ctx->fp3_p1[j - 1]);
198  for (int k = 0; k < (j * ctx->frb3[0]) % 3; k++) {
199  fp3_mul_nor(c, c);
200  }
201  break;
202  case 2:
203  fp_mul(c[0], c[0], ctx->fp3_p2[j - 1]);
204  fp_mul(c[1], c[1], ctx->fp3_p2[j - 1]);
205  fp_mul(c[2], c[2], ctx->fp3_p2[j - 1]);
206  for (int k = 0; k < ctx->frb3[j]; k++) {
207  fp3_mul_nor(c, c);
208  }
209  break;
210  }
