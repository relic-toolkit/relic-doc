82 int g1_is_valid(g1_t a) {
83  bn_t n;
84  g1_t u;
85  int r;
86 
87  bn_null(n);
88  g1_null(u);
89 
90  TRY {
91  bn_new(n);
92  g1_new(u);
93 
94  ep_curve_get_cof(n);
95  if (bn_cmp_dig(n, 1) == RLC_EQ) {
96  /* If curve has prime order, simpler to check if point on curve. */
97  return ep_is_valid(a);
98  } else {
99  /* Otherwise, check order explicitly. */
100  g1_get_ord(n);
101  /* Multiply by (n-1)/2 to prevent weird interactions with recoding. */
102  bn_add_dig(n, n, 1);
103  bn_hlv(n, n);
104  g1_mul(u, a, n);
105  g1_dbl(u, u);
106  r = (g1_cmp(u, a) == RLC_EQ);
107  }
108  } CATCH_ANY {
109  THROW(ERR_CAUGHT);
110  } FINALLY {
111  bn_free(n);
112  g1_free(u);
113  }
114 
115  return r;
116 }

118 int g2_is_valid(g2_t a) {
119  bn_t p, n;
120  g2_t u, v;
121  int r;
122 
123  bn_null(n);
124  bn_null(p);
125  g2_null(u);
126  g2_null(v);
127 
128  TRY {
129  bn_new(n);
130  bn_new(p);
131  g2_new(u);
132  g2_new(v);
133 
134  g2_get_ord(n);
135  ep_curve_get_cof(p);
136 
137  if (bn_cmp_dig(p, 1) == RLC_EQ) {
138  /* Trick for curves of prime order or subgroup-secure. */
139  bn_mul(n, n, p);
140  dv_copy(p->dp, fp_prime_get(), RLC_FP_DIGS);
141  p->used = RLC_FP_DIGS;
142  p->sign = RLC_POS;
143  /* Compute trace t = p - n + 1. */
144  bn_sub(n, p, n);
145  bn_add_dig(n, n, 1);
146  /* Compute u = a^t. */
147  g2_mul(u, a, n);
148  /* Compute v = a^(p + 1). */
149  ep2_frb(v, a, 1);
150  g2_add(v, v, a);
151  /* Check if a^(p + 1) = a^t. */
152  r = (g2_cmp(u, v) == RLC_EQ);
153  } else {
154  /* Common case. */
155  bn_sub_dig(n, n, 1);
156  bn_hlv(n, n);
157  g2_mul(u, a, n);
158  g2_dbl(u, u);
159  g2_neg(u, u);
160  r = (g2_cmp(u, a) == RLC_EQ);
161  }
162  } CATCH_ANY {
163  THROW(ERR_CAUGHT);
164  } FINALLY {
165  bn_free(p);
166  bn_free(n);
167  g2_free(u);
168  g2_free(v);
169  }
170 
171  return r;
172 }
