50void gt_rand(gt_t a) {
51 gt_rand_imp(a);
52#if FP_PRIME < 1536
53#if FP_PRIME == 575
54 pp_exp_k48(a, a);
55#elif FP_PRIME == 315 || FP_PRIME == 317 || FP_PRIME == 509
56 pp_exp_k24(a, a);
57#elif FP_PRIME == 330 || FP_PRIME == 510 || FP_PRIME == 765 || FP_PRIME == 766
58 pp_exp_k16(a, a);
59#elif FP_PRIME == 354 || FP_PRIME == 508 || FP_PRIME == 768 || FP_PRIME == 638 && !defined(FP_QNRES)
60 pp_exp_k18(a, a);
61#elif FP_PRIME == 544
62 pp_exp_k8(a, a);
63#else
64 pp_exp_k12(a, a);
65#endif
66#else
67#if FP_PRIME == 1536
68 pp_exp_k2(a, a);
69#else
70 pp_exp_k1(a, a);
71#endif
72#endif
73}

79int g1_is_valid(const g1_t a) {
80 bn_t n, t;
81 g1_t u, v, w;
82 size_t l0, l1, r = 0;
83 int8_t naf0[RLC_FP_BITS + 1], naf1[RLC_FP_BITS + 1];
84
85 if (g1_is_infty(a)) {
86 return 0;
87 }
88
89 bn_null(n);
90 bn_null(t);
91 g1_null(u);
92 g1_null(v);
93 g1_null(w);
94
95 RLC_TRY {
96 bn_new(n);
97 bn_new(t);
98 g1_new(u);
99 g1_new(v);
100 g1_new(w);
101
102 ep_curve_get_cof(n);
103 if (bn_cmp_dig(n, 1) == RLC_EQ) {
104 /* If curve has prime order, simpler to check if point on curve. */
105 r = g1_on_curve(a);
106 } else {
107 fp_prime_get_par(n);
108 switch (ep_curve_is_pairf()) {
109#if defined(EP_ENDOM)
110 /* Formulas from "Co-factor clearing and subgroup membership
111 * testing on pairing-friendly curves" by El Housni, Guillevic,
112 * Piellard. https://eprint.iacr.org/2022/352.pdf */
113 case EP_B12:
114 case EP_B24:
115 case EP_B48:
116 /* Check [\psi(P) == [z^2 - 1]P. */
117 bn_sqr(n, n);
118 if (ep_curve_is_pairf() == EP_B24) {
119 /* Check [\psi(P) == [z^4 - 1]P. */
120 bn_sqr(n, n);
121 }
122 if (ep_curve_is_pairf() == EP_B48) {
123 /* Check [\psi(P) == [z^8 - 1]P. */
124 bn_sqr(n, n);
125 bn_sqr(n, n);
126 }
127 bn_sub_dig(n, n, 1);
128 g1_mul_any(u, a, n);
129 ep_psi(v, a);
130 r = g1_on_curve(a) && (g1_cmp(v, u) == RLC_EQ);
131 break;
132 /* if (u % 2) == 0, check (u**4)*\psi(P) == P
133 * else check (u**4-1)//2 * (\psi(P) - P) == P */
134 case EP_AFG16:
135 bn_sqr(n, n);
136 bn_sqr(n, n);
137 ep_psi(u, a);
138 if (!bn_is_even(n)) {
139 bn_sub_dig(n, n, 1);
140 bn_hlv(n, n);
141 g1_sub(u, u, a);
142 g1_norm(u, u);
143 }
144 g1_mul_any(u, u, n);
145 r = g1_on_curve(a) && (g1_cmp(u, a) == RLC_EQ);
146 break;
147 /* Formulas from "Fast Subgroup Membership Testings on Pairing-
148 * friendly Curves" by Yu Dai, Kaizhan Lin, Chang-An Zhao,
149 * Zijian Zhou. https://eprint.iacr.org/2022/348.pdf */
150 case EP_K16:
151 /* If u = 25 or 45 mod 70 then a1 = ((u//5)**4 + 5)//14
152 * is an integer by definition. */
153 bn_div_dig(n, n, 5);
154 bn_sqr(n, n);
155 bn_sqr(n, n);
156 bn_add_dig(n, n, 5);
157 bn_div_dig(n, n, 14);
158 bn_mul_dig(n, n, 17);
159 bn_neg(n, n);
160 bn_add_dig(n, n, 6);
161 /* Compute P1 = a1*P. */
162 g1_mul_any(w, a, n);
163 /* Compute \psi([17]P1) - [31]P1 */
164 g1_dbl(u, w);
165 g1_dbl(u, u);
166 g1_dbl(u, u);
167 g1_dbl(v, u);
168 g1_add(u, v, w);
169 g1_dbl(v, v);
170 g1_sub(v, v, w);
171 ep_psi(u, u);
172 g1_add(u, u, v);
173 g1_neg(u, u);
174 r = g1_on_curve(a) && (g1_cmp(u, a) == RLC_EQ);
175 break;
176 case EP_FM16:
177 /* Check that P == (u**4)*\phi(P). */
178 g1_mul_any(u, a, n);
179 g1_mul_any(u, u, n);
180 g1_mul_any(u, u, n);
181 g1_mul_any(u, u, n);
182 ep_psi(u, u);
183 r = g1_on_curve(a) && (g1_cmp(u, a) == RLC_EQ);
184 break;
185 case EP_K18:
186 /* Check that [a_0]P + [a_1]\psi(P)) == O, for
187 * a_0 = 19a_1 + 1, a_1 = (x/7)^3 */
188 bn_div_dig(n, n, 7);
189 bn_sqr(t, n);
190 bn_mul(n, n, t);
191 bn_mul_dig(t, n, 19);
192 bn_add_dig(t, t, 1);
193 ep_psi(v, a);
194
195 l0 = l1 = RLC_FP_BITS + 1;
196 bn_rec_naf(naf0, &l0, t, 2);
197 bn_rec_naf(naf1, &l1, n, 2);
198
199 ep_copy(u, a);
200 for (int i = RLC_MAX(l0, l1) - 2; i >= 0; i--) {
201 g1_dbl(u, u);
202 if (naf0[i] > 0) {
203 g1_add(u, u, a);
204 }
205 if (naf0[i] < 0) {
206 g1_sub(u, u, a);
207 }
208 if (naf1[i] > 0) {
209 g1_add(u, u, v);
210 }
211 if (naf1[i] < 0) {
212 g1_sub(u, u, v);
213 }
214 }
215 if (bn_sign(n) == RLC_NEG) {
216 g1_neg(u, u);
217 }
218 r = g1_on_curve(a) && g1_is_infty(u);
219 break;
220 case EP_FM18:
221 /* Check that [u^3 - 1]P == \psi(P). */
222 ep_psi(u, a);
223 bn_sqr(t, n);
224 bn_mul(t, t, n);
225 bn_sub_dig(t, t, 1);
226 g1_mul_any(v, a, t);
227 r = g1_on_curve(a) && (g1_cmp(u, v) == RLC_EQ);
228 break;
229 case EP_SG18:
230 /* Check that [9u^3+2]\psi(P) == -P. */
231 /* Apply \psi twice to get the other beta. */
232 ep_psi(u, a);
233 ep_psi(u, u);
234 g1_mul_any(v, u, n);
235 g1_mul_any(v, v, n);
236 g1_mul_any(v, v, n);
237 g1_mul_dig(v, v, 9);
238 g1_dbl(u, u);
239 g1_add(v, v, u);
240 g1_norm(v, v);
241 g1_neg(v, v);
242 r = g1_on_curve(a) && (g1_cmp(a, v) == RLC_EQ);
243 break;
244#endif
245 default:
246 pc_get_ord(n);
247 bn_sub_dig(n, n, 1);
248 /* Otherwise, check order explicitly. */
249 g1_mul_any(u, a, n);
250 g1_neg(u, u);
251 r = g1_on_curve(a) && (g1_cmp(u, a) == RLC_EQ);
252 break;
253 }
254 }
255 } RLC_CATCH_ANY {
256 RLC_THROW(ERR_CAUGHT);
257 } RLC_FINALLY {
258 bn_free(n);
259 bn_free(t);
260 g1_free(u);
261 g1_free(v);
262 g1_free(w);
263 }
264
265 return r;
266}

268int g2_is_valid(const g2_t a) {
269#if FP_PRIME >= 1536
270 return g1_is_valid(a);
271#else
272 g2_t s, t, u, v, w;
273 bn_t n;
274 dig_t rem;
275 int r = 0;
276
277 if (g2_is_infty(a)) {
278 return 0;
279 }
280
281 bn_null(n);
282 g2_null(s);
283 g2_null(t);
284 g2_null(u);
285 g2_null(v);
286 g2_null(w);
287
288 RLC_TRY {
289 bn_new(n);
290 g2_new(s);
291 g2_new(t);
292 g2_new(u);
293 g2_new(v);
294 g2_new(w);
295
296 fp_prime_get_par(n);
297 switch (ep_curve_is_pairf()) {
298#if defined(EP_ENDOM)
299 /* Formulas from "Co-factor clearing and subgroup membership
300 * testing on pairing-friendly curves" by El Housni, Guillevic,
301 * Piellard. https://eprint.iacr.org/2022/352.pdf */
302 case EP_B12:
303 case EP_B24:
304 case EP_B48:
305 if (core_get()->ep_id == B12_383) {
306 /* Since p mod n = r, we can check instead that
307 * psi^4(P) + P == \psi^2(P). */
308 g2_frb(u, a, 4);
309 g2_add(u, u, a);
310 g2_frb(v, a, 2);
311 } else {
312 /* Check \psi(P) == [z]P. */
313 fp_prime_get_par(n);
314 g2_mul_any(u, a, n);
315 g2_frb(v, a, 1);
316 }
317 r = g2_on_curve(a) && (g2_cmp(u, v) == RLC_EQ);
318 break;
319 /* Formulas from "Fast Subgroup Membership Testings for G1,
320 * G2 and GT on Pairing-friendly Curves" by Dai et al.
321 * https://eprint.iacr.org/2022/348.pdf */
322 case EP_BN:
323 /*Check that [z+1]P+[z]\psi(P)+[z]\psi^2(P)=[2z]\psi^3(P)*/
324 g2_mul_any(u, a, n);
325 g2_frb(v, u, 1);
326 g2_add(u, u, a);
327 g2_add(u, u, v);
328 g2_frb(v, v, 1);
329 g2_add(u, u, v);
330 g2_frb(v, v, 1);
331 g2_dbl(v, v);
332 r = g2_on_curve(a) && (g2_cmp(u, v) == RLC_EQ);
333 break;
334 /* If u is even, check that [u*p^3]P = P
335 * else check [p^5]P = [u]P. */
336 case EP_AFG16:
337 g2_mul_any(u, a, n);
338 if (bn_is_even(n)) {
339 g2_frb(v, u, 3);
340 g2_copy(u, a);
341 } else {
342 g2_frb(v, a, 5);
343 }
344 r = g2_on_curve(a) && (g2_cmp(u, v) == RLC_EQ);
345 break;
346 /* Formulas from "Fast Subgroup Membership Testings for G1,
347 * G2 and GT on Pairing-friendly Curves" by Dai et al.
348 * https://eprint.iacr.org/2022/348.pdf
349 * Paper has u = 45 mod 70, we ran their code for u = 25 mod 70. */
350 case EP_K16:
351 bn_mod_dig(&rem, n, 70);
352 if (rem == 45) {
353 bn_neg(n, n);
354 }
355 /* Compute s = (\pm u - 25)/70. */
356 bn_sub_dig(n, n, 25);
357 bn_div_dig(n, n, 70);
358 /* [11s+4, 9s+3, 3s+1, -(3s+1), -13*u-5, -7*u-3, u, -11s-4] */
359 /* or [11s+4, -9s-3, 3s+1, 3s+1, -13*u-5, 7*u+3, u, 11s+4]. */
360 g2_mul_any(u, a, n); /* u = a^s*/
361 g2_frb(w, u, 6);
362 g2_dbl(s, u);
363 g2_add(v, s, a);
364 g2_add(t, v, u); /* t = a^(3s + 1) */
365 g2_copy(u, v); /* u = a^(2s + 1)*/
366 g2_frb(v, t, 2);
367 g2_add(w, w, v);
368 g2_frb(v, t, 3);
369 if (rem == 45) {
370 g2_add(w, w, v);
371 } else {
372 g2_sub(w, w, v);
373 }
374 g2_dbl(v, t);
375 g2_add(t, t, v); /* t = a^(9s + 3). */
376 g2_frb(v, t, 1);
377 if (rem == 45) {
378 g2_neg(v, v);
379 }
380 g2_add(w, w, v);
381 g2_sub(s, t, s); /* s = a^(7s + 3). */
382 g2_frb(v, s, 5);
383 if (rem == 45) {
384 g2_add(w, w, v);
385 } else {
386 g2_sub(w, w, v);
387 }
388 g2_add(t, t, u); /* t = a^(11s + 4). */
389 g2_add(w, w, t);
390 g2_frb(v, t, 7);
391 if (rem == 45) {
392 g2_add(w, w, v);
393 } else {
394 g2_sub(w, w, v);
395 }
396 g2_add(t, t, u); /* t = a^(13s + 5). */
397 g2_frb(t, t, 4);
398 r = g2_on_curve(a) && (g2_cmp(w, t) == RLC_EQ);
399 break;
400 case EP_FM16:
401 /* Check that u*Q == psi(Q). */
402 g2_mul_any(u, a, n);
403 g2_frb(v, a, 1);
404 r = g2_on_curve(a) && (g2_cmp(u, v) == RLC_EQ);
405 break;
406 case EP_K18:
407 /* Check that P + u*psi2P + 2*psi3P == \mathcal{O}. */
408 g2_frb(u, a, 2);
409 g2_frb(v, u, 1);
410 g2_dbl(v, v);
411 g2_mul_any(u, u, n);
412 g2_add(v, v, u);
413 g2_neg(u, v);
414 r = g2_on_curve(a) && (g2_cmp(u, a) == RLC_EQ);
415 break;
416 case EP_FM18:
417 /* Check that Q == -u*\psi^2(Q). */
418 bn_neg(n, n);
419 g2_mul_any(u, a, n);
420 g2_frb(u, u, 2);
421 r = g2_on_curve(a) && (g2_cmp(u, a) == RLC_EQ);
422 break;
423 case EP_SG18:
424 /* Check that 3u*P + 2\psi^2(P) == \psi^5P] and [3]P \eq O. */
425 bn_mul_dig(n, n, 3);
426 g2_mul_any(u, a, n);
427 r = g2_is_infty(a) == 0;
428 g2_frb(v, a, 2);
429 g2_add(u, u, v);
430 g2_add(u, u, v);
431 g2_frb(v, a, 5);
432 r &= g2_on_curve(a) && (g2_cmp(u, v) == RLC_EQ);
433 break;
434#endif
435 default:
436 pc_get_ord(n);
437 bn_sub_dig(n, n, 1);
438 /* Otherwise, check order explicitly. */
439 g2_mul_any(u, a, n);
440 g2_neg(u, u);
441 r = g2_on_curve(a) && (g2_cmp(u, a) == RLC_EQ);
442 break;
443 }
444 } RLC_CATCH_ANY {
445 RLC_THROW(ERR_CAUGHT);
446 } RLC_FINALLY {
447 bn_free(n);
448 g2_free(s);
449 g2_free(t);
450 g2_free(u);
451 g2_free(v);
452 g2_free(w);
453 }
454
455 return r;
456#endif
457}

459int gt_is_valid(const gt_t a) {
460 bn_t n;
461 gt_t s, t, u, v, w;
462 int l, r = 0;
463 const int *b;
464 dig_t rem;
465
466 if (gt_is_unity(a)) {
467 return 0;
468 }
469
470 bn_null(n);
471 gt_null(s);
472 gt_null(t);
473 gt_null(u);
474 gt_null(v);
475 gt_null(w);
476
477 RLC_TRY {
478 bn_new(n);
479 gt_new(s);
480 gt_new(t);
481 gt_new(u);
482 gt_new(v);
483 gt_new(w);
484
485 fp_prime_get_par(n);
486 b = fp_prime_get_par_sps(&l);
487 switch (ep_curve_is_pairf()) {
488 /* Formulas from "Families of SNARK-friendly 2-chains of
489 * elliptic curves" by Housni and Guillevic.
490 * https://eprint.iacr.org/2021/1359.pdf */
491 case EP_B12:
492 if (core_get()->ep_id == B12_383) {
493 /* GT-strong, so test for cyclotomic only. */
494 r = 1;
495 } else {
496 /* Check that a^u = a^p. */
497 gt_frb(u, a, 1);
498 fp12_exp_cyc_sps((void *)v, (void *)a, b, l, bn_sign(n));
499 r = (gt_cmp(u, v) == RLC_EQ);
500 }
501 r &= fp12_test_cyc((void *)a);
502 break;
503 case EP_B24:
504 /* Check that a^u = a^p. */
505 gt_frb(u, a, 1);
506 fp24_exp_cyc_sps((void *)v, (void *)a, b, l, bn_sign(n));
507 r = (gt_cmp(u, v) == RLC_EQ);
508 r &= fp24_test_cyc((void *)a);
509 break;
510 case EP_B48:
511 /* Check that a^u = a^p. */
512 gt_frb(u, a, 1);
513 fp48_exp_cyc_sps((void *)v, (void *)a, b, l, bn_sign(n));
514 r = (gt_cmp(u, v) == RLC_EQ);
515 r &= fp48_test_cyc((void *)a);
516 break;
517 /* Formulas from "Fast Subgroup Membership Testings for G1,
518 * G2 and GT on Pairing-friendly Curves" by Dai et al.
519 * https://eprint.iacr.org/2022/348.pdf */
520 case EP_BN:
521 /*Check that [z+1]P+[z]\psi(P)+[z]\psi^2(P)=[2z]\psi^3(P)*/
522 fp12_exp_cyc_sps((void *)u, (void *)a, b, l, bn_sign(n));
523 gt_frb(v, u, 1);
524 gt_mul(u, u, a);
525 gt_mul(u, u, v);
526 gt_frb(v, v, 1);
527 gt_mul(u, u, v);
528 gt_frb(v, v, 1);
529 gt_sqr(v, v);
530 r = (gt_cmp(u, v) == RLC_EQ);
531 r &= fp12_test_cyc((void *)a);
532 break;
533 /* If u is even, check that [u*p^3]P = P
534 * else check [p^5]P = [u]P. */
535 case EP_AFG16:
536 fp_prime_get_par(n);
537 gt_exp(u, a, n);
538 if (bn_is_even(n)) {
539 gt_frb(v, u, 3);
540 gt_copy(u, a);
541 } else {
542 gt_frb(v, a, 5);
543 }
544 r = (gt_cmp(u, v) == RLC_EQ);
545 r &= fp16_test_cyc((void *)a);
546 break;
547 case EP_K16:
548 fp_prime_get_par(n);
549 bn_mod_dig(&rem, n, 70);
550 if (rem == 45) {
551 bn_neg(n, n);
552 }
553 /* Compute s = (u - 25)/70. */
554 bn_sub_dig(n, n, 25);
555 bn_div_dig(n, n, 70);
556 /* Vectors for u = 25 or 45 mod 70 below, respectively: */
557 /* [11s+4, 9s+3, 3s+1, -(3s+1), -13*u-5, -7*u-3, u, -11s-4] */
558 /* or [11s+4, -9s-3, 3s+1, 3s+1, -13*u-5, 7*u+3, u, 11s+4]. */
559 gt_exp(u, a, n); /* u = a^s*/
560 gt_frb(w, u, 6);
561 gt_sqr(s, u);
562 gt_mul(v, s, a);
563 gt_mul(t, v, u); /* t = a^(3s + 1) */
564 gt_copy(u, v); /* u = a^(2s + 1)*/
565 gt_frb(v, t, 2);
566 gt_mul(w, w, v);
567 gt_frb(v, t, 3);
568 if (rem != 45) {
569 gt_inv(v, v);
570 }
571 gt_mul(w, w, v);
572 gt_sqr(v, t);
573 gt_mul(t, t, v); /* t = a^(9s + 3). */
574 gt_frb(v, t, 1);
575 if (rem == 45) {
576 gt_inv(v, v);
577 }
578 gt_mul(w, w, v);
579 gt_inv(s, s);
580 gt_mul(s, t, s); /* s = a^(7s + 3). */
581 gt_frb(v, s, 5);
582 if (rem != 45) {
583 gt_inv(v, v);
584 }
585 gt_mul(w, w, v);
586 gt_mul(t, t, u); /* t = a^(11s + 4). */
587 gt_mul(w, w, t);
588 gt_frb(v, t, 7);
589 if (rem != 45) {
590 gt_inv(v, v);
591 }
592 gt_mul(w, w, v);
593 gt_mul(t, t, u); /* t = a^(13s + 5). */
594 gt_frb(t, t, 4);
595 r = (gt_cmp(w, t) == RLC_EQ);
596 r &= fp16_test_cyc((void *)a);
597 break;
598 case EP_FM16:
599 /* Check that u*Q == psi(Q). */
600 fp_prime_get_par(n);
601 gt_exp(u, a, n);
602 gt_frb(v, a, 1);
603 r = (gt_cmp(u, v) == RLC_EQ);
604 r &= fp16_test_cyc((void *)a);
605 break;
606 case EP_K18:
607 /* Check that P + u*psi2P + 2*psi3P == \mathcal{O}. */
608 gt_frb(u, a, 2);
