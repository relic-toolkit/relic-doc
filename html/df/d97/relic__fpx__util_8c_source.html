75 void fp2_read_bin(fp2_t a, const uint8_t *bin, int len) {
76  if (len != RLC_FP_BYTES + 1 && len != 2 * RLC_FP_BYTES) {
77  THROW(ERR_NO_BUFFER);
78  }
79  if (len == RLC_FP_BYTES + 1) {
80  fp_read_bin(a[0], bin, RLC_FP_BYTES);
81  fp_zero(a[1]);
82  fp_set_bit(a[1], 0, bin[RLC_FP_BYTES]);
83  fp2_upk(a, a);
84  }
85  if (len == 2 * RLC_FP_BYTES) {
86  fp_read_bin(a[0], bin, RLC_FP_BYTES);
87  fp_read_bin(a[1], bin + RLC_FP_BYTES, RLC_FP_BYTES);
88  }
89 }

91 void fp2_write_bin(uint8_t *bin, int len, fp2_t a, int pack) {
92  fp2_t t;
93 
94  fp2_null(t);
95 
96  TRY {
97  fp2_new(t);
98 
99  if (pack && fp2_test_cyc(a)) {
100  if (len < RLC_FP_BYTES + 1) {
101  THROW(ERR_NO_BUFFER);
102  } else {
103  fp2_pck(t, a);
104  fp_write_bin(bin, RLC_FP_BYTES, t[0]);
105  bin[RLC_FP_BYTES] = fp_get_bit(t[1], 0);
106  }
107  } else {
108  if (len < 2 * RLC_FP_BYTES) {
109  THROW(ERR_NO_BUFFER);
110  } else {
111  fp_write_bin(bin, RLC_FP_BYTES, a[0]);
112  fp_write_bin(bin + RLC_FP_BYTES, RLC_FP_BYTES, a[1]);
113  }
114  }
115  } CATCH_ANY {
116  THROW(ERR_CAUGHT);
117  }
118  FINALLY {
119  fp2_free(t);
120  }
121 }

438 void fp12_read_bin(fp12_t a, const uint8_t *bin, int len) {
439  if (len != 8 * RLC_FP_BYTES && len != 12 * RLC_FP_BYTES) {
440  THROW(ERR_NO_BUFFER);
441  }
442  if (len == 8 * RLC_FP_BYTES) {
443  fp2_zero(a[0][0]);
444  fp2_read_bin(a[0][1], bin, 2 * RLC_FP_BYTES);
445  fp2_read_bin(a[0][2], bin + 2 * RLC_FP_BYTES, 2 * RLC_FP_BYTES);
446  fp2_read_bin(a[1][0], bin + 4 * RLC_FP_BYTES, 2 * RLC_FP_BYTES);
447  fp2_zero(a[1][1]);
448  fp2_read_bin(a[1][2], bin + 6 * RLC_FP_BYTES, 2 * RLC_FP_BYTES);
449  fp12_back_cyc(a, a);
450  }
451  if (len == 12 * RLC_FP_BYTES) {
452  fp6_read_bin(a[0], bin, 6 * RLC_FP_BYTES);
453  fp6_read_bin(a[1], bin + 6 * RLC_FP_BYTES, 6 * RLC_FP_BYTES);
454  }
