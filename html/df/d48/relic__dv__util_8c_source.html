73void dv_copy_sec(dig_t *c, const dig_t *a, size_t digits, dig_t bit) {
74 dig_t mask, t;
75
76 mask = -bit;
77 for (size_t i = 0; i < digits; i++) {
78 t = (a[i] ^ c[i]) & mask;
79 c[i] ^= t;

94int dv_cmp(const dig_t *a, const dig_t *b, size_t size) {
95 int r;
96
97 a += (size - 1);
98 b += (size - 1);
99
100 r = RLC_EQ;
101 for (size_t i = 0; i < size; i++, --a, --b) {
102 if (*a != *b && r == RLC_EQ) {
103 r = (*a > *b ? RLC_GT : RLC_LT);
104 }
105 }
106 return r;
107}

109int dv_cmp_sec(const dig_t *a, const dig_t *b, size_t size) {
110 dig_t r = 0;
111
112 for (size_t i = 0; i < size; i++) {
113 r |= a[i] ^ b[i];
114 }
115
116 return (r == 0 ? RLC_EQ : RLC_NE);
117}

119void dv_rshd(dig_t *c, const dig_t *a, size_t size, uint_t digits) {
120 const dig_t *top;
121 dig_t *bot;
122 size_t i;
123
124 top = a + digits;
125 bot = c;
126
127 for (i = 0; i < size - digits; i++, top++, bot++) {
128 *bot = *top;
129 }
130 for (; i < size; i++, bot++) {
131 *bot = 0;
132 }
133}
