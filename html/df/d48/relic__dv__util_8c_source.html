93 int dv_cmp(const dig_t *a, const dig_t *b, int size) {
94  int i, r;
95 
96  a += (size - 1);

98 
99  r = RLC_EQ;
100  for (i = 0; i < size; i++, --a, --b) {

102  r = (*a > *b ? RLC_GT : RLC_LT);
103  }
104  }
