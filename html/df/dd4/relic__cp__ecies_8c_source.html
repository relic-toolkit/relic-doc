93  l = bn_size_bin(x);
94  if (bn_bits(x) % 8 == 0) {
95  /* Compatibility with BouncyCastle. */
96  l = l + 1;
97  }
98  bn_write_bin(_x, l, x);
99  md_kdf(key, 2 * size, _x, l);

102  != RLC_OK || (*out_len + RLC_MD_LEN) > l) {
103  result = RLC_ERR;
104  } else {
105  md_hmac(out + *out_len, out, *out_len, key + size, size);
106  *out_len += RLC_MD_LEN;
107  }
108  }
109  CATCH_ANY {
110  result = RLC_ERR;
111  }
112  FINALLY {
113  bn_free(k);
114  bn_free(n);
115  bn_free(x);
116  ec_free(p);
