42void fp16_mul_basic(fp16_t c, const fp16_t a, const fp16_t b) {
43 fp8_t t0, t1, t4;
44
45 fp8_null(t0);
46 fp8_null(t1);
47 fp8_null(t4);
48
49 RLC_TRY {
50 fp8_new(t0);
51 fp8_new(t1);
52 fp8_new(t4);
53
54 /* Karatsuba algorithm. */
55
56 /* t0 = a_0 * b_0. */
57 fp8_mul(t0, a[0], b[0]);
58 /* t1 = a_1 * b_1. */
59 fp8_mul(t1, a[1], b[1]);
60 /* t4 = b_0 + b_1. */
61 fp8_add(t4, b[0], b[1]);
62
63 /* c_1 = a_0 + a_1. */
64 fp8_add(c[1], a[0], a[1]);
65
66 /* c_1 = (a_0 + a_1) * (b_0 + b_1) */
67 fp8_mul(c[1], c[1], t4);
68 fp8_sub(c[1], c[1], t0);
69 fp8_sub(c[1], c[1], t1);
70
71 /* c_0 = a_0b_0 + v * a_1b_1. */
72 fp8_mul_art(t4, t1);
73 fp8_add(c[0], t0, t4);
74 } RLC_CATCH_ANY {
75 RLC_THROW(ERR_CAUGHT);
76 } RLC_FINALLY {
77 fp8_free(t0);
78 fp8_free(t1);
79 fp8_free(t4);
80 }
81}

83void fp16_mul_dxs_basic(fp16_t c, const fp16_t a, const fp16_t b) {
84 fp8_t t0, t1, t4;
85
86 fp8_null(t0);
87 fp8_null(t1);
88 fp8_null(t4);
89
90 RLC_TRY {
91 fp8_new(t0);
92 fp8_new(t1);
93 fp8_new(t4);
94
95 /* Karatsuba algorithm. */
96
97 if (fp4_is_zero(b[1][0])) {
98 /* t0 = a_0 * b_0. */
99 fp8_mul(t0, a[0], b[0]);
100
101 /* t1 = a_1 * b_1. */
102 fp4_mul(t1[0], a[1][1], b[1][1]);
103 fp4_add(t1[1], a[1][0], a[1][1]);
104 fp4_mul(t1[1], t1[1], b[1][1]);
105 fp4_sub(t1[1], t1[1], t1[0]);
106 fp4_mul_art(t1[0], t1[0]);
107 } else {
108#if EP_ADD == BASIC
109 /* t0 = a_0 * b_0. */
110 for (int i = 0; i < 2; i++) {
111 for (int j = 0; j < 2; j++) {
112 for (int k = 0; k < 2; k++) {
113 fp_mul(t0[i][j][k], a[0][i][j][k], b[0][0][0][0]);
114 }
115 }
116 }
117#else
118 /* t0 = a_0 * b_0. */
119 for (int i = 0; i < 2; i++) {
120 fp4_mul(t0[i], a[0][i], b[0][0]);
121 }
122#endif
123 /* t1 = a_1 * b_1. */
124 fp8_mul(t1, a[1], b[1]);
125 }
126 /* t4 = b_0 + b_1. */
127 fp8_add(t4, b[0], b[1]);
128
129 /* c_1 = a_0 + a_1. */
130 fp8_add(c[1], a[0], a[1]);
131
132 /* c_1 = (a_0 + a_1) * (b_0 + b_1) */
133 fp8_mul(c[1], c[1], t4);
134 fp8_sub(c[1], c[1], t0);
135 fp8_sub(c[1], c[1], t1);
136
137 /* c_0 = a_0b_0 + v * a_1b_1. */
138 fp8_mul_art(t4, t1);
139 fp8_add(c[0], t0, t4);
140 } RLC_CATCH_ANY {
141 RLC_THROW(ERR_CAUGHT);
142 } RLC_FINALLY {
143 fp8_free(t0);
144 fp8_free(t1);
145 fp8_free(t4);
146 }
147}

153void fp16_mul_unr(dv16_t c, const fp16_t a, const fp16_t b) {
154 fp8_t t0, t1;
155 dv8_t u0, u1, u2, u3;
156
157 fp8_null(t0);
158 fp8_null(t1);
159 dv8_null(u0);
160 dv8_null(u1);
161 dv8_null(u2);
162 dv8_null(u3);
163
164 RLC_TRY {
165 fp8_new(t0);
166 fp8_new(t1);
167 dv8_new(u0);
168 dv8_new(u1);
169 dv8_new(u2);
170 dv8_new(u3);
171
172 /* Karatsuba algorithm. */
173
174 /* u0 = a_0 * b_0. */
175 fp8_mul_unr(u0, a[0], b[0]);
176 /* u1 = a_1 * b_1. */
177 fp8_mul_unr(u1, a[1], b[1]);
178 /* t1 = a_0 + a_1. */
179 fp8_add(t0, a[0], a[1]);
180 /* t0 = b_0 + b_1. */
181 fp8_add(t1, b[0], b[1]);
182 /* u2 = (a_0 + a_1) * (b_0 + b_1) */
183 fp8_mul_unr(u2, t0, t1);
184 /* c_1 = u2 - a_0b_0 - a_1b_1. */
185 for (int i = 0; i < 2; i++) {
186 for (int j = 0; j < 2; j++) {
187 fp2_subc_low(c[1][i][j], u2[i][j], u0[i][j]);
188 fp2_subc_low(c[1][i][j], c[1][i][j], u1[i][j]);
189 }
190 }
191 /* c_0 = a_0b_0 + v * a_1b_1. */
192 fp2_nord_low(u2[0][0], u1[1][1]);
193 dv_copy(u2[0][1][0], u1[1][0][0], 2 * RLC_FP_DIGS);
194 dv_copy(u2[0][1][1], u1[1][0][1], 2 * RLC_FP_DIGS);
195 dv_copy(u2[1][0][0], u1[0][0][0], 2 * RLC_FP_DIGS);
196 dv_copy(u2[1][0][1], u1[0][0][1], 2 * RLC_FP_DIGS);
197 dv_copy(u2[1][1][0], u1[0][1][0], 2 * RLC_FP_DIGS);
198 dv_copy(u2[1][1][1], u1[0][1][1], 2 * RLC_FP_DIGS);
199 for (int i = 0; i < 2; i++) {
200 for (int j = 0; j < 2; j++) {
201 fp2_addc_low(c[0][i][j], u0[i][j], u2[i][j]);
202 }
203 }
204 } RLC_CATCH_ANY {
205 RLC_THROW(ERR_CAUGHT);
206 } RLC_FINALLY {
207 fp8_free(t0);
208 fp8_free(t1);
209 dv8_free(u0);
210 dv8_free(u1);
211 dv8_free(u2);
212 dv8_free(u3);
213 }
214}

343void fp16_mul_art(fp16_t c, const fp16_t a) {
344 fp8_t t0;
345
346 fp8_null(t0);
347
348 RLC_TRY {
349 fp8_new(t0);
350
351 /* (a_0 + a_1 * v) * v = a_0 * v + a_1 * v^4 */
352 fp8_copy(t0, a[0]);
353 fp8_mul_art(c[0], a[1]);
354 fp8_copy(c[1], t0);
355 } RLC_CATCH_ANY {
356 RLC_THROW(ERR_CAUGHT);
357 } RLC_FINALLY {
358 fp8_free(t0);
359 }
