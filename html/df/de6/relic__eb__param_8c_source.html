253 void eb_param_set(int param) {

255  fb_t a, b;
256  eb_t g;
257  bn_t r;
258  bn_t h;
259 
260  fb_null(a);
261  fb_null(b);
262  eb_null(g);
263  bn_null(r);
264  bn_null(h);
265 
266  TRY {
267  fb_new(a);
268  fb_new(b);
269  eb_new(g);
270  bn_new(r);
271  bn_new(h);
272 
273  core_get()->eb_id = 0;
274 
275  switch (param) {
276 #if defined(EB_PLAIN) && FB_POLYN == 163

295 #endif

298  ASSIGN(SECG_K239, SECG_239);

323  ASSIGN(NIST_B409, NIST_409);

325 #endif
326 #if defined(EB_KBLTZ) && FB_POLYN == 409
327  case NIST_K409:
328  ASSIGN(NIST_K409, NIST_409);

335 #endif

340 #endif
341  default:
342  (void)str;
343  THROW(ERR_NO_VALID);
344  break;
345  }
346  fb_zero(g->z);
347  fb_set_bit(g->z, 0, 1);
348  g->norm = 1;
349 
350  eb_curve_set(a, b, g, r, h);
351  core_get()->eb_id = param;
352  }
353  CATCH_ANY {
354  THROW(ERR_CAUGHT);
355  }
356  FINALLY {
357  fb_free(a);
358  fb_free(b);
359  eb_free(g);
360  bn_free(r);
361  bn_free(h);
362  }
363 }
