253void eb_param_set(int param) {
254 char str[2 * RLC_FB_BYTES + 1];
255 fb_t a, b;
256 eb_t g;
257 bn_t r;
258 bn_t h;
259
260 fb_null(a);
261 fb_null(b);
262 eb_null(g);
263 bn_null(r);
264 bn_null(h);
265
266 RLC_TRY {
267 fb_new(a);
268 fb_new(b);
269 eb_new(g);
270 bn_new(r);
271 bn_new(h);
272
273 core_get()->eb_id = 0;
274
275 switch (param) {
276#if defined(EB_PLAIN) && FB_POLYN == 163
277 case NIST_B163:
278 ASSIGN(NIST_B163, NIST_163);
279 break;
280#endif
281#if defined(EB_KBLTZ) && FB_POLYN == 163
282 case NIST_K163:
283 ASSIGN(NIST_K163, NIST_163);
284 break;
285#endif
286#if defined(EB_PLAIN) && FB_POLYN == 233
287 case NIST_B233:
288 ASSIGN(NIST_B233, NIST_233);
289 break;
290#endif
291#if defined(EB_KBLTZ) && FB_POLYN == 233
292 case NIST_K233:
293 ASSIGN(NIST_K233, NIST_233);
294 break;
295#endif
296#if defined(EB_KBLTZ) && FB_POLYN == 239
297 case SECG_K239:
298 ASSIGN(SECG_K239, SECG_239);
299 break;
300#endif
301#if defined(EB_PLAIN) && FB_POLYN == 251
302 case EBACS_B251:
303 ASSIGN(EBACS_B251, PENTA_251);
304 break;
305#endif
306#if defined(EB_PLAIN) && FB_POLYN == 257
307 case HALVE_B257:
308 ASSIGN(HALVE_B257, TRINO_257);
309 break;
310#endif
311#if defined(EB_PLAIN) && FB_POLYN == 283
312 case NIST_B283:
313 ASSIGN(NIST_B283, NIST_283);
314 break;
315#endif
316#if defined(EB_KBLTZ) && FB_POLYN == 283
317 case NIST_K283:
318 ASSIGN(NIST_K283, NIST_283);
319 break;
320#endif
321#if defined(EB_PLAIN) && FB_POLYN == 409
322 case NIST_B409:
323 ASSIGN(NIST_B409, NIST_409);
324 break;
325#endif
326#if defined(EB_KBLTZ) && FB_POLYN == 409
327 case NIST_K409:
328 ASSIGN(NIST_K409, NIST_409);
329 break;
330#endif
331#if defined(EB_PLAIN) && FB_POLYN == 571
332 case NIST_B571:
333 ASSIGN(NIST_B571, NIST_571);
334 break;
335#endif
336#if defined(EB_KBLTZ) && FB_POLYN == 571
337 case NIST_K571:
338 ASSIGN(NIST_K571, NIST_571);
339 break;
340#endif
341 default:
342 (void)str;
343 RLC_THROW(ERR_NO_VALID);
344 break;
345 }
346 fb_zero(g->z);
347 fb_set_bit(g->z, 0, 1);
348 g->coord = BASIC;
349
350 eb_curve_set(a, b, g, r, h);
351 core_get()->eb_id = param;
352 }
353 RLC_CATCH_ANY {
354 RLC_THROW(ERR_CAUGHT);
355 }
356 RLC_FINALLY {
357 fb_free(a);
358 fb_free(b);
359 eb_free(g);
360 bn_free(r);
361 bn_free(h);
362 }
363}
