47int cp_shipsi_gen(bn_t g, crt_t crt, size_t bits) {
48 /* Generate different primes p and q. */
49 do {
50 bn_gen_prime(crt->p, bits / 2);
51 bn_gen_prime(crt->q, bits / 2);
52 } while (bn_cmp(crt->p, crt->q) == RLC_EQ);
53
54 /* Swap p and q so that p is smaller. */
55 if (bn_cmp(crt->p, crt->q) != RLC_LT) {
56 bn_copy(g, crt->p);
57 bn_copy(crt->p, crt->q);
58 bn_copy(crt->q, g);
59 }
60
61 /* n = pq. */
62 bn_mul(crt->n, crt->p, crt->q);
63
64 do {
65 bn_rand_mod(g, crt->n);
66 bn_gcd(crt->qi, g, crt->n);
67 } while (bn_cmp_dig(crt->qi, 1) != RLC_EQ);
68
69 /* phi(n) = (p - 1)(q - 1). */
70 bn_sub_dig(crt->dp, crt->p, 1);
71 bn_sub_dig(crt->dq, crt->q, 1);
72 bn_mod_inv(crt->qi, crt->q, crt->p);
73
74 return RLC_OK;
75}

77int cp_shipsi_ask(bn_t d, bn_t r, bn_t p[], const bn_t g, const bn_t n,
78 const bn_t x[], size_t m) {
79 int i, result = RLC_OK, len = RLC_CEIL(RLC_BN_BITS, 8);
80 uint8_t h[RLC_MD_LEN], bin[RLC_CEIL(RLC_BN_BITS, 8)];
81
82 /* Compute R = g^r mod N. */
83 bn_rand_mod(r, n);
84 bn_mxp(d, g, r, n);
85
86 /* Now hash all x_i and accmulate on R. */
87 for (i = 0; i < m; i++) {
88 bn_write_bin(bin, len, x[i]);
89 md_map(h, bin, len);
90 bn_read_bin(p[i], h, 2 * STAT_SEC / 8);
91 if (bn_is_even(p[i])) {
92 bn_add_dig(p[i], p[i], 1);
93 }
94 do {
95 bn_add_dig(p[i], p[i], 2);
96 } while (!bn_is_prime(p[i]));
97 bn_mxp(d, d, p[i], n);
98 }
99
100 return result;
101}

103int cp_shipsi_ans(bn_t t[], bn_t u, bn_t d, const bn_t g, const crt_t crt,
104 const bn_t y[], size_t n) {
105 int j, result = RLC_OK, len = RLC_CEIL(RLC_BN_BITS, 8);
106 uint8_t h[RLC_MD_LEN], bin[RLC_CEIL(RLC_BN_BITS, 8)];
107 uint_t *shuffle = RLC_ALLOCA(uint_t, n);
108 bn_t p, q;
109
110 bn_null(p);
111 bn_null(q);
112
113 RLC_TRY {
114 bn_new(p);
115 bn_new(q);
116 if (shuffle == NULL) {
117 RLC_THROW(ERR_NO_MEMORY);
118 }
119
120 util_perm(shuffle, n);
121
122 bn_rand_mod(u, crt->n);
123 for (j = 0; j < n; j++) {
124 bn_write_bin(bin, len, y[shuffle[j]]);
125 md_map(h, bin, len);
126 bn_read_bin(p, h, 2 * STAT_SEC / 8);
127 if (bn_is_even(p)) {
128 bn_add_dig(p, p, 1);
129 }
130 do {
131 bn_add_dig(p, p, 2);
132 } while (!bn_is_prime(p));
133
134#if !defined(CP_CRT)
135 bn_mul(q, crt->dp, crt->dq);
136 bn_mod_inv(p, p, q);
137 bn_mul(p, p, u);
138 bn_mod(p, p, q);
139 bn_mxp(t[j], d, p, crt->n);
140#else
141 bn_mod_inv(q, p, crt->dq);
142 bn_mul(q, q, u);
143 bn_mod(q, q, crt->dq);
144
145 bn_mod_inv(p, p, crt->dp);
146 bn_mul(p, p, u);
147 bn_mod(p, p, crt->dp);
148
149 bn_mxp_crt(t[j], d, p, q, crt, 0);
150#endif /* CP_CRT */
151 }
152
153#if !defined(CP_CRT)
154 bn_mxp(u, g, u, crt->n);
155#else
156 bn_mod(p, u, crt->dp);
157 bn_mod(q, u, crt->dq);
158
159 bn_mxp_crt(u, g, p, q, crt, 0);
160#endif
161 }
162 RLC_CATCH_ANY {
163 result = RLC_ERR;
164 }
165 RLC_FINALLY {
166 bn_free(p);
167 bn_free(q);
168 RLC_FREE(shuffle);
169 }
170 return result;
171}

173int cp_shipsi_int(bn_t z[], size_t *len, const bn_t r, const bn_t p[],
174 const bn_t n, const bn_t x[], size_t m, const bn_t t[], const bn_t u,
175 size_t l) {
176 int i, j, k, result = RLC_OK;
177 bn_t e, f;
178
179 bn_null(e);
180 bn_null(f);
181
182 RLC_TRY {
183 bn_new(e);
184 bn_new(f);
185
186 *len = 0;
187 if (m > 0) {
188 bn_mxp(f, u, r, n);
189 for (k = 0; k < m; k++) {
190 bn_copy(e, f);
191 for (i = 0; i < m; i++) {
192 if (i != k) {
193 bn_mxp(e, e, p[i], n);
194 }
195 }
196 for (j = 0; j < l; j++) {
197 if (bn_cmp(e, t[j]) == RLC_EQ) {
198 bn_copy(z[*len], x[k]);
199 (*len)++;
200 }
201 }
202 }
203 }
204 }
205 RLC_CATCH_ANY {
206 result = RLC_ERR;
207 }
208 RLC_FINALLY {
209 bn_free(e);
210 bn_free(f);
211 }
212 return result;
