44int ed_cmp(const ed_t p, const ed_t q) {
45 ed_t r, s;
46 int result = RLC_EQ;
47
48 ed_null(r);
49 ed_null(s);
50
51 RLC_TRY {
52 ed_new(r);
53 ed_new(s);
54
55 if ((p->coord != BASIC) && (q->coord != BASIC)) {
56 /* If the two points are not normalized, it is faster to compare
57 * x1 * z2 == x2 * z1 and y1 * z2 == y2 * z1. */
58 fp_mul(r->x, p->x, q->z);
59 fp_mul(s->x, q->x, p->z);
60 fp_mul(r->y, p->y, q->z);
61 fp_mul(s->y, q->y, p->z);
62#if ED_ADD == EXTND
63 fp_mul(r->t, p->t, q->z);
64 fp_mul(s->t, q->t, p->z);
65 if (fp_cmp(r->t, s->t) != RLC_EQ) {
66 result = RLC_NE;
67 }
68#endif
69 } else {
70 ed_copy(r, p);
71 ed_copy(s, q);
72 if (p->coord != BASIC) {
73 ed_norm(r, p);
74 }
75 if (q->coord != BASIC) {
76 ed_norm(s, q);
77 }
78 }
79
80 if (fp_cmp(r->x, s->x) != RLC_EQ) {
81 result = RLC_NE;
82 }
83 if (fp_cmp(r->y, s->y) != RLC_EQ) {
84 result = RLC_NE;
85 }
86 } RLC_CATCH_ANY {
87 RLC_THROW(ERR_CAUGHT);
88 } RLC_FINALLY {
89 ep_free(r);
90 ep_free(s);
91 }
92
93 return result;
94}
