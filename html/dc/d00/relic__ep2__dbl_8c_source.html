49 static void ep2_dbl_basic_imp(ep2_t r, fp2_t s, ep2_t p) {
50  fp2_t t0, t1, t2;
51 
52  fp2_null(t0);
53  fp2_null(t1);
54  fp2_null(t2);
55 
56  TRY {
57  fp2_new(t0);
58  fp2_new(t1);
59  fp2_new(t2);
60 
61  /* t0 = 1/(2 * y1). */
62  fp2_dbl(t0, p->y);
63  fp2_inv(t0, t0);
64 
65  /* t1 = 3 * x1^2 + a. */
66  fp2_sqr(t1, p->x);
67  fp2_copy(t2, t1);
68  fp2_dbl(t1, t1);
69  fp2_add(t1, t1, t2);
70 
71  ep2_curve_get_a(t2);
72  fp2_add(t1, t1, t2);
73 
74  /* t1 = (3 * x1^2 + a)/(2 * y1). */
75  fp2_mul(t1, t1, t0);
76 
77  if (s != NULL) {
78  fp2_copy(s, t1);
79  }
80 
81  /* t2 = t1^2. */
82  fp2_sqr(t2, t1);
83 
84  /* x3 = t1^2 - 2 * x1. */
85  fp2_dbl(t0, p->x);
86  fp2_sub(t0, t2, t0);
87 
88  /* y3 = t1 * (x1 - x3) - y1. */
89  fp2_sub(t2, p->x, t0);
90  fp2_mul(t1, t1, t2);
91 
92  fp2_sub(r->y, t1, p->y);
93 
94  fp2_copy(r->x, t0);
95  fp2_copy(r->z, p->z);
96 
97  r->norm = 1;
98  }
99  CATCH_ANY {
100  THROW(ERR_CAUGHT);
101  }
102  FINALLY {
103  fp2_free(t0);
104  fp2_free(t1);
105  fp2_free(t2);
106  }
107 }

120 static void ep2_dbl_projc_imp(ep2_t r, ep2_t p) {
121  fp2_t t0, t1, t2, t3, t4, t5;
122 
123  fp2_null(t0);
124  fp2_null(t1);
125  fp2_null(t2);
126  fp2_null(t3);
127  fp2_null(t4);
128  fp2_null(t5);
129 
130  TRY {
131  if (ep_curve_opt_a() == RLC_ZERO) {
132  fp2_new(t0);
133  fp2_new(t1);
134  fp2_new(t2);
135  fp2_new(t3);
136  fp2_new(t4);
137  fp2_new(t5);
138 
139  fp2_sqr(t0, p->x);
140  fp2_add(t2, t0, t0);
141  fp2_add(t0, t2, t0);
142 
143  fp2_sqr(t3, p->y);
144  fp2_mul(t1, t3, p->x);
145  fp2_add(t1, t1, t1);
146  fp2_add(t1, t1, t1);
147  fp2_sqr(r->x, t0);
148  fp2_add(t2, t1, t1);
149  fp2_sub(r->x, r->x, t2);
150  fp2_mul(r->z, p->z, p->y);
151  fp2_add(r->z, r->z, r->z);
152  fp2_add(t3, t3, t3);
153 
154  fp2_sqr(t3, t3);
155  fp2_add(t3, t3, t3);
156  fp2_sub(t1, t1, r->x);
157  fp2_mul(r->y, t0, t1);
158  fp2_sub(r->y, r->y, t3);
159  } else {
160  /* dbl-2007-bl formulas: 1M + 8S + 1*a + 10add + 1*8 + 2*2 + 1*3 */
161  /* http://www.hyperelliptic.org/EFD/g1p/auto-shortw-jacobian.html#doubling-dbl-2007-bl */
162 
163  /* t0 = x1^2, t1 = y1^2, t2 = y1^4. */
164  fp2_sqr(t0, p->x);
165  fp2_sqr(t1, p->y);
166  fp2_sqr(t2, t1);
167 
168  if (!p->norm) {
169  /* t3 = z1^2. */
170  fp2_sqr(t3, p->z);
171 
172  if (ep_curve_get_a() == RLC_ZERO) {
173  /* z3 = 2 * y1 * z1. */
174  fp2_mul(r->z, p->y, p->z);
175  fp2_dbl(r->z, r->z);
176  } else {
177  /* z3 = (y1 + z1)^2 - y1^2 - z1^2. */
178  fp2_add(r->z, p->y, p->z);
179  fp2_sqr(r->z, r->z);
180  fp2_sub(r->z, r->z, t1);
181  fp2_sub(r->z, r->z, t3);
182  }
183  } else {
184  /* z3 = 2 * y1. */
185  fp2_dbl(r->z, p->y);
186  }
187 
188  /* t4 = S = 2*((x1 + y1^2)^2 - x1^2 - y1^4). */
189  fp2_add(t4, p->x, t1);
190  fp2_sqr(t4, t4);
191  fp2_sub(t4, t4, t0);
192  fp2_sub(t4, t4, t2);
193  fp2_dbl(t4, t4);
194 
195  /* t5 = M = 3 * x1^2 + a * z1^4. */
196  fp2_dbl(t5, t0);
197  fp2_add(t5, t5, t0);
198  ep2_curve_get_a(t0);
199  if (!p->norm) {
200  fp2_sqr(t3, t3);
201  fp2_mul(t1, t0, t3);
202  fp2_add(t5, t5, t1);
203  } else {
204  fp2_add(t5, t5, t0);
205  }
206 
207  /* x3 = T = M^2 - 2 * S. */
208  fp2_sqr(r->x, t5);
209  fp2_dbl(t1, t4);
210  fp2_sub(r->x, r->x, t1);
211 
212  /* y3 = M * (S - T) - 8 * y1^4. */
213  fp2_dbl(t2, t2);
214  fp2_dbl(t2, t2);
215  fp2_dbl(t2, t2);
216  fp2_sub(t4, t4, r->x);
217  fp2_mul(t5, t5, t4);
218  fp2_sub(r->y, t5, t2);
219  }
220 
221  r->norm = 0;
222 
223 
224  r->norm = 0;
225  }
226  CATCH_ANY {
227  THROW(ERR_CAUGHT);
228  }
229  FINALLY {
230  fp2_free(t0);
231  fp2_free(t1);
232  fp2_free(t2);
233  fp2_free(t3);
234  fp2_free(t4);
235  fp2_free(t5);
236  }
237 }
