51static void eb_add_basic_imp(eb_t r, const eb_t p, const eb_t q) {
52 fb_t t0, t1, t2;
53
54 fb_null(t0);
55 fb_null(t1);
56 fb_null(t2);
57
58 RLC_TRY {
59 fb_new(t0);
60 fb_new(t1);
61 fb_new(t2);
62
63 /* t0 = (y1 + y2). */
64 fb_add(t0, p->y, q->y);
65 /* t1 = (x1 + x2). */
66 fb_add(t1, p->x, q->x);
67
68 if (fb_is_zero(t1)) {
69 if (fb_is_zero(t0)) {
70 /* If t1 is zero and t0 is zero, p = q, should have doubled. */
71 eb_dbl_basic(r, p);
72 } else {
73 /* If t0 is not zero and t1 is zero, q = -p and r = infinity. */
74 eb_set_infty(r);
75 }
76 } else {
77 /* t2 = 1/(x1 + x2). */
78 fb_inv(t2, t1);
79 /* t0 = lambda = (y1 + y2)/(x1 + x2). */
80 fb_mul(t0, t0, t2);
81 /* t2 = lambda^2. */
82 fb_sqr(t2, t0);
83
84 /* t2 = lambda^2 + lambda + x1 + x2 + a. */
85 fb_add(t2, t2, t0);
86 fb_add(t2, t2, t1);
87
88 switch (eb_curve_opt_a()) {
89 case RLC_ZERO:
90 break;
91 case RLC_ONE:
92 fb_add_dig(t2, t2, (dig_t)1);
93 break;
94 case RLC_TINY:
95 fb_add_dig(t2, t2, eb_curve_get_a()[0]);
96 break;
97 default:
98 fb_add(t2, t2, eb_curve_get_a());
99 break;
100 }
101
102 /* y3 = lambda*(x3 + x1) + x3 + y1. */
103 fb_add(t1, t2, p->x);
104 fb_mul(t1, t1, t0);
105 fb_add(t1, t1, t2);
106 fb_add(r->y, p->y, t1);
107
108 /* x3 = lambda^2 + lambda + x1 + x2 + a. */
109 fb_copy(r->x, t2);
110 fb_copy(r->z, p->z);
111
112 r->coord = BASIC;
113 }
114 }
115 RLC_CATCH_ANY {
116 RLC_THROW(ERR_CAUGHT);
117 }
118 RLC_FINALLY {
119 fb_free(t0);
120 fb_free(t1);
121 fb_free(t2);
122 }
123}

139static void eb_add_projc_mix(eb_t r, const eb_t p, const eb_t q) {
140 fb_t t0, t1, t2, t3, t4, t5;
141
142 fb_null(t0);
143 fb_null(t1);
144 fb_null(t2);
145 fb_null(t3);
146 fb_null(t4);
147 fb_null(t5);
148
149 RLC_TRY {
150 fb_new(t0);
151 fb_new(t1);
152 fb_new(t2);
153 fb_new(t3);
154 fb_new(t4);
155 fb_new(t5);
156
157 /* madd-2005-dl formulas: 7M + 4S + 9add + 1*4 + 3*2. */
158 /* http://www.hyperelliptic.org/EFD/g12o/auto-shortw-lopezdahab-1.html#addition-madd-2005-dl */
159
160 if (p->coord != BASIC) {
161 /* A = y1 + y2 * z1^2. */
162 fb_sqr(t0, p->z);
163 fb_mul(t0, t0, q->y);
164 fb_add(t0, t0, p->y);
165 /* B = x1 + x2 * z1. */
166 fb_mul(t1, p->z, q->x);
167 fb_add(t1, t1, p->x);
168 } else {
169 /* t0 = A = y1 + y2. */
170 fb_add(t0, p->y, q->y);
171 /* t1 = B = x1 + x2. */
172 fb_add(t1, p->x, q->x);
173 }
174
175 if (fb_is_zero(t1)) {
176 if (fb_is_zero(t0)) {
177 /* If t0 = 0 and t1 = 0, p = q, should have doubled! */
178 eb_dbl_projc(r, p);
179 } else {
180 /* If t0 = 0, r is infinity. */
181 eb_set_infty(r);
182 }
183 } else {
184 if (p->coord != BASIC) {
185 /* t2 = C = B * z1. */
186 fb_mul(t2, p->z, t1);
187 /* z3 = C^2. */
188 fb_sqr(r->z, t2);
189 /* t1 = B^2. */
190 fb_sqr(t1, t1);
191 /* t1 = A + B^2. */
192 fb_add(t1, t0, t1);
193 } else {
194 /* If z1 = 0, t2 = C = B. */
195 fb_copy(t2, t1);
196 /* z3 = B^2. */
197 fb_sqr(r->z, t1);
198 /* t1 = A + z3. */
199 fb_add(t1, t0, r->z);
200 }
201
202 /* t3 = D = x2 * z3. */
203 fb_mul(t3, r->z, q->x);
204
205 /* t4 = (y2 + x2). */
206 fb_add(t4, q->x, q->y);
207
208 /* z3 = A^2. */
209 fb_sqr(r->x, t0);
210
211 /* t1 = A + B^2 + a2 * C. */
212 switch (eb_curve_opt_a()) {
213 case RLC_ZERO:
214 break;
215 case RLC_ONE:
216 fb_add(t1, t1, t2);
217 break;
218 case RLC_TINY:
219 /* t5 = a2 * C. */
220 fb_mul_dig(t5, t2, eb_curve_get_a()[0]);
221 fb_add(t1, t1, t5);
222 break;
223 default:
224 /* t5 = a2 * C. */
225 fb_mul(t5, eb_curve_get_a(), t2);
226 fb_add(t1, t1, t5);
227 break;
228 }
229
230 /* t1 = C * (A + B^2 + a2 * C). */
231 fb_mul(t1, t1, t2);
232 /* x3 = A^2 + C * (A + B^2 + a2 * C). */
233 fb_add(r->x, r->x, t1);
234
235 /* t3 = D + x3. */
236 fb_add(t3, t3, r->x);
237 /* t2 = A * B. */
238 fb_mul(t2, t0, t2);
239 /* y3 = (D + x3) * (A * B + z3). */
240 fb_add(r->y, t2, r->z);
241 fb_mul(r->y, r->y, t3);
242 /* t0 = z3^2. */
243 fb_sqr(t0, r->z);
244 /* t0 = (y2 + x2) * z3^2. */
245 fb_mul(t0, t0, t4);
246 /* y3 = (D + x3) * (A * B + z3) + (y2 + x2) * z3^2. */
247 fb_add(r->y, r->y, t0);
248 }
249
250 r->coord = PROJC;
251 }
252 RLC_CATCH_ANY {
253 RLC_THROW(ERR_CAUGHT);
254 }
255 RLC_FINALLY {
256 fb_free(t0);
257 fb_free(t1);
258 fb_free(t2);
259 fb_free(t3);
260 fb_free(t4);
261 fb_free(t5);
262 }
263}

275static void eb_add_projc_imp(eb_t r, const eb_t p, const eb_t q) {
276#if defined(EB_MIXED) && defined(STRIP)
277 /* If code size is a problem, leave only the mixed version. */
278 eb_add_projc_mix(r, p, q);
279#else /* General addition. */
280
281#if defined(EB_MIXED) || !defined(STRIP)
282 /* Test if z2 = 1 only if mixed coordinates are turned on. */
283 if (q->coord == BASIC) {
284 eb_add_projc_mix(r, p, q);
285 return;
286 }
287#endif
288 fb_t t0, t1, t2, t3, t4, t5, t6, t7;
289
290 fb_null(t0);
291 fb_null(t1);
292 fb_null(t2);
293 fb_null(t3);
294 fb_null(t4);
295 fb_null(t5);
296 fb_null(t6);
297 fb_null(t7);
298
299 RLC_TRY {
300 fb_new(t0);
301 fb_new(t1);
302 fb_new(t2);
303 fb_new(t3);
304 fb_new(t4);
305 fb_new(t5);
306 fb_new(t6);
307 fb_new(t7);
308
309 /* t0 = B = x2 * z1. */
310 fb_mul(t0, q->x, p->z);
311
312 /* A = x1 * z2 */
313 fb_mul(t1, p->x, q->z);
314
315 /* t2 = E = A + B. */
316 fb_add(t2, t1, t0);
317
318 /* t3 = D = B^2. */
319 fb_sqr(t3, t0);
320 /* t4 = C = A^2. */
321 fb_sqr(t4, t1);
322 /* t5 = F = C + D. */
323 fb_add(t5, t3, t4);
324
325 /* t6 = H = y2 * z1^2. */
326 fb_sqr(t6, p->z);
327 fb_mul(t6, t6, q->y);
328
329 /* t7 = G = y1 * z2^2. */
330 fb_sqr(t7, q->z);
331 fb_mul(t7, t7, p->y);
332
333 /* t3 = D + H. */
334 fb_add(t3, t3, t6);
335 /* t4 = C + G. */
336 fb_add(t4, t4, t7);
337 /* t6 = I = G + H. */
338 fb_add(t6, t7, t6);
339
340 /* If E is zero. */
341 if (fb_is_zero(t2)) {
342 if (fb_is_zero(t6)) {
343 /* If I is zero, p = q, should have doubled. */
344 eb_dbl_projc(r, p);
345 } else {
346 /* If I is not zero, q = -p, r = infinity. */
347 eb_set_infty(r);
348 }
349 } else {
350 /* t6 = J = I * E. */
351 fb_mul(t6, t6, t2);
352
353 /* z3 = F * z1 * z2. */
354 fb_mul(r->z, p->z, q->z);
355 fb_mul(r->z, t5, r->z);
356
357 /* t4 = B * (C + G). */
358 fb_mul(t4, t0, t4);
359 /* t2 = A * J. */
360 fb_mul(t2, t1, t6);
361 /* x3 = A * (D + H) + B * (C + G). */
362 fb_mul(r->x, t1, t3);
363 fb_add(r->x, r->x, t4);
364
365 /* t7 = F * G. */
366 fb_mul(t7, t7, t5);
367 /* Y3 = (A * J + F * G) * F + (J + z3) * x3. */
368 fb_add(r->y, t2, t7);
369 fb_mul(r->y, r->y, t5);
370 /* t7 = (J + z3) * x3. */
371 fb_add(t7, t6, r->z);
372 fb_mul(t7, t7, r->x);
373 fb_add(r->y, r->y, t7);
374 }
375
376 r->coord = PROJC;
377 }
378 RLC_CATCH_ANY {
379 RLC_THROW(ERR_CAUGHT);
380 }
381 RLC_FINALLY {
382 fb_free(t0);
383 fb_free(t1);
384 fb_free(t2);
385 fb_free(t3);
386 fb_free(t4);
387 fb_free(t5);
388 fb_free(t6);
389 fb_free(t7);
390 }
391#endif
392}

416void eb_sub_basic(eb_t r, const eb_t p, const eb_t q) {
417 eb_t t;
418
419 eb_null(t);
420
421 if (p == q) {
422 eb_set_infty(r);
423 return;
424 }
425
426 RLC_TRY {
427 eb_new(t);
428
429 eb_neg_basic(t, q);
430 eb_add_basic(r, p, t);

433 }

435 RLC_THROW(ERR_CAUGHT);
