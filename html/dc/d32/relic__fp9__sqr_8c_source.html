42 void fp9_sqr_basic(fp9_t c, fp9_t a) {
43  fp3_t t0, t1, t2, t3, t4;
44 
45  fp3_null(t0);
46  fp3_null(t1);
47  fp3_null(t2);
48  fp3_null(t3);
49  fp3_null(t4);
50 
51  TRY {
52  fp3_new(t0);
53  fp3_new(t1);
54  fp3_new(t2);
55  fp3_new(t3);
56  fp3_new(t4);
57 
58  /* t0 = a_0^2 */
59  fp3_sqr(t0, a[0]);
60 
61  /* t1 = 2 * a_1 * a_2 */
62  fp3_mul(t1, a[1], a[2]);
63  fp3_dbl(t1, t1);
64 
65  /* t2 = a_2^2. */
66  fp3_sqr(t2, a[2]);
67 
68  /* c2 = a_0 + a_2. */
69  fp3_add(c[2], a[0], a[2]);
70 
71  /* t3 = (a_0 + a_2 + a_1)^2. */
72  fp3_add(t3, c[2], a[1]);
73  fp3_sqr(t3, t3);
74 
75  /* c2 = (a_0 + a_2 - a_1)^2. */
76  fp3_sub(c[2], c[2], a[1]);
77  fp3_sqr(c[2], c[2]);
78 
79  /* c2 = (c2 + t3)/2. */
80  fp3_add(c[2], c[2], t3);
81  fp_hlv(c[2][0], c[2][0]);
82  fp_hlv(c[2][1], c[2][1]);
83  fp_hlv(c[2][2], c[2][2]);
84 
85  /* t3 = t3 - c2 - t1. */
86  fp3_sub(t3, t3, c[2]);
87  fp3_sub(t3, t3, t1);
88 
89  /* c2 = c2 - t0 - t2. */
90  fp3_sub(c[2], c[2], t0);
91  fp3_sub(c[2], c[2], t2);
92 
93  /* c0 = t0 + t1 * E. */
94  fp3_mul_nor(t4, t1);
95  fp3_add(c[0], t0, t4);
96 
97  /* c1 = t3 + t2 * E. */
98  fp3_mul_nor(t4, t2);
99  fp3_add(c[1], t3, t4);
100  } CATCH_ANY {
101  THROW(ERR_CAUGHT);
102  } FINALLY {
103  fp3_free(t0);
104  fp3_free(t1);
105  fp3_free(t2);
106  fp3_free(t3);
107  fp3_free(t4);
108  }
109 }

143  fp3_sqrn_low(u0, a[0]);
144 
145  /* t1 = 2 * a_1 * a_2 */
146  fp3_dblm_low(t0, a[1]);
147  fp3_muln_low(u1, t0, a[2]);
148 
149  /* u2 = a_2^2. */
150  fp3_sqrn_low(u2, a[2]);
151

153  fp3_addm_low(t3, a[0], a[2]);
154 
155  /* u3 = (a_0 + a_2 + a_1)^2. */
156  fp3_addm_low(t2, t3, a[1]);
157  fp3_sqrn_low(u3, t2);
158 
159  /* u4 = (a_0 + a_2 - a_1)^2. */
160  fp3_subm_low(t1, t3, a[1]);
161  fp3_sqrn_low(u4, t1);
162 
163  /* u4 = (u4 + u3)/2. */
164 #ifdef RLC_FP_ROOM
165  fp3_addc_low(u4, u4, u3);
166 #else
167  fp3_addc_low(u4, u4, u3);
168 #endif
169  fp_hlvd_low(u4[0], u4[0]);
170  fp_hlvd_low(u4[1], u4[1]);
171  fp_hlvd_low(u4[2], u4[2]);
172 
173  /* u3 = u3 - u4 - u1. */
174  fp3_addc_low(u5, u1, u4);
175  fp3_subc_low(u3, u3, u5);
176 
177  /* c2 = u4 - u0 - u2. */
178  fp3_addc_low(u5, u0, u2);
179  fp3_subc_low(c[2], u4, u5);
180 
181  /* c0 = u0 + u1 * E. */
182  fp3_nord_low(u4, u1);
183  fp3_addc_low(c[0], u0, u4);
184 
185  /* c1 = u3 + u2 * E. */
186  fp3_nord_low(u4, u2);
187  fp3_addc_low(c[1], u3, u4);
188  } CATCH_ANY {
189  THROW(ERR_CAUGHT);
190  } FINALLY {
191  dv3_free(u0);
192  dv3_free(u1);
193  dv3_free(u2);
194  dv3_free(u3);
195  dv3_free(u4);
