93  /* t0 = z1^2. */
94  fp_sqr(t0, r->z);
95 
96  /* t3 = x2 * z1^2. */
97  fp_mul(t3, p->x, t0);
98 
99  /* t1 = y2 * z1^3. */
100  fp_mul(t1, t0, r->z);
101  fp_mul(t1, t1, p->y);
102 
103  /* t2 = x1 - t3. */
104  fp_sub(t2, r->x, t3);
105 
106  /* t4 = y1 - t1. */
107  fp_sub(t4, r->y, t1);
108 
109  /* l0 = slope * (x2 + xq) - z3 * y2. */
110  fp_add(l[0], p->x, q->x);
111  fp_mul(l[0], l[0], t4);
112 
113  fp_dbl(t0, t3);
114  fp_add(t3, t0, t2);
115  fp_dbl(t0, t1);
116  fp_add(t1, t0, t4);
117 
118  fp_mul(r->z, t2, r->z);
119  fp_sqr(t0, t2);
120  fp_mul(t2, t0, t2);
121  fp_mul(t0, t0, t3);
122  fp_sqr(t3, t4);
123 
124  fp_sub(r->x, t3, t0);
125  fp_sub(t0, t0, r->x);
126  fp_sub(t0, t0, r->x);
127  fp_mul(t5, t0, t4);
128  fp_mul(t2, t2, t1);
129  fp_sub(t1, t5, t2);
130 
131  fp_mul(t5, r->z, p->y);
132  fp_sub(l[0], l[0], t5);
133

135 
136  fp_hlv(r->y, t1);
137 
138  r->norm = 0;
139  }

142  }

144  fp_free(t0);
