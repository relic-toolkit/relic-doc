33static BLAKE2_INLINE uint32_t load32( const void *src )
34{
35#if defined(NATIVE_LITTLE_ENDIAN)
36 uint32_t w;
37 memcpy(&w, src, sizeof w);
38 return w;
39#else
40 const uint8_t *p = ( const uint8_t * )src;
41 return (( uint32_t )( p[0] ) << 0) |
42 (( uint32_t )( p[1] ) << 8) |
43 (( uint32_t )( p[2] ) << 16) |
44 (( uint32_t )( p[3] ) << 24) ;
45#endif
46}

80static BLAKE2_INLINE void store16( void *dst, uint16_t w )
81{
82#if defined(NATIVE_LITTLE_ENDIAN)
83 memcpy(dst, &w, sizeof w);
84#else
85 uint8_t *p = ( uint8_t * )dst;
86 *p++ = ( uint8_t )w; w >>= 8;
87 *p++ = ( uint8_t )w;
88#endif
89}
