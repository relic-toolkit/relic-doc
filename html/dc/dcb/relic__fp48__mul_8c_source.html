42 void fp48_mul_basic(fp48_t c, fp48_t a, fp48_t b) {
43  fp24_t t0, t1, t2;
44 
45  fp24_null(t0);
46  fp24_null(t1);
47  fp24_null(t2);
48 
49  TRY {
50  fp24_new(t0);
51  fp24_new(t1);
52  fp24_new(t2);
53 
54  /* Karatsuba algorithm. */
55 
56  /* t0 = a_0 * b_0. */
57  fp24_mul(t0, a[0], b[0]);
58  /* t1 = a_1 * b_1. */
59  fp24_mul(t1, a[1], b[1]);
60  /* t2 = b_0 + b_1. */
61  fp24_add(t2, b[0], b[1]);
62 
63  /* c_1 = a_0 + a_1. */
64  fp24_add(c[1], a[0], a[1]);
65 
66  /* c_1 = (a_0 + a_1) * (b_0 + b_1) */
67  fp24_mul(c[1], c[1], t2);
68  fp24_sub(c[1], c[1], t0);
69  fp24_sub(c[1], c[1], t1);
70 
71  /* c_0 = a_0b_0 + v * a_1b_1. */
72  fp24_mul_art(t1, t1);
73  fp24_add(c[0], t0, t1);
74  } CATCH_ANY {
75  THROW(ERR_CAUGHT);
76  } FINALLY {
77  fp24_free(t0);
78  fp24_free(t1);
79  fp24_free(t2);
80  }
81 }

111 #if EP_ADD == BASIC

117  fp_mul(t1[2][i][j][k], a[1][2][i][j][k], b[1][1][0][0][0]);
118  }
119  }
120  }
121 #else

123  fp8_mul(t1[1], a[1][1], b[1][1]);
