111static void configure_rdtsc() {
112 if (kpc_force_all_ctrs_set(1)) {
113 printf("kpc_force_all_ctrs_set failed\n");
114 return;
115 }
116
117 if (kpc_set_config(KPC_MASK, g_config)) {
118 printf("kpc_set_config failed\n");
119 return;
120 }
121
122 if (kpc_set_counting(KPC_MASK)) {
123 printf("kpc_set_counting failed\n");
124 return;
125 }
126
127 if (kpc_set_thread_counting(KPC_MASK)) {
128 printf("kpc_set_thread_counting failed\n");
129 return;
130 }
131}

133static void init_rdtsc() {
134 void *kperf = dlopen(
135 "/System/Library/PrivateFrameworks/kperf.framework/Versions/A/kperf",
136 RTLD_LAZY);
137 if (!kperf) {
138 printf("kperf = %p\n", kperf);
139 return;
140 }
141
142#define F(ret, name, ...) \
143 name = (name##proc *)(dlsym(kperf, #name)); \
144 if (!name) { \
145 printf("%s = %p\n", #name, (void *)name); \
146 return; \
147 } \
148
149 KPERF_LIST
150#undef F
151
152 if (kpc_get_counter_count(KPC_MASK) != COUNTERS_COUNT) {
153 printf("wrong fixed counters count\n");
154 return;
155 }
156
157 if (kpc_get_config_count(KPC_MASK) != CONFIG_COUNT) {
158 printf("wrong fixed config count\n");
159 return;
160 }
161 g_config[0] = CPMU_CORE_CYCLE | CFGWORD_EL0A64EN_MASK;
162 g_config[3] = CPMU_INST_BRANCH | CFGWORD_EL0A64EN_MASK;
163 g_config[4] = CPMU_SYNC_BR_ANY_MISP | CFGWORD_EL0A64EN_MASK;
164 g_config[5] = CPMU_INST_A64 | CFGWORD_EL0A64EN_MASK;
165
166 configure_rdtsc();
167}

176void arch_init(void) {
177 ctx_t *ctx = core_get();
178 if (ctx != NULL) {
179 core_get()->lzcnt_ptr =
180 (has_lzcnt_hard() ? lzcnt64_hard : lzcnt64_soft);
181 core_get()->tzcnt_ptr =
182 (has_lzcnt_hard() ? tzcnt64_hard : tzcnt64_soft);
183 }
184
185#if TIMER == CYCLE
186
187#if OPSYS == MACOSX
188 int test_high_perf_cores = 1;
189 if (test_high_perf_cores) {
190 pthread_set_qos_class_self_np(QOS_CLASS_USER_INTERACTIVE, 0);
191 } else {
192 pthread_set_qos_class_self_np(QOS_CLASS_BACKGROUND, 0);
193 }
194 init_rdtsc();
195#endif
196
197#endif /* TIMER = CYCLE */
198}

204 core_get()->tzcnt_ptr = NULL;
