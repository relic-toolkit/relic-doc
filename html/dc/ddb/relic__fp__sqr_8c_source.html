52static void fp_sqr_karat_imp(dv_t c, const fp_t a, size_t size, uint_t level) {
53 int i, h, h1;
54 dv_t t0, t1, a0a0, a1a1;
55 dig_t carry;
56
57 /* Compute half the digits of a or b. */
58 h = size >> 1;
59 h1 = size - h;
60
61 dv_null(t0);
62 dv_null(t1);
63 dv_null(a0a0);
64 dv_null(a1a1);
65
66 RLC_TRY {
67 /* Allocate the temp variables. */
68 dv_new(t0);
69 dv_new(t1);
70 dv_new(a0a0);
71 dv_new(a1a1);
72 dv_zero(t0, 2 * h1);
73 dv_zero(t1, 2 * (h1 + 1));
74 dv_zero(a0a0, 2 * h);
75 dv_zero(a1a1, 2 * h1);
76
77 if (level <= 1) {
78 /* a0a0 = a0 * a0 and a1a1 = a1 * a1 */
79#if FP_SQR == BASIC
80 for (i = 0; i < h - 1; i++) {
81 a0a0[h + i + 1] = bn_sqra_low(a0a0 + 2 * i, a + i, h - i);
82 }
83 bn_sqra_low(a0a0 + 2 * i, a + i, 1);
84 for (i = 0; i < h1 - 1; i++) {
85 a1a1[h1 + i + 1] = bn_sqra_low(a1a1 + 2 * i, a + h + i, h1 - i);
86 }
87 bn_sqra_low(a1a1 + 2 * i, a + h + i, 1);
88#elif FP_SQR == COMBA || FP_SQR == INTEG
89 bn_sqrn_low(a0a0, a, h);
90 bn_sqrn_low(a1a1, a + h, h1);
91#elif FP_SQR == MULTP
92 bn_muln_low(a0a0, a, a, h);
93 bn_muln_low(a1a1, a + h, a + h, h1);
94#endif
95 } else {
96 fp_sqr_karat_imp(a0a0, a, h, level - 1);
97 fp_sqr_karat_imp(a1a1, a + h, h1, level - 1);
98 }
99
100 /* t2 = a1 * a1 << 2*h digits + a0 * a0. */
101 for (i = 0; i < 2 * h; i++) {
102 c[i] = a0a0[i];
103 }
104 for (i = 0; i < 2 * h1; i++) {
105 c[2 * h + i] = a1a1[i];
106 }
107
108 /* t = (a1 + a0) */
109 carry = bn_addn_low(t0, a, a + h, h);
110 carry = bn_add1_low(t0 + h, t0 + h, carry, 2);
111 if (h1 > h) {
112 carry = bn_add1_low(t0 + h, t0 + h, *(a + 2 * h), 2);
113 }
114
115 if (level <= 1) {
116 /* a1a1 = (a1 + a0)*(a1 + a0) */
117#if FP_SQR == BASIC
118 for (i = 0; i < h1; i++) {
119 t1[h1 + i + 2] = bn_sqra_low(t1 + (2 * i), t0 + i, h1 + 1 - i);
120 }
121 bn_sqra_low(t1 + (2 * i), t0 + i, 1);
122#elif FP_SQR == COMBA || FP_SQR == INTEG
123 bn_sqrn_low(t1, t0, h1 + 1);
124#elif FP_SQR == MULTP
125 bn_muln_low(t1, t0, t0, h1 + 1);
126#endif
127 } else {
128 fp_sqr_karat_imp(t1, t0, h1 + 1, level - 1);
129 }
130
131 /* t = t - (a0*a0 << h digits) */
132 carry = bn_subn_low(t1, t1, a0a0, 2 * h);
133 bn_sub1_low(t1 + 2 * h, t1 + 2 * h, carry, 2 * (h1 + 1) - 2 * h);
134
135 /* t = t - (a1*a1 << h digits) */
136 carry = bn_subn_low(t1, t1, a1a1, 2 * h1);
137 bn_sub1_low(t1 + 2 * h, t1 + 2 * h, carry, 2 * (h1 + 1) - 2 * h);
138
139 /* c = c + [(a1 + a0)*(a1 + a0) << digits] */
140 c += h;
141 carry = bn_addn_low(c, c, t1, 2 * (h1 + 1));
142 c += 2 * (h1 + 1);
143 if (2 * size > h + 2 * (h1 + 1)) {
144 carry = bn_add1_low(c, c, carry, 2 * size - h - 2 * (h1 + 1));
145 }
146 }
147 RLC_CATCH_ANY {
148 RLC_THROW(ERR_CAUGHT);
149 }
150 RLC_FINALLY {
151 dv_free(t0);
152 dv_free(t1);
153 dv_free(a0a0);
154 dv_free(a1a1);
155 }
156}
157#endif

165void fp_sqr_basic(fp_t c, const fp_t a) {
166 int i;
167 dv_t t;
168
169 dv_null(t);
170
171 RLC_TRY {
172 dv_new(t);
173 dv_zero(t, 2 * RLC_FP_DIGS);
174
175 for (i = 0; i < RLC_FP_DIGS - 1; i++) {
176 t[RLC_FP_DIGS + i + 1] =
177 bn_sqra_low(t + 2 * i, a + i, RLC_FP_DIGS - i);
178 }
179 bn_sqra_low(t + 2 * i, a + i, 1);
180
181 fp_rdc(c, t);
182 }
183 RLC_CATCH_ANY {
184 RLC_THROW(ERR_CAUGHT);
185 }
186 RLC_FINALLY {
187 fp_free(t);
188 }
189}
