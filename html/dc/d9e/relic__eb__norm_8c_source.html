50 static void eb_norm_imp(eb_t r, const eb_t p, int inverted) {
51  if (!p->norm) {
52  if (inverted) {
53  fb_copy(r->z, p->z);
54  } else {
55  fb_inv(r->z, p->z);
56  }
57  fb_mul(r->x, p->x, r->z);
58  fb_sqr(r->z, r->z);
59  fb_mul(r->y, p->y, r->z);
60  fb_set_dig(r->z, 1);
61  }
62 
63  r->norm = 1;
64 }

85 void eb_norm(eb_t r, const eb_t p) {
86  if (eb_is_infty(p)) {
87  eb_set_infty(r);
88  return;
89  }
90 
91  if (p->norm == 1) {
92  /* If the point is represented in affine coordinates, just copy it. */
93  eb_copy(r, p);
94  return;
95  }
96 
97  if (p->norm == 2) {
98  eb_norm_hlv(r, p);
99  return;
100  }
101 
102 #if EB_ADD == PROJC || !defined(STRIP)
103  eb_norm_imp(r, p, 0);
104 #endif /* EB_ADD == PROJC */
105 }

107 void eb_norm_sim(eb_t *r, const eb_t *t, int n) {
108  fb_t *a = RLC_ALLOCA(fb_t, n);
109 
110  if (n == 1) {
111  eb_norm(r[0], t[0]);
112  return;
113  }
114 
115  TRY {
116  if (a == NULL) {
117  THROW(ERR_NO_MEMORY);
118  }
119  for (int i = 0; i < n; i++) {
120  fb_null(a[i]);
121  fb_new(a[i]);
122  if (!eb_is_infty(t[i])) {
123  fb_copy(a[i], t[i]->z);
124  } else {
125  fb_set_dig(a[i], 1);
126  }
127  }
128 
129  fb_inv_sim(a, (const fb_t *)a, n);
130 
131  for (int i = 0; i < n; i++) {
132  fb_copy(r[i]->x, t[i]->x);
133  fb_copy(r[i]->y, t[i]->y);
134  if (!eb_is_infty(t[i])) {
135  fb_copy(r[i]->z, a[i]);
136  }
137  }
138 
139  for (int i = 0; i < n; i++) {
140  eb_norm_imp(r[i], r[i], 1);
141  }
142  }
143  CATCH_ANY {
144  THROW(ERR_CAUGHT);
145  }
146  FINALLY {
147  for (int i = 0; i < n; i++) {
148  fb_free(a[i]);
149  }
150  RLC_FREE(a);
151  }
152 }
