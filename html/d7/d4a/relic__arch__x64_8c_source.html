50void arch_init(void) {
51 ctx_t *ctx = core_get();
52 if (ctx != NULL) {
53 core_get()->lzcnt_ptr =
54 (has_lzcnt_hard() ? lzcnt64_hard : lzcnt64_soft);
55 core_get()->tzcnt_ptr =
56 (has_tzcnt_hard() ? tzcnt64_hard : tzcnt64_soft);

70ull_t arch_cycles(void) {
71 uint32_t hi, lo;
72 asm (
73 "cpuid\n\t"/*serialize*/
74 "rdtsc\n\t"/*read the clock*/
75 "mov %%edx, %0\n\t"
76 "mov %%eax, %1\n\t"
77 : "=r" (hi), "=r" (lo):: "%rax", "%rbx", "%rcx", "%rdx"
78 );
79 return ((ull_t) lo) | (((ull_t) hi) << 32);
80}
81
82#elif TIMER == PERF
83
84ull_t arch_cycles(void) {
85 uint_t seq;
86 ull_t index, offset, result = 0;
87 if (core_get()->perf_buf != NULL) {
88 do {
89 seq = core_get()->perf_buf->lock;
90 asm("" ::: "memory");
91 index = core_get()->perf_buf->index;
92 offset = core_get()->perf_buf->offset;
93 asm(
94 "rdpmc; shlq $32, %%rdx; orq %%rdx,%%rax"
95 : "=a" (result) : "c" (index - 1) : "%rdx"
96 );
97 asm("" ::: "memory");
98 } while (core_get()->perf_buf->lock != seq);
99
100 result += offset;
101 result &= RLC_MASK(48); /* Get lower 48 bits only. */
102 }
103 return result;
104}
