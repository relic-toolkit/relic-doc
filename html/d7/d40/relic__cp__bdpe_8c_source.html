46 int cp_bdpe_gen(bdpe_t pub, bdpe_t prv, dig_t block, int bits) {
47  bn_t t, r;
48  int result = RLC_OK;
49 
50  bn_null(t);
51  bn_null(r);
52 
53  TRY {
54  bn_new(t);
55  bn_new(r);
56 
57  prv->t = pub->t = block;
58 
59  /* Make sure that block size is prime. */
60  bn_set_dig(t, block);
61  if (bn_is_prime_basic(t) == 0) {
62  THROW(ERR_NO_VALID);
63  }
64 
65  /* Generate prime q such that gcd(block, (q - 1)) = 1. */
66  do {
67  bn_gen_prime(prv->q, bits / 2);
68  bn_sub_dig(prv->q, prv->q, 1);
69  bn_gcd_dig(t, prv->q, block);
70  bn_add_dig(prv->q, prv->q, 1);
71  } while (bn_cmp_dig(t, 1) != RLC_EQ);
72 
73  /* Generate different primes p and q. */
74  do {
75  /* Compute p = block * (x * block + b) + 1, 0 < b < block random. */
76  bn_rand(prv->p, RLC_POS, bits / 2 - 2 * util_bits_dig(block));
77  bn_mul_dig(prv->p, prv->p, block);
78  bn_rand(t, RLC_POS, util_bits_dig(block));
79  bn_add_dig(prv->p, prv->p, t->dp[0]);
80 
81  /* We know that block divides (p-1). */
82  bn_gcd_dig(t, prv->p, block);
83  bn_mul_dig(prv->p, prv->p, block);
84  bn_add_dig(prv->p, prv->p, 1);
85  } while (bn_cmp_dig(t, 1) != RLC_EQ || bn_is_prime(prv->p) == 0);
86 
87  /* Compute t = (p-1)*(q-1). */
88  bn_sub_dig(prv->q, prv->q, 1);
89  bn_sub_dig(prv->p, prv->p, 1);
90  bn_mul(t, prv->p, prv->q);
91  bn_div_dig(t, t, block);
92 
93  /* Restore factors p and q and compute n = p * q. */
94  bn_add_dig(prv->p, prv->p, 1);
95  bn_add_dig(prv->q, prv->q, 1);
96  bn_mul(pub->n, prv->p, prv->q);
97  bn_copy(prv->n, pub->n);
98 
99  /* Select random y such that y^{(p-1)(q-1)}/block \neq 1 mod N. */
100  do {
101  bn_rand(pub->y, RLC_POS, bits);
102  bn_mxp(r, pub->y, t, pub->n);
103  } while (bn_cmp_dig(r, 1) == RLC_EQ);
104 
105  bn_copy(prv->y, pub->y);
106  }
107  CATCH_ANY {
108  result = RLC_ERR;
109  }
110  FINALLY {
111  bn_free(t);
112  bn_free(r);
113  }
114 
115  return result;
116 }

118 int cp_bdpe_enc(uint8_t *out, int *out_len, dig_t in, bdpe_t pub) {
119  bn_t m, u;

121 
122  bn_null(m);
123  bn_null(u);
124 
125  size = bn_size_bin(pub->n);
126 
127  if (in > pub->t) {
128  return RLC_ERR;
129  }
130 
131  TRY {
132  bn_new(m);

134 
135  bn_set_dig(m, in);
136 
137  bn_rand_mod(u, pub->n);
138  bn_mxp(m, pub->y, m, pub->n);
139  bn_mxp_dig(u, u, pub->t, pub->n);
140  bn_mul(m, m, u);
141  bn_mod(m, m, pub->n);
142 
143  if (size <= *out_len) {
144  *out_len = size;
145  memset(out, 0, *out_len);

152  result = RLC_ERR;

154  FINALLY {

162 int cp_bdpe_dec(dig_t *out, uint8_t *in, int in_len, bdpe_t prv) {
163  bn_t m, t, z;
164  int size, result = RLC_OK;
165  dig_t i;
166 
167  size = bn_size_bin(prv->n);
168 
169  if (in_len < 0 || in_len != size) {
170  return RLC_ERR;
171  }
172 
173  bn_null(m);
174  bn_null(t);
175  bn_null(z);
176 
177  TRY {
178  bn_new(m);
179  bn_new(t);
180  bn_new(z);
181 
182  /* Compute t = (p-1)(q-1)/block. */
183  bn_mul(t, prv->p, prv->q);
184  bn_sub(t, t, prv->p);
185  bn_sub(t, t, prv->q);
186  bn_add_dig(t, t, 1);
187  bn_div_dig(t, t, prv->t);
188  bn_read_bin(m, in, in_len);
189  bn_mxp(m, m, t, prv->n);
190  bn_mxp(t, prv->y, t, prv->n);
191 
192  for (i = 0; i < prv->t; i++) {
193  bn_mxp_dig(z, t, i, prv->n);
194  if (bn_cmp(z, m) == RLC_EQ) {
195  *out = i;
196  break;
197  }
198  }
199 
200  if (i == prv->t) {
201  result = RLC_ERR;
202  }
203  } CATCH_ANY {
204  result = RLC_ERR;
205  }
206  FINALLY {
207  bn_free(m);
208  bn_free(t);
209  bn_free(z);
210  }
211 
212  return result;
213 }
