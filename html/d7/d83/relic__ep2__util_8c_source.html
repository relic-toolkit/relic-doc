81void ep2_blind(ep2_t r, const ep2_t p) {
82 fp2_t rand;
83
84 fp2_null(rand);
85
86 RLC_TRY {
87 fp2_new(rand);
88 fp2_rand(rand);
89#if EP_ADD == BASIC
90 (void)rand;
91 ep2_copy(r, p);
92#elif EP_ADD == PROJC
93 fp2_mul(r->x, p->x, rand);
94 fp2_mul(r->y, p->y, rand);
95 fp2_mul(r->z, p->z, rand);
96 r->coord = PROJC;
97#elif EP_ADD == JACOB
98 fp2_mul(r->z, p->z, rand);
99 fp2_mul(r->y, p->y, rand);
100 fp2_sqr(rand, rand);
101 fp2_mul(r->x, r->x, rand);
102 fp2_mul(r->y, r->y, rand);
103 r->coord = JACOB;
104#endif
105 } RLC_CATCH_ANY {
106 RLC_THROW(ERR_CAUGHT);
107 } RLC_FINALLY {
108 fp2_free(rand);
109 }
110}

112void ep2_rhs(fp2_t rhs, const fp2_t x) {
113 fp2_t t0;
114
115 fp2_null(t0);
116
117 RLC_TRY {
118 fp2_new(t0);
119
120 fp2_sqr(t0, x); /* x1^2 */
121
122 switch (ep2_curve_opt_a()) {
123 case RLC_ZERO:
124 break;
125#if FP_RDC != MONTY
126 case RLC_MIN3:
127 fp2_sub_dig(t0, t0, 3);
128 break;
129 case RLC_ONE:
130 fp2_add_dig(t0, t0, 1);
131 break;
132 case RLC_TWO:
133 fp2_add_dig(t0, t0, 2);
134 break;
135 case RLC_TINY:
136 fp2_mul_dig(t0, t0, ep2_curve_get_a()[0][0]);
137 break;
138#endif
139 default:
140 fp2_add(t0, t0, ep2_curve_get_a());
141 break;
142 }
143
144 fp2_mul(t0, t0, x); /* x1^3 + a * x */
145
146 switch (ep2_curve_opt_b()) {
147 case RLC_ZERO:
148 break;
149#if FP_RDC != MONTY
150 case RLC_MIN3:
151 fp2_sub_dig(t0, t0, 3);
152 break;
153 case RLC_ONE:
154 fp2_add_dig(t0, t0, 1);
155 break;
156 case RLC_TWO:
157 fp2_add_dig(t0, t0, 2);
158 break;
159 case RLC_TINY:
160 fp2_mul_dig(t0, t0, ep2_curve_get_b()[0][0]);
161 break;
162#endif
163 default:
164 fp2_add(t0, t0, ep2_curve_get_b());
165 break;
166 }
167
168 fp2_copy(rhs, t0);
169 } RLC_CATCH_ANY {
170 RLC_THROW(ERR_CAUGHT);
171 } RLC_FINALLY {
172 fp2_free(t0);
173 }
174}

177int ep2_on_curve(const ep2_t p) {
178 ep2_t t;
179 int r = 0;
180
181 ep2_null(t);
182
183 RLC_TRY {
184 ep2_new(t);
185
186 ep2_norm(t, p);
187
188 ep2_rhs(t->x, t->x);
189 fp2_sqr(t->y, t->y);
190
191 r = (fp2_cmp(t->x, t->y) == RLC_EQ) || ep2_is_infty(p);
192 } RLC_CATCH_ANY {
193 RLC_THROW(ERR_CAUGHT);
194 } RLC_FINALLY {
195 ep2_free(t);
196 }
197 return r;
198}

200void ep2_tab(ep2_t *t, const ep2_t p, int w) {
201 if (w > 2) {
202 ep2_dbl(t[0], p);
203#if defined(EP_MIXED)
204 ep2_norm(t[0], t[0]);
205#endif
206 ep2_add(t[1], t[0], p);
207 for (int i = 2; i < (1 << (w - 2)); i++) {
208 ep2_add(t[i], t[i - 1], t[0]);
209 }
210#if defined(EP_MIXED)
211 ep2_norm_sim(t + 1, t + 1, (1 << (w - 2)) - 1);
212#endif
213 }
214#if defined(EP_MIXED)
215 ep2_norm(t[0], p);
216#else
217 ep2_copy(t[0], p);
218#endif
219}

227size_t ep2_size_bin(const ep2_t a, int pack) {
228 ep2_t t;
229 size_t size = 0;
230
231 ep2_null(t);
232
233 if (ep2_is_infty(a)) {
234 return 1;
235 }
236
237 RLC_TRY {
238 ep2_new(t);
239
240 ep2_norm(t, a);
241
242 size = 1 + 2 * RLC_FP_BYTES;
243 if (!pack) {
244 size += 2 * RLC_FP_BYTES;
245 }
246 } RLC_CATCH_ANY {
247 RLC_THROW(ERR_CAUGHT);
248 } RLC_FINALLY {
249 ep2_free(t);
250 }
251
252 return size;
253}

255void ep2_read_bin(ep2_t a, const uint8_t *bin, size_t len) {
256 if (len == 1) {
257 if (bin[0] == 0) {
258 ep2_set_infty(a);
259 return;
260 } else {
261 RLC_THROW(ERR_NO_BUFFER);
262 return;
263 }
264 }
265
266 if (len != (2 * RLC_FP_BYTES + 1) && len != (4 * RLC_FP_BYTES + 1)) {
267 RLC_THROW(ERR_NO_BUFFER);
268 return;
269 }
270
271 a->coord = BASIC;
272 fp2_set_dig(a->z, 1);
273 fp2_read_bin(a->x, bin + 1, 2 * RLC_FP_BYTES);
274 if (len == 2 * RLC_FP_BYTES + 1) {
275 switch(bin[0]) {
276 case 2:
277 fp2_zero(a->y);
278 break;
279 case 3:
280 fp2_zero(a->y);
281 fp_set_bit(a->y[0], 0, 1);
282 fp_zero(a->y[1]);
283 break;
284 default:
285 RLC_THROW(ERR_NO_VALID);
286 break;
287 }
288 ep2_upk(a, a);
289 }
290
291 if (len == 4 * RLC_FP_BYTES + 1) {
292 if (bin[0] == 4) {
293 fp2_read_bin(a->y, bin + 2 * RLC_FP_BYTES + 1, 2 * RLC_FP_BYTES);
294 } else {
295 RLC_THROW(ERR_NO_VALID);
296 return;
297 }
298 }
299
300 if (!ep2_on_curve(a)) {
301 RLC_THROW(ERR_NO_VALID);
302 }
303}

305void ep2_write_bin(uint8_t *bin, size_t len, const ep2_t a, int pack) {
306 ep2_t t;
307
308 ep2_null(t);
309
310 memset(bin, 0, len);
311
312 if (ep2_is_infty(a)) {
313 if (len < 1) {
314 RLC_THROW(ERR_NO_BUFFER);
315 return;
316 } else {
317 return;
318 }
319 }
320
321 RLC_TRY {
322 ep2_new(t);
323
324 ep2_norm(t, a);
325
326 if (pack) {
327 if (len < 2 * RLC_FP_BYTES + 1) {
328 RLC_THROW(ERR_NO_BUFFER);
329 } else {
330 ep2_pck(t, t);
331 bin[0] = 2 | fp_get_bit(t->y[0], 0);
332 fp2_write_bin(bin + 1, 2 * RLC_FP_BYTES, t->x, 0);
333 }
334 } else {
335 if (len < 4 * RLC_FP_BYTES + 1) {
336 RLC_THROW(ERR_NO_BUFFER);
337 } else {
338 bin[0] = 4;
339 fp2_write_bin(bin + 1, 2 * RLC_FP_BYTES, t->x, 0);
340 fp2_write_bin(bin + 2 * RLC_FP_BYTES + 1, 2 * RLC_FP_BYTES, t->y, 0);
341 }
342 }
343 } RLC_CATCH_ANY {
344 RLC_THROW(ERR_CAUGHT);
345 } RLC_FINALLY {
346 ep2_free(t);
347 }
