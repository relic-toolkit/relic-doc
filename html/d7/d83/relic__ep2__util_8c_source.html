57 int ep2_cmp(ep2_t p, ep2_t q) {
58  ep2_t r, s;
59  int result = RLC_EQ;
60 
61  ep2_null(r);
62  ep2_null(s);
63 
64  TRY {
65  ep2_new(r);
66  ep2_new(s);
67 
68  if ((!p->norm) && (!q->norm)) {
69  /* If the two points are not normalized, it is faster to compare
70  * x1 * z2^2 == x2 * z1^2 and y1 * z2^3 == y2 * z1^3. */
71  fp2_sqr(r->z, p->z);
72  fp2_sqr(s->z, q->z);
73  fp2_mul(r->x, p->x, s->z);
74  fp2_mul(s->x, q->x, r->z);
75  fp2_mul(r->z, r->z, p->z);
76  fp2_mul(s->z, s->z, q->z);
77  fp2_mul(r->y, p->y, s->z);
78  fp2_mul(s->y, q->y, r->z);
79  } else {
80  if (!p->norm) {
81  ep2_norm(r, p);
82  } else {
83  ep2_copy(r, p);
84  }
85 
86  if (!q->norm) {
87  ep2_norm(s, q);
88  } else {
89  ep2_copy(s, q);
90  }
91  }
92 
93  if (fp2_cmp(r->x, s->x) != RLC_EQ) {
94  result = RLC_NE;
95  }
96 
97  if (fp2_cmp(r->y, s->y) != RLC_EQ) {
98  result = RLC_NE;
99  }
100  } CATCH_ANY {
101  THROW(ERR_CAUGHT);
102  } FINALLY {
103  ep2_free(r);
104  ep2_free(s);
105  }
106 
107  return result;
108 }

134 void ep2_rhs(fp2_t rhs, ep2_t p) {
135  fp2_t t0;
136  fp2_t t1;
137 
138  fp2_null(t0);
139  fp2_null(t1);
140 
141  TRY {
142  fp2_new(t0);
143  fp2_new(t1);
144 
145  /* t0 = x1^2. */
146  fp2_sqr(t0, p->x);
147  /* t1 = x1^3. */
148  fp2_mul(t1, t0, p->x);
149 
150  ep2_curve_get_a(t0);
151  fp2_mul(t0, p->x, t0);
152  fp2_add(t1, t1, t0);
153 
154  ep2_curve_get_b(t0);
155  fp2_add(t1, t1, t0);
156 
157  fp2_copy(rhs, t1);
158 
159  } CATCH_ANY {
160  THROW(ERR_CAUGHT);
161  } FINALLY {
162  fp2_free(t0);
163  fp2_free(t1);
164  }
165 }

168 int ep2_is_valid(ep2_t p) {
169  ep2_t t;
170  int r = 0;
171 
172  ep2_null(t);
173 
174  TRY {
175  ep2_new(t);
176 
177  ep2_norm(t, p);
178 
179  ep2_rhs(t->x, t);
180  fp2_sqr(t->y, t->y);
181 
182  r = (fp2_cmp(t->x, t->y) == RLC_EQ) || ep2_is_infty(p);
183  } CATCH_ANY {
184  THROW(ERR_CAUGHT);
185  } FINALLY {
186  ep2_free(t);
187  }
188  return r;
189 }

191 void ep2_tab(ep2_t *t, ep2_t p, int w) {
192  if (w > 2) {
193  ep2_dbl(t[0], p);
194 #if defined(EP_MIXED)
195  ep2_norm(t[0], t[0]);
196 #endif
197  ep2_add(t[1], t[0], p);
198  for (int i = 2; i < (1 << (w - 2)); i++) {
199  ep2_add(t[i], t[i - 1], t[0]);
200  }
201 #if defined(EP_MIXED)
202  ep2_norm_sim(t + 1, t + 1, (1 << (w - 2)) - 1);
203 #endif
204  }
205  ep2_copy(t[0], p);
206 }

242 void ep2_read_bin(ep2_t a, const uint8_t *bin, int len) {
243  if (len == 1) {
244  if (bin[0] == 0) {
245  ep2_set_infty(a);
246  return;
247  } else {
248  THROW(ERR_NO_BUFFER);
249  return;
250  }
251  }
252 
253  if (len != (2 * RLC_FP_BYTES + 1) && len != (4 * RLC_FP_BYTES + 1)) {
254  THROW(ERR_NO_BUFFER);
255  return;
256  }
257 
258  a->norm = 1;
259  fp_set_dig(a->z[0], 1);
260  fp_zero(a->z[1]);
261  fp2_read_bin(a->x, bin + 1, 2 * RLC_FP_BYTES);
262  if (len == 2 * RLC_FP_BYTES + 1) {
263  switch(bin[0]) {
264  case 2:
265  fp2_zero(a->y);
266  break;
267  case 3:
268  fp2_zero(a->y);
269  fp_set_bit(a->y[0], 0, 1);
270  fp_zero(a->y[1]);
271  break;
272  default:
273  THROW(ERR_NO_VALID);
274  break;
275  }
276  ep2_upk(a, a);
277  }
278 
279  if (len == 4 * RLC_FP_BYTES + 1) {
280  if (bin[0] == 4) {
281  fp2_read_bin(a->y, bin + 2 * RLC_FP_BYTES + 1, 2 * RLC_FP_BYTES);
282  } else {
283  THROW(ERR_NO_VALID);
284  }
285  }
286 }

288 void ep2_write_bin(uint8_t *bin, int len, ep2_t a, int pack) {
289  ep2_t t;
290 
291  ep2_null(t);
292 
293  if (ep2_is_infty(a)) {
294  if (len < 1) {
295  THROW(ERR_NO_BUFFER);
296  } else {
297  bin[0] = 0;
298  return;
299  }
300  }
301 
302  TRY {
303  ep2_new(t);
304 
305  ep2_norm(t, a);
306 
307  if (pack) {
308  if (len < 2 * RLC_FP_BYTES + 1) {
309  THROW(ERR_NO_BUFFER);
310  } else {
311  ep2_pck(t, t);
312  bin[0] = 2 | fp_get_bit(t->y[0], 0);
313  fp2_write_bin(bin + 1, 2 * RLC_FP_BYTES, t->x, 0);
314  }
315  } else {
316  if (len < 4 * RLC_FP_BYTES + 1) {
317  THROW(ERR_NO_BUFFER);
318  } else {
319  bin[0] = 4;
320  fp2_write_bin(bin + 1, 2 * RLC_FP_BYTES, t->x, 0);
321  fp2_write_bin(bin + 2 * RLC_FP_BYTES + 1, 2 * RLC_FP_BYTES, t->y, 0);
322  }
323  }
324  } CATCH_ANY {
325  THROW(ERR_CAUGHT);
326  } FINALLY {
327  ep2_free(t);
328  }
