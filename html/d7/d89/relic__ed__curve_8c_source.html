35void ed_curve_init(void) {
36 ctx_t *ctx = core_get();
37#ifdef ED_PRECO
38 for (int i = 0; i < RLC_ED_TABLE; i++) {
39 ctx->ed_ptr[i] = &(ctx->ed_pre[i]);
40 }
41#endif
42 ed_set_infty(&ctx->ed_g);
43 bn_make(&ctx->ed_r, RLC_FP_DIGS);
44 bn_make(&ctx->ed_h, RLC_FP_DIGS);
45}
