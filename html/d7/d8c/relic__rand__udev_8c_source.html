54void rand_bytes(uint8_t *buf, size_t size) {
55 int c, l, *fd = (int *)&(core_get()->rand);
56
57 l = 0;
58 do {
59 c = read(*fd, buf + l, size - l);
60 l += c;
61 if (c == -1) {
62 RLC_THROW(ERR_NO_READ);
63 return;
64 }
65 } while (l < size);
66}
