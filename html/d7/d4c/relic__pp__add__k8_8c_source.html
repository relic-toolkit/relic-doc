44void pp_add_k8_basic(fp8_t l, ep2_t r, const ep2_t q, const ep_t p) {
45 int one = 1, zero = 0;
46 fp2_t s;
47 ep2_t t;
48
49 fp2_null(s);
50 ep2_null(t);
51
52 RLC_TRY {
53 fp2_new(s);
54 ep2_new(t);
55
56 ep2_copy(t, r);
57 ep2_add_slp_basic(r, s, r, q);
58
59 if (ep2_curve_is_twist() == RLC_EP_MTYPE) {
60 one ^= 1;
61 zero ^= 1;
62 }
63
64 fp_mul(l[one][zero][0], s[0], p->x);
65 fp_mul(l[one][zero][1], s[1], p->x);
66 fp2_mul(l[one][one], s, t->x);
67 fp2_sub(l[one][one], t->y, l[one][one]);
68 fp_copy(l[zero][zero][0], p->y);
69 }
70 RLC_CATCH_ANY {
71 RLC_THROW(ERR_CAUGHT);
72 }
73 RLC_FINALLY {
74 fp2_free(s);
75 ep2_free(t);
76 }
77}

85void pp_add_k8_projc_basic(fp8_t l, ep2_t r, const ep2_t q, const ep_t p) {
86 fp2_t t0, t1, t2, t3, t4, t5;
87 int one = 1, zero = 0;
88
89 fp2_null(t0);
90 fp2_null(t1);
91 fp2_null(t2);
92 fp2_null(t3);
93 fp2_null(t4);
94
95 RLC_TRY {
96 fp2_new(t0);
97 fp2_new(t1);
98 fp2_new(t2);
99 fp2_new(t3);
100 fp2_new(t4);
101 fp2_new(t5);
102
103 if (ep2_curve_is_twist() == RLC_EP_MTYPE) {
104 one ^= 1;
105 zero ^= 1;
106 }
107
108 /* t0 = A = Z1^2, t1 = B = X2*Z1. */
109 fp2_sqr(t0, r->z);
110 fp2_mul(t1, r->z, q->x);
111
112 /* t0 = C = y2*A, t2 = D = (x1 - B) */
113 fp2_mul(t0, t0, q->y);
114 fp2_sub(t2, r->x, t1);
115
116 /* t3 = E = 2*(y1 - C), t4 = F = 2*D*z1, t2 = G = 4*D*F. */
117 fp2_sub(t3, r->y, t0);
118 fp2_dbl(t3, t3);
119 fp2_dbl(t2, t2);
120 fp2_mul(t4, t2, r->z);
121 fp2_mul(t2, t2, t4);
122 fp2_dbl(t2, t2);
123
124 /* l = E*X2 - F*Y2 - E*xQ + F*yQ. */
125 fp2_mul(l[one][one], t3, q->x);
126 fp2_mul(t0, t4, q->y);
127 fp2_sub(l[one][one], l[one][one], t0);
128 fp_mul(l[one][zero][0], t3[0], p->x);
129 fp_mul(l[one][zero][1], t3[1], p->x);
130 fp_mul(l[zero][zero][0], t4[0], p->y);
131 fp_mul(l[zero][zero][1], t4[1], p->y);
132
133 /* z3 = F^2, t4 = (F + E)^2, t3 = E^2. */
134 fp2_sqr(r->z, t4);
135 fp2_add(t4, t4, t3);
136 fp2_sqr(t4, t4);
137 fp2_sqr(t3, t3);
138
139 /* t5 = x3 = 2*E^2 - (x1 + B)*G. */
140 fp2_add(t1, t1, r->x);
141 fp2_mul(t1, t1, t2);
142 fp2_dbl(t5, t3);
143 fp2_sub(t5, t5, t1);
144
145 /* y3 = ((F + E)^2 - E^2 - F^2)*(x1*G - x3) - y1*G^2. */
146 fp2_sub(t4, t4, r->z);
147 fp2_sub(t4, t4, t3);
148 fp2_mul(t1, r->x, t2);
149 fp2_sub(t1, t1, t5);
150 fp2_mul(t4, t4, t1);
151 fp2_sqr(t2, t2);
152 fp2_mul(r->y, r->y, t2);
153 fp2_sub(r->y, t4, r->y);
154
155 /* Z3 = 2*F^2. */
156 fp2_dbl(r->z, r->z);
157 fp2_copy(r->x, t5);
158
159 r->coord = PROJC;
160 }
161 RLC_CATCH_ANY {
162 RLC_THROW(ERR_CAUGHT);
163 }
164 RLC_FINALLY {
165 fp2_free(t0);
166 fp2_free(t1);
167 fp2_free(t2);
168 fp2_free(t3);
169 fp2_free(t4);
170 fp2_free(t5);
171 }
172}
