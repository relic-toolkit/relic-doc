51static void eb_mul_fix_kbltz(eb_t r, const eb_t *t, const bn_t k) {
52 int i, n;
53 int8_t u, tnaf[RLC_FB_BITS + 8];
54 size_t l;
55
56 if (bn_is_zero(k)) {
57 eb_set_infty(r);
58 return;
59 }
60
61 /* Compute the w-TNAF representation of k. */
62 if (eb_curve_opt_a() == RLC_ZERO) {
63 u = -1;
64 } else {
65 u = 1;
66 }
67
68 /* Compute the w-TNAF representation of k. */
69 l = sizeof(tnaf);
70 bn_rec_tnaf(tnaf, &l, k, u, RLC_FB_BITS, RLC_DEPTH);
71
72 n = tnaf[l - 1];
73 if (n > 0) {
74 eb_copy(r, t[n / 2]);
75 } else {
76 eb_neg(r, t[-n / 2]);
77 }
78
79 for (i = l - 2; i >= 0; i--) {
80 eb_frb(r, r);
81
82 n = tnaf[i];
83 if (n > 0) {
84 eb_add(r, r, t[n / 2]);
85 }
86 if (n < 0) {
87 eb_sub(r, r, t[-n / 2]);
88 }
89 }
90 /* Convert r to affine coordinates. */
91 eb_norm(r, r);
92 if (bn_sign(k) == RLC_NEG) {
93 eb_neg(r, r);
94 }
95}
96
97#endif /* EB_KBLTZ */

109static void eb_mul_fix_plain(eb_t r, const eb_t *t, const bn_t k) {
110 int i, n;
111 int8_t naf[RLC_FB_BITS + 1];
112 size_t l;
113
114 if (bn_is_zero(k)) {
115 eb_set_infty(r);
116 return;
117 }
118
119 /* Compute the w-TNAF representation of k. */
120 l = RLC_FB_BITS + 1;
121 bn_rec_naf(naf, &l, k, RLC_DEPTH);
122
123 n = naf[l - 1];
124 if (n > 0) {
125 eb_copy(r, t[n / 2]);
126 }
127
128 for (i = l - 2; i >= 0; i--) {
129 eb_dbl(r, r);
130
131 n = naf[i];
132 if (n > 0) {
133 eb_add(r, r, t[n / 2]);
134 }
135 if (n < 0) {
136 eb_sub(r, r, t[-n / 2]);
137 }
138 }
139 /* Convert r to affine coordinates. */
140 eb_norm(r, r);
141 if (bn_sign(k) == RLC_NEG) {
142 eb_neg(r, r);
143 }
144}

181void eb_mul_fix_basic(eb_t r, const eb_t *t, const bn_t k) {
182 if (bn_is_zero(k)) {
183 eb_set_infty(r);
184 return;
185 }
186
187 eb_set_infty(r);
188 for (int i = 0; i < bn_bits(k); i++) {
189 if (bn_get_bit(k, i)) {
190 eb_add(r, r, t[i]);
191 }
192 }
193 eb_norm(r, r);
194 if (bn_sign(k) == RLC_NEG) {
195 eb_neg(r, r);

203void eb_mul_pre_combs(eb_t *t, const eb_t p) {
204 int i, j, l;
205 bn_t ord;
206
207 bn_null(ord);
208
209 RLC_TRY {
210 bn_new(ord);
211
212 eb_curve_get_ord(ord);

214
215 eb_set_infty(t[0]);
216
217 eb_copy(t[1], p);
218 for (j = 1; j < RLC_DEPTH; j++) {
219 eb_dbl(t[1 << j], t[1 << (j - 1)]);
220 for (i = 1; i < l; i++) {
221 eb_dbl(t[1 << j], t[1 << j]);
222 }
223#if defined(EB_MIXED)
224 eb_norm(t[1 << j], t[1 << j]);
225#endif
226 for (i = 1; i < (1 << j); i++) {
227 eb_add(t[(1 << j) + i], t[i], t[1 << j]);
228 }
229 }

231 eb_norm_sim(t + 2, (const eb_t *)t + 2, RLC_EB_TABLE_COMBS - 2);
232 }

234 RLC_THROW(ERR_CAUGHT);
235 }
236 RLC_FINALLY {

241void eb_mul_fix_combs(eb_t r, const eb_t *t, const bn_t k) {
242 int i, j, l, w, n, p0, p1;
243 bn_t ord;
244
245 if (bn_is_zero(k)) {
246 eb_set_infty(r);
247 return;
248 }
249
250 bn_null(ord);
251
252 RLC_TRY {
253 bn_new(ord);
254
255 eb_curve_get_ord(ord);
256 l = RLC_CEIL(bn_bits(ord), RLC_DEPTH);
257
258 n = bn_bits(k);
259 p0 = (RLC_DEPTH) * l - 1;
260
261 w = 0;
262 p1 = p0--;
263 for (j = RLC_DEPTH - 1; j >= 0; j--, p1 -= l) {
264 w = w << 1;
265 if (p1 < n && bn_get_bit(k, p1)) {
266 w = w | 1;
267 }
268 }
269 eb_copy(r, t[w]);
270
271 for (i = l - 2; i >= 0; i--) {
272 eb_dbl(r, r);
273
274 w = 0;
275 p1 = p0--;
276 for (j = RLC_DEPTH - 1; j >= 0; j--, p1 -= l) {
277 w = w << 1;
278 if (p1 < n && bn_get_bit(k, p1)) {
279 w = w | 1;
280 }
281 }
282 if (w > 0) {
283 eb_add(r, r, t[w]);
284 }
285 }
286 eb_norm(r, r);
287 if (bn_sign(k) == RLC_NEG) {
288 eb_neg(r, r);
289 }
290 }
291 RLC_CATCH_ANY {
292 RLC_THROW(ERR_CAUGHT);
293 }
294 RLC_FINALLY {
295 bn_free(ord);
296 }

303void eb_mul_pre_combd(eb_t *t, const eb_t p) {
304 int i, j, d, e;
305 bn_t n;
306
307 bn_null(n);
308
309 RLC_TRY {
310 bn_new(n);
311
312 eb_curve_get_ord(n);
313 d = RLC_CEIL(bn_bits(n), RLC_DEPTH);
314 e = (d % 2 == 0 ? (d / 2) : (d / 2) + 1);
315
316 eb_set_infty(t[0]);
317 eb_copy(t[1], p);
318 for (j = 1; j < RLC_DEPTH; j++) {
319 eb_dbl(t[1 << j], t[1 << (j - 1)]);
320 for (i = 1; i < d; i++) {
321 eb_dbl(t[1 << j], t[1 << j]);
322 }
323 for (i = 1; i < (1 << j); i++) {
324 eb_add(t[(1 << j) + i], t[1 << j], t[i]);
325 }
326 }
327 eb_set_infty(t[1 << RLC_DEPTH]);
328 for (j = 1; j < (1 << RLC_DEPTH); j++) {
329 eb_dbl(t[(1 << RLC_DEPTH) + j], t[j]);
330 for (i = 1; i < e; i++) {
331 eb_dbl(t[(1 << RLC_DEPTH) + j], t[(1 << RLC_DEPTH) + j]);
332 }
333 }
334
335 eb_norm_sim(t + 2, (const eb_t *)t + 2, (1 << RLC_DEPTH) - 2);
336 eb_norm_sim(t + (1 << RLC_DEPTH) + 1,
337 (const eb_t *)t + (1 << RLC_DEPTH) + 1, (1 << RLC_DEPTH) - 1);
338 }
339 RLC_CATCH_ANY {
340 RLC_THROW(ERR_CAUGHT);
341 }
342 RLC_FINALLY {
343 bn_free(n);
344 }

347void eb_mul_fix_combd(eb_t r, const eb_t *t, const bn_t k) {
348 int i, j, d, e, w0, w1, n0, p0, p1;
349 bn_t n;
350
351 if (bn_is_zero(k)) {
352 eb_set_infty(r);
353 return;
354 }
355
356 bn_null(n);
357
358 RLC_TRY {
359 bn_new(n);
360
361 eb_curve_get_ord(n);
362
363 d = RLC_CEIL(bn_bits(n), RLC_DEPTH);
364 e = (d % 2 == 0 ? (d / 2) : (d / 2) + 1);
365
366 eb_set_infty(r);
367 n0 = bn_bits(k);
368
369 p1 = (e - 1) + (RLC_DEPTH - 1) * d;
370 for (i = e - 1; i >= 0; i--) {
371 eb_dbl(r, r);
372
373 w0 = 0;
374 p0 = p1;
375 for (j = RLC_DEPTH - 1; j >= 0; j--, p0 -= d) {
376 w0 = w0 << 1;
377 if (p0 < n0 && bn_get_bit(k, p0)) {
378 w0 = w0 | 1;
379 }
380 }
381
382 w1 = 0;
383 p0 = p1-- + e;
384 for (j = RLC_DEPTH - 1; j >= 0; j--, p0 -= d) {
385 w1 = w1 << 1;
386 if (i + e < d && p0 < n0 && bn_get_bit(k, p0)) {
387 w1 = w1 | 1;
388 }
389 }
390
391 eb_add(r, r, t[w0]);
392 eb_add(r, r, t[(1 << RLC_DEPTH) + w1]);
393 }
394 eb_norm(r, r);
395 if (bn_sign(k) == RLC_NEG) {
396 eb_neg(r, r);
397 }
398 }
399 RLC_CATCH_ANY {
400 RLC_THROW(ERR_CAUGHT);
401 }
402 RLC_FINALLY {
403 bn_free(n);
404 }
405}
