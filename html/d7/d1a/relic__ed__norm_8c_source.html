46static void ed_norm_imp(ed_t r, const ed_t p, int inverted) {
47 if (p->coord != BASIC) {
48 if (inverted) {
49 fp_copy(r->z, p->z);
50 } else {
51 fp_inv(r->z, p->z);
52 }
53 fp_mul(r->x, p->x, r->z);
54 fp_mul(r->y, p->y, r->z);
55#if ED_ADD == EXTND
56 fp_mul(r->t, p->t, r->z);
57#endif
58 fp_set_dig(r->z, 1);
59
60 r->coord = BASIC;
61 }
62}

64#endif /* ED_ADD == PROJC */

70void ed_norm(ed_t r, const ed_t p) {
71 if (ed_is_infty(p)) {
72 ed_set_infty(r);
73 return;
74 }
75
76 if (p->coord == BASIC) {
77 /* If the point is represented in affine coordinates, just copy it. */
78 ed_copy(r, p);
79 return;
80 }
81#if ED_ADD == PROJC || ED_ADD == EXTND || !defined(STRIP)
82 ed_norm_imp(r, p, 0);
83#endif /* ED_ADD != BASIC*/
84}

86void ed_norm_sim(ed_t *r, const ed_t *t, int n) {
87 fp_t* a = RLC_ALLOCA(fp_t, n);
88
89 RLC_TRY {
90 if (a == NULL) {
91 RLC_THROW(ERR_NO_MEMORY);
92 }
93 for (int i = 0; i < n; i++) {
94 fp_null(a[i]);
95 fp_new(a[i]);
96 fp_copy(a[i], t[i]->z);
97 }
98
99 fp_inv_sim(a, (const fp_t *)a, n);
100
101 for (int i = 0; i < n; i++) {
102 fp_copy(r[i]->x, t[i]->x);
103 fp_copy(r[i]->y, t[i]->y);
104#if ED_ADD == EXTND
105 fp_copy(r[i]->t, t[i]->t);
106#endif
107 if (!ed_is_infty(t[i])) {
108 fp_copy(r[i]->z, a[i]);
109 }
110 }
111
112#if ED_ADD == PROJC || ED_ADD == EXTND || !defined(STRIP)
113 for (int i = 0; i < n; i++) {
114 ed_norm_imp(r[i], r[i], 1);
115 }
116#endif /* ED_ADD != BASIC*/
117 }
118 RLC_CATCH_ANY {
119 RLC_THROW(ERR_CAUGHT);
120 }
121 RLC_FINALLY {
122 for (int i = 0; i < n; i++) {
123 fp_free(a[i]);
124 }
125 RLC_FREE(a);
126 }
127}
