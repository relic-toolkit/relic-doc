84 void pp_dbl_k8_projc_new(fp8_t l, ep2_t r, ep2_t q, ep_t p) {
85  fp2_t t0, t1, t2, t3, t4;
86  int one = 1, zero = 0;
87 
88  fp2_null(t0);
89  fp2_null(t1);
90  fp2_null(t2);
91  fp2_null(t3);
92  fp2_null(t4);
93 
94  TRY {
95  fp2_new(t0);
96  fp2_new(t1);
97  fp2_new(t2);
98  fp2_new(t3);
99  fp2_new(t4);
100 
101  if (ep2_curve_is_twist() == EP_MTYPE) {
102  one ^= 1;
103  zero ^= 1;
104  }
105 
106  /* t0 = z2 = z1^2, t1 = y1^2, t2 = x1*t1. */
107  fp2_sqr(t0, q->z);
108  fp2_sqr(t1, q->y);
109  fp2_mul(t2, q->x, t1);
110 
111  /* t3 = 3/2*x1^2 + (a = 2)/2*Z2^2. */
112  fp2_sqr(t3, q->x);
113  fp_hlv(t4[0], t3[0]);
114  fp_hlv(t4[1], t3[1]);
115  fp2_add(t3, t3, t4);
116  fp2_sqr(t4, t0);
117  fp2_mul_art(t4, t4);
118  fp2_add(t3, t3, t4);
119 
120  /* z3 := z1*y1, l := z3*z2*yP + (t3*x1 - t1) + t3*z2*xP. */
121  fp2_mul(r->z, q->y, q->z);
122  fp2_mul(l[one][one], t3, q->x);
123  fp2_sub(l[one][one], l[one][one], t1);
124  fp2_mul(l[zero][zero], r->z, t0);
125  fp_mul(l[zero][zero][0], l[zero][zero][0], p->y);
126  fp_mul(l[zero][zero][1], l[zero][zero][1], p->y);
127  fp2_mul(l[one][zero], t3, t0);
128  fp_mul(l[one][zero][0], l[one][zero][0], p->x);
129  fp_mul(l[one][zero][1], l[one][zero][1], p->x);
130 
131  /* x3 := t3^2 - 2*t2. */
132  fp2_sqr(r->x, t3);
133  fp2_sub(r->x, r->x, t2);
134  fp2_sub(r->x, r->x, t2);
135  /* y3 := t3*(t2 - x3) - t1^2. */
136  fp2_sub(t2, t2, r->x);
137  fp2_mul(r->y, t3, t2);
138  fp2_sqr(t1, t1);
139  fp2_sub(r->y, r->y, t1);
140  }
141  CATCH_ANY {
142  THROW(ERR_CAUGHT);
143  }
144  FINALLY {
145  fp2_free(t0);
146  fp2_free(t1);
147  fp2_free(t2);
148  fp2_free(t3);
149  fp2_free(t4);
150  }
151 }

153 void pp_dbl_k8_projc_basic(fp8_t l, ep2_t r, ep2_t q, ep_t p) {
154  fp2_t t0, t1, t2, t3, t4, t5, t6;
155  int one = 1, zero = 0;
156 
157  fp2_null(t0);
158  fp2_null(t1);
159  fp2_null(t2);
160  fp2_null(t3);
161  fp2_null(t4);
162  fp2_null(t5);
163  fp2_null(t6);
164 
165  TRY {
166  fp2_new(t0);
167  fp2_new(t1);
168  fp2_new(t2);
169  fp2_new(t3);
170  fp2_new(t4);
171  fp2_new(t5);
172  fp2_new(t6);
173 
174  if (ep2_curve_is_twist() == EP_MTYPE) {
175  one ^= 1;
176  zero ^= 1;
177  }
178 
179  /* t0 = A = X1^2, t1 = B = Y1^2, t2 = C = Z1^2, t3 = D = a*C. */
180  fp2_sqr(t0, q->x);
181  fp2_sqr(t1, q->y);
182  fp2_sqr(t2, q->z);
183  switch (ep_curve_opt_a()) {
184  case RLC_ZERO:
185  fp2_zero(t3);
186  break;
187  case RLC_ONE:
188  fp2_copy(t3, t2);
189  break;
190 #if FP_RDC != MONTY
191  case RLC_TINY:
192  fp_mul_dig(t3[0], t2[0], ep_curve_get_a()[0]);
193  fp_mul_dig(t3[1], t2[1], ep_curve_get_a()[0]);
194  break;
195 #endif
196  default:
197  fp_mul(t3[0], t2[0], ep_curve_get_a());
198  fp_mul(t3[1], t2[1], ep_curve_get_a());
199  break;
200  }
201  fp2_mul_art(t3, t3);
202 
203  /* x3 = (A - D)^2, l11 = (A - D + x1)^2 - x3 - A. */
204  fp2_sub(t5, t0, t3);
205  fp2_add(l[one][one], t5, q->x);
206  fp2_sqr(r->x, t5);
207  fp2_sqr(l[one][one], l[one][one]);
208  fp2_sub(l[one][one], l[one][one], r->x);
209  fp2_sub(l[one][one], l[one][one], t0);
210 
211  /* l10 := -xp*z1*2*(3A + D). */
212  fp2_add(t6, t0, t3);
213  fp2_dbl(t0, t0);
214  fp2_add(t0, t0, t6);
215  fp2_dbl(t0, t0);
216  fp2_mul(l[one][zero], t0, q->z);
217  fp_mul(l[one][zero][0], l[one][zero][0], p->x);
218  fp_mul(l[one][zero][1], l[one][zero][1], p->x);
219 
220  /* l01 = 2*((y1 + z1)^2 - B - C)*yP. */
221  fp2_add(l[zero][zero], q->y, q->z);
222  fp2_sqr(l[zero][zero], l[zero][zero]);
223  fp2_sub(l[zero][zero], l[zero][zero], t1);
224  fp2_sub(l[zero][zero], l[zero][zero], t2);
225  fp2_dbl(l[zero][zero], l[zero][zero]);
226  fp_mul(l[zero][zero][0], l[zero][zero][0], p->y);
227  fp_mul(l[zero][zero][1], l[zero][zero][1], p->y);
228 
229  /* t4 = E = 2*(A + D)^2 - x3. */
230  fp2_sqr(t4, t6);
231  fp2_dbl(t4, t4);
232  fp2_sub(t4, t4, r->x);
233  /* y3 = E * ((A - D + y1)^2 - B - x3). */
234  fp2_add(r->y, t5, q->y);
235  fp2_sqr(r->y, r->y);
236  fp2_sub(r->y, r->y, t1);
237  fp2_sub(r->y, r->y, r->x);
238  fp2_mul(r->y, r->y, t4);
239  /* z3 = 4*B. */
240  fp2_dbl(r->z, t1);
241  fp2_dbl(r->z, r->z);
242  }
243  CATCH_ANY {
244  THROW(ERR_CAUGHT);
245  }
246  FINALLY {
247  fp2_free(t0);
248  fp2_free(t1);
249  fp2_free(t2);
250  fp2_free(t3);
251  fp2_free(t4);
252  fp2_free(t5);
253  fp2_free(t6);
254  }
255 }
