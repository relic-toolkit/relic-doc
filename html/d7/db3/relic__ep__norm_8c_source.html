47static void ep_norm_imp(ep_t r, const ep_t p, int inv) {
48 if (p->coord != BASIC) {
49 fp_t t;
50
51 fp_null(t);
52
53 RLC_TRY {
54 fp_new(t);
55
56 if (inv) {
57 fp_copy(r->z, p->z);
58 } else {
59 fp_inv(r->z, p->z);
60 }
61
62 switch (p->coord) {
63 case PROJC:
64 fp_mul(r->x, p->x, r->z);
65 fp_mul(r->y, p->y, r->z);
66 break;
67 case JACOB:
68 fp_sqr(t, r->z);
69 fp_mul(r->x, p->x, t);
70 fp_mul(t, t, r->z);
71 fp_mul(r->y, p->y, t);
72 break;
73 default:
74 ep_copy(r, p);
75 break;
76 }
77 fp_set_dig(r->z, 1);
78 }
79 RLC_CATCH_ANY {
80 RLC_THROW(ERR_CAUGHT);
81 }
82 RLC_FINALLY {
83 fp_free(t);
84 }
85 }
86
87 r->coord = BASIC;
88}

96void ep_norm(ep_t r, const ep_t p) {
97 if (ep_is_infty(p)) {
98 ep_set_infty(r);
99 return;
100 }
101
102 if (p->coord == BASIC) {
103 /* If the point is represented in affine coordinates, just copy it. */
104 ep_copy(r, p);
105 return;
106 }
107#if EP_ADD == PROJC || EP_ADD == JACOB || !defined(STRIP)
108 ep_norm_imp(r, p, 0);
109#endif /* EP_ADD == PROJC */
110}

112void ep_norm_sim(ep_t *r, const ep_t *t, int n) {
113 int i;
114 fp_t* a = RLC_ALLOCA(fp_t, n);
115
116 RLC_TRY {
117 if (a == NULL) {
118 RLC_THROW(ERR_NO_MEMORY);
119 }
120 for (i = 0; i < n; i++) {
121 fp_null(a[i]);
122 fp_new(a[i]);
123 fp_copy(a[i], t[i]->z);
124 }
125
126 fp_inv_sim(a, (const fp_t *)a, n);
127
128 for (i = 0; i < n; i++) {
129 fp_copy(r[i]->x, t[i]->x);
130 fp_copy(r[i]->y, t[i]->y);
131 if (!ep_is_infty(t[i])) {
132 fp_copy(r[i]->z, a[i]);
133 }
134 }
135#if EP_ADD == PROJC || EP_ADD == JACOB || !defined(STRIP)
136 for (i = 0; i < n; i++) {
137 ep_norm_imp(r[i], r[i], 1);
138 }
139#endif /* EP_ADD == PROJC */
140 }
141 RLC_CATCH_ANY {
142 RLC_THROW(ERR_CAUGHT);
143 }
144 RLC_FINALLY {
145 for (i = 0; i < n; i++) {
146 fp_free(a[i]);
147 }
148 RLC_FREE(a);
149 }
150}
