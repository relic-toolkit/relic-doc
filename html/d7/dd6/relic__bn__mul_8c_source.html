50 static void bn_mul_karat_imp(bn_t c, const bn_t a, const bn_t b, int level) {
51  int h;
52  bn_t a0, a1, b0, b1, a0b0, a1b1;
53  bn_t t;
54  const dig_t *tmpa, *tmpb;
55  dig_t *t0;
56 
57  bn_null(a0);
58  bn_null(a1);
59  bn_null(b0);
60  bn_null(b1);
61  bn_null(a0b0);
62  bn_null(a1b1);
63  bn_null(t);
64 
65  /* Compute half the digits of a or b. */
66  h = RLC_MIN(a->used, b->used) >> 1;
67 
68  TRY {
69  /* Allocate the temp variables. */
70  bn_new(a0);
71  bn_new(a1);
72  bn_new(b0);
73  bn_new(b1);
74  bn_new(a0b0);
75  bn_new(a1b1);
76  bn_new(t);
77 
78  a0->used = b0->used = h;
79  a1->used = a->used - h;
80  b1->used = b->used - h;
81 
82  tmpa = a->dp;
83  tmpb = b->dp;
84 
85  /* a = a1 || a0 */
86  t0 = a0->dp;
87  for (int i = 0; i < h; i++, t0++, tmpa++)
88  *t0 = *tmpa;
89  t0 = a1->dp;
90  for (int i = 0; i < a1->used; i++, t0++, tmpa++)
91  *t0 = *tmpa;
92 
93  /* b = b1 || b0 */
94  t0 = b0->dp;
95  for (int i = 0; i < h; i++, t0++, tmpb++)
96  *t0 = *tmpb;
97  t0 = b1->dp;
98  for (int i = 0; i < b1->used; i++, t0++, tmpb++)
99  *t0 = *tmpb;
100 
101  bn_trim(a0);
102  bn_trim(b0);
103  bn_trim(a1);
104  bn_trim(b1);
105 
106  /* a0b0 = a0 * b0 and a1b1 = a1 * b1 */
107  if (level <= 1) {
108 #if BN_MUL == BASIC
109  bn_mul_basic(a0b0, a0, b0);
110  bn_mul_basic(a1b1, a1, b1);
111 #elif BN_MUL == COMBA
112  bn_mul_comba(a0b0, a0, b0);
113  bn_mul_comba(a1b1, a1, b1);
114 #endif
115  } else {
116  bn_mul_karat_imp(a0b0, a0, b0, level - 1);
117  bn_mul_karat_imp(a1b1, a1, b1, level - 1);
118  }
119 
120  /* t = (a1 + a0) */
121  bn_add(a1, a1, a0);
122  /* t2 = (b1 + b0) */
123  bn_add(b1, b1, b0);
124 
125  /* t = (a1 + a0)*(b1 + b0) */
126  if (level <= 1) {
127 #if BN_MUL == BASIC
128  bn_mul_basic(t, a1, b1);
129 #elif BN_MUL == COMBA
130  bn_mul_comba(t, a1, b1);
131 #endif
132  } else {
133  bn_mul_karat_imp(t, a1, b1, level - 1);
134  }
135  /* t2 = (a0*b0 + a1*b1) */
136  bn_sub(t, t, a0b0);
137 
138  /* t = (a1 + a0)*(b1 + b0) - (a0*b0 + a1*b1) */
139  bn_sub(t, t, a1b1);
140 
141  /* t = (a1 + a0)*(b1 + b0) - (a0*b0 + a1*b1) << h digits */
142  bn_lsh(t, t, h * RLC_DIG);
143 
144  /* t2 = a1 * b1 << 2*h digits */
145  bn_lsh(a1b1, a1b1, 2 * h * RLC_DIG);
146 
147  /* t = t + a0*b0 */
148  bn_add(t, t, a0b0);
149 
150  /* c = t + a1*b1 */
151  bn_add(t, t, a1b1);
152 
153  t->sign = a->sign ^ b->sign;
154  bn_copy(c, t);
155  bn_trim(c);
156  }
157  CATCH_ANY {
158  THROW(ERR_CAUGHT);
159  }
160  FINALLY {
161  bn_free(a0);
162  bn_free(a1);
163  bn_free(b0);
164  bn_free(b1);
165  bn_free(a0b0);
166  bn_free(a1b1);
167  bn_free(t);
168  }
169 }

177 void bn_mul_dig(bn_t c, const bn_t a, dig_t b) {
178  bn_grow(c, a->used + 1);
179  c->sign = a->sign;
180  c->dp[a->used] = bn_mul1_low(c->dp, a->dp, b, a->used);
181  c->used = a->used + 1;
182  bn_trim(c);
183 }

187 void bn_mul_basic(bn_t c, const bn_t a, const bn_t b) {
188  int i;
189  bn_t t;
190  dig_t carry;
191 
192  bn_null(t);
193 
194  TRY {
195  /* We need a temporary variable so that c can be a or b. */
196  bn_new_size(t, a->used + b->used);
197  bn_zero(t);
198  t->used = a->used + b->used;
199 
200  for (i = 0; i < a->used; i++) {
201  carry = bn_mula_low(t->dp + i, b->dp, *(a->dp + i), b->used);
202  *(t->dp + i + b->used) = carry;
203  }
204  t->sign = a->sign ^ b->sign;
205  bn_trim(t);
206 
207  /* Swap c and t. */
208  bn_copy(c, t);
209  }
210  CATCH_ANY {
211  THROW(ERR_CAUGHT);
212  }
213  FINALLY {
214  bn_free(t);
215  }
216 }

223  int digits;

225 
226  bn_null(t);
227 
228  TRY {
229  digits = a->used + b->used;
230 
231  /* We need a temporary variable so that c can be a or b. */
232  bn_new_size(t, digits);
233  t->used = digits;
234 
235  if (a->used == b->used) {
236  bn_muln_low(t->dp, a->dp, b->dp, a->used);
237  } else {

239  bn_muld_low(t->dp, a->dp, a->used, b->dp, b->used, 0,
240  a->used + b->used);

243  a->used + b->used);
244  }
245  }
246
