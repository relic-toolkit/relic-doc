47 static void ep2_norm_imp(ep2_t r, ep2_t p, int inverted) {
48  if (!p->norm) {
49  fp2_t t0, t1;
50 
51  fp2_null(t0);
52  fp2_null(t1);
53 
54  TRY {
55 
56  fp2_new(t0);
57  fp2_new(t1);
58 
59  if (inverted) {
60  fp2_copy(t1, p->z);
61  } else {
62  fp2_inv(t1, p->z);
63  }
64  fp2_sqr(t0, t1);
65  fp2_mul(r->x, p->x, t0);
66  fp2_mul(t0, t0, t1);
67  fp2_mul(r->y, p->y, t0);
68  fp2_set_dig(r->z, 1);
69  }
70  CATCH_ANY {
71  THROW(ERR_CAUGHT);
72  }
73  FINALLY {
74  fp2_free(t0);
75  fp2_free(t1);
76  }
77  }
78 
79  r->norm = 1;
80 }

88 void ep2_norm(ep2_t r, ep2_t p) {
89  if (ep2_is_infty(p)) {
90  ep2_set_infty(r);
91  return;
92  }
93 
94  if (p->norm) {
95  /* If the point is represented in affine coordinates, we just copy it. */
96  ep2_copy(r, p);
97  }
98 #if EP_ADD == PROJC || !defined(STRIP)
99  ep2_norm_imp(r, p, 0);
100 #endif
101 }
