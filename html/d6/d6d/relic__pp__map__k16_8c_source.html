51static void pp_mil_k16(fp16_t r, ep4_t *t, ep4_t *q, ep_t *p, int m, bn_t a) {
52 fp16_t l;
53 ep_t *_p = RLC_ALLOCA(ep_t, m);
54 ep4_t *_q = RLC_ALLOCA(ep4_t, m);
55 int i, j;
56 size_t len = bn_bits(a) + 1;
57 int8_t s[RLC_FP_BITS + 1];
58
59 if (m == 0) {
60 return;
61 }
62
63 fp16_null(l);
64
65 RLC_TRY {
66 fp16_new(l);
67 if (_p == NULL || _q == NULL) {
68 RLC_THROW(ERR_NO_MEMORY);
69 }
70
71 for (j = 0; j < m; j++) {
72 ep_null(_p[j]);
73 ep4_null(_q[j]);
74 ep_new(_p[j]);
75 ep4_new(_q[j]);
76 ep4_copy(t[j], q[j]);
77 ep4_neg(_q[j], q[j]);
78#if EP_ADD == BASIC
79 ep_neg(_p[j], p[j]);
80#else
81 fp_neg(_p[j]->x, p[j]->x);
82 fp_copy(_p[j]->y, p[j]->y);
83#endif
84 }
85
86 fp16_zero(l);
87 bn_rec_naf(s, &len, a, 2);
88 pp_dbl_k16(r, t[0], t[0], _p[0]);
89 for (j = 1; j < m; j++) {
90 pp_dbl_k16(l, t[j], t[j], _p[j]);
91 fp16_mul_dxs(r, r, l);
92 }
93 if (s[len - 2] > 0) {
94 for (j = 0; j < m; j++) {
95 pp_add_k16(l, t[j], q[j], _p[j]);
96 fp16_mul_dxs(r, r, l);
97 }
98 }
99 if (s[len - 2] < 0) {
100 for (j = 0; j < m; j++) {
101 pp_add_k16(l, t[j], _q[j], _p[j]);
102 fp16_mul_dxs(r, r, l);
103 }
104 }
105
106 for (i = len - 3; i >= 0; i--) {
107 fp16_sqr(r, r);
108 for (j = 0; j < m; j++) {
109 pp_dbl_k16(l, t[j], t[j], _p[j]);
110 fp16_mul_dxs(r, r, l);
111 if (s[i] > 0) {
112 pp_add_k16(l, t[j], q[j], _p[j]);
113 fp16_mul_dxs(r, r, l);
114 }
115 if (s[i] < 0) {
116 pp_add_k16(l, t[j], _q[j], _p[j]);
117 fp16_mul_dxs(r, r, l);
118 }
119 }
120 }
121 }
122 RLC_CATCH_ANY {
123 RLC_THROW(ERR_CAUGHT);
124 }
125 RLC_FINALLY {
126 fp16_free(l);
127 for (j = 0; j < m; j++) {
128 ep_free(_p[j]);
129 ep4_free(_q[j]);
130 }
131 RLC_FREE(_p);
132 RLC_FREE(_q);
133 }
134}

147static void pp_mil_lit_k16(fp16_t r, ep_t *t, ep_t *p, ep4_t *q, int m, bn_t a) {
148 fp16_t l;
149 ep4_t *_q = RLC_ALLOCA(ep4_t, m);
150 int j;
151
152 fp16_null(l);
153
154 RLC_TRY {
155 if (_q == NULL) {
156 RLC_THROW(ERR_NO_MEMORY);
157 }
158 fp16_new(l);
159
160 for (j = 0; j < m; j++) {
161 ep4_null(_q[j]);
162 ep4_new(_q[j]);
163 ep_copy(t[j], p[j]);
164 ep4_neg(_q[j], q[j]);
165 }
166
167 fp16_zero(l);
168 for (int i = bn_bits(a) - 2; i >= 0; i--) {
169 fp16_sqr(r, r);
170 for (j = 0; j < m; j++) {
171 pp_dbl_lit_k16(l, t[j], t[j], _q[j]);
172 fp16_mul(r, r, l);
173 if (bn_get_bit(a, i)) {
174 pp_add_lit_k16(l, t[j], p[j], q[j]);
175 fp16_mul(r, r, l);
176 }
177 }
178 }
179 }
180 RLC_CATCH_ANY {
181 RLC_THROW(ERR_CAUGHT);
182 }
183 RLC_FINALLY {
184 fp16_free(l);
185 for (j = 0; j < m; j++) {
186 ep4_free(_q[j]);
187 }
188 RLC_FREE(_q);
189 }
190}

201static void pp_fin_k16_oatep(fp16_t r, ep4_t t, const ep4_t q, const ep_t p) {
202 ep4_t q1, q2;
203 fp16_t tmp;
204 ep_t _p;
205
206 fp16_null(tmp);
207 ep4_null(q1);
208 ep4_null(q2);
209 ep_null(_p);
210
211 RLC_TRY {
212 ep4_new(q1);
213 ep4_new(q2);
214 fp16_new(tmp);
215 ep_new(_p);
216
217#if EP_ADD == BASIC
218 ep_neg(_p, p);
219#else
220 fp_neg(_p->x, p->x);
221 fp_copy(_p->y, p->y);
222#endif
223
224 fp16_zero(tmp);
225 ep4_frb(q1, q, 1);
226 pp_add_k16(tmp, t, q1, _p);
227 fp16_frb(tmp, tmp, 3);
228 fp16_mul_dxs(r, r, tmp);
229
230 fp16_zero(tmp);
231 pp_dbl_k16(tmp, q2, q, _p);
232 fp16_mul_dxs(r, r, tmp);
233 } RLC_CATCH_ANY {
234 RLC_THROW(ERR_CAUGHT);
235 } RLC_FINALLY {
236 fp16_free(tmp);
237 ep4_free(q1);
238 ep4_free(q2);
239 ep_free(_p);
240 }
