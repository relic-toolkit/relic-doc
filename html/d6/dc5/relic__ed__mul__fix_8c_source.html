49static void ed_mul_fix_plain(ed_t r, const ed_t * t, const bn_t k) {
50 int i, n;
51 int8_t naf[RLC_FP_BITS + 1], *_k;
52 size_t l;
53
54 /* Compute the w-TNAF representation of k. */
55 l = RLC_FP_BITS + 1;
56 bn_rec_naf(naf, &l, k, RLC_DEPTH);
57
58 _k = naf + l - 1;
59 ed_set_infty(r);
60 for (i = l - 1; i >= 0; i--, _k--) {
61 n = *_k;
62 if (n == 0) {
63 /* doubling is followed by another doubling */
64 if (i > 0) {
65 r->coord = EXTND;
66 ed_dbl(r, r);
67 } else {
68 /* use full extended coordinate doubling for last step */
69 ed_dbl(r, r);
70 }
71 } else {
72 ed_dbl(r, r);
73 if (n > 0) {
74 ed_add(r, r, t[n / 2]);
75 } else if (n < 0) {
76 ed_sub(r, r, t[-n / 2]);
77 }
78 }
79 }
80 /* Convert r to affine coordinates. */
81 ed_norm(r, r);
82 if (bn_sign(k) == RLC_NEG) {
83 ed_neg(r, r);
84 }
85}
86
87#endif /* ED_FIX == LWNAF */

99static void ed_mul_combs_plain(ed_t r, const ed_t * t, const bn_t k) {
100 int i, j, l, w, n0, p0, p1;
101 bn_t n;
102
103 bn_null(n);
104
105 RLC_TRY {
106 bn_new(n);
107
108 ed_curve_get_ord(n);
109 l = bn_bits(n);
110 l = ((l % RLC_DEPTH) == 0 ? (l / RLC_DEPTH) : (l / RLC_DEPTH) + 1);
111
112 n0 = bn_bits(k);
113
114 p0 = (RLC_DEPTH) * l - 1;
115
116 w = 0;
117 p1 = p0--;
118 for (j = RLC_DEPTH - 1; j >= 0; j--, p1 -= l) {
119 w = w << 1;
120 if (p1 < n0 && bn_get_bit(k, p1)) {
121 w = w | 1;
122 }
123 }
124 ed_copy(r, t[w]);
125
126 for (i = l - 2; i >= 0; i--) {
127 ed_dbl(r, r);
128
129 w = 0;
130 p1 = p0--;
131 for (j = RLC_DEPTH - 1; j >= 0; j--, p1 -= l) {
132 w = w << 1;
133 if (p1 < n0 && bn_get_bit(k, p1)) {
134 w = w | 1;
135 }
136 }
137 if (w > 0) {
138 ed_add(r, r, t[w]);
139 }
140 }
141 ed_norm(r, r);
142 if (bn_sign(k) == RLC_NEG) {
143 ed_neg(r, r);
144 }
145 }
146 RLC_CATCH_ANY {
147 RLC_THROW(ERR_CAUGHT);
148 }
149 RLC_FINALLY {
150 bn_free(n);
151 }
152}

187void ed_mul_fix_basic(ed_t r, const ed_t *t, const bn_t k) {
188 if (bn_is_zero(k)) {
189 ed_set_infty(r);
190 return;
191 }
192
193 ed_set_infty(r);
194
195 for (int i = 0; i < bn_bits(k); i++) {
196 if (bn_get_bit(k, i)) {
197 ed_add(r, r, t[i]);
198 }
199 }
200 ed_norm(r, r);
201 if (bn_sign(k) == RLC_NEG) {
202 ed_neg(r, r);
203 }
204}

212 bn_t n;

304void ed_mul_fix_combd(ed_t r, const ed_t * t, const bn_t k) {
305 int i, j, d, e, w0, w1, n0, p0, p1;
306 bn_t n;
307
308 bn_null(n);
309
310 RLC_TRY {
311 bn_new(n);
312
313 ed_curve_get_ord(n);
314 d = bn_bits(n);
315 d = ((d % RLC_DEPTH) == 0 ? (d / RLC_DEPTH) : (d / RLC_DEPTH) + 1);
316 e = (d % 2 == 0 ? (d / 2) : (d / 2) + 1);
317
318 ed_set_infty(r);
319 n0 = bn_bits(k);
320
321 p1 = (e - 1) + (RLC_DEPTH - 1) * d;
322 for (i = e - 1; i >= 0; i--) {
323 ed_dbl(r, r);
324
325 w0 = 0;
326 p0 = p1;
327 for (j = RLC_DEPTH - 1; j >= 0; j--, p0 -= d) {
328 w0 = w0 << 1;
329 if (p0 < n0 && bn_get_bit(k, p0)) {
330 w0 = w0 | 1;
331 }
332 }
333
334 w1 = 0;
335 p0 = p1-- + e;
336 for (j = RLC_DEPTH - 1; j >= 0; j--, p0 -= d) {
337 w1 = w1 << 1;
338 if (i + e < d && p0 < n0 && bn_get_bit(k, p0)) {
339 w1 = w1 | 1;
340 }
341 }
342
343 ed_add(r, r, t[w0]);
344 ed_add(r, r, t[(1 << RLC_DEPTH) + w1]);
345 }
346 ed_norm(r, r);
347 if (bn_sign(k) == RLC_NEG) {
348 ed_neg(r, r);
349 }
350 }
351 RLC_CATCH_ANY {
352 RLC_THROW(ERR_CAUGHT);
353 }
354 RLC_FINALLY {
355 bn_free(n);
356 }
357}
