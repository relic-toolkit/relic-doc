50static void eb_dbl_basic_imp(eb_t r, const eb_t p) {
51 fb_t t0, t1, t2;
52
53 fb_null(t0);
54 fb_null(t1);
55 fb_null(t2);
56
57 RLC_TRY {
58 fb_new(t0);
59 fb_new(t1);
60 fb_new(t2);
61
62 /* t0 = 1/x1. */
63 fb_inv(t0, p->x);
64 /* t0 = y1/x1. */
65 fb_mul(t0, t0, p->y);
66 /* t0 = lambda = x1 + y1/x1. */
67 fb_add(t0, t0, p->x);
68 /* t1 = lambda^2. */
69 fb_sqr(t1, t0);
70 /* t2 = lambda^2 + lambda. */
71 fb_add(t2, t1, t0);
72
73 /* t2 = lambda^2 + lambda + a2. */
74 switch (eb_curve_opt_a()) {
75 case RLC_ZERO:
76 break;
77 case RLC_ONE:
78 fb_add_dig(t2, t2, (dig_t)1);
79 break;
80 case RLC_TINY:
81 fb_add_dig(t2, t2, eb_curve_get_a()[0]);
82 break;
83 default:
84 fb_add(t2, t2, eb_curve_get_a());
85 break;
86 }
87
88 /* t1 = x1 + x3. */
89 fb_add(t1, t2, p->x);
90
91 /* t1 = lambda * (x1 + x3). */
92 fb_mul(t1, t0, t1);
93
94 fb_copy(r->x, t2);
95 /* y3 = lambda * (x1 + x3) + x3 + y1. */
96 fb_add(t1, t1, r->x);
97 fb_add(r->y, t1, p->y);
98
99 fb_copy(r->z, p->z);
100
101 r->coord = BASIC;
102 }
103 RLC_CATCH_ANY {
104 RLC_THROW(ERR_CAUGHT);
105 }
106 RLC_FINALLY {
107 fb_free(t0);
108 fb_free(t1);
109 fb_free(t2);
110 }
111}

113#endif /* EB_ADD == BASIC */

124static void eb_dbl_projc_imp(eb_t r, const eb_t p) {
125 fb_t t0, t1;
126
127 fb_null(t0);
128 fb_null(t1);
129
130 RLC_TRY {
131 fb_new(t0);
132 fb_new(t1);
133
134 /* t0 = B = x1^2. */
135 fb_sqr(t0, p->x);
136 /* C = B + y1. */
137 fb_add(r->y, t0, p->y);
138
139 if (p->coord != BASIC) {
140 /* A = x1 * z1. */
141 fb_mul(t1, p->x, p->z);
142 /* z3 = A^2. */
143 fb_sqr(r->z, t1);
144 } else {
145 /* if z1 = 1, A = x1. */
146 fb_copy(t1, p->x);
147 /* if z1 = 1, z3 = x1^2. */
148 fb_copy(r->z, t0);
149 }
150
151 /* t1 = D = A * C. */
152 fb_mul(t1, t1, r->y);
153
154 /* C^2 + D. */
155 fb_sqr(r->y, r->y);
156 fb_add(r->x, t1, r->y);
157
158 /* C^2 + D + a2 * z3. */
159 switch (eb_curve_opt_a()) {
160 case RLC_ZERO:
161 break;
162 case RLC_ONE:
163 fb_add(r->x, r->z, r->x);
164 break;
165 case RLC_TINY:
166 fb_mul_dig(r->y, r->z, eb_curve_get_a()[0]);
167 fb_add(r->x, r->y, r->x);
168 break;
169 default:
170 fb_mul(r->y, r->z, eb_curve_get_a());
171 fb_add(r->x, r->y, r->x);
172 break;
173 }
174
175 /* t1 = (D + z3). */
176 fb_add(t1, t1, r->z);
177 /* t0 = B^2. */
178 fb_sqr(t0, t0);
179 /* t0 = B^2 * z3. */
180 fb_mul(t0, t0, r->z);
181 /* y3 = (D + z3) * r3 + B^2 * z3. */
182 fb_mul(r->y, t1, r->x);
183 fb_add(r->y, r->y, t0);
184
185 r->coord = PROJC;
186 }
187 RLC_CATCH_ANY {
188 RLC_THROW(ERR_CAUGHT);
189 }
190 RLC_FINALLY {
191 fb_free(t0);
192 fb_free(t1);
193 }
194}

196#endif /* EB_ADD == PROJC */
