84void pp_add_k48_projc(fp48_t l, ep8_t r, const ep8_t q, const ep_t p) {

88 fp8_null(t1);
89 fp8_null(t2);
90 fp8_null(t3);

95 fp8_new(t1);
96 fp8_new(t2);
97 fp8_new(t3);
98 fp8_new(t4);
99
100 /* B = t0 = x1 - x2 * z1. */
101 fp8_mul(t0, r->z, q->x);
102 fp8_sub(t0, r->x, t0);
103 /* A = t1 = y1 - y2 * z1. */
104 fp8_mul(t1, r->z, q->y);
105 fp8_sub(t1, r->y, t1);
106
107 /* D = B^2. */
108 fp8_sqr(t2, t0);
109 /* G = x1 * D. */
110 fp8_mul(r->x, r->x, t2);
111 /* E = B^3. */
112 fp8_mul(t2, t2, t0);
113 /* C = A^2. */
114 fp8_sqr(t3, t1);
115 /* F = E + z1 * C. */
116 fp8_mul(t3, t3, r->z);
117 fp8_add(t3, t2, t3);
118
119 /* l10 = - (A * xp). */
120 fp_neg(t4[0][0][0], p->x);
121 fp_mul(l[0][1][0][0][0], t1[0][0][0], t4[0][0][0]);
122 fp_mul(l[0][1][0][0][1], t1[0][0][1], t4[0][0][0]);
123 fp_mul(l[0][1][0][1][0], t1[0][1][0], t4[0][0][0]);

129
130 /* t4 = B * x2. */
131 fp8_mul(t4, q->x, t1);
132
133 /* H = E + F - 2 * G. */
134 fp8_sub(t3, t3, r->x);
135 fp8_sub(t3, t3, r->x);
136 /* y3 = A * (G - H) - y1 * E. */
137 fp8_sub(r->x, r->x, t3);
138 fp8_mul(t1, t1, r->x);
139 fp8_mul(r->y, t2, r->y);
140 fp8_sub(r->y, t1, r->y);
141 /* x3 = B * H. */
142 fp8_mul(r->x, t0, t3);
143 /* z3 = z1 * E. */
144 fp8_mul(r->z, r->z, t2);
145
146 /* l11 = J = B * x2 - A * y2. */
147 fp8_mul(t2, q->y, t0);
148 fp8_sub(l[0][0], t4, t2);
149
150 /* l00 = B * yp. */
151 fp_mul(l[1][1][0][0][0], p->y, t0[0][0][0]);
152 fp_mul(l[1][1][0][0][1], p->y, t0[0][0][1]);

156 fp_mul(l[1][1][1][0][1], p->y, t0[1][0][1]);
157 fp_mul(l[1][1][1][1][0], p->y, t0[1][1][0]);
158 fp_mul(l[1][1][1][1][1], p->y, t0[1][1][1]);
159 }
160 RLC_CATCH_ANY {
161 RLC_THROW(ERR_CAUGHT);
162 } RLC_FINALLY {
163 fp8_free(t0);
164 fp8_free(t1);
165 fp8_free(t2);
166 fp8_free(t3);
167 fp8_free(t4);
168 }
169}
