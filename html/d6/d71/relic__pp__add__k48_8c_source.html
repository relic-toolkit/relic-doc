40 static void ep8_add_basic(fp8_t s, fp8_t rx, fp8_t ry, fp8_t qx, fp8_t qy) {
41  fp8_t t0, t1, t2;
42 
43  fp8_null(t0);
44  fp8_null(t1);
45  fp8_null(t2);
46 
47  TRY {
48  fp8_new(t0);
49  fp8_new(t1);
50  fp8_new(t2);
51 
52  /* t0 = x2 - x1. */
53  fp8_sub(t0, qx, rx);
54  /* t1 = y2 - y1. */
55  fp8_sub(t1, qy, ry);
56 
57  /* If t0 is zero. */
58  if (fp8_is_zero(t0)) {
59  if (fp8_is_zero(t1)) {
60  /* If t1 is zero, q = p, should have doubled. */
61  //ep8_dbl_basic(s, rx, ry);
62  THROW(ERR_NO_VALID);
63  } else {
64  /* If t1 is not zero and t0 is zero, q = -p and r = infty. */
65  fp8_zero(rx);
66  fp8_zero(ry);
67  }
68  } else {
69  /* t2 = 1/(x2 - x1). */
70  fp8_inv(t2, t0);
71  /* t2 = lambda = (y2 - y1)/(x2 - x1). */
72  fp8_mul(t2, t1, t2);
73 
74  /* x3 = lambda^2 - x2 - x1. */
75  fp8_sqr(t1, t2);
76  fp8_sub(t0, t1, rx);
77  fp8_sub(t0, t0, qx);
78 
79  /* y3 = lambda * (x1 - x3) - y1. */
80  fp8_sub(t1, rx, t0);
81  fp8_mul(t1, t2, t1);
82  fp8_sub(ry, t1, ry);
83 
84  fp8_copy(rx, t0);
85 
86  if (s != NULL) {
87  fp8_copy(s, t2);
88  }
89  }
90  }
91  CATCH_ANY {
92  THROW(ERR_CAUGHT);
93  }
94  FINALLY {
95  fp8_free(t0);
96  fp8_free(t1);
97  fp8_free(t2);
98  }
99 }
