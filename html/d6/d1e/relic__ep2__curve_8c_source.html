329 void ep2_curve_get_vs(bn_t *v) {
330  bn_t x, t;
331 
332  bn_null(x);
333  bn_null(t);
334 
335  TRY {
336  bn_new(x);
337  bn_new(t);
338 
339  fp_prime_get_par(x);
340  bn_copy(v[1], x);
341  bn_copy(v[2], x);
342  bn_copy(v[3], x);
343 
344  /* t = 2x^2. */
345  bn_sqr(t, x);
346  bn_dbl(t, t);
347 
348  /* v0 = 2x^2 + 3x + 1. */
349  bn_mul_dig(v[0], x, 3);
350  bn_add_dig(v[0], v[0], 1);
351  bn_add(v[0], v[0], t);
352 
353  /* v3 = -(2x^2 + x). */
354  bn_add(v[3], v[3], t);
355  bn_neg(v[3], v[3]);
356 
357  /* v1 = 12x^3 + 8x^2 + x, v2 = 6x^3 + 4x^2 + x. */
358  bn_dbl(t, t);
359  bn_add(v[2], v[2], t);
360  bn_dbl(t, t);
361  bn_add(v[1], v[1], t);
362  bn_rsh(t, t, 2);
363  bn_mul(t, t, x);
364  bn_mul_dig(t, t, 3);
365  bn_add(v[2], v[2], t);
366  bn_dbl(t, t);
367  bn_add(v[1], v[1], t);
368  } CATCH_ANY {
369  THROW(ERR_CAUGHT);
370  } FINALLY {
371  bn_free(x);
372  bn_free(t);
373  }
