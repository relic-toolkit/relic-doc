364static int ep2_curve_get_coeffs(fp2_t *coeffs, const char *str) {
365 int degree = 0;
366 unsigned offset = 0;
367
368 if (str[0] == '\0') {
369 /* need nonzero strlen */
370 RLC_THROW(ERR_NO_VALID);
371 return 0;
372 }
373 for (; degree < RLC_EPX_CTMAP_MAX; ++degree) {
374 /* first coeff */
375 const char *end = strchr(str + offset, ',');
376 if (end == NULL) {
377 /* should not happen --- means there's no second coeff */
378 RLC_THROW(ERR_NO_VALID);
379 }
380 unsigned len = end - str - offset;
381 fp_read_str(coeffs[degree][0], str + offset, len, 16);
382 offset += len + 1; /* move to after ',' */
383
384 /* second coeff */
385 end = strchr(str + offset, ';');
386 if (end == NULL) {
387 /* last one */
388 fp_read_str(coeffs[degree][1], str + offset, strlen(str + offset), 16);
389 break;
390 }
391 len = end - str - offset;
392 fp_read_str(coeffs[degree][1], str + offset, len, 16);
393 offset += len + 1; /* move to after ';' */
394 }
395 if (degree == RLC_EPX_CTMAP_MAX) {
396 /* ran out of space before converting all coeffs */
397 RLC_THROW(ERR_NO_VALID);
398 }
399 return degree;
400}

438static void ep2_curve_set_map(void) {
439 bn_t t;
440 bn_null(t);
441
442 const int abNeq0 = (ep2_curve_opt_a() != RLC_ZERO) && (ep2_curve_opt_b() != RLC_ZERO);
443
444 ctx_t *ctx = core_get();
445 fp_t *c1 = ctx->ep2_map_c[0];
446 fp_t *c2 = ctx->ep2_map_c[1];
447 fp_t *c3 = ctx->ep2_map_c[2];
448 fp_t *c4 = ctx->ep2_map_c[3];
449
450 RLC_TRY {
451 bn_new(t);
452
453 if (ep2_curve_is_ctmap() || abNeq0) {
454 /* SSWU map constants */
455 /* constants 3 and 4 are a and b for the curve or isogeny */
456#ifdef EP_CTMAP
457 if (ep2_curve_is_ctmap()) {
458 fp2_copy(c3, ctx->ep2_iso.a);
459 fp2_copy(c4, ctx->ep2_iso.b);
460 } else {
461#endif
462 fp2_copy(c3, ctx->ep2_a);
463 fp2_copy(c4, ctx->ep2_b);
464#ifdef EP_CTMAP
465 }
466#endif
467 /* constant 1: -b / a */
468 fp2_neg(c1, c3); /* c1 = -a */
469 fp2_inv(c1, c1); /* c1 = -1 / a */
470 fp2_mul(c1, c1, c4); /* c1 = -b / a */
471
472 /* constant 2 is unused in this case */
473 } else {
474 /* SvdW map constants */
475 /* constant 1: g(u) = u^3 + a * u + b */
476 fp2_sqr(c1, ctx->ep2_map_u);
477 fp2_add(c1, c1, ctx->ep2_a);
478 fp2_mul(c1, c1, ctx->ep2_map_u);
479 fp2_add(c1, c1, ctx->ep2_b);
480
481 /* constant 2: -u / 2 */
482 fp2_set_dig(c2, 2);
483 fp2_neg(c2, c2); /* -2 */
484 fp2_inv(c2, c2); /* -1 / 2 */
485 fp2_mul(c2, c2, ctx->ep2_map_u); /* -u / 2 */
486
487 /* constant 3: sqrt(-g(u) * (3 * u^2 + 4 * a)) */
488 fp2_sqr(c3, ctx->ep2_map_u); /* u^2 */
489 fp2_mul_dig(c3, c3, 3); /* 3 * u^2 */
490 fp2_dbl(c4, ctx->ep2_a); /* 4 * a */
491 fp2_dbl(c4, c4);
492 fp2_add(c4, c3, c4); /* 3 * u^2 + 4 * a */
493 fp2_neg(c4, c4); /* -(3 * u^2 + 4 * a) */
494 fp2_mul(c3, c4, c1); /* -g(u) * (3 * u^2 + 4 * a) */
495 if (!fp2_srt(c3, c3)) {
496 RLC_THROW(ERR_NO_VALID);
497 }
498 /* make sure sgn0(c3) == 0 */
499 const int c30_z = fp_is_zero(c3[0]);
500 fp_prime_back(t, c3[0]);
501 const int c30_n = bn_get_bit(t, 0);
502 fp_prime_back(t, c3[1]);
503 const int c31_n = bn_get_bit(t, 0);
504 if (c30_n | (c30_z & c31_n)) {
505 /* set sgn0(c3) == 0 */
506 fp2_neg(c3, c3);
507 }
508
509 /* constant 4: -4 * g(u) / (3 * u^2 + 4 * a) */
510 fp2_inv(c4, c4); /* -1 / (3 * u^2 + 4 * a */
511 fp2_mul(c4, c4, c1); /* -g(u) / (3 * u^2 + 4 * a) */
512 fp2_dbl(c4, c4); /* -4 * g(u) / (3 * u^2 + 4 * a) */
513 fp2_dbl(c4, c4);
514 }
515 }
516 RLC_CATCH_ANY {
517 RLC_THROW(ERR_CAUGHT);
518 }
519 RLC_FINALLY {
520 bn_free(t);
521 }
522}

527static void detect_opt(int *opt, fp2_t a) {
528 fp2_t t;
529 fp2_null(t);
530
531 RLC_TRY {
532 fp2_new(t);
533 fp2_set_dig(t, 3);
534 fp2_neg(t, t);
535
536 if (fp2_cmp(a, t) == RLC_EQ) {
537 *opt = RLC_MIN3;
538 } else if (fp2_is_zero(a)) {
539 *opt = RLC_ZERO;
540 } else if (fp2_cmp_dig(a, 1) == RLC_EQ) {
541 *opt = RLC_ONE;
542 } else if (fp2_cmp_dig(a, 2) == RLC_EQ) {
543 *opt = RLC_TWO;
544 } else if ((fp_bits(a[0]) <= RLC_DIG) && fp_is_zero(a[1])) {
545 *opt = RLC_TINY;
546 } else {
547 *opt = RLC_HUGE;
548 }
549 }
550 RLC_CATCH_ANY {
551 RLC_THROW(ERR_CAUGHT);
552 }
553 RLC_FINALLY {
554 fp2_free(t);
555 }
556}
