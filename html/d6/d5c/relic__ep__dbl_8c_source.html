48 static void ep_dbl_basic_imp(ep_t r, fp_t s, const ep_t p) {
49  fp_t t0, t1, t2;
50 
51  fp_null(t0);
52  fp_null(t1);
53  fp_null(t2);
54 
55  TRY {
56  fp_new(t0);
57  fp_new(t1);
58  fp_new(t2);
59 
60  /* t0 = 1/2 * y1. */
61  fp_dbl(t0, p->y);
62  fp_inv(t0, t0);
63 
64  /* t1 = 3 * x1^2 + a. */
65  fp_sqr(t1, p->x);
66  fp_copy(t2, t1);
67  fp_dbl(t1, t1);
68  fp_add(t1, t1, t2);
69 
70  switch (ep_curve_opt_a()) {
71  case RLC_ZERO:
72  break;
73  case RLC_ONE:
74  fp_add_dig(t1, t1, (dig_t)1);
75  break;
76 #if FP_RDC != MONTY
77  case RLC_TINY:
78  fp_add_dig(t1, t1, ep_curve_get_a()[0]);
79  break;
80 #endif
81  default:
82  fp_add(t1, t1, ep_curve_get_a());
83  break;
84  }
85 
86  /* t1 = (3 * x1^2 + a)/(2 * y1). */
87  fp_mul(t1, t1, t0);
88 
89  if (s != NULL) {
90  fp_copy(s, t1);
91  }
92 
93  /* t2 = t1^2. */
94  fp_sqr(t2, t1);
95 
96  /* x3 = t1^2 - 2 * x1. */
97  fp_dbl(t0, p->x);
98  fp_sub(t0, t2, t0);
99 
100  /* y3 = t1 * (x1 - x3) - y1. */
101  fp_sub(t2, p->x, t0);
102  fp_mul(t1, t1, t2);
103  fp_sub(r->y, t1, p->y);
104 
105  fp_copy(r->x, t0);
106  fp_copy(r->z, p->z);
107 
108  r->norm = 1;
109  }
110  CATCH_ANY {
111  THROW(ERR_CAUGHT);
112  }
113  FINALLY {
114  fp_free(t0);
115  fp_free(t1);
116  fp_free(t2);
117  }
118 }

131 static void ep_dbl_projc_imp(ep_t r, const ep_t p) {
132  fp_t t0, t1, t2, t3, t4, t5;
133 
134  fp_null(t1);
135  fp_null(t2);
136  fp_null(t3);
137  fp_null(t4);
138  fp_null(t5);
139 
140  TRY {
141  fp_new(t0);
142  fp_new(t1);
143  fp_new(t2);
144  fp_new(t3);
145  fp_new(t4);
146  fp_new(t5);
147 
148  if (!p->norm && ep_curve_opt_a() == RLC_MIN3) {
149  /* dbl-2001-b formulas: 3M + 5S + 8add + 1*4 + 2*8 + 1*3 */
150  /* http://www.hyperelliptic.org/EFD/g1p/auto-shortw-jacobian-3.html#doubling-dbl-2001-b */
151 
152  /* t0 = delta = z1^2. */
153  fp_sqr(t0, p->z);
154 
155  /* t1 = gamma = y1^2. */
156  fp_sqr(t1, p->y);
157 
158  /* t2 = beta = x1 * y1^2. */
159  fp_mul(t2, p->x, t1);
160 
161  /* t3 = alpha = 3 * (x1 - z1^2) * (x1 + z1^2). */
162  fp_sub(t3, p->x, t0);
163  fp_add(t4, p->x, t0);
164  fp_mul(t4, t3, t4);
165  fp_dbl(t3, t4);
166  fp_add(t3, t3, t4);
167 
168  /* x3 = alpha^2 - 8 * beta. */
169  fp_dbl(t2, t2);
170  fp_dbl(t2, t2);
171  fp_dbl(t5, t2);
172  fp_sqr(r->x, t3);
173  fp_sub(r->x, r->x, t5);
174 
175  /* z3 = (y1 + z1)^2 - gamma - delta. */
176  fp_add(r->z, p->y, p->z);
177  fp_sqr(r->z, r->z);
178  fp_sub(r->z, r->z, t1);
179  fp_sub(r->z, r->z, t0);
180 
181  /* y3 = alpha * (4 * beta - x3) - 8 * gamma^2. */
182  fp_dbl(t1, t1);
183  fp_sqr(t1, t1);
184  fp_dbl(t1, t1);
185  fp_sub(r->y, t2, r->x);
186  fp_mul(r->y, r->y, t3);
187  fp_sub(r->y, r->y, t1);
188  } else if (ep_curve_opt_a() == RLC_ZERO) {
189  /* dbl-2009-l formulas: 2M + 5S + 6add + 1*8 + 3*2 + 1*3. */
190  /* http://www.hyperelliptic.org/EFD/g1p/auto-shortw-jacobian-0.html#doubling-dbl-2009-l */
191 
192  /* A = X1^2 */
193  fp_sqr(t0, p->x);
194 
195  /* B = Y1^2 */
196  fp_sqr(t1, p->y);
197 
198  /* C = B^2 */
199  fp_sqr(t2, t1);
200 
201  /* D = 2*((X1+B)^2-A-C) */
202  fp_add(t1, t1, p->x);
203  fp_sqr(t1, t1);
204  fp_sub(t1, t1, t0);
205  fp_sub(t1, t1, t2);
206  fp_dbl(t1, t1);
207 
208  /* E = 3*A */
209  fp_dbl(t3, t0);
210  fp_add(t0, t3, t0);
211 
212  /* F = E^2 */
213  fp_sqr(t3, t0);
214 
215  /* Z3 = 2*Y1*Z1 */
216  fp_mul(r->z, p->y, p->z);
217  fp_dbl(r->z, r->z);
218 
219  /* X3 = F-2*D */
220  fp_sub(r->x, t3, t1);
221  fp_sub(r->x, r->x, t1);
222 
223  /* Y3 = E*(D-X3)-8*C */
224  fp_sub(r->y, t1, r->x);
225  fp_mul(r->y, r->y, t0);
226  fp_dbl(t2, t2);
227  fp_dbl(t2, t2);
228  fp_dbl(t2, t2);
229  fp_sub(r->y, r->y, t2);
230  } else {
231  /* dbl-2007-bl formulas: 1M + 8S + 1*a + 10add + 1*8 + 2*2 + 1*3 */
232  /* http://www.hyperelliptic.org/EFD/g1p/auto-shortw-jacobian.html#doubling-dbl-2007-bl */
233 
234  /* t0 = x1^2, t1 = y1^2, t2 = y1^4. */
235  fp_sqr(t0, p->x);
236  fp_sqr(t1, p->y);
237  fp_sqr(t2, t1);
238 
239  if (!p->norm) {
240  /* t3 = z1^2. */
241  fp_sqr(t3, p->z);
242 
243  if (ep_curve_opt_a() == RLC_ZERO) {
244  /* z3 = 2 * y1 * z1. */
245  fp_mul(r->z, p->y, p->z);
246  fp_dbl(r->z, r->z);
247  } else {
248  /* z3 = (y1 + z1)^2 - y1^2 - z1^2. */
249  fp_add(r->z, p->y, p->z);
250  fp_sqr(r->z, r->z);
251  fp_sub(r->z, r->z, t1);
252  fp_sub(r->z, r->z, t3);
253  }
254  } else {
255  /* z3 = 2 * y1. */
256  fp_dbl(r->z, p->y);
257  }
258 
259  /* t4 = S = 2*((x1 + y1^2)^2 - x1^2 - y1^4). */
260  fp_add(t4, p->x, t1);
261  fp_sqr(t4, t4);
262  fp_sub(t4, t4, t0);
263  fp_sub(t4, t4, t2);
264  fp_dbl(t4, t4);
265 
266  /* t5 = M = 3 * x1^2 + a * z1^4. */
267  fp_dbl(t5, t0);
268  fp_add(t5, t5, t0);
269  if (!p->norm) {
270  fp_sqr(t3, t3);
271  switch (ep_curve_opt_a()) {
272  case RLC_ZERO:
273  break;
274  case RLC_ONE:
275  fp_add(t5, t5, t3);
276  break;
277  case RLC_TINY:
278  fp_mul_dig(t1, t3, ep_curve_get_a()[0]);
279  fp_add(t5, t5, t1);
280  break;
281  default:
282  fp_mul(t1, t3, ep_curve_get_a());
283  fp_add(t5, t5, t1);
284  break;
285  }
286  } else {
287  switch (ep_curve_opt_a()) {
288  case RLC_ZERO:
289  break;
290  case RLC_ONE:
291  fp_add_dig(t5, t5, (dig_t)1);
292  break;
293  case RLC_TINY:
294  fp_add_dig(t5, t5, ep_curve_get_a()[0]);
295  break;
296  default:
297  fp_add(t5, t5, ep_curve_get_a());
298  break;
299  }
300  }
301 
302  /* x3 = T = M^2 - 2 * S. */
303  fp_sqr(r->x, t5);
304  fp_dbl(t1, t4);
305  fp_sub(r->x, r->x, t1);
306 
307  /* y3 = M * (S - T) - 8 * y1^4. */
308  fp_dbl(t2, t2);
309  fp_dbl(t2, t2);
310  fp_dbl(t2, t2);
311  fp_sub(t4, t4, r->x);
312  fp_mul(t5, t5, t4);
313  fp_sub(r->y, t5, t2);
314  }
315 
316  r->norm = 0;
317  }
318  CATCH_ANY {
319  THROW(ERR_CAUGHT);
320  }
321  FINALLY {
322  fp_free(t0);
323  fp_free(t1);
324  fp_free(t2);
325  fp_free(t3);
326  fp_free(t4);
327  fp_free(t5);
328  }
329 }
