46static void ep3_mul_cof_k18(ep3_t r, const ep3_t p) {
47 ep3_t t0, t1, t2, t3, t4, t5;
48 bn_t x;
49
50 ep3_null(t0);
51 ep3_null(t1);
52 ep3_null(t2);
53 ep3_null(t3);
54 ep3_null(t4);
55 ep3_null(t5);
56 bn_null(x);
57
58 RLC_TRY {
59 ep3_new(t0);
60 ep3_new(t1);
61 ep3_new(t2);
62 ep3_new(t3);
63 ep3_new(t4);
64 ep3_new(t5);
65 bn_new(x);
66
67 /* Method from "Faster Hashing to G2" by Laura Fuentes-Castañeda,
68 * Edward Knapp and Francisco Rodríguez-Henríquez.
69 . */
70
71 /* Compute multi-addition-subtraction chain \sum \lambda_i \psi^P, where
72 * \lambda_0 = 5u + 18
73 * \lambda_1 = (u^3+3u^2+1)
74 * \lambda_2 = -(3u^2+8*u)
75 * \lambda_3 = (3u+1)
76 * \lambda_4 = -(u^2+2)
77 * \lambda_5 = (u^2+5u)
78
79 * We will write the subscalars below as vectors for simplicity.
80 */
81
82 fp_prime_get_par(x);
83
84 /* t0 = [u]P, t4 = [u^2]P, later t2 = [u^3]P. */
85 ep3_mul_basic(t0, p, x);
86 ep3_mul_basic(t4, t0, x);
87
88 /* t1 = [1, 0, -u, 0, 0, 0]. */
89 ep3_frb(t1, t0, 2);
90 /* t2 = [u, 0, 0, 0, 0, u]. */
91 ep3_frb(t2, t1, 3);
92 ep3_add(t2, t2, t0);
93 /* t3 = [0, 0, 0, -u, 0, 0]. */
94 ep3_frb(t3, t1, 1);
95 ep3_neg(t1, t1);
96 ep3_add(t1, t1, p);
97 /* t0 = [u+3, 0, -u, 0, 0, u]. */
98 ep3_dbl(t0, p);
99 ep3_add(t0, t0, t2);
100 ep3_add(t0, t0, t1);
101
102 /* t2 = [0, 0, u^2, 0, 0, 0], t3 = [0, u^2, -u^2, -u, 0, 0]. */
103 ep3_frb(t2, t4, 1);
104 ep3_add(t3, t3, t2);
105 ep3_frb(t2, t2, 1);
106 ep3_sub(t3, t3, t2);
107
108 /* t5 = [0, u^3 + 1, 0, 1, -u^2, u^2] */
109 ep3_frb(t5, p, 1);
110 ep3_frb(t2, t5, 2);
111 ep3_add(t5, t5, t2);
112 ep3_frb(t2, t4, 4);
113 ep3_sub(t5, t5, t2);
114 ep3_frb(t2, t2, 1);
115 ep3_add(t5, t5, t2);
116 ep3_mul_basic(t2, t4, x);
117 ep3_frb(t2, t2, 1);
118 ep3_add(t5, t5, t2);
119
120 /* t4 = [0, 0, 0, 0, -1, 0], t3 = [1, u^2, -u^2-u, -u, 0, 0]. */
121 ep3_frb(t4, p, 4);
122 ep3_neg(t4, t4);
123 ep3_add(t3, t3, t1);
124 /* t4 = [u+3, 0, -u, 0, -1, u]. */
125 ep3_add(t4, t4, t0);
126 /* t3 = [u+4, u^2, -u^2-2u, -u, -1, u]. */
127 ep3_add(t3, t3, t0);
128 /* t4 = [2u+7, u^2, -u^2-3u, -u, -1, 2u]. */
129 ep3_add(t4, t4, t3);
130 /* t3 = [u+4, u^3+u^2+1, -u^2-2u, -u+1, -u^2, u^2+u]. */
131 ep3_add(t3, t3, t5);
132 /* t4 = [4u+14, 2u^2, -2u^2-6u, -2u, -2, 4u]. */
133 ep3_dbl(t4, t4);
134 /* r = [5u+18, u^3+3u^2+1, -3u^2-8u, -3u+1, -u^2-2, u^2+5u]. */
135 ep3_add(r, t4, t3);
136 ep3_norm(r, r);
137 } RLC_CATCH_ANY {
138 RLC_THROW(ERR_CAUGHT);
139 } RLC_FINALLY {
140 ep3_free(t0);
141 ep3_free(t1);
142 ep3_free(t2);
143 ep3_free(t3);
144 ep3_free(t4);
145 ep3_free(t5);
146 bn_free(x);
147 }
148}

156static void ep3_mul_cof_sg18(ep3_t r, const ep3_t p) {
157 ep3_t t0, t1, t2, t3, t4;
158 bn_t x;
159
160 ep3_null(t0);
161 ep3_null(t1);
162 ep3_null(t2);
163 ep3_null(t3);
164 ep3_null(t4);
165 bn_null(x);
166
167 RLC_TRY {
168 ep3_new(t0);
169 ep3_new(t1);
170 ep3_new(t2);
171 ep3_new(t3);
172 ep3_new(t4);
173 bn_new(x);
174
175 /* Vector computed by Guillevic's MAGMA script:
176 [9*u^4-3*u^2+u, 3*u^2-1, -6*u^3 + 2*u, -2*u, 0, 3*u^3-u+1] */
177 fp_prime_get_par(x);
178
179 /* t0 = [u]P, t1 = [3u^2]P, t2 = [3u^3]P, t3 = [9u^4]P. */
180 ep3_mul_basic(t0, p, x);
181 bn_mul_dig(x, x, 3);
182 ep3_mul_basic(t1, t0, x);
183 bn_div_dig(x, x, 3);
184 ep3_mul_basic(t2, t1, x);
185 bn_mul_dig(x, x, 3);
186 ep3_mul_basic(t3, t2, x);
187 ep3_sub(t3, t3, t1);
188 ep3_add(t3, t3, t0);
189
190 ep3_sub(t4, t1, p),
191 ep3_frb(t4, t4, 1);
192 ep3_add(t3, t3, t4);
193
194 ep3_sub(t2, t2, t0);
195 ep3_frb(t4, t2, 2);
196 ep3_dbl(t4, t4);
197 ep3_sub(t3, t3, t4);
198 ep3_add(t2, t2, p);
199 ep3_frb(t2, t2, 5);
200 ep3_add(t3, t3, t2);
201
202 ep3_dbl(t4, t0);
203 ep3_frb(t4, t4, 3);
204 ep3_sub(r, t3, t4);
205 ep3_norm(r, r);
206 } RLC_CATCH_ANY {
207 RLC_THROW(ERR_CAUGHT);
208 } RLC_FINALLY {
209 ep3_free(t0);
210 ep3_free(t1);
211 ep3_free(t2);
212 ep3_free(t3);
213 ep3_free(t4);
214 bn_free(x);
215 }

224static void ep3_mul_cof_fm18(ep3_t r, const ep3_t p) {
225 ep3_t t0, t1, t2, t3, t4;
226 bn_t x;
227
228 ep3_null(t0);
229 ep3_null(t1);
230 ep3_null(t2);
231 ep3_null(t3);
232 ep3_null(t4);
233 bn_null(x);
234
235 RLC_TRY {
236 ep3_new(t0);
237 ep3_new(t1);
238 ep3_new(t2);
239 ep3_new(t3);
240 ep3_new(t4);
241 bn_new(x);
242
243 /* Vector computed by Guillevic's MAGMA script:
244 [2*x*(x+2)/3, x^3-(x+2)/3, -2*x^2*(x+2)/3, -x*(x^3+(x+2)/3), 2*(x+2)/3, x^2*(x^3+(x+2)/3)-1] */
245 fp_prime_get_par(x);
246
247 /* t0 = [(x+2)/3]P, t1 = [x]P. */
248 bn_add_dig(x, x, 2);
249 bn_div_dig(x, x, 3);
250 ep3_mul_basic(t0, p, x);
251 ep3_dbl(t1, t0);
252 ep3_add(t1, t1, t0);
253 ep3_dbl(t2, p);
254 ep3_sub(t1, t1, t2);
255
256 /* Compute t2 = [x*(x+2)/3]P, t1 = [3*x*(x+2)/3-2x]P = [x^2]P. */
257 fp_prime_get_par(x);
258 ep3_frb(t3, t0, 4);
259 ep3_mul_basic(t2, t0, x);
260 ep3_add(t3, t3, t2);
261 ep3_dbl(t4, t2);
262 ep3_add(t4, t4, t2);
263 ep3_dbl(t1, t1);
264 ep3_sub(t1, t4, t1);
265 ep3_norm(t1, t1);
266 /* Compute t2 = [x^2*(x+2)/3]P, */
267 ep3_mul_basic(t2, t2, x);
268 ep3_frb(t4, t2, 2);
269 ep3_sub(t3, t3, t4);
270 ep3_dbl(t3, t3);
271 ep3_mul_basic(t2, t1, x);
272 ep3_sub(t4, t2, t0);
273 ep3_frb(t4, t4, 1);
274 ep3_add(t3, t3, t4);
275 ep3_add(t4, t2, t0);
276 ep3_norm(t4, t4);
277 ep3_mul_basic(t2, t4, x);
278 ep3_frb(t4, t2, 3);
279 ep3_sub(t3, t3, t4);
280 ep3_mul_basic(t2, t2, x);
281 ep3_sub(t2, t2, p);
282 ep3_frb(t2, t2, 5);
283 ep3_add(t3, t3, t2);
284 ep3_norm(r, t3);
285 } RLC_CATCH_ANY {
286 RLC_THROW(ERR_CAUGHT);
287 } RLC_FINALLY {
288 ep3_free(t0);
289 ep3_free(t1);
290 ep3_free(t2);
291 ep3_free(t3);
292 ep3_free(t4);
293 bn_free(x);
294 }
295}

301void ep3_mul_cof(ep3_t r, const ep3_t p) {
302 bn_t k;
303
304 bn_null(k);
305
306 RLC_TRY {
307 switch (ep_curve_is_pairf()) {
308 case EP_K18:
309 ep3_mul_cof_k18(r, p);
310 break;
311 case EP_SG18:
312 ep3_mul_cof_sg18(r, p);
313 break;
314 case EP_FM18:
315 ep3_mul_cof_fm18(r, p);
316 break;
317 default:
318 /* Now, multiply by cofactor to get the correct group. */
319 ep3_curve_get_cof(k);
320 ep3_mul_big(r, p, k);
321 break;
322 }
323 } RLC_CATCH_ANY {
324 RLC_THROW(ERR_CAUGHT);
325 } RLC_FINALLY {
326 bn_free(k);
327 }
