175void ep3_sub(ep3_t r, const ep3_t p, const ep3_t q) {
176 ep3_t t;
177
178 ep3_null(t);
179
180 if (p == q) {
181 ep3_set_infty(r);
182 return;
183 }
184
185 RLC_TRY {
186 ep3_new(t);
187 ep3_neg(t, q);
188 ep3_add(r, p, t);
189 }
190 RLC_CATCH_ANY {
191 RLC_THROW(ERR_CAUGHT);
192 }
193 RLC_FINALLY {
194 ep3_free(t);
195 }
