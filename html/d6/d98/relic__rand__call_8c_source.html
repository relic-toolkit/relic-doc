47static void rand_stub(uint8_t *buf, size_t size, void *args) {
48 int c, l, fd = open("/dev/urandom", O_RDONLY);
49
50 if (fd == -1) {
51 RLC_THROW(ERR_NO_FILE);
52 return;
53 }
54
55 l = 0;
56 do {
57 c = read(fd, buf + l, size - l);
58 l += c;
59 if (c == -1) {
60 RLC_THROW(ERR_NO_READ);
61 return;
62 }
63 } while (l < size);
64
65 close(fd);
66}
