40void bn_gcd_basic(bn_t c, const bn_t a, const bn_t b) {
41 bn_t u, v;
42
43 if (bn_is_zero(a)) {
44 bn_abs(c, b);
45 return;
46 }
47
48 if (bn_is_zero(b)) {
49 bn_abs(c, a);
50 return;
51 }
52
53 bn_null(u);
54 bn_null(v);
55
56 RLC_TRY {
57 bn_new(u);
58 bn_new(v);
59
60 bn_abs(u, a);
61 bn_abs(v, b);
62 while (!bn_is_zero(v)) {
63 bn_copy(c, v);
64 bn_mod(v, u, v);
65 bn_copy(u, c);
66 }
67 }
68 RLC_CATCH_ANY {
69 RLC_THROW(ERR_CAUGHT);
70 }
71 RLC_FINALLY {
72 bn_free(u);
73 bn_free(v);
74 }
75}

77void bn_gcd_ext_basic(bn_t c, bn_t d, bn_t e, const bn_t a, const bn_t b) {
78 bn_t u, v, x_1, y_1, q, r;
79
80 if (bn_is_zero(a)) {
81 bn_abs(c, b);
82 bn_zero(d);
83 if (e != NULL) {
84 bn_set_dig(e, 1);
85 }
86 return;
87 }
88
89 if (bn_is_zero(b)) {
90 bn_abs(c, a);
91 bn_set_dig(d, 1);
92 if (e != NULL) {
93 bn_zero(e);
94 }
95 return;
96 }
97
98 bn_null(u);
99 bn_null(v);
100 bn_null(x_1);
101 bn_null(y_1);
102 bn_null(q);
103 bn_null(r);
104
105 RLC_TRY {
106 bn_new(u);
107 bn_new(v);
108 bn_new(x_1);
109 bn_new(y_1);
110 bn_new(q);
111 bn_new(r);
112
113 bn_abs(u, a);
114 bn_abs(v, b);
115
116 bn_zero(x_1);
117 bn_set_dig(y_1, 1);
118 bn_set_dig(d, 1);
119 if (e != NULL) {
120 bn_zero(e);
121 }
122
123 while (!bn_is_zero(v)) {
124 bn_div_rem(q, r, u, v);
125
126 bn_copy(u, v);
127 bn_copy(v, r);
128
129 bn_mul(c, q, x_1);
130 bn_sub(r, d, c);
131 bn_copy(d, x_1);
132 bn_copy(x_1, r);
133
134 if (e != NULL) {
135 bn_mul(c, q, y_1);
136 bn_sub(r, e, c);
137 bn_copy(e, y_1);
138 bn_copy(y_1, r);
139 }
140 }
141 bn_copy(c, u);
142 }
143 RLC_CATCH_ANY {
144 RLC_THROW(ERR_CAUGHT);
145 }
146 RLC_FINALLY {
147 bn_free(u);
148 bn_free(v);
149 bn_free(x_1);
150 bn_free(y_1);
151 bn_free(q);
152 bn_free(r);
153 }
154}

160void bn_gcd_lehme(bn_t c, const bn_t a, const bn_t b) {
161 bn_t x, y, u, v, t0, t1, t2, t3;
162 dig_t _x, _y, q, _q, t, _t;
163 dis_t _a, _b, _c, _d;
164
165 if (bn_is_zero(a)) {
166 bn_abs(c, b);
167 return;
168 }
169
170 if (bn_is_zero(b)) {
171 bn_abs(c, a);
172 return;
173 }
174
175 bn_null(x);
176 bn_null(y);
177 bn_null(u);
178 bn_null(v);
179 bn_null(t0);
180 bn_null(t1);
181 bn_null(t2);
182 bn_null(t3);
183
184 /*
185 * Taken from Handbook of Hyperelliptic and Elliptic Cryptography.
186 */
187 RLC_TRY {
188 bn_new(x);
189 bn_new(y);
190 bn_new(u);
191 bn_new(v);
192 bn_new(t0);
193 bn_new(t1);
194 bn_new(t2);
195 bn_new(t3);
196
197 if (bn_cmp_abs(a, b) == RLC_GT) {
198 bn_abs(x, a);
199 bn_abs(y, b);
200 } else {
201 bn_abs(x, b);
202 bn_abs(y, a);
203 }
204 while (y->used > 1) {
205 if (bn_bits(x) > RLC_DIG) {
206 bn_rsh(u, x, bn_bits(x) - RLC_DIG);
207 bn_rsh(v, y, bn_bits(x) - RLC_DIG);
208 } else {
209 bn_copy(u, x);
210 bn_copy(v, y);
211 }
212 _x = u->dp[0];
213 _y = v->dp[0];
214 _a = _d = 1;
215 _b = _c = 0;
216 t = 0;
217 if (_y != 0) {
218 q = _x / _y;
219 t = _x % _y;
220 }
221 if (t >= ((dig_t)1 << (RLC_DIG / 2))) {
222 while (1) {
223 _q = _y / t;
224 _t = _y % t;
225 if (_t < ((dig_t)1 << (RLC_DIG / 2))) {
226 break;
227 }
228 _x = _y;
229 _y = t;
230 t = _a - q * _c;
231 _a = _c;
232 _c = t;
233 t = _b - q * _d;
234 _b = _d;
235 _d = t;
236 t = _t;
237 q = _q;
238 }
239 }
240 if (_b == 0) {
241 bn_mod(t0, x, y);
242 bn_copy(x, y);
243 bn_copy(y, t0);
244 } else {
245 if (bn_bits(x) > 2 * RLC_DIG) {
246 bn_rsh(u, x, bn_bits(x) - 2 * RLC_DIG);
247 bn_rsh(v, y, bn_bits(x) - 2 * RLC_DIG);
248 } else {
249 bn_copy(u, x);
250 bn_copy(v, y);
251 }
252 if (_a < 0) {
253 bn_mul_dig(t0, u, -_a);
254 bn_neg(t0, t0);
255 } else {
256 bn_mul_dig(t0, u, _a);
257 }
258 if (_b < 0) {
259 bn_mul_dig(t1, v, -_b);
260 bn_neg(t1, t1);
261 } else {
262 bn_mul_dig(t1, v, _b);
263 }
264 if (_c < 0) {
265 bn_mul_dig(t2, u, -_c);
266 bn_neg(t2, t2);
267 } else {
268 bn_mul_dig(t2, u, _c);
269 }
270 if (_d < 0) {
271 bn_mul_dig(t3, v, -_d);
272 bn_neg(t3, t3);
273 } else {
274 bn_mul_dig(t3, v, _d);
275 }
276 bn_add(u, t0, t1);
277 bn_add(v, t2, t3);
278 if (bn_bits(u) > RLC_DIG) {
279 bn_rsh(t0, u, bn_bits(u) - RLC_DIG);
280 bn_rsh(t1, v, bn_bits(u) - RLC_DIG);
281 } else {
282 bn_copy(t0, u);
283 bn_copy(t1, v);
284 }
285 _x = t0->dp[0];
286 _y = t1->dp[0];
287 t = 0;
288 if (_y != 0) {
289 q = _x / _y;
290 t = _x % _y;
291 }
292 if (t >= ((dig_t)1 << RLC_DIG / 2)) {
293 while (1) {
294 _q = _y / t;
295 _t = _y % t;
296 if (_t < ((dig_t)1 << RLC_DIG / 2)) {
297 break;
298 }
299 _x = _y;
300 _y = t;
301 t = _a - q * _c;
302 _a = _c;
303 _c = t;
304 t = _b - q * _d;
305 _b = _d;
306 _d = t;
307 t = _t;
308 q = _q;
309 }
310 }
311 if (_a < 0) {
312 bn_mul_dig(t0, x, -_a);
313 bn_neg(t0, t0);
314 } else {
315 bn_mul_dig(t0, x, _a);
316 }
317 if (_b < 0) {
318 bn_mul_dig(t1, y, -_b);
319 bn_neg(t1, t1);
320 } else {
321 bn_mul_dig(t1, y, _b);
322 }
323 if (_c < 0) {
324 bn_mul_dig(t2, x, -_c);
325 bn_neg(t2, t2);
326 } else {
327 bn_mul_dig(t2, x, _c);
328 }
329 if (_d < 0) {
330 bn_mul_dig(t3, y, -_d);
331 bn_neg(t3, t3);
332 } else {
333 bn_mul_dig(t3, y, _d);
334 }
335 bn_add(x, t0, t1);
336 bn_add(y, t2, t3);
337 }
338 }
339 bn_gcd_ext_dig(c, u, v, x, y->dp[0]);
340 }
341 RLC_CATCH_ANY {
342 RLC_THROW(ERR_CAUGHT);
343 }
344 RLC_FINALLY {
345 bn_free(x);
346 bn_free(y);
347 bn_free(u);
348 bn_free(v);
349 bn_free(t0);
350 bn_free(t1);
351 bn_free(t2);
352 bn_free(t3);
353 }
354}

356void bn_gcd_ext_lehme(bn_t c, bn_t d, bn_t e, const bn_t a, const bn_t b) {
357 bn_t x, y, u, v, t0, t1, t2, t3, t4;
358 dig_t _x, _y, q, _q, t, _t;
359 dis_t _a, _b, _c, _d;
360 int swap;
361
362 if (bn_is_zero(a)) {
363 bn_abs(c, b);
364 bn_zero(d);
365 if (e != NULL) {
366 bn_set_dig(e, 1);
367 }
368 return;
369 }
370
371 if (bn_is_zero(b)) {
372 bn_abs(c, a);
373 bn_set_dig(d, 1);
374 if (e != NULL) {
375 bn_zero(e);
376 }
377 return;
378 }
379
380 bn_null(x);
381 bn_null(y);
382 bn_null(u);
383 bn_null(v);
384 bn_null(t0);
385 bn_null(t1);
386 bn_null(t2);
387 bn_null(t3);
388 bn_null(t4);
389
390 /*
391 * Taken from Handbook of Hyperelliptic and Elliptic Cryptography.
392 */
393 RLC_TRY {
394 bn_new(x);
395 bn_new(y);
396 bn_new(u);
397 bn_new(v);
398 bn_new(t0);
399 bn_new(t1);
400 bn_new(t2);
401 bn_new(t3);
402 bn_new(t4);
403
404 if (bn_cmp_abs(a, b) != RLC_LT) {
405 bn_abs(x, a);
406 bn_abs(y, b);
407 swap = 0;
408 } else {
409 bn_abs(x, b);
410 bn_abs(y, a);
411 swap = 1;
412 }
413
414 bn_zero(t4);
415 bn_set_dig(d, 1);
416
417 while (y->used > 1) {
418 if (bn_bits(x) > RLC_DIG) {
419 bn_rsh(u, x, bn_bits(x) - RLC_DIG);
420 bn_rsh(v, y, bn_bits(x) - RLC_DIG);
421 } else {
422 bn_copy(u, x);
423 bn_copy(v, y);
424 }
425 _x = u->dp[0];
426 _y = v->dp[0];
427 _a = _d = 1;
428 _b = _c = 0;
429 t = 0;
430 if (_y != 0) {
431 q = _x / _y;
432 t = _x % _y;
433 }
434 if (t >= ((dig_t)1 << (RLC_DIG / 2))) {
435 while (1) {
436 _q = _y / t;
437 _t = _y % t;
438 if (_t < ((dig_t)1 << (RLC_DIG / 2))) {
439 break;
440 }
441 _x = _y;
442 _y = t;
443 t = _a - q * _c;
444 _a = _c;
445 _c = t;
446 t = _b - q * _d;
447 _b = _d;
448 _d = t;
449 t = _t;
450 q = _q;
451 }
452 }
453 if (_b == 0) {
454 bn_div_rem(t1, t0, x, y);
455 bn_copy(x, y);
456 bn_copy(y, t0);
457 bn_mul(t1, t1, d);
458 bn_sub(t1, t4, t1);
459 bn_copy(t4, d);
460 bn_copy(d, t1);
461 } else {
462 if (bn_bits(x) > 2 * RLC_DIG) {
463 bn_rsh(u, x, bn_bits(x) - 2 * RLC_DIG);
464 bn_rsh(v, y, bn_bits(x) - 2 * RLC_DIG);
465 } else {
466 bn_copy(u, x);
467 bn_copy(v, y);
468 }
469 if (_a < 0) {
470 bn_mul_dig(t0, u, -_a);
471 bn_neg(t0, t0);
472 } else {
473 bn_mul_dig(t0, u, _a);
474 }
475 if (_b < 0) {
476 bn_mul_dig(t1, v, -_b);
477 bn_neg(t1, t1);
478 } else {
479 bn_mul_dig(t1, v, _b);
480 }
481 if (_c < 0) {
482 bn_mul_dig(t2, u, -_c);
483 bn_neg(t2, t2);
484 } else {
485 bn_mul_dig(t2, u, _c);
486 }
487 if (_d < 0) {
488 bn_mul_dig(t3, v, -_d);
489 bn_neg(t3, t3);
490 } else {
491 bn_mul_dig(t3, v, _d);
492 }
493 bn_add(u, t0, t1);
494 bn_add(v, t2, t3);
495 if (bn_bits(u) > RLC_DIG) {
496 bn_rsh(t0, u, bn_bits(u) - RLC_DIG);
497 bn_rsh(t1, v, bn_bits(u) - RLC_DIG);
498 } else {
499 bn_copy(t0, u);
500 bn_copy(t1, v);
501 }
502 _x = t0->dp[0];
503 _y = t1->dp[0];
504 t = 0;
505 if (_y != 0) {
506 q = _x / _y;
507 t = _x % _y;
508 }
509 if (t >= ((dig_t)1 << RLC_DIG / 2)) {
510 while (1) {
511 _q = _y / t;
512 _t = _y % t;
513 if (_t < ((dig_t)1 << RLC_DIG / 2)) {
514 break;
515 }
516 _x = _y;
517 _y = t;
518 t = _a - q * _c;
519 _a = _c;
520 _c = t;
521 t = _b - q * _d;
522 _b = _d;
523 _d = t;
524 t = _t;
525 q = _q;
526 }
527 }
528 if (_a < 0) {
529 bn_mul_dig(t0, x, -_a);
530 bn_neg(t0, t0);
531 } else {
532 bn_mul_dig(t0, x, _a);
533 }
534 if (_b < 0) {
535 bn_mul_dig(t1, y, -_b);
536 bn_neg(t1, t1);
537 } else {
538 bn_mul_dig(t1, y, _b);
539 }
540 if (_c < 0) {
541 bn_mul_dig(t2, x, -_c);
542 bn_neg(t2, t2);
543 } else {
544 bn_mul_dig(t2, x, _c);
545 }
546 if (_d < 0) {
547 bn_mul_dig(t3, y, -_d);
548 bn_neg(t3, t3);
549 } else {
550 bn_mul_dig(t3, y, _d);
551 }
552 bn_add(x, t0, t1);
553 bn_add(y, t2, t3);
554
555 if (_a < 0) {
556 bn_mul_dig(t0, t4, -_a);
557 bn_neg(t0, t0);
558 } else {
559 bn_mul_dig(t0, t4, _a);
560 }
561 if (_b < 0) {
562 bn_mul_dig(t1, d, -_b);
563 bn_neg(t1, t1);
564 } else {
565 bn_mul_dig(t1, d, _b);
566 }
567 if (_c < 0) {
568 bn_mul_dig(t2, t4, -_c);
569 bn_neg(t2, t2);
570 } else {
571 bn_mul_dig(t2, t4, _c);
572 }
573 if (_d < 0) {
574 bn_mul_dig(t3, d, -_d);
575 bn_neg(t3, t3);
576 } else {
577 bn_mul_dig(t3, d, _d);
578 }
579 bn_add(t4, t0, t1);
580 bn_add(d, t2, t3);
581 }
582 }
583 bn_gcd_ext_dig(c, u, v, x, y->dp[0]);
584 if (!swap) {
585 bn_mul(t0, t4, u);
586 bn_mul(t1, d, v);
587 bn_add(t4, t0, t1);
588 bn_mul(x, b, t4);
589 bn_sub(x, c, x);
590 bn_div(d, x, a);
591 } else {
592 bn_mul(t0, t4, u);
593 bn_mul(t1, d, v);
594 bn_add(d, t0, t1);
595 bn_mul(x, a, d);
596 bn_sub(x, c, x);
597 bn_div(t4, x, b);
598 }
599 if (e != NULL) {
600 bn_copy(e, t4);
601 }
602 }
603 RLC_CATCH_ANY {
604 RLC_THROW(ERR_CAUGHT);
605 }
606 RLC_FINALLY {
607 bn_free(x);
608 bn_free(y);
609 bn_free(u);
610 bn_free(v);
611 bn_free(t0);
612 bn_free(t1);
613 bn_free(t2);
614 bn_free(t3);
615 bn_free(t4);
616 }
617}

623void bn_gcd_binar(bn_t c, const bn_t a, const bn_t b) {
624 bn_t u, v, t;
625 int shift;
626
627 if (bn_is_zero(a)) {
628 bn_abs(c, b);
629 return;
630 }
631
632 if (bn_is_zero(b)) {
633 bn_abs(c, a);
634 return;
635 }
636
637 bn_null(u);
638 bn_null(v);
639 bn_null(t);
640
641 RLC_TRY {
642 bn_new(u);
643 bn_new(v);
644 bn_new(t);
645
646 bn_abs(u, a);
647 bn_abs(v, b);
648
649 shift = 0;
650 while (bn_is_even(u) && bn_is_even(v)) {
651 bn_hlv(u, u);
652 bn_hlv(v, v);
653 shift++;
654 }
655 while (!bn_is_zero(u)) {
656 while (bn_is_even(u)) {
657 bn_hlv(u, u);
658 }
659 while (bn_is_even(v)) {
660 bn_hlv(v, v);
661 }
662 bn_sub(t, u, v);
663 bn_abs(t, t);
664 bn_hlv(t, t);
665 if (bn_cmp(u, v) != RLC_LT) {
666 bn_copy(u, t);
667 } else {
668 bn_copy(v, t);
669 }
670 }
671 bn_lsh(c, v, shift);
672 }
673 RLC_CATCH_ANY {
674 RLC_THROW(ERR_CAUGHT);
675 }
676 RLC_FINALLY {
677 bn_free(u);
678 bn_free(v);
679 bn_free(t);
680 }
681}

683void bn_gcd_ext_binar(bn_t c, bn_t d, bn_t e, const bn_t a, const bn_t b) {
684 bn_t x, y, t, u, v, _a, _b, _e;
685 int shift;
686
687 if (bn_is_zero(a)) {
688 bn_abs(c, b);
689 bn_zero(d);
690 if (e != NULL) {
691 bn_set_dig(e, 1);
692 }
693 return;
694 }
695
696 if (bn_is_zero(b)) {
697 bn_abs(c, a);
698 bn_set_dig(d, 1);
699 if (e != NULL) {
700 bn_zero(e);
701 }
702 return;
703 }
704
705 bn_null(x);
706 bn_null(y);
707 bn_null(t);
708 bn_null(u);
709 bn_null(v);
710 bn_null(_a);
711 bn_null(_b);
712 bn_null(_e);
713
714 RLC_TRY {
715 bn_new(x);
716 bn_new(y);
717 bn_new(t);
718 bn_new(u);
719 bn_new(v);
720 bn_new(_a);
721 bn_new(_b);
722 bn_new(_e);
723
724 bn_abs(x, a);
725 bn_abs(y, b);
726
727 /* g = 1. */
728 shift = 0;
729 /* While x and y are both even, x = x/2 and y = y/2, g = 2g. */
730 while (bn_is_even(x) && bn_is_even(y)) {
731 bn_hlv(x, x);
732 bn_hlv(y, y);
733 shift++;
734 }
735
736 bn_copy(u, x);
737 bn_copy(v, y);
738
739 /* u = x, y = v, A = 1, B = 0, C = 0, D = 1. */
740 bn_set_dig(_a, 1);
741 bn_zero(_b);
742 bn_zero(d);
743 bn_set_dig(_e, 1);
744
745 /* While u is even, u = u/2. */
746 while (bn_is_even(u)) {
747 bn_hlv(u, u);
748 /* If A = B = 0 (mod 2) then A = A/2, B = B/2. */
749 if ((_a->dp[0] & 0x01) == 0 && (_b->dp[0] & 0x01) == 0) {
750 bn_hlv(_a, _a);
751 bn_hlv(_b, _b);
752 } else {
753 /* Otherwise A = (A + y)/2, B = (B - x)/2. */
