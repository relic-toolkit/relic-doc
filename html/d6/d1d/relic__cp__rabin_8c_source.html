60 int cp_rabin_gen(rabin_t pub, rabin_t prv, int bits) {
61  bn_t r;
62  int result = RLC_OK;
63 
64  bn_null(r);
65 
66  TRY {
67  bn_new(r);
68 
69  /* Generate different primes p and q. */
70  do {
71  bn_gen_prime(prv->p, bits / 2);
72  bn_mod_2b(r, prv->p, 2);
73  } while (bn_cmp_dig(r, 3) != RLC_EQ);
74 
75  do {
76  bn_gen_prime(prv->q, bits / 2);
77  bn_mod_2b(r, prv->q, 2);
78  } while (bn_cmp(prv->p, prv->q) == RLC_EQ ||
79  bn_cmp_dig(r, 3) != RLC_EQ);
80 
81  /* Swap p and q so that p is smaller. */
82  if (bn_cmp(prv->p, prv->q) == RLC_LT) {
83  bn_copy(r, prv->p);
84  bn_copy(prv->p, prv->q);
85  bn_copy(prv->q, r);
86  }
87 
88  bn_gcd_ext(r, prv->dp, prv->dq, prv->p, prv->q);
89  if (bn_cmp_dig(r, 1) != RLC_EQ) {
90  result = RLC_ERR;
91  }
92 
93  bn_mul(prv->n, prv->p, prv->q);
94  bn_copy(pub->n, prv->n);
95  }
96  CATCH_ANY {
97  result = RLC_ERR;
98  }
99  FINALLY {
100  bn_free(r);
101  }
102 
103  return result;
104 }

106 int cp_rabin_enc(uint8_t *out, int *out_len, uint8_t *in, int in_len,
107  rabin_t pub) {
108  bn_t m, t;
109  int size, result = RLC_OK;
110 
111  bn_null(m);
112  bn_null(t);
113 
114  size = bn_size_bin(pub->n);
115 
116  if (in_len <= 0 || in_len > (size - RABIN_PAD_LEN - 2)) {
117  return RLC_ERR;
118  }
119 
120  TRY {
121  bn_new(m);
122  bn_new(t);
123  bn_zero(m);
124 
125  /* EB = 00 | FF || D || SUFFIX */
126  bn_lsh(m, m, 8);
127  bn_add_dig(m, m, RABIN_PAD);
128  bn_lsh(m, m, in_len * 8);
129  bn_read_bin(t, in, in_len);
130  bn_add(m, m, t);
131  bn_mod_2b(t, m, 8 * RABIN_PAD_LEN);
132  bn_lsh(m, m, 8 * RABIN_PAD_LEN);
133  bn_add(m, m, t);
134 
135  bn_sqr(m, m);
136  bn_mod(m, m, pub->n);
137 
138  if (size <= *out_len) {
139  *out_len = size;
140  memset(out, 0, *out_len);
141  bn_write_bin(out, size, m);
142  } else {
143  result = RLC_ERR;
144  }
145  }
146  CATCH_ANY {
147  result = RLC_ERR;
148  }
149  FINALLY {
150  bn_free(m);
151  bn_free(t);
152  }
153 
154  return result;
155 }

157 int cp_rabin_dec(uint8_t *out, int *out_len, uint8_t *in, int in_len,
158  rabin_t prv) {
159  bn_t m, m0, m1, t, n;
160  int size, result = RLC_OK;
161  uint8_t pad;
162 
163  if (in_len < RABIN_PAD_LEN) {
164  return RLC_ERR;
165  }
166 
167  bn_null(m);
168  bn_null(m0);
169  bn_null(m1);
170  bn_null(n);
171  bn_null(t);
172 
173  TRY {
174  bn_new(m);
175  bn_new(m0);
176  bn_new(m1);
177  bn_new(n);
178  bn_new(t);
179 
180  bn_read_bin(m, in, in_len);
181 
182  bn_add_dig(t, prv->p, 1);
183  bn_rsh(t, t, 2);
184  bn_mxp(m0, m, t, prv->p);
185 
186  bn_add_dig(t, prv->q, 1);
187  bn_rsh(t, t, 2);
188  bn_mxp(m1, m, t, prv->q);
189 
190  bn_mul(m, prv->dp, prv->p);
191  bn_mul(m, m, m1);
192  bn_mul(t, prv->dq, prv->q);
193  bn_mul(t, t, m0);
194  bn_add(m0, m, t);
195  bn_mod(m0, m0, prv->n);
196  if (bn_sign(m0) == RLC_NEG) {
197  bn_add(m0, m0, prv->n);
198  }
199  bn_sub(m1, m, t);
200  bn_mod(m1, m1, prv->n);
201  if (bn_sign(m1) == RLC_NEG) {
202  bn_add(m1, m1, prv->n);
203  }
204 
205  bn_mod_2b(m, m0, 8 * RABIN_PAD_LEN);
206  bn_rsh(t, m0, 8 * RABIN_PAD_LEN);
207  bn_mod_2b(t, t, 8 * RABIN_PAD_LEN);
208  if (bn_cmp(t, m) == RLC_EQ) {
209  bn_rsh(m, m0, 8 * RABIN_PAD_LEN);
210  } else {
211  bn_sub(m0, prv->n, m0);
212  bn_mod_2b(m, m0, 8 * RABIN_PAD_LEN);
213  bn_rsh(t, m0, 8 * RABIN_PAD_LEN);
214  bn_mod_2b(t, t, 8 * RABIN_PAD_LEN);
215  if (bn_cmp(t, m) == RLC_EQ) {
216  bn_rsh(m, m0, 8 * RABIN_PAD_LEN);
217  } else {
218  bn_mod_2b(m, m1, 8 * RABIN_PAD_LEN);
219  bn_rsh(t, m1, 8 * RABIN_PAD_LEN);
220  bn_mod_2b(t, t, 8 * RABIN_PAD_LEN);
221  if (bn_cmp(t, m) == RLC_EQ) {
222  bn_rsh(m, m1, 8 * RABIN_PAD_LEN);
223  } else {
224  bn_sub(m1, prv->n, m1);
225  bn_mod_2b(m, m1, 8 * RABIN_PAD_LEN);
226  bn_rsh(t, m1, 8 * RABIN_PAD_LEN);
227  bn_mod_2b(t, t, 8 * RABIN_PAD_LEN);
228  if (bn_cmp(t, m) == RLC_EQ) {
229  bn_rsh(m, m1, 8 * RABIN_PAD_LEN);
230  } else {
231  result = RLC_ERR;
232  }
233  }
234  }
235  }
236 
237  if (result == RLC_OK) {
238  size = bn_size_bin(prv->n);
239  size--;
240  bn_rsh(t, m, 8 * size);
241 
242  if (!bn_is_zero(t)) {
243  result = RLC_ERR;
244  } else {
245  do {
246  size--;
247  bn_rsh(t, m, 8 * size);
248  pad = (uint8_t)t->dp[0];
249  } while (pad == 0);
250 
251  if (pad != RABIN_PAD) {
252  result = RLC_ERR;
253  } else {
254  bn_mod_2b(m, m, size * 8);
255  }
256  }
257 
258  if (size <= *out_len) {
259  *out_len = size;
260  memset(out, 0, size);
261  bn_write_bin(out, size, m);
262  } else {
263  result = RLC_ERR;
264  }
265  }
266  }
267  CATCH_ANY {
268  result = RLC_ERR;
269  }
270  FINALLY {
271  bn_free(m);
272  bn_free(m0);
273  bn_free(m1);
274  bn_free(n);
275  bn_free(t);
276  }
277 
278  return result;
279 }
