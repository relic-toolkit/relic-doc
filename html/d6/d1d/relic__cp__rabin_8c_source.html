52int cp_rabin_gen(rabin_t pub, rabin_t prv, size_t bits) {
53 bn_t r;
54 int result = RLC_OK;
55
56 bn_null(r);
57
58 RLC_TRY {
59 bn_new(r);
60
61 /* Generate different primes p and q. */
62 do {
63 bn_gen_prime(prv->p, bits / 2);
64 bn_mod_2b(r, prv->p, 2);
65 } while (bn_cmp_dig(r, 3) != RLC_EQ);
66
67 do {
68 bn_gen_prime(prv->q, bits / 2);
69 bn_mod_2b(r, prv->q, 2);
70 } while (bn_cmp(prv->p, prv->q) == RLC_EQ ||
71 bn_cmp_dig(r, 3) != RLC_EQ);
72
73 /* Swap p and q so that p is smaller. */
74 if (bn_cmp(prv->p, prv->q) != RLC_LT) {
75 bn_copy(r, prv->p);
76 bn_copy(prv->p, prv->q);
77 bn_copy(prv->q, r);
78 }
79
80 bn_gcd_ext(r, prv->dp, prv->dq, prv->p, prv->q);
81 if (bn_cmp_dig(r, 1) != RLC_EQ) {
82 result = RLC_ERR;
83 }
84
85 bn_mul(prv->n, prv->p, prv->q);
86 bn_copy(pub->n, prv->n);
87 }
88 RLC_CATCH_ANY {
89 result = RLC_ERR;
90 }
91 RLC_FINALLY {
92 bn_free(r);
93 }
94
95 return result;
96}

98int cp_rabin_enc(uint8_t *out, size_t *out_len, const uint8_t *in,
99 size_t in_len, const rabin_t pub) {
100 bn_t m, t;
101 size_t size;
102 int result = RLC_OK;
103
104 bn_null(m);
105 bn_null(t);
106
107 size = bn_size_bin(pub->n);
108
109 if (in_len <= 0 || in_len > (size - RABIN_PAD_LEN - 2)) {
110 return RLC_ERR;
111 }
112
113 RLC_TRY {
114 bn_new(m);
115 bn_new(t);
116 bn_zero(m);
117
118 /* EB = 00 | FF || D || SUFFIX */
119 bn_lsh(m, m, 8);
120 bn_add_dig(m, m, RABIN_PAD);
121 bn_lsh(m, m, in_len * 8);
122 bn_read_bin(t, in, in_len);
123 bn_add(m, m, t);
124 bn_mod_2b(t, m, 8 * RABIN_PAD_LEN);
125 bn_lsh(m, m, 8 * RABIN_PAD_LEN);
126 bn_add(m, m, t);
127
128 bn_sqr(m, m);
129 bn_mod(m, m, pub->n);
130
131 if (size <= *out_len) {
132 *out_len = size;
133 memset(out, 0, *out_len);
134 bn_write_bin(out, size, m);
135 } else {
136 result = RLC_ERR;
137 }
138 }
139 RLC_CATCH_ANY {
140 result = RLC_ERR;
141 }
142 RLC_FINALLY {
143 bn_free(m);
144 bn_free(t);
145 }
146
147 return result;
148}

150int cp_rabin_dec(uint8_t *out, size_t *out_len, const uint8_t *in,
151 size_t in_len, const rabin_t prv) {
152 bn_t m, m0, m1, t, n;
153 size_t size;
154 int result = RLC_OK;
155 uint8_t pad;
156
157 if (in_len < RABIN_PAD_LEN) {
158 return RLC_ERR;
159 }
160
161 bn_null(m);
162 bn_null(m0);
163 bn_null(m1);
164 bn_null(n);
165 bn_null(t);
166
167 RLC_TRY {
168 bn_new(m);
169 bn_new(m0);
170 bn_new(m1);
171 bn_new(n);
172 bn_new(t);
173
174 bn_read_bin(m, in, in_len);
175
176 bn_add_dig(t, prv->p, 1);
177 bn_rsh(t, t, 2);
178 bn_mxp(m0, m, t, prv->p);
179
180 bn_add_dig(t, prv->q, 1);
181 bn_rsh(t, t, 2);
182 bn_mxp(m1, m, t, prv->q);
183
184 bn_mul(m, prv->dp, prv->p);
185 bn_mul(m, m, m1);
186 bn_mul(t, prv->dq, prv->q);
187 bn_mul(t, t, m0);
188 bn_add(m0, m, t);
189 bn_mod(m0, m0, prv->n);
190 if (bn_sign(m0) == RLC_NEG) {
191 bn_add(m0, m0, prv->n);
192 }
193 bn_sub(m1, m, t);
194 bn_mod(m1, m1, prv->n);
195 if (bn_sign(m1) == RLC_NEG) {
196 bn_add(m1, m1, prv->n);
197 }
198
199 bn_mod_2b(m, m0, 8 * RABIN_PAD_LEN);
200 bn_rsh(t, m0, 8 * RABIN_PAD_LEN);
201 bn_mod_2b(t, t, 8 * RABIN_PAD_LEN);
202 if (bn_cmp(t, m) == RLC_EQ) {
203 bn_rsh(m, m0, 8 * RABIN_PAD_LEN);
204 } else {
205 bn_sub(m0, prv->n, m0);
206 bn_mod_2b(m, m0, 8 * RABIN_PAD_LEN);
207 bn_rsh(t, m0, 8 * RABIN_PAD_LEN);
208 bn_mod_2b(t, t, 8 * RABIN_PAD_LEN);
209 if (bn_cmp(t, m) == RLC_EQ) {
210 bn_rsh(m, m0, 8 * RABIN_PAD_LEN);
211 } else {
212 bn_mod_2b(m, m1, 8 * RABIN_PAD_LEN);
213 bn_rsh(t, m1, 8 * RABIN_PAD_LEN);
214 bn_mod_2b(t, t, 8 * RABIN_PAD_LEN);
215 if (bn_cmp(t, m) == RLC_EQ) {
216 bn_rsh(m, m1, 8 * RABIN_PAD_LEN);
217 } else {
218 bn_sub(m1, prv->n, m1);
219 bn_mod_2b(m, m1, 8 * RABIN_PAD_LEN);
220 bn_rsh(t, m1, 8 * RABIN_PAD_LEN);
221 bn_mod_2b(t, t, 8 * RABIN_PAD_LEN);
222 if (bn_cmp(t, m) == RLC_EQ) {
223 bn_rsh(m, m1, 8 * RABIN_PAD_LEN);
224 } else {
225 result = RLC_ERR;
226 }
227 }
228 }
229 }
230
231 if (result == RLC_OK) {
232 size = bn_size_bin(prv->n);
233 size--;
234 bn_rsh(t, m, 8 * size);
235
236 if (!bn_is_zero(t)) {
237 result = RLC_ERR;
238 } else {
239 do {
240 size--;
241 bn_rsh(t, m, 8 * size);
242 pad = (uint8_t)t->dp[0];
243 } while (pad == 0);
244
245 if (pad != RABIN_PAD) {
246 result = RLC_ERR;
247 } else {
248 bn_mod_2b(m, m, size * 8);
249 }
250 }
251
252 if (size <= *out_len) {
253 *out_len = size;
254 memset(out, 0, size);
255 bn_write_bin(out, size, m);
256 } else {
257 result = RLC_ERR;
258 }
259 }
260 }
261 RLC_CATCH_ANY {
262 result = RLC_ERR;
263 }
264 RLC_FINALLY {
265 bn_free(m);
266 bn_free(m0);
267 bn_free(m1);
268 bn_free(n);
269 bn_free(t);
270 }
271
272 return result;
273}
