176static void detect_opt(int *opt, fp3_t a) {
177 fp3_t t;
178 fp3_null(t);
179
180 RLC_TRY {
181 fp3_new(t);
182 fp3_set_dig(t, 3);
183 fp3_neg(t, t);
184
185 if (fp3_cmp(a, t) == RLC_EQ) {
186 *opt = RLC_MIN3;
187 } else if (fp3_is_zero(a)) {
188 *opt = RLC_ZERO;
189 } else if (fp3_cmp_dig(a, 1) == RLC_EQ) {
190 *opt = RLC_ONE;
191 } else if (fp3_cmp_dig(a, 2) == RLC_EQ) {
192 *opt = RLC_TWO;
193 } else if ((fp_bits(a[0]) <= RLC_DIG) && fp_is_zero(a[1]) &&
194 fp_is_zero(a[2])) {
195 *opt = RLC_TINY;
196 } else {
197 *opt = RLC_HUGE;
198 }
199 }
200 RLC_CATCH_ANY {
201 RLC_THROW(ERR_CAUGHT);
202 }
203 RLC_FINALLY {
204 fp3_free(t);
205 }
206}

272void ep3_curve_mul_a(fp3_t c, const fp3_t a) {
273 ctx_t *ctx = core_get();
274 switch (ctx->ep3_opt_a) {
275 case RLC_ZERO:
276 fp3_zero(c);
277 break;
278 case RLC_ONE:
279 fp3_copy(c, a);
280 break;
281 case RLC_TWO:
282 fp3_dbl(c, a);
283 break;
284#if FP_RDC != MONTY
285 case RLC_TINY:
286 fp3_mul_dig(c, a, ctx->ep3_a[0][0]);
287 break;
288#endif
289 default:
290 fp3_mul(c, a, ctx->ep3_a);
291 break;
292 }
293}
