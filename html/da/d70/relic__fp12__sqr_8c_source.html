69void fp12_sqr_cyc_basic(fp12_t c, const fp12_t a) {
70 fp2_t t0, t1, t2, t3, t4, t5, t6;
71
72 fp2_null(t0);
73 fp2_null(t1);
74 fp2_null(t2);
75 fp2_null(t3);
76 fp2_null(t4);
77 fp2_null(t5);
78 fp2_null(t6);
79
80 RLC_TRY {
81 fp2_new(t0);
82 fp2_new(t1);
83 fp2_new(t2);
84 fp2_new(t3);
85 fp2_new(t4);
86 fp2_new(t5);
87 fp2_new(t6);
88
89 /* Define z = sqrt(E) */
90
91 /* Now a is seen as (t0,t1) + (t2,t3) * w + (t4,t5) * w^2 */
92
93 /* (t0, t1) = (a00 + a11*z)^2. */
94 fp2_sqr(t2, a[0][0]);
95 fp2_sqr(t3, a[1][1]);
96 fp2_add(t1, a[0][0], a[1][1]);
97
98 fp2_mul_nor(t0, t3);
99 fp2_add(t0, t0, t2);
100
101 fp2_sqr(t1, t1);
102 fp2_sub(t1, t1, t2);
103 fp2_sub(t1, t1, t3);
104
105 fp2_sub(c[0][0], t0, a[0][0]);
106 fp2_add(c[0][0], c[0][0], c[0][0]);
107 fp2_add(c[0][0], t0, c[0][0]);
108
109 fp2_add(c[1][1], t1, a[1][1]);
110 fp2_add(c[1][1], c[1][1], c[1][1]);
111 fp2_add(c[1][1], t1, c[1][1]);
112
113 fp2_sqr(t0, a[0][1]);
114 fp2_sqr(t1, a[1][2]);
115 fp2_add(t5, a[0][1], a[1][2]);
116 fp2_sqr(t2, t5);
117
118 fp2_add(t3, t0, t1);
119 fp2_sub(t5, t2, t3);
120
121 fp2_add(t6, a[1][0], a[0][2]);
122 fp2_sqr(t3, t6);
123 fp2_sqr(t2, a[1][0]);
124
125 fp2_mul_nor(t6, t5);
126 fp2_add(t5, t6, a[1][0]);
127 fp2_dbl(t5, t5);
128 fp2_add(c[1][0], t5, t6);
129
130 fp2_mul_nor(t4, t1);
131 fp2_add(t5, t0, t4);
132 fp2_sub(t6, t5, a[0][2]);
133
134 fp2_sqr(t1, a[0][2]);
135
136 fp2_dbl(t6, t6);
137 fp2_add(c[0][2], t6, t5);
138
139 fp2_mul_nor(t4, t1);
140 fp2_add(t5, t2, t4);
141 fp2_sub(t6, t5, a[0][1]);
142 fp2_dbl(t6, t6);
143 fp2_add(c[0][1], t6, t5);
144
145 fp2_add(t0, t2, t1);
146 fp2_sub(t5, t3, t0);
147 fp2_add(t6, t5, a[1][2]);
148 fp2_dbl(t6, t6);
149 fp2_add(c[1][2], t5, t6);
150 } RLC_CATCH_ANY {
151 RLC_THROW(ERR_CAUGHT);
152 } RLC_FINALLY {
153 fp2_free(t0);
154 fp2_free(t1);
155 fp2_free(t2);
156 fp2_free(t3);
157 fp2_free(t4);
158 fp2_free(t5);
159 fp2_free(t6);
160 }

237void fp12_sqr_unr(dv12_t c, const fp12_t a) {
238 fp4_t t0, t1;
239 dv4_t u0, u1, u2, u3, u4;
240
241 fp4_null(t0);
242 fp4_null(t1);
243 dv4_null(u0);
244 dv4_null(u1);
245 dv4_null(u2);
246 dv4_null(u3);
247 dv4_null(u4);
248
249 RLC_TRY {
250 fp4_new(t0);
251 fp4_new(t1);
252 dv4_new(u0);
253 dv4_new(u1);
254 dv4_new(u2);
255 dv4_new(u3);
256 dv4_new(u4);
257
258 /* a0 = (a00, a11). */
259 /* a1 = (a10, a02). */
260 /* a2 = (a01, a12). */
261
262 /* (t0,t1) = a0^2 */
263 fp2_copy(t0[0], a[0][0]);
264 fp2_copy(t0[1], a[1][1]);
265 fp4_sqr_unr(u0, t0);
266
267 /* (t2,t3) = 2 * a1 * a2 */
268 fp2_copy(t0[0], a[1][0]);
269 fp2_copy(t0[1], a[0][2]);
270 fp2_copy(t1[0], a[0][1]);
271 fp2_copy(t1[1], a[1][2]);
272 fp4_mul_unr(u1, t0, t1);
273 fp2_addc_low(u1[0], u1[0], u1[0]);
274 fp2_addc_low(u1[1], u1[1], u1[1]);
275
276 /* (t4,t5) = a2^2. */
277 fp4_sqr_unr(u2, t1);
278
279 /* c2 = a0 + a2. */
280 fp2_addm_low(t1[0], a[0][0], a[0][1]);
281 fp2_addm_low(t1[1], a[1][1], a[1][2]);
282
283 /* (t6,t7) = (a0 + a2 + a1)^2. */
284 fp2_addm_low(t0[0], t1[0], a[1][0]);
285 fp2_addm_low(t0[1], t1[1], a[0][2]);
286 fp4_sqr_unr(u3, t0);
287
288 /* c2 = (a0 + a2 - a1)^2. */
289 fp2_subm_low(t0[0], t1[0], a[1][0]);
290 fp2_subm_low(t0[1], t1[1], a[0][2]);
291 fp4_sqr_unr(u4, t0);
292
293 /* c2 = (c2 + (t6,t7))/2. */
294#ifdef RLC_FP_ROOM
295 fp2_addd_low(u4[0], u4[0], u3[0]);
296 fp2_addd_low(u4[1], u4[1], u3[1]);
297#else
298 fp2_addc_low(u4[0], u4[0], u3[0]);
299 fp2_addc_low(u4[1], u4[1], u3[1]);
300#endif
301 fp_hlvd_low(u4[0][0], u4[0][0]);
302 fp_hlvd_low(u4[0][1], u4[0][1]);
303 fp_hlvd_low(u4[1][0], u4[1][0]);
304 fp_hlvd_low(u4[1][1], u4[1][1]);
305
306 /* (t6,t7) = (t6,t7) - c2 - (t2,t3). */
307 fp2_subc_low(u3[0], u3[0], u4[0]);
308 fp2_subc_low(u3[1], u3[1], u4[1]);
309 fp2_subc_low(u3[0], u3[0], u1[0]);
310 fp2_subc_low(u3[1], u3[1], u1[1]);
311
312 /* c2 = c2 - (t0,t1) - (t4,t5). */
313 fp2_subc_low(u4[0], u4[0], u0[0]);
314 fp2_subc_low(u4[1], u4[1], u0[1]);
315 fp2_subc_low(c[0][1], u4[0], u2[0]);
316 fp2_subc_low(c[1][2], u4[1], u2[1]);
317
318 /* c1 = (t6,t7) + (t4,t5) * E. */
319 fp2_nord_low(u4[1], u2[1]);
320 fp2_addc_low(c[1][0], u3[0], u4[1]);
321 fp2_addc_low(c[0][2], u3[1], u2[0]);
322
323 /* c0 = (t0,t1) + (t2,t3) * E. */
324 fp2_nord_low(u4[1], u1[1]);
325 fp2_addc_low(c[0][0], u0[0], u4[1]);
326 fp2_addc_low(c[1][1], u0[1], u1[0]);
327 } RLC_CATCH_ANY {
328 RLC_THROW(ERR_CAUGHT);
329 } RLC_FINALLY {
330 fp4_free(t0);
331 fp4_free(t1);
332 dv4_free(u0);
333 dv4_free(u1);
334 dv4_free(u2);
335 dv4_free(u3);
336 dv4_free(u4);
337 }
338}

341 dv12_t t;
342
343 dv12_null(t);
344
345 RLC_TRY {
346 dv12_new(t);
347 fp12_sqr_unr(t, a);
