40void pp_exp_k24(fp24_t c, fp24_t a) {
41 fp24_t t0, t1, t2;
42 const int *b;
43 bn_t x;
44 int l;
45
46 bn_null(x);
47 fp24_null(t0);
48 fp24_null(t1);
49 fp24_null(t2);
50
51 RLC_TRY {
52 bn_new(x);
53 fp24_new(t0);
54 fp24_new(t1);
55 fp24_new(t2);
56
57 /*
58 * Final exponentiation following Hayashida, Hayasaka and Teruya:
59 * Efficient Final Exponentiation via Cyclotomic Structure for Pairings
60 * over Families of Elliptic Curves: https://eprint.iacr.org/2020/875
61 */
62 fp_prime_get_par(x);
63 b = fp_prime_get_par_sps(&l);
64 /* First, compute m^(p^12 - 1)(p^4 + 1). */
65 fp24_conv_cyc(c, a);
66
67 /* t0 = f^(x-1)^2. */
68 if (bn_sign(x) == RLC_NEG && b[0] == 0 && b[1] == -1) {
69 fp24_exp_cyc_sps(t0, c, b+2, l-2, RLC_POS);
70 fp24_exp_cyc_sps(t0, t0, b+2, l-2, RLC_POS);
71 } else {
72 fp24_exp_cyc_sps(t1, c, b, l, bn_sign(x));
73 fp24_inv_cyc(t0, c);
74 fp24_mul(t1, t1, t0);
75 fp24_exp_cyc_sps(t0, t1, b, l, bn_sign(x));
76 fp24_inv_cyc(t1, t1);
77 fp24_mul(t0, t0, t1);
78 }
79
80 /* t1 = t0^(x + p). */
81 fp24_exp_cyc_sps(t1, t0, b, l, bn_sign(x));
82 fp24_frb(t0, t0, 1);
83 fp24_mul(t1, t1, t0);
84
85 /* t0 = t1^(x^2 + p^2). */
86 fp24_exp_cyc_sps(t0, t1, b, l, RLC_POS);
87 fp24_exp_cyc_sps(t0, t0, b, l, RLC_POS);
88 fp24_frb(t1, t1, 2);
89 fp24_mul(t0, t0, t1);
90
91 /* t1 = t0^(x^4 + p^4 - 1). */
92 fp24_exp_cyc_sps(t1, t0, b, l, RLC_POS);
93 fp24_exp_cyc_sps(t1, t1, b, l, RLC_POS);
94 fp24_exp_cyc_sps(t1, t1, b, l, RLC_POS);
95 fp24_exp_cyc_sps(t1, t1, b, l, RLC_POS);
96 fp24_inv_cyc(t2, t0);
97 fp24_frb(t0, t0, 4);
98 fp24_mul(t1, t1, t0);
99 fp24_mul(t1, t1, t2);
100
101 /* c = c^3. */
102 fp24_sqr_cyc(t0, c);
103 fp24_mul(c, c, t0);
104 fp24_mul(c, c, t1);
105 }
106 RLC_CATCH_ANY {
107 RLC_THROW(ERR_CAUGHT);
108 }
109 RLC_FINALLY {
110 bn_free(x);
111 fp24_free(t0);
112 fp24_free(t1);
113 fp24_free(t2);
114 }
115}
