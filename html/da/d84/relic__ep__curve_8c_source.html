44static void detect_opt(int *opt, fp_t a) {
45 fp_t t;
46
47 fp_null(t);
48
49 RLC_TRY {
50 fp_new(t);
51 fp_prime_conv_dig(t, 3);
52 fp_neg(t, t);
53
54 if (fp_cmp(a, t) == RLC_EQ) {
55 *opt = RLC_MIN3;
56 } else if (fp_is_zero(a)) {
57 *opt = RLC_ZERO;
58 } else if (fp_cmp_dig(a, 1) == RLC_EQ) {
59 *opt = RLC_ONE;
60 } else if (fp_cmp_dig(a, 2) == RLC_EQ) {
61 *opt = RLC_TWO;
62 } else if (fp_bits(a) <= RLC_DIG) {
63 *opt = RLC_TINY;
64 } else {
65 *opt = RLC_HUGE;
66 }
67 }
68 RLC_CATCH_ANY {
69 RLC_THROW(ERR_CAUGHT);
70 }
71 RLC_FINALLY {
72 fp_free(t);
73 }
74}

82static void ep_curve_set_map(void) {
83 bn_t t;
84 bn_null(t);
85
86 const int abNeq0 = (ep_curve_opt_a() * ep_curve_opt_b()) != RLC_ZERO;
87
88 ctx_t *ctx = core_get();
89 dig_t *c0 = ctx->ep_map_c[0];
90 dig_t *c1 = ctx->ep_map_c[1];
91 dig_t *c2 = ctx->ep_map_c[2];
92 dig_t *c3 = ctx->ep_map_c[3];
93 dig_t *c4 = ctx->ep_map_c[4];
94
95 RLC_TRY {
96 bn_new(t);
97
98 if (ep_curve_is_ctmap() || abNeq0) {
99 /* SSWU map constants */
100 /* constants 3 and 4: a and b for either the curve or the isogeny */
101#ifdef EP_CTMAP
102 if (ep_curve_is_ctmap()) {
103 fp_copy(c2, ctx->ep_iso.a);
104 fp_copy(c3, ctx->ep_iso.b);
105 } else {
106#endif
107 fp_copy(c2, ctx->ep_a);
108 fp_copy(c3, ctx->ep_b);
109 /* Generate a non-square u to define the map. */
110 fp_set_dig(ctx->ep_map_u, 0);
111 do {
112 fp_add_dig(ctx->ep_map_u, ctx->ep_map_u, 1);
113 /* Check that g(b/ua) = u^3 + a * u + b is square*/
114 fp_mul(c1, ctx->ep_a, ctx->ep_map_u);
115 fp_inv(c1, c1);
116 fp_mul(c1, c1, ctx->ep_b);
117 fp_sqr(c0, c1);
118 fp_add(c0, c0, ctx->ep_a);
119 fp_mul(c0, c0, ctx->ep_map_u);
120 fp_add(c0, c0, ctx->ep_b);
121 } while (fp_is_sqr(ctx->ep_map_u) || !fp_is_sqr(c0));
122#ifdef EP_CTMAP
123 }
124#endif
125 /* constant 1: -b / a */
126 fp_neg(c0, c2); /* c0 = -a */
127 fp_inv(c0, c0); /* c0 = -1 / a */
128 fp_mul(c0, c0, c3); /* c0 = -b / a */
129
130 /* constant 2 is unused in this case */
131 } else {
132 /* SvdW map constants */
133 fp_set_dig(ctx->ep_map_u, 0);
134 do {
135 /* Generate u by trial and error. */
136 fp_add_dig(ctx->ep_map_u, ctx->ep_map_u, 1);
137
138 /* constant 1: g(u) = u^3 + a * u + b */
139 fp_sqr(c0, ctx->ep_map_u);
140 fp_add(c0, c0, ctx->ep_a);
141 fp_mul(c0, c0, ctx->ep_map_u);
142 fp_add(c0, c0, ctx->ep_b);
143
144 /* constant 2: -u / 2 */
145 fp_set_dig(c1, 1);
146 fp_neg(c1, c1); /* -1 */
147 fp_hlv(c1, c1); /* -1/2 */
148 fp_mul(c1, c1, ctx->ep_map_u); /* c1 = -1/2 * u */
149
150 /* constant 3: sqrt(-g(u) * (3 * u^2 + 4 * a)) */
151 fp_sqr(c2, ctx->ep_map_u); /* c2 = u^2 */
152 fp_mul_dig(c2, c2, 3); /* c2 = 3 * u^2 */
153 fp_mul_dig(c3, ctx->ep_a, 4); /* c3 = 4 * a */
154 fp_add(c3, c2, c3); /* c3 = 3 * u^2 + 4 * a */
155 fp_neg(c3, c3); /* c3 = -(3 * u^2 + 4 * a) */
156 fp_mul(c2, c3, c0); /* c2 = -g(u) * (3 * u^2 + 4 * a) */
157 } while (fp_is_zero(c2) || !fp_is_sqr(c2));
158 if (!fp_srt(c2, c2)) { /* c2 = sqrt(-g(u) * (3 * u^2 + 4 * a)) */
159 RLC_THROW(ERR_NO_VALID);
160 }
161 /* make sure sgn0(c2) == 0 */
162 fp_prime_back(t, c2);
163 if (bn_get_bit(t, 0) != 0) {
164 /* set sgn0(c2) == 0 */
165 fp_neg(c2, c2);
166 }
167
168 /* constant 4: -4 * g(u) / (3 * u^2 + 4 * a) */
169 fp_inv(c3, c3); /* c3 = -1 / (3 * u^2 + 4 * a) */
170 fp_mul(c3, c3, c0); /* c3 *= g(u) */
171 fp_mul_dig(c3, c3, 4); /* c3 *= 4 */
172 }
173
174 /* If curve is not supersingular, precompute and store sqrt(-3)
175 * neeed for hashing using the SwiftEC algorithm and variants. */
176 if (!ep_curve_is_super() && (ep_curve_opt_a() == RLC_ZERO || ep_curve_opt_b() == RLC_ZERO)) {
177 fp_set_dig(c4, 3);
178 fp_neg(c4, c4);
179 if (!fp_srt(c4, c4)) {
180 RLC_THROW(ERR_NO_VALID);
181 }
182 }
183 }
184 RLC_CATCH_ANY {
185 RLC_THROW(ERR_CAUGHT);
186 }
187 RLC_FINALLY {
188 bn_free(t);
189 }
190}

203static void ep_curve_set(const fp_t a, const fp_t b, const ep_t g, const bn_t r,
204 const bn_t h, int ctmap) {
205 ctx_t *ctx = core_get();
206
207 fp_copy(ctx->ep_a, a);
208 fp_copy(ctx->ep_b, b);
209
210 detect_opt(&(ctx->ep_opt_a), ctx->ep_a);
211 detect_opt(&(ctx->ep_opt_b), ctx->ep_b);
212
213 ctx->ep_is_ctmap = ctmap;
214 ep_curve_set_map();
215
216 ep_norm(&(ctx->ep_g), g);
217 bn_copy(&(ctx->ep_r), r);
218 bn_copy(&(ctx->ep_h), h);
219
220#if defined(EP_PRECO)
221 ep_mul_pre((ep_t *)ep_curve_get_tab(), &(ctx->ep_g));
222#endif
223}

299void ep_curve_mul_a(fp_t c, const fp_t a) {
300 ctx_t *ctx = core_get();
301 switch (ctx->ep_opt_a) {
302 case RLC_ZERO:
303 fp_zero(c);
304 break;
305 case RLC_ONE:
306 fp_copy(c, a);
307 break;
308 case RLC_TWO:
309 fp_dbl(c, a);
310 break;
311#if FP_RDC != MONTY
312 case RLC_TINY:
313 fp_mul_dig(c, a, ctx->ep_a[0]);
314 break;
315#endif
316 default:
317 fp_mul(c, a, ctx->ep_a);
318 break;
319 }
320}

397void ep_curve_set_plain(const fp_t a, const fp_t b, const ep_t g, const bn_t r,
398 const bn_t h, int ctmap) {
399 ctx_t *ctx = core_get();
400 ctx->ep_is_endom = 0;
401 ctx->ep_is_super = 0;
402
403#ifdef EP_ENDOM
404 /* We do not use beta due to lack of endomorphisms so compute and cache
405 * square root of -1 for evaluating the distortion map in pairing-friendly
406 * curves with embedding degree 1. */
407 if (ctx->ep_is_pairf) {
408 fp_set_dig(ctx->beta, 1);
409 fp_neg(ctx->beta, ctx->beta);
410 fp_srt(ctx->beta, ctx->beta);
411 }
412#endif
413
414 ep_curve_set(a, b, g, r, h, ctmap);
415}

434void ep_curve_set_endom(const fp_t a, const fp_t b, const ep_t g, const bn_t r,
435 const bn_t h, const fp_t beta, const bn_t l, int ctmap) {
436 uint_t bits = bn_bits(r);
437 ctx_t *ctx = core_get();
438 ctx->ep_is_endom = 1;
439 ctx->ep_is_super = 0;
440
441 ep_curve_set(a, b, g, r, h, ctmap);
442
443 /* Precompute endomorphism constants. */
444#if EP_MUL == LWNAF || EP_FIX == COMBS || EP_FIX == LWNAF || EP_SIM == INTER || !defined(STRIP)
445 ep_t p, q;
446 bn_t m;
447
448 ep_null(p);
449 ep_null(q);
450 bn_null(m);
451
452 RLC_TRY {
453 ep_new(p);
454 ep_new(q);
455 bn_new(m);
456
457 /* Check if [m]P = \psi(P). */
458 fp_copy(ctx->beta, beta);
459 /* Fix lambda in case it is negative. */
460 if (bn_sign(l) == RLC_NEG) {
461 bn_add(m, l, r);
462 } else {
463 bn_copy(m, l);
464 }
465 /* Now check that beta and lambda match each other. */
466 ep_psi(p, g);
467 ep_mul_basic(q, g, m);
468 /* Fix beta in case it is the wrong value. */
469 if (ep_cmp(q, p) != RLC_EQ) {
470 fp_neg(ctx->beta, ctx->beta);
471 if (fp_is_zero(a)) {
472 /* In this case, look for other choice of beta. */
473 fp_sub_dig(ctx->beta, ctx->beta, 1);
474 }
475 ep_psi(p, g);
476 ep_mul_basic(q, g, m);
477 if (ep_cmp(q, p) != RLC_EQ) {
478 RLC_THROW(ERR_NO_VALID);
479 }
480 }
481 bn_gcd_ext_mid(&(ctx->ep_v1[1]), &(ctx->ep_v1[2]), &(ctx->ep_v2[1]),
482 &(ctx->ep_v2[2]), m, r);
483 /* m = (v1[1] * v2[2] - v1[2] * v2[1]) / 2. */
484 bn_mul(&(ctx->ep_v1[0]), &(ctx->ep_v1[1]), &(ctx->ep_v2[2]));
485 bn_mul(&(ctx->ep_v2[0]), &(ctx->ep_v1[2]), &(ctx->ep_v2[1]));
486 bn_sub(m, &(ctx->ep_v1[0]), &(ctx->ep_v2[0]));
487 bn_hlv(m, m);
488 /* v1[0] = round(v2[2] * 2^|n| / m). */
489 bn_lsh(&(ctx->ep_v1[0]), &(ctx->ep_v2[2]), bits + 1);
490 if (bn_sign(&(ctx->ep_v1[0])) == RLC_POS) {
491 bn_add(&(ctx->ep_v1[0]), &(ctx->ep_v1[0]), m);
492 } else {
493 bn_sub(&(ctx->ep_v1[0]), &(ctx->ep_v1[0]), m);
494 }
495 bn_dbl(m, m);
496 bn_div(&(ctx->ep_v1[0]), &(ctx->ep_v1[0]), m);
497 if (bn_sign(&ctx->ep_v1[0]) == RLC_NEG) {
498 bn_add_dig(&(ctx->ep_v1[0]), &(ctx->ep_v1[0]), 1);
499 }
500 /* v2[0] = round(v1[2] * 2^|n| / m). */
501 bn_lsh(&(ctx->ep_v2[0]), &(ctx->ep_v1[2]), bits + 1);
502 if (bn_sign(&(ctx->ep_v2[0])) == RLC_POS) {
503 bn_add(&(ctx->ep_v2[0]), &(ctx->ep_v2[0]), m);
504 } else {
505 bn_sub(&(ctx->ep_v2[0]), &(ctx->ep_v2[0]), m);
506 }
507 bn_div(&(ctx->ep_v2[0]), &(ctx->ep_v2[0]), m);
508 if (bn_sign(&ctx->ep_v2[0]) == RLC_NEG) {
509 bn_add_dig(&(ctx->ep_v2[0]), &(ctx->ep_v2[0]), 1);
510 }
511 bn_neg(&(ctx->ep_v2[0]), &(ctx->ep_v2[0]));
512 } RLC_CATCH_ANY {
513 RLC_THROW(ERR_CAUGHT);
514 } RLC_FINALLY {
515 ep_free(p);
516 ep_free(q);
517 bn_free(m);
518 }
519
520#endif
521}
