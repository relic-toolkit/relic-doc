44 static void detect_opt(int *opt, fp_t a) {
45  fp_t t;
46 
47  fp_null(t);
48 
49  TRY {
50  fp_new(t);
51  fp_prime_conv_dig(t, 3);
52  fp_neg(t, t);
53 
54  if (fp_cmp(a, t) == RLC_EQ) {
55  *opt = RLC_MIN3;
56  } else {
57  if (fp_is_zero(a)) {
58  *opt = RLC_ZERO;
59  } else {
60  fp_set_dig(t, 1);
61  if (fp_cmp_dig(a, 1) == RLC_EQ) {
62  *opt = RLC_ONE;
63  } else {
64  if (fp_cmp_dig(a, 2) == RLC_EQ) {
65  *opt = RLC_TWO;
66  } else {
67  if (fp_bits(a) <= RLC_DIG) {
68  *opt = RLC_TINY;
69  } else {
70  *opt = RLC_HUGE;
71  }
72  }
73  }
74  }
75  }
76  }
77  CATCH_ANY {
78  THROW(ERR_CAUGHT);
79  }
80  FINALLY {
81  fp_free(t);
82  }
83 }

249 void ep_curve_set_endom(const fp_t a, const fp_t b, const ep_t g, const bn_t r,
250  const bn_t h, const fp_t beta, const bn_t l) {
251  int bits = bn_bits(r);
252  ctx_t *ctx = core_get();
253  ctx->ep_is_endom = 1;
254  ctx->ep_is_super = 0;
255 
256  fp_copy(ctx->ep_a, a);
257  fp_copy(ctx->ep_b, b);
258 
259  detect_opt(&(ctx->ep_opt_a), ctx->ep_a);
260  detect_opt(&(ctx->ep_opt_b), ctx->ep_b);
261 
262 #if EP_MUL == LWNAF || EP_FIX == COMBS || EP_FIX == LWNAF || EP_SIM == INTER || !defined(STRIP)
263  fp_copy(ctx->beta, beta);
264  bn_gcd_ext_mid(&(ctx->ep_v1[1]), &(ctx->ep_v1[2]), &(ctx->ep_v2[1]),
265  &(ctx->ep_v2[2]), l, r);
266  /* r = (v1[1] * v2[2] - v1[2] * v2[1]) / 2. */
267  bn_mul(&(ctx->ep_v1[0]), &(ctx->ep_v1[1]), &(ctx->ep_v2[2]));
268  bn_mul(&(ctx->ep_v2[0]), &(ctx->ep_v1[2]), &(ctx->ep_v2[1]));
269  bn_sub(&(ctx->ep_r), &(ctx->ep_v1[0]), &(ctx->ep_v2[0]));
270  bn_hlv(&(ctx->ep_r), &(ctx->ep_r));
271  /* v1[0] = round(v2[2] * 2^|n| / l). */
272  bn_lsh(&(ctx->ep_v1[0]), &(ctx->ep_v2[2]), bits + 1);
273  if (bn_sign(&(ctx->ep_v1[0])) == RLC_POS) {
274  bn_add(&(ctx->ep_v1[0]), &(ctx->ep_v1[0]), &(ctx->ep_r));
275  } else {
276  bn_sub(&(ctx->ep_v1[0]), &(ctx->ep_v1[0]), &(ctx->ep_r));
277  }
278  bn_dbl(&(ctx->ep_r), &(ctx->ep_r));
279  bn_div(&(ctx->ep_v1[0]), &(ctx->ep_v1[0]), &(ctx->ep_r));
280  if (bn_sign(&ctx->ep_v1[0]) == RLC_NEG) {
281  bn_add_dig(&(ctx->ep_v1[0]), &(ctx->ep_v1[0]), 1);
282  }
283  /* v2[0] = round(v1[2] * 2^|n| / l). */
284  bn_lsh(&(ctx->ep_v2[0]), &(ctx->ep_v1[2]), bits + 1);
285  if (bn_sign(&(ctx->ep_v2[0])) == RLC_POS) {
286  bn_add(&(ctx->ep_v2[0]), &(ctx->ep_v2[0]), &(ctx->ep_r));
287  } else {
288  bn_sub(&(ctx->ep_v2[0]), &(ctx->ep_v2[0]), &(ctx->ep_r));
289  }
290  bn_div(&(ctx->ep_v2[0]), &(ctx->ep_v2[0]), &(ctx->ep_r));
291  if (bn_sign(&ctx->ep_v2[0]) == RLC_NEG) {
292  bn_add_dig(&(ctx->ep_v2[0]), &(ctx->ep_v2[0]), 1);
293  }
294  bn_neg(&(ctx->ep_v2[0]), &(ctx->ep_v2[0]));
295 #endif
296 
297  ep_norm(&(ctx->ep_g), g);
298  bn_copy(&(ctx->ep_r), r);
299  bn_copy(&(ctx->ep_h), h);
300 
301 #if defined(EP_PRECO)
302  ep_mul_pre((ep_t *)ep_curve_get_tab(), &(ctx->ep_g));
303 #endif
304 }
