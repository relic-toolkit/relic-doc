52 static void pp_mil_k8(fp8_t r, ep2_t *t, ep2_t *q, ep_t *p, int m, bn_t a) {
53  fp8_t l;
54  ep_t *_p = RLC_ALLOCA(ep_t, m);
55  ep2_t *_q = RLC_ALLOCA(ep2_t, m);
56  int i, j, len = bn_bits(a) + 1;
57  int8_t s[RLC_FP_BITS + 1];
58 
59  if (m == 0) {
60  return;
61  }
62 
63  fp8_null(l);
64 
65  TRY {
66  fp8_new(l);
67  if (_p == NULL || _q == NULL) {
68  THROW(ERR_NO_MEMORY);
69  }
70 
71  for (j = 0; j < m; j++) {
72  ep_null(_p[j]);
73  ep2_null(_q[j]);
74  ep_new(_p[j]);
75  ep2_new(_q[j]);
76  ep2_copy(t[j], q[j]);
77  ep2_neg(_q[j], q[j]);
78 
79 #if EP_ADD == BASIC
80  ep_neg(_p[j], p[j]);
81 #else
82  fp_neg(_p[j]->x, p[j]->x);
83  fp_copy(_p[j]->y, p[j]->y);
84 #endif
85  }
86 
87  fp8_zero(l);
88  bn_rec_naf(s, &len, a, 2);
89  for (i = len - 2; i >= 0; i--) {
90  fp8_sqr(r, r);
91  for (j = 0; j < m; j++) {
92  pp_dbl_k8(l, t[j], t[j], _p[j]);
93  fp8_mul(r, r, l);
94  if (s[i] > 0) {
95  pp_add_k8(l, t[j], q[j], _p[j]);
96  fp8_mul_dxs(r, r, l);
97  }
98  if (s[i] < 0) {
99  pp_add_k8(l, t[j], _q[j], _p[j]);
100  fp8_mul_dxs(r, r, l);
101  }
102  }
103  }
104  }
105  CATCH_ANY {
106  THROW(ERR_CAUGHT);
107  }
108  FINALLY {
109  fp8_free(l);
110  for (j = 0; j < m; j++) {
111  ep_free(_p[j]);
112  ep2_free(_q[j]);
113  }
114  RLC_FREE(_p);
115  RLC_FREE(_q);
116  }
117 }
