42void fp54_mul_basic(fp54_t c, const fp54_t a, const fp54_t b) {
43 fp18_t t0, t1, t2, t3, t4, t5;
44
45 fp18_null(t0);
46 fp18_null(t1);
47 fp18_null(t2);
48 fp18_null(t3);
49 fp18_null(t4);
50 fp18_null(t5);
51
52 RLC_TRY {
53 fp18_new(t0);
54 fp18_new(t1);
55 fp18_new(t2);
56 fp18_new(t3);
57 fp18_new(t4);
58 fp18_new(t5);
59
60 /* Karatsuba algorithm. */
61
62 /* t0 = a_0 * b_0. */
63 fp18_mul(t0, a[0], b[0]);
64 /* t1 = a_1 * b_1. */
65 fp18_mul(t1, a[1], b[1]);
66 /* t2 = a_2 * b_2. */
67 fp18_mul(t2, a[2], b[2]);
68
69 fp18_add(t3, a[1], a[2]);
70 fp18_add(t4, b[1], b[2]);
71 fp18_mul(t3, t3, t4);
72 fp18_sub(t3, t3, t1);
73 fp18_sub(t3, t3, t2);
74 fp18_mul_art(t3, t3);
75 fp18_add(t3, t3, t0);
76
77 fp18_add(t4, a[0], a[1]);
78 fp18_add(t5, b[0], b[1]);
79 fp18_mul(t4, t4, t5);
80 fp18_sub(t4, t4, t0);
81 fp18_sub(t4, t4, t1);
82 fp18_mul_art(t5, t2);
83 fp18_add(c[1], t4, t5);
84
85 fp18_add(t4, a[0], a[2]);
86 fp18_add(t5, b[0], b[2]);
87 fp18_mul(c[2], t4, t5);
88 fp18_sub(c[2], c[2], t0);
89 fp18_add(c[2], c[2], t1);
90 fp18_sub(c[2], c[2], t2);
91
92 fp18_copy(c[0], t3);
93 } RLC_CATCH_ANY {
94 RLC_THROW(ERR_CAUGHT);
95 } RLC_FINALLY {
96 fp18_free(t0);
97 fp18_free(t1);
98 fp18_free(t2);
99 fp18_free(t3);
100 fp18_free(t4);
101 fp18_free(t5);
102 }
103}

175void fp54_mul_dxs(fp54_t c, const fp54_t a, const fp54_t b) {
176 fp18_t t0, t1, t2, t3, t4;
177
178 fp18_null(t0);
179 fp18_null(t1);
180 fp18_null(t2);
181 fp18_null(t3);
182 fp18_null(t4);
183
184 RLC_TRY {
185 fp18_new(t0);
186 fp18_new(t1);
187 fp18_new(t2);
188 fp18_new(t3);
189 fp18_new(t4);
190
191 /* Karatsuba algorithm. */
192
193 /* t0 = a_0 * b_0. */
194#if EP_ADD == BASIC
195 fp18_mul_dxs(t0, a[0], b[0]);
196#else
197 fp18_mul(t0, a[0], b[0]);
198#endif
199 /* t2 = a_2 * b_2. */
200 fp9_mul(t2[0], a[2][0], b[2][0]);
201 fp9_mul(t2[1], a[2][1], b[2][0]);
202
203 fp18_add(t3, a[1], a[2]);
204 fp9_mul(t3[0], t3[0], b[2][0]);
205 fp9_mul(t3[1], t3[1], b[2][0]);
206 fp18_sub(t3, t3, t2);
207 fp18_mul_art(t3, t3);
208 fp18_add(t3, t3, t0);
209
210 fp18_add(t4, a[0], a[1]);
211#if EP_ADD == BASIC
212 fp18_mul_dxs(t4, t4, b[0]);
213#else
214 fp18_mul(t4, t4, b[0]);
215#endif
216 fp18_sub(t4, t4, t0);
217 fp18_mul_art(t1, t2);
218 fp18_add(c[1], t4, t1);
219
220 fp18_add(t4, a[0], a[2]);
221 fp9_add(t1[0], b[0][0], b[2][0]);
222 fp9_copy(t1[1], b[0][1]);
223#if EP_ADD == BASIC
224 fp18_mul_dxs(c[2], t4, t1);
225#else
226 fp18_mul(c[2], t4, t1);
227#endif
228 fp18_sub(c[2], c[2], t0);
229 fp18_sub(c[2], c[2], t2);
230
231 fp18_copy(c[0], t3);
232 } RLC_CATCH_ANY {
233 RLC_THROW(ERR_CAUGHT);
234 } RLC_FINALLY {
235 fp18_free(t0);
236 fp18_free(t1);
237 fp18_free(t2);
238 fp18_free(t3);
239 fp18_free(t4);
240 }

243void fp54_mul_art(fp54_t c, const fp54_t a) {
244 fp18_t t0;
245
246 fp18_null(t0);
247
248 RLC_TRY {
249 fp18_new(t0);
250
251 /* (a_0 + a_1 * v + a_2 * v^2) * v = a_2 + a_0 * v + a_1 * v^2 */
252 fp18_copy(t0, a[0]);
253 fp18_mul_art(c[0], a[2]);
254 fp18_copy(c[2], a[1]);
255 fp18_copy(c[1], t0);
256 } RLC_CATCH_ANY {
257 RLC_THROW(ERR_CAUGHT);
258 } RLC_FINALLY {
259 fp18_free(t0);
260 }
