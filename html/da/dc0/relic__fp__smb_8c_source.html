44static dig_t porninstep(dis_t m[4],const dig_t f[2], const dig_t g[2],
45 dig_t k, size_t s) {
46 dig_t limbx, ai = 1, bi = 0, ci = 0, di = 1;
47 dig_t g_lo = g[0], g_hi = g[1], f_lo = f[0], f_hi = f[1];
48 dig_t t_lo, t_hi, odd, borrow, xorm;
49
50 /* Unrolling twice gives some small speedup. */
51 for (size_t i = 0; i < s; i+=2) {
52 odd = 0 - (g_lo & 1);
53
54 /* g_ -= f_ if g_ is odd */
55 t_lo = g_lo, t_hi = g_hi;
56
57 borrow = 0;
58 limbx = g_lo - (f_lo & odd);
59 borrow = (g_lo < limbx);
60 g_lo = limbx;
61
62 limbx = g_hi - (f_hi & odd);
63 xorm = limbx - borrow;
64 borrow = -((g_hi < limbx) || (borrow && !limbx));
65 g_hi = xorm;
66
67 k += ((t_lo & f_lo) >> 1) & borrow;
68
69 /* negate g_-f_ if it borrowed */
70 g_lo ^= borrow;
71 g_hi ^= borrow;
72 limbx = g_lo + (borrow & 1);
73 g_hi += (g_lo < limbx);
74 g_lo = limbx;
75
76 /* f_=g_ if g_-f_ borrowed */
77 f_lo = ((t_lo ^ f_lo) & borrow) ^ f_lo;
78 f_hi = ((t_hi ^ f_hi) & borrow) ^ f_hi;
79
80 /* exchange ai and ci if g_-f_ borrowed */
81 xorm = (ai ^ ci) & borrow;
82 ai ^= xorm;
83 ci ^= xorm;
84
85 /* exchange bi and di if g_-f_ borrowed */
86 xorm = (bi ^ di) & borrow;
87 bi ^= xorm;
88 di ^= xorm;
89
90 /* subtract if g_ was odd */
91 ai -= ci & odd;
92 bi -= di & odd;
93
94 ci <<= 1;
95 di <<= 1;
96 g_lo >>= 1;
97 g_lo |= g_hi << (RLC_DIG - 1);
98 g_hi >>= 1;
99
100 k += (f_lo + 2) >> 2;
101
102 odd = 0 - (g_lo & 1);
103
104 /* g_ -= f_ if g_ is odd */
105 t_lo = g_lo, t_hi = g_hi;
106
107 borrow = 0;
108 limbx = g_lo - (f_lo & odd);
109 borrow = (g_lo < limbx);
110 g_lo = limbx;
111
112 limbx = g_hi - (f_hi & odd);
113 xorm = limbx - borrow;
114 borrow = -((g_hi < limbx) || (borrow && !limbx));
115 g_hi = xorm;
116
117 k += ((t_lo & f_lo) >> 1) & borrow;
118
119 /* negate g_-f_ if it borrowed */
120 g_lo ^= borrow;
121 g_hi ^= borrow;
122 limbx = g_lo + (borrow & 1);
123 g_hi += (g_lo < limbx);
124 g_lo = limbx;
125
126 /* f_=g_ if g_-f_ borrowed */
127 f_lo = ((t_lo ^ f_lo) & borrow) ^ f_lo;
128 f_hi = ((t_hi ^ f_hi) & borrow) ^ f_hi;
129
130 /* exchange ai and ci if g_-f_ borrowed */
131 xorm = (ai ^ ci) & borrow;
132 ai ^= xorm;
133 ci ^= xorm;
134
135 /* exchange bi and di if g_-f_ borrowed */
136 xorm = (bi ^ di) & borrow;
137 bi ^= xorm;
138 di ^= xorm;
139
140 /* subtract if g_ was odd */
141 ai -= ci & odd;
142 bi -= di & odd;
143
144 ci <<= 1;
145 di <<= 1;
146 g_lo >>= 1;
147 g_lo |= g_hi << (RLC_DIG - 1);
148 g_hi >>= 1;
149
150 k += (f_lo + 2) >> 2;
151 }
152
153 m[0] = ai;
154 m[1] = bi;
155 m[2] = ci;
156 m[3] = di;
157
158 return k;
159}

165static dis_t jumpdivstep(dis_t m[4], dig_t *k, dis_t delta, dis_t y, dis_t x,
166 int s) {
167 dig_t t0, t1, t2, c0, c1, yi, ai = 1, bi = 0, ci = 0, di = 1, u = 0;
168
169 /* Unrolling twice makes it faster. */
170 for (s -= 2; s >= 0; s -= 2) {
171 yi = y;
172
173 c0 = delta >> (RLC_DIG - 1);
174 c1 = -(x & 1);
175 c0 &= c1;
176
177 t0 = (y ^ c0) - c0;
178 t1 = (ci ^ c0) - c0;
179 t2 = (di ^ c0) - c0;
180 x += t0 & c1;
181 ai += t1 & c1;
182 bi += t2 & c1;
183
184 /* delta = RLC_SEL(2 + delta, 2 - delta, c0) */
185 y += x & c0;
186 ci += ai & c0;
187 di += bi & c0;
188
189 x >>= 1;
190 ci <<= 1;
191 di <<= 1;
192 delta = (delta ^ c0) - 1;
193
194 u += ((yi & y) ^ (y >> 1)) & 2;
195 u += (u & 1) ^ RLC_SIGN(ci);
196
197 yi = y;
198
199 c0 = delta >> (RLC_DIG - 1);
200 c1 = -(x & 1);
201 c0 &= c1;
202
203 t0 = (y ^ c0) - c0;
204 t1 = (ci ^ c0) - c0;
205 t2 = (di ^ c0) - c0;
206 x += t0 & c1;
207 ai += t1 & c1;
208 bi += t2 & c1;
209
210 /* delta = RLC_SEL(2 + delta, 2 - delta, c0) */
211 y += x & c0;
212 ci += ai & c0;
213 di += bi & c0;
214
215 x >>= 1;
216 ci <<= 1;
217 di <<= 1;
218 delta = (delta ^ c0) - 1;
219
220 u += ((yi & y) ^ (y >> 1)) & 2;
221 u += (u & 1) ^ RLC_SIGN(ci);
222 }
223 m[0] = ai;
224 m[1] = bi;
225 m[2] = ci;
226 m[3] = di;
227 *k = u;
228 return delta;
229}

239int fp_smb_basic(const fp_t a) {
240 bn_t t;
241 int r = 0;
242
243 bn_null(t);
244
245 RLC_TRY {
246 bn_new(t);
247
248 /* t = (b - 1)/2. */
249 t->sign = RLC_POS;
250 t->used = RLC_FP_DIGS;
251 dv_copy(t->dp, fp_prime_get(), RLC_FP_DIGS);
252 bn_sub_dig(t, t, 1);
253 bn_hlv(t, t);
254
255 fp_exp(t->dp, a, t);
256 r = (fp_cmp_dig(t->dp, 1) == RLC_EQ);
257 fp_neg(t->dp, t->dp);
258 r = RLC_SEL(r, -(fp_cmp_dig(t->dp, 1) == RLC_EQ), !r);
259 }
260 RLC_CATCH_ANY {
261 RLC_THROW(ERR_CAUGHT);
262 }
263 RLC_FINALLY {
264 bn_free(t);
265 }
266 return r;
267}

276int fp_smb_binar(const fp_t a) {
277 const size_t s = RLC_DIG - 2;
278 dv_t f, g, t, t0, t1;
279 dig_t g_[2], f_[2], neg, l, g_hi, g_lo, f_hi, f_lo, mask, k = 0;
280 dis_t m[4];
281 int iterations = 2 * RLC_FP_DIGS * RLC_DIG;
282
283 if (fp_is_zero(a)) {
284 return 0;
285 }
286
287 dv_null(f);
288 dv_null(g);
289 dv_null(t);
290 dv_null(t0);
291 dv_null(t1);
292
293 RLC_TRY {
294 dv_new(f);
295 dv_new(g);
296 dv_new(t);
297 dv_new(t0);
298 dv_new(t1);
299
300 dv_zero(t, 2 * RLC_FP_DIGS);
301 dv_copy(f, fp_prime_get(), RLC_FP_DIGS);
302#if FP_RDC == MONTY
303 /* Convert a from Montgomery form. */
304 fp_copy(t, a);
305 fp_rdcn_low(g, t);
306#else
307 fp_copy(g, a);
308#endif
309
310 for (size_t len, i = 0; i < iterations / s; i++) {
311 f_hi = f[RLC_FP_DIGS - 1], f_lo = f[RLC_FP_DIGS - 2];
312 g_hi = g[RLC_FP_DIGS - 1], g_lo = g[RLC_FP_DIGS - 2];
313 for (int j = RLC_FP_DIGS - 2; j >= 0; j--) {
314 l = (f_hi | g_hi);
315 l = ~l & (l - 1);
316 mask = -(l >> (RLC_DIG - 1));
317 f_hi = ((f_lo ^ f_hi) & mask) ^ f_hi;
318 g_hi = ((g_lo ^ g_hi) & mask) ^ g_hi;
319 f_lo = ((f[j] ^ f_lo) & mask) ^ f_lo;
320 g_lo = ((g[j] ^ g_lo) & mask) ^ g_lo;
321 }
322 len = RLC_DIG - util_bits_dig(f_hi | g_hi);
323 f_[0] = f[0], f_[1] = RLC_LSH(f_hi, f_lo, len);
324 g_[0] = g[0], g_[1] = RLC_LSH(g_hi, g_lo, len);
325
326 k = porninstep(m, f_, g_, k, s);
327
328 t0[RLC_FP_DIGS] = bn_muls_low(t0, g, RLC_POS, m[0], RLC_FP_DIGS);
329 t1[RLC_FP_DIGS] = bn_muls_low(t1, f, RLC_POS, m[1], RLC_FP_DIGS);
330 bn_addn_low(t0, t0, t1, RLC_FP_DIGS + 1);
331 neg = RLC_SIGN(t0[RLC_FP_DIGS]);
332 bn_rshb_low(t, t0, RLC_FP_DIGS + 1, (RLC_DIG - 2));
333 bn_negs_low(t, t, neg, RLC_FP_DIGS);
334
335 t0[RLC_FP_DIGS] = bn_muls_low(t0, g, RLC_POS, m[2], RLC_FP_DIGS);
336 t1[RLC_FP_DIGS] = bn_muls_low(t1, f, RLC_POS, m[3], RLC_FP_DIGS);
337 bn_addn_low(t1, t1, t0, RLC_FP_DIGS + 1);
338 bn_rshb_low(f, t1, RLC_FP_DIGS + 1, (RLC_DIG - 2));
339 bn_negs_low(f, f, RLC_SIGN(t1[RLC_FP_DIGS]), RLC_FP_DIGS);
340
341 fp_copy(g, t);
342 k += (f[0] >> 1) & neg;
343 }
344
345 k = porninstep(m, g, f, k, iterations % s);
346
347 } RLC_CATCH_ANY {
348 RLC_THROW(ERR_CAUGHT)
349 } RLC_FINALLY {
350 dv_free(f);
351 dv_free(g);
352 dv_free(t);
353 dv_free(t0);
354 dv_free(t1);
355 }
356
357 return (k & 1 ? -1 : 1);
358}

364int fp_smb_divst(const fp_t a) {
365 /* Compute number of iterations based on modulus size. */
366#if FP_PRIME < 46
367 const int d = (49 * FP_PRIME + 80) / 17;
368#else
369 const int d = (49 * FP_PRIME + 57) / 17;
370#endif
371 dig_t delta = 1, g0, d0, fs, gs, k, mask, s;
372 bn_t _t;
373 dv_t f, g, t;
374 int r = 0;
375
376 bn_null(_t);
377 dv_null(f);
378 dv_null(g);
379 dv_null(t);
380
381 RLC_TRY {
382 bn_new(_t);
383 dv_new(f);
384 dv_new(g);
385 dv_new(t);
386
387#if WSIZE == 8
388 bn_set_dig(_t, d >> 8);
389 bn_lsh(_t, _t, 8);
390 bn_add_dig(_t, _t, d & 0xFF);
391#else
392 bn_set_dig(_t, d);
393#endif
394
395 k = 0;
396 dv_copy(f, fp_prime_get(), RLC_FP_DIGS);
397#if FP_RDC == MONTY
398 /* Convert a from Montgomery form. */
399 dv_zero(t, 2 * RLC_FP_DIGS);
400 fp_copy(t, a);
401 fp_rdcn_low(g, t);
402#else
403 fp_copy(g, a);
404#endif
405 r = dv_cmp(g, f, RLC_FP_DIGS);
406 fp_subn_low(t, g, f);
407 fp_copy_sec(g, t, r != RLC_LT);
408
409 fs = gs = RLC_POS;
410
411 for (int i = 0; i < d; i++) {
412 d0 = g[0] & ((int)delta > 0);
413 /* Conditionally negate delta if d0 is set. */
414 delta = (delta ^ -d0) + d0;
415 /* Conditionally swap and negate based on d0. */
416 mask = -d0;
417 s = (fs ^ gs) & mask;
418 fs ^= s;
419 gs ^= s ^ d0;
420 for (int j = 0; j < RLC_FP_DIGS; j++) {
421 s = (f[j] ^ g[j]) & mask;
422 f[j] ^= s;
423 g[j] ^= s ^ (-d0);
424 }
425 fp_add1_low(g, g, d0);
426
427 k ^= (((g[0] & f[0]) >> (dig_t)1) ^ (fs & gs)) & d0;
428 k ^= (f[0] >> 1) ^ (f[0] >> 2);
429 k &= 1;
430
431 delta++;
432 g0 = g[0] & 1;
433 for (int j = 0; j < RLC_FP_DIGS; j++) {
434 t[j] = f[j] & (-g0);
435 }
436
437 /* Compute g = (g + g0*f) div 2 by conditionally copying f to u and
438 * updating the sign of g. */
439 gs ^= g0 & (fs ^ bn_addn_low(g, g, t, RLC_FP_DIGS));
440 /* Shift and restore the sign. */
441 fp_rsh1_low(g, g);
442 g[RLC_FP_DIGS - 1] |= (dig_t)gs << (RLC_DIG - 1);
443 }
444
445 k = (2*k) % 4;
446 fp_zero(t);
447 t[0] = 1;
448 bn_negs_low(f, f, fs, RLC_FP_DIGS);
449
450 r = RLC_SEL(r, 1 - k, dv_cmp_sec(f, t, RLC_FP_DIGS) == RLC_EQ);
451 bn_negs_low(t, t, 1, RLC_FP_DIGS);
452 r = RLC_SEL(r, 1 - k, dv_cmp_sec(f, t, RLC_FP_DIGS) == RLC_EQ);
453 r = RLC_SEL(r, 1 - k, fp_is_zero(f));
454 r = RLC_SEL(r, 0, fp_is_zero(a));
455 } RLC_CATCH_ANY {
456 RLC_THROW(ERR_CAUGHT)
457 } RLC_FINALLY {
458 bn_free(_t);
459 dv_free(f);
460 dv_free(g);
461 dv_free(t);
462 }
463 return r;
464}

470int fp_smb_jmpds(const fp_t a) {
471 dis_t m[4], d = -1;
472 /* Iterations taken directly from https://github.com/sipa/safegcd-bounds */
473 const int iterations = (45907 * FP_PRIME + 26313) / 19929;
474 int loops, precision, i, r = 0, s = RLC_DIG - 2;
475 dv_t f, g, t0, t1, u0, u1;
476 dig_t sf, sg, j, k;
477
478 dv_null(f);
479 dv_null(g);
480 dv_null(t0);
481 dv_null(t1);
482 dv_null(u0);
483 dv_null(u1);
484
485 RLC_TRY {
486 dv_new(f);
487 dv_new(g);
488 dv_new(t0);
489 dv_new(t1);
490 dv_new(u0);
491 dv_new(u1);
492
493 dv_copy(f, fp_prime_get(), RLC_FP_DIGS);
494 f[RLC_FP_DIGS] = 0;
495 dv_zero(g, RLC_FP_DIGS + 1);
496 dv_zero(t0 + RLC_FP_DIGS, RLC_FP_DIGS);
497
498#if FP_RDC == MONTY
499 /* Convert a from Montgomery form. */
500 fp_copy(t0, a);
501 fp_rdcn_low(g, t0);
502#else
503 fp_copy(g, a);
504#endif
505 precision = RLC_FP_DIGS;
506 loops = iterations / s;
507 loops = (iterations % s == 0 ? loops - 1 : loops);
508
509 for (i = j = k = 0; i < loops; i++) {
510 d = jumpdivstep(m, &k, d, f[0], g[0], s);
511
512 sf = RLC_SIGN(f[precision]);
513 sg = RLC_SIGN(g[precision]);
514 bn_negs_low(u0, f, sf, precision);
515 bn_negs_low(u1, g, sg, precision);
516
517 t0[precision] = bn_muls_low(t0, u0, sf, m[3], precision);
518 t1[precision] = bn_muls_low(t1, u1, sg, m[2], precision);
519 bn_addn_low(t0, t0, t1, precision + 1);
520 bn_rshs_low(f, t0, precision + 1, s);
521
522 t0[precision] = bn_muls_low(t0, u0, sf, m[1], precision);
523 t1[precision] = bn_muls_low(t1, u1, sg, m[0], precision);
524 bn_addn_low(t1, t1, t0, precision + 1);
525 bn_rshs_low(g, t1, precision + 1, s);
526
527 j = (j + k) % 4;
528 j = (j + ((j & 1) ^ (RLC_SIGN(f[precision])))) % 4;
529 }
530
531 s = iterations - loops * s;
532 d = jumpdivstep(m, &k, d, f[0], g[0], s);
533
534 sf = RLC_SIGN(f[precision]);
535 sg = RLC_SIGN(g[precision]);
536 bn_negs_low(u0, f, sf, precision);
537 bn_negs_low(u1, g, sg, precision);
538
539 t0[precision] = bn_muls_low(t0, u0, sf, m[3], precision);
540 t1[precision] = bn_muls_low(t1, u1, sg, m[2], precision);
541 bn_addn_low(t0, t0, t1, precision + 1);
542 bn_rshs_low(f, t0, precision + 1, s);
543
544 j = (j + k) % 4;
545 j = (j + ((j & 1) ^ (RLC_SIGN(f[precision])))) % 4;
546 j = (j + (j & 1)) % 4;
547
548 fp_zero(t0);
549 r = RLC_SEL(r, 1 - j, dv_cmp_sec(f, t0, RLC_FP_DIGS) == RLC_EQ);
550 t0[0] = 1;
551 r = RLC_SEL(r, 1 - j, dv_cmp_sec(f, t0, RLC_FP_DIGS) == RLC_EQ);
552 bn_negs_low(t0, t0, 1, RLC_FP_DIGS);
553 r = RLC_SEL(r, 1 - j, dv_cmp_sec(f, t0, RLC_FP_DIGS) == RLC_EQ);
554 }
555 RLC_CATCH_ANY {
556 RLC_THROW(ERR_CAUGHT);
557 }
558 RLC_FINALLY {
559 dv_free(f);
560 dv_free(g);
561 dv_free(t0);
562 dv_free(t1);
563 dv_free(u0);
564 dv_free(u1);
565 }
566
567 return r;
568}
