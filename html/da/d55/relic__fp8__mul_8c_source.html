42 void fp8_mul_basic(fp8_t c, fp8_t a, fp8_t b) {
43  fp4_t t0, t1, t4;
44 
45  fp4_null(t0);
46  fp4_null(t1);
47  fp4_null(t4);
48 
49  TRY {
50  fp4_new(t0);
51  fp4_new(t1);
52  fp4_new(t4);
53 
54  /* Karatsuba algorithm. */
55 
56  /* t0 = a_0 * b_0. */
57  fp4_mul(t0, a[0], b[0]);
58  /* t1 = a_1 * b_1. */
59  fp4_mul(t1, a[1], b[1]);
60  /* t4 = b_0 + b_1. */
61  fp4_add(t4, b[0], b[1]);
62 
63  /* c_1 = a_0 + a_1. */
64  fp4_add(c[1], a[0], a[1]);
65 
66  /* c_1 = (a_0 + a_1) * (b_0 + b_1) */
67  fp4_mul(c[1], c[1], t4);
68  fp4_sub(c[1], c[1], t0);
69  fp4_sub(c[1], c[1], t1);
70 
71  /* c_0 = a_0b_0 + v * a_1b_1. */
72  fp4_mul_art(t4, t1);
73  fp4_add(c[0], t0, t4);
74  } CATCH_ANY {
75  THROW(ERR_CAUGHT);
76  } FINALLY {
77  fp4_free(t0);
78  fp4_free(t1);
79  fp4_free(t4);
80  }
81 }

124 void fp8_mul_dxs(fp8_t c, fp8_t a, fp8_t b) {
125  fp4_t t0, t1;
126  dv4_t u0, u1, u2, u3;
127 
128  fp4_null(t0);
129  fp4_null(t1);
130  dv4_null(u0);
131  dv4_null(u1);
132  dv4_null(u2);
133  dv4_null(u3);
134 
135  TRY {
136  fp4_new(t0);
137  fp4_new(t1);
138  dv4_new(u0);
139  dv4_new(u1);
140  dv4_new(u2);
141  dv4_new(u3);
142 
143  /* Karatsuba algorithm. */
144 
145  /* u0 = a_0 * b_0. */
146  fp4_mul_unr(u0, a[0], b[0]);
147  /* u1 = a_1 * b_1. */
148  fp4_mul_dxs_unr(u1, a[1], b[1]);
149 
150  /* t1 = a_0 + a_1. */
151  fp4_add(t0, a[0], a[1]);
152  /* t0 = b_0 + b_1. */
153  fp4_add(t1, b[0], b[1]);
154  /* u2 = (a_0 + a_1) * (b_0 + b_1) */
155  fp4_mul_unr(u2, t0, t1);
156  /* c_1 = u2 - a_0b_0 - a_1b_1. */
157  for (int i = 0; i < 2; i++) {
158  fp2_addc_low(u3[i], u0[i], u1[i]);
159  fp2_subc_low(u2[i], u2[i], u3[i]);
160  fp2_rdcn_low(c[1][i], u2[i]);
161  }
162  /* c_0 = a_0b_0 + v * a_1b_1. */
163  fp2_nord_low(u2[0], u1[1]);
164  dv_copy(u2[1][0], u1[0][0], 2 * RLC_FP_DIGS);
165  dv_copy(u2[1][1], u1[0][1], 2 * RLC_FP_DIGS);
166  for (int i = 0; i < 2; i++) {
167  fp2_addc_low(u2[i], u0[i], u2[i]);
168  fp2_rdcn_low(c[0][i], u2[i]);
169  }
170  } CATCH_ANY {

172  } FINALLY {
