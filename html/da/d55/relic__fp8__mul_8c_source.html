42void fp8_mul_basic(fp8_t c, const fp8_t a, const fp8_t b) {
43 fp4_t t0, t1, t4;
44
45 fp4_null(t0);
46 fp4_null(t1);
47 fp4_null(t4);
48
49 RLC_TRY {
50 fp4_new(t0);
51 fp4_new(t1);
52 fp4_new(t4);
53
54 /* Karatsuba algorithm. */
55
56 /* t0 = a_0 * b_0. */
57 fp4_mul(t0, a[0], b[0]);
58 /* t1 = a_1 * b_1. */
59 fp4_mul(t1, a[1], b[1]);
60 /* t4 = b_0 + b_1. */
61 fp4_add(t4, b[0], b[1]);
62
63 /* c_1 = a_0 + a_1. */
64 fp4_add(c[1], a[0], a[1]);
65
66 /* c_1 = (a_0 + a_1) * (b_0 + b_1) */
67 fp4_mul(c[1], c[1], t4);
68 fp4_sub(c[1], c[1], t0);
69 fp4_sub(c[1], c[1], t1);
70
71 /* c_0 = a_0b_0 + v * a_1b_1. */
72 fp4_mul_art(t4, t1);
73 fp4_add(c[0], t0, t4);
74 } RLC_CATCH_ANY {
75 RLC_THROW(ERR_CAUGHT);
76 } RLC_FINALLY {
77 fp4_free(t0);
78 fp4_free(t1);
79 fp4_free(t4);
80 }
81}

119void fp8_mul_dxs(fp8_t c, const fp8_t a, const fp8_t b) {
120 fp4_t t0, t1;
121 dv4_t u0, u1, u2, u3;
122
123 fp4_null(t0);
124 fp4_null(t1);
125 dv4_null(u0);
126 dv4_null(u1);
127 dv4_null(u2);
128 dv4_null(u3);
129
130 RLC_TRY {
131 fp4_new(t0);
132 fp4_new(t1);
133 dv4_new(u0);
134 dv4_new(u1);
135 dv4_new(u2);
136 dv4_new(u3);
137
138 /* Karatsuba algorithm. */
139
140 /* u0 = a_0 * b_0. */
141 fp4_mul_unr(u0, a[0], b[0]);
142 /* u1 = a_1 * b_1. */
143 fp4_mul_dxs_unr(u1, a[1], b[1]);
144
145 /* t1 = a_0 + a_1. */
146 fp4_add(t0, a[0], a[1]);
147 /* t0 = b_0 + b_1. */
148 fp4_add(t1, b[0], b[1]);
149 /* u2 = (a_0 + a_1) * (b_0 + b_1) */
150 fp4_mul_unr(u2, t0, t1);
151 /* c_1 = u2 - a_0b_0 - a_1b_1. */
152 for (int i = 0; i < 2; i++) {
153 fp2_addc_low(u3[i], u0[i], u1[i]);
154 fp2_subc_low(u2[i], u2[i], u3[i]);
155 fp2_rdcn_low(c[1][i], u2[i]);
156 }
157 /* c_0 = a_0b_0 + v * a_1b_1. */
158 fp2_nord_low(u2[0], u1[1]);
159 dv_copy(u2[1][0], u1[0][0], 2 * RLC_FP_DIGS);
160 dv_copy(u2[1][1], u1[0][1], 2 * RLC_FP_DIGS);
161 for (int i = 0; i < 2; i++) {
162 fp2_addc_low(u2[i], u0[i], u2[i]);
163 fp2_rdcn_low(c[0][i], u2[i]);
164 }
165 } RLC_CATCH_ANY {
166 RLC_THROW(ERR_CAUGHT);
167 } RLC_FINALLY {
168 fp4_free(t0);
169 dv4_free(t1);
170 dv4_free(u0);
171 dv4_free(u1);
172 dv4_free(u2);
173 dv4_free(u3);
174 }
175}

177void fp8_mul_unr(dv8_t c, const fp8_t a, const fp8_t b) {
178 fp4_t t0, t1;
179 dv4_t u0, u1, u2;
180
181 fp4_null(t0);
182 fp4_null(t1);
183 dv4_null(u0);
184 dv4_null(u1);
185 dv4_null(u2);
186
187 RLC_TRY {
188 fp4_new(t0);
189 fp4_new(t1);
190 dv4_new(u0);
191 dv4_new(u1);
192 dv4_new(u2);
193
194 /* Karatsuba algorithm. */
195
196 /* u0 = a_0 * b_0. */
197 fp4_mul_unr(u0, a[0], b[0]);
198 /* u1 = a_1 * b_1. */
199 fp4_mul_unr(u1, a[1], b[1]);
200 /* t1 = a_0 + a_1. */
201 fp4_add(t0, a[0], a[1]);
202 /* t0 = b_0 + b_1. */
203 fp4_add(t1, b[0], b[1]);
204 /* u2 = (a_0 + a_1) * (b_0 + b_1) */
205 fp4_mul_unr(u2, t0, t1);
206 /* c_1 = u2 - a_0b_0 - a_1b_1. */
207 for (int i = 0; i < 2; i++) {
208 fp2_subc_low(c[1][i], u2[i], u0[i]);
209 fp2_subc_low(c[1][i], c[1][i], u1[i]);
210 }
211 /* c_0 = a_0b_0 + v * a_1b_1. */
212 fp2_nord_low(u2[0], u1[1]);
213 dv_copy(u2[1][0], u1[0][0], 2 * RLC_FP_DIGS);
214 dv_copy(u2[1][1], u1[0][1], 2 * RLC_FP_DIGS);
215 for (int i = 0; i < 2; i++) {
216 fp2_addc_low(c[0][i], u0[i], u2[i]);
217 }
218 } RLC_CATCH_ANY {
219 RLC_THROW(ERR_CAUGHT);
220 } RLC_FINALLY {
221 fp4_free(t0);
222 fp4_free(t1);
223 dv4_free(u0);
224 dv4_free(u1);
225 dv4_free(u2);
226 }

250void fp8_mul_art(fp8_t c, const fp8_t a) {
251 fp4_t t0;
252
253 fp4_null(t0);
254
255 RLC_TRY {
256 fp4_new(t0);
257
258 /* (a_0 + a_1 * v) * v = a_0 * v + a_1 * v^4 */
259 fp4_copy(t0, a[0]);
260 fp4_mul_art(c[0], a[1]);
261 fp4_copy(c[1], t0);
262 } RLC_CATCH_ANY {
263 RLC_THROW(ERR_CAUGHT);
264 } RLC_FINALLY {
265 fp4_free(t0);

269void fp8_mul_frb(fp8_t c, const fp8_t a, int i, int j) {
270 fp2_t t;
271
272 fp2_null(t);
273
274 RLC_TRY {
275 fp4_new(t);
276
277 fp_copy(t[0], core_get()->fp8_p1[0]);
278 fp_copy(t[1], core_get()->fp8_p1[1]);
279
280 if (i == 1) {
281 fp8_copy(c, a);
282 for (int k = 0; k < j; k++) {
283 fp2_mul(c[0][0], c[0][0], t);
284 fp2_mul(c[0][1], c[0][1], t);
285 fp2_mul(c[1][0], c[1][0], t);
286 fp2_mul(c[1][1], c[1][1], t);
287 /* If constant in base field, then second component is zero. */
288 if (core_get()->frb8 == 1) {
289 fp8_mul_art(c, c);
290 }
291 }
292 } else {
293 RLC_THROW(ERR_NO_VALID);
294 }
295 } RLC_CATCH_ANY {
296 RLC_THROW(ERR_CAUGHT);
297 } RLC_FINALLY {
298 fp4_free(t);
299 }
