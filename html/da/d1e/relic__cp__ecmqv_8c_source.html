38int cp_ecmqv_gen(bn_t d, ec_t q) {
39 bn_t n;
40 int result = RLC_OK;
41
42 bn_null(n);
43
44 RLC_TRY {
45 bn_new(n);
46
47 ec_curve_get_ord(n);
48 bn_rand_mod(d, n);
49 ec_mul_gen(q, d);
50 }
51 RLC_CATCH_ANY {
52 result = RLC_ERR;
53 }
54 RLC_FINALLY {
55 bn_free(n);
56 }
57
58 return result;
59}

61int cp_ecmqv_key(uint8_t *key, size_t key_len, const bn_t d1, const bn_t d2,
62 const ec_t q2u, const ec_t q1v, const ec_t q2v) {
63 ec_t p;
64 bn_t x, n, s;
65 int l, result = RLC_OK;
66 uint8_t _x[RLC_FC_BYTES];
67
68 ec_null(p);
69 bn_null(x);
70 bn_null(n);
71 bn_null(s);
72
73 RLC_TRY {
74 ec_new(p);
75 bn_new(x);
76 bn_new(n);
77 bn_new(s);
78
79 ec_curve_get_ord(n);
80 l = bn_bits(n);
81 l = (l % 2) ? (l + 1) / 2 : l / 2;
82
83 ec_get_x(x, q2u);
84 bn_mod_2b(x, x, l);
85
86 bn_set_bit(x, l, 1);
87
88 bn_mul(s, x, d1);
89 bn_mod(s, s, n);
90 bn_add(s, s, d2);
91 bn_mod(s, s, n);
92
93 ec_get_x(x, q2v);
94 bn_mod_2b(x, x, l);
95
96 bn_set_bit(x, l, 1);
97
98 bn_mul(x, x, s);
99 bn_mod(x, x, n);
100
101 ec_mul_sim(p, q2v, s, q1v, x);
102
103 ec_get_x(x, p);
104 l = bn_size_bin(x);
105 bn_write_bin(_x, l, x);
106 md_kdf(key, key_len, _x, l);
107 }
108 RLC_CATCH_ANY {
109 result = RLC_ERR;
110 }
111 RLC_FINALLY {
112 ec_free(p);
113 bn_free(x);
114 bn_free(n);
115 bn_free(s);
116 }
117 return result;
118}
