40 void ed_map(ed_t p, const uint8_t *msg, int len) {
41  bn_t h;
42  fp_t t, u, v;
43  uint8_t digest[RLC_MD_LEN];
44 
45  bn_null(h);
46  fp_null(t);
47  fp_null(u);
48  fp_null(v);
49 
50  TRY {
51  bn_new(h);
52  fp_new(t);
53  fp_new(u);
54  fp_new(v);
55 
56  md_map(digest, msg, len);
57  bn_read_bin(h, digest, RLC_MIN(RLC_FP_BYTES, RLC_MD_LEN));
58 
59  fp_prime_conv(p->y, h);
60  fp_set_dig(p->z, 1);
61 
62  /* Make e = p. */
63  h->used = RLC_FP_DIGS;
64  dv_copy(h->dp, fp_prime_get(), RLC_FP_DIGS);
65 
66  /* Compute a^((p - 5)/8). */
67  bn_sub_dig(h, h, 5);
68  bn_rsh(h, h, 3);
69 
70  /* Decode using Elligator 2. */
71  while (1) {
72  /* u = y^2 - 1, v = d * y^2 + 1. */
73  fp_sqr(u, p->y);
74  fp_mul(v, u, core_get()->ed_d);
75  fp_sub_dig(u, u, 1);
76  fp_add_dig(v, v, 1);
77 
78  /* t = v^3, x = uv^7. */
79  fp_sqr(t, v);
80  fp_mul(t, t, v);
81  fp_sqr(p->x, t);
82  fp_mul(p->x, p->x, v);
83  fp_mul(p->x, p->x, u);
84 
85  /* x = uv^3 * (uv^7)^((p - 5)/8). */
86  fp_exp(p->x, p->x, h);
87  fp_mul(p->x, p->x, t);
88  fp_mul(p->x, p->x, u);
89 
90  /* Check if vx^2 == u. */
91  fp_sqr(t, p->x);
92  fp_mul(t, t, v);
93 
94  if (fp_cmp(t, u) != RLC_EQ) {
95  fp_neg(u, u);
96  /* Check if vx^2 == -u. */
97  if (fp_cmp(t, u) != RLC_EQ) {
98  fp_add_dig(p->y, p->y, 1);
99  } else {
100  fp_mul(p->x, p->x, core_get()->srm1);
101  break;
102  }
103  } else {
104  break;
105  }
106  }
107 
108  /* By Elligator convention. */
109  if (p->x[RLC_FP_DIGS - 1] >> (RLC_DIG - 1) == 1) {
110  fp_neg(p->x, p->x);
111  }
112 
113  /* Multiply by cofactor. */
114  ed_dbl(p, p);
115  ed_dbl(p, p);
116  ed_dbl(p, p);
117  ed_norm(p, p);
118 
119 #if ED_ADD == EXTND
120  fp_mul(p->t, p->x, p->y);
121 #endif
122  p->norm = 1;
123  }
124  CATCH_ANY {
125  THROW(ERR_CAUGHT);
126  }
127  FINALLY {
128  bn_free(h);
129  fp_free(t);
130  fp_free(u);
131  fp_free(v);
132  }
133 }
