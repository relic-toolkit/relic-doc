40void ed_map_ell2_5mod8(ed_t p, fp_t t) {
41 bn_t h;
42 fp_t tv1, tv2, tv3, tv4, tv5;
43 bn_null(h);
44 fp_null(tv1);
45 fp_null(tv2);
46 fp_null(tv3);
47 fp_null(tv4);
48 fp_null(tv5);
49
50 /* aliases to make code easier to read */
51 ctx_t *ctx = core_get();
52 dig_t *c_2exp = ctx->ed_map_c[0];
53 dig_t *sqrt_M1 = ctx->ed_map_c[1];
54 dig_t *sqrt_M486664 = ctx->ed_map_c[2];
55 dig_t *c_486662 = ctx->ed_map_c[3];
56
57 RLC_TRY {
58 bn_new(h);
59 fp_new(tv1);
60 fp_new(tv2);
61 fp_new(tv3);
62 fp_new(tv4);
63 fp_new(tv5);
64
65 /* set h = (p - 5) / 8 */
66 h->used = RLC_FP_DIGS;
67 h->sign = RLC_POS;
68 dv_copy(h->dp, fp_prime_get(), RLC_FP_DIGS); /* p */
69 bn_sub_dig(h, h, 5); /* p - 5 */
70 bn_rsh(h, h, 3); /* (p - 5) / 8 */
71
72 /* start evaluating map */
73 fp_sqr(tv1, t);
74 fp_dbl(tv1, tv1);
75 fp_add_dig(tv3, tv1, 1);
76 fp_sqr(tv2, tv3);
77 fp_mul(p->z, tv2, tv3);
78
79 /* compute numerator of g(x1) */
80 fp_sqr(tv4, c_486662);
81 fp_mul(tv4, tv4, tv1);
82 fp_sub(tv4, tv4, tv2);
83 fp_mul(tv4, tv4, c_486662);
84
85 /* compute divsrqt */
86 fp_sqr(tv3, p->z);
87 fp_sqr(tv2, tv3);
88 fp_mul(tv3, tv3, p->z);
89 fp_mul(tv3, tv3, tv4);
90 fp_mul(tv2, tv2, tv3);
91 fp_exp(tv2, tv2, h);
92 fp_mul(tv2, tv2, tv3);
93
94 /* figure out which sqrt we should keep */
95 fp_mul(p->y, tv2, sqrt_M1);
96 fp_sqr(p->x, tv2);
97 fp_mul(p->x, p->x, p->z);
98 {
99 const int e1 = fp_cmp(p->x, tv4);
100 fp_copy_sec(p->y, tv2, e1 == RLC_EQ);
101 } /* e1 goes out of scope */
102
103 /* compute numerator of g(x2) */
104 fp_mul(tv3, tv2, t);
105 fp_mul(tv3, tv3, c_2exp);
106 fp_mul(tv5, tv3, sqrt_M1);
107
108 /* figure out which sqrt we should keep */
109 fp_mul(p->x, tv4, tv1);
110 fp_sqr(tv2, tv3);
111 fp_mul(tv2, tv2, p->z);
112 {
113 const int e2 = fp_cmp(p->x, tv2);
114 fp_copy_sec(tv5, tv3, e2 == RLC_EQ);
115 } /* e2 goes out of scope */
116
117 /* figure out whether we wanted y1 or y2 and x1 or x2 */
118 fp_sqr(tv2, p->y);
119 fp_mul(tv2, tv2, p->z);
120 {
121 const int e3 = fp_cmp(tv2, tv4);
122 fp_set_dig(p->x, 1);
123 fp_copy_sec(p->x, tv1, e3 != RLC_EQ);
124 fp_mul(p->x, p->x, c_486662);
125 fp_neg(p->x, p->x);
126 fp_copy_sec(p->y, tv5, e3 != RLC_EQ);
127
128 /* fix sign of y */
129 fp_prime_back(h, p->y);
130 const int e4 = bn_get_bit(h, 0);
131 fp_neg(tv2, p->y);
132 fp_copy_sec(p->y, tv2, (e3 == RLC_EQ) ^ (e4 == 1));
133 } /* e3 and e4 go out of scope */
134 fp_add_dig(p->z, tv1, 1);
135
136 /* convert to an Edwards point */
137 /* tmp1 = xnumerator = sqrt_M486664 * x */
138 fp_mul(tv1, p->x, sqrt_M486664); /* xn = sqrt(-486664) * x */
139 fp_mul(tv2, p->y, p->z); /* xd = y * z */
140 fp_sub(tv3, p->x, p->z); /* yn = x - z */
141 fp_add(tv4, p->x, p->z); /* yd = x + z */
142
143 fp_mul(p->z, tv2, tv4);
144 fp_mul(p->x, tv1, tv4);
145 fp_mul(p->y, tv2, tv3);
146 {
147 /* exceptional case: either denominator == 0 */
148 const int e4 = fp_is_zero(p->z);
149 fp_set_dig(tv5, 1);
150 dv_copy_sec(p->x, p->z, RLC_FP_DIGS, e4); /* set x to 0 */
151 dv_copy_sec(p->y, tv5, RLC_FP_DIGS, e4);
152 dv_copy_sec(p->z, tv5, RLC_FP_DIGS, e4);
153 } /* e4 goes out of scope */
154
155 /* clear denominator / compute extended coordinates if necessary */
156#if ED_ADD == EXTND || ED_ADD == PROJC
157 p->coord = PROJC;
158#if ED_ADD == EXTND
159 /* extended coordinates: T * Z == X * Y */
160 fp_mul(p->t, p->x, p->y);
161 fp_mul(p->x, p->x, p->z);
162 fp_mul(p->y, p->y, p->z);
163 fp_sqr(p->z, p->z);
164#endif /* ED_ADD == EXTND */
165#else /* ED_ADD == BASIC */
166 fp_inv(tv1, p->z);
167 fp_mul(p->x, p->x, tv1);
168 fp_mul(p->y, p->y, tv1);
169 fp_set_dig(p->z, 1);
170 p->coord = BASIC;
171#endif /* ED_ADD */
172 }
173 RLC_CATCH_ANY {
174 RLC_THROW(ERR_CAUGHT)
175 }
176 RLC_FINALLY {
177 bn_free(h);
178 fp_free(tv1);
179 fp_free(tv2);
180 fp_free(tv3);
181 fp_free(tv4);
182 fp_free(tv5);
183 }
184}

190void ed_map_dst(ed_t p, const uint8_t *msg, size_t len, const uint8_t *dst,
191 size_t dst_len) {
192 bn_t k;
193 fp_t t;
194 ed_t q;
195 /* enough space for two field elements plus extra bytes for uniformity */
196 const int len_per_elm = (FP_PRIME + ed_param_level() + 7) / 8;
197 uint8_t *pseudo_random_bytes = RLC_ALLOCA(uint8_t, 2 * len_per_elm);
198
199 bn_null(k);
200 fp_null(t);
201 ed_null(q);
202
203 RLC_TRY {
204 bn_new(k);
205 fp_new(t);
206 ed_new(q);
207
208 /* pseudorandom string */
209 md_xmd(pseudo_random_bytes, 2 * len_per_elm, msg, len, dst, dst_len);
210
211#define ED_MAP_CONVERT_BYTES(IDX) \
212 do { \
213 bn_read_bin(k, pseudo_random_bytes + IDX * len_per_elm, len_per_elm); \
214 fp_prime_conv(t, k); \
215 } while (0)
216
217 /* first map invocation */
218 ED_MAP_CONVERT_BYTES(0);
219 ed_map_ell2_5mod8(p, t);
220
221 /* second map invocation */
222 ED_MAP_CONVERT_BYTES(1);
223 ed_map_ell2_5mod8(q, t);
224
225#undef ED_MAP_CONVERT_BYTES
226
227 ed_add(p, p, q);
228
229 /* clear cofactor */
230 switch (ed_param_get()) {
231 case CURVE_ED25519:
232 ed_dbl(p, p);
233 ed_dbl(p, p);
234 ed_dbl(p, p);
235 break;
236 default:
237 RLC_THROW(ERR_NO_VALID);
238 break;
239 }
240
241 ed_norm(p, p);
242#if ED_ADD == EXTND
243 fp_mul(p->t, p->x, p->y);
244#endif
245 p->coord = BASIC;
246 }
247 RLC_CATCH_ANY {
248 RLC_THROW(ERR_CAUGHT);
249 }
250 RLC_FINALLY {
251 bn_free(k);
252 fp_free(t);
253 ed_free(q);
254 RLC_FREE(pseudo_random_bytes);
255 }
256}
